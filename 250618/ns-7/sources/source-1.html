 package vn.momo.compose.payment.bankOtp.presentation.model
 
 import kotlinx.serialization.SerialName
 import kotlinx.serialization.Serializable
 import kotlinx.serialization.json.JsonObject
 import kotlinx.serialization.json.contentOrNull
 import kotlinx.serialization.json.jsonPrimitive
 import vn.momo.compose.payment.bankOtp.domain.entities.EOtpType
 import vn.momo.compose.payment.bankOtp.domain.entities.PaymentBankOtpConfirmExtras
 
 
 @Serializable
 data class PaymentBankOtpParams(
     @SerialName("bankName") val bankNameParams: String? = null,
     val initSource: JsonObject? = null,
     val confirmExtras: PaymentBankOtpConfirmExtras? = null,
     val paymentType: String? = null,
     val typeOTP: String? = null,
     val shouldHandleBackPress: Boolean? = null,
     val extras: JsonObject? = null,
     val otpInputOptions: JsonObject? = null,
 ) {
     val bankCode: String?
         get() = initSource?.let {
             it["bankCode"]?.jsonPrimitive?.contentOrNull
         }
     val bankName: String
         get() = bankNameParams ?: initSource?.let {
             listOf(
                 "bankName",
                 "title",
                 "displayName"
             ).firstNotNullOfOrNull { key -> it[key]?.jsonPrimitive?.contentOrNull }
         } ?: ""
 
     val guideOtpUrl: String?
         get() = confirmExtras?.guideOtpUrl
             ?: extras?.get("guideOtpUrl")?.jsonPrimitive?.contentOrNull
             ?: null
 
     val otpType: EOtpType
         get() = when (typeOTP) {
             "D-OTP" -> EOtpType.DOTP
             "SMS" -> EOtpType.SMS
             else -> EOtpType.SMS
         }
 
     val otpURL: String get() = confirmExtras?.otpURL ?: ""
 }
