 package vn.momo.compose.payment.payx.presentation.viewModel.reducers
 
 import vn.momo.compose.payment.base.domain.entities.MoneySource
 import vn.momo.compose.payment.base.domain.entities.uiKit.PayXComponent
 import vn.momo.compose.payment.base.presentation.viewModel.reducer.Reducer
 import vn.momo.compose.payment.payx.domain.entities.PayXResult
 import vn.momo.compose.payment.payx.domain.entities.reward.RewardConfig
 import vn.momo.compose.payment.payx.presentation.viewModel.reducers.RewardReducer.RewardEffect.UpdateCoinToSof
 import vn.momo.core.features.annotation.Immutable
 
 class RewardReducer :
     Reducer<RewardReducer.RewardState, RewardReducer.RewardAction, RewardReducer.RewardEffect> {
 
     @Immutable
     data class RewardState(
         /*
         Success: show
         Fail: hide
         Loading: loading
          */
         val missions: PayXResult<List<PayXComponent>, Throwable> = PayXResult.Fail(error = Throwable()),
         val coins: Map<MoneySource, Int> = emptyMap(),
         val missionId: List<String> = emptyList()
     ) : Reducer.State
 
     @Immutable
     sealed class RewardAction : Reducer.Action {
         object StartLoading : RewardAction()
         data class UpdateConfig(val config: RewardConfig) : RewardAction()
         object ResetConfig : RewardAction()
     }
 
     @Immutable
     sealed class RewardEffect : Reducer.Effect {
         data class UpdateCoinToSof(val reward: Map<MoneySource, Int>) : RewardEffect()
     }
 
 
     override fun reduce(
         prev: RewardState,
         action: RewardAction
     ): Pair<RewardState, RewardEffect?> {
         return when (action) {
             is RewardAction.UpdateConfig -> prev.copy(
                 missions = if (action.config.missionSummaryUi.isNotEmpty()) PayXResult.Success(
                     action.config.missionSummaryUi
                 ) else PayXResult.Fail(error = Throwable()),
                 coins = action.config.rewardBySof,
                 missionId = action.config.missionIds
             ) to UpdateCoinToSof(action.config.rewardBySof)
 
             RewardAction.ResetConfig -> RewardState() to UpdateCoinToSof(
                 emptyMap()
             )
 
             RewardAction.StartLoading -> prev.copy(
                 missions = if (prev.missions is PayXResult.Success) PayXResult.Loading else prev.missions
             ) to null
         }
     }
 }
