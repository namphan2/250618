 package vn.momo.compose.payment.payx.presentation.viewModel.reducers
 
 import kotlinx.serialization.json.JsonObject
 import vn.momo.compose.payment.base.presentation.viewModel.reducer.Reducer
 import vn.momo.compose.payment.payx.domain.entities.autodebit.AutoDebitItem
 import vn.momo.compose.utils.toJsonObject
 import vn.momo.core.features.annotation.Immutable
 
 class AutoDebitReducer :
     Reducer<AutoDebitReducer.AutoDebitState, AutoDebitReducer.AutoDebitAction, AutoDebitReducer.AutoDebitEffect> {
     @Immutable
     data class AutoDebitState(
         val items: List<AutoDebitItem> = emptyList(),
         val serviceInfo: JsonObject? = null
     ) : Reducer.State
 
     @Immutable
     sealed class AutoDebitAction : Reducer.Action {
         data class UpdateData(val items: List<AutoDebitItem>?) : AutoDebitAction()
         data class ChangeToggleItem(val index: Int) : AutoDebitAction()
         data class OnClickTnC(val title: String, val url: String) : AutoDebitAction()
         data class UpdateSkipLoading(val skipLoading: Boolean?): AutoDebitAction()
     }
 
     @Immutable
     sealed class AutoDebitEffect : Reducer.Effect {
         data class OnConfigUpdate(val serviceInfo: JsonObject?) : AutoDebitEffect()
         data class OnClickTnC(val title: String, val url: String) : AutoDebitEffect()
         data class OnSkipLoadingUpdate(val skipLoading: Boolean?) : AutoDebitEffect()
     }
 
     override fun reduce(
         prev: AutoDebitState, action: AutoDebitAction
     ): Pair<AutoDebitState, AutoDebitEffect?> {
         return when (action) {
             is AutoDebitAction.UpdateData -> onConfigUpdate(prev, action)
             is AutoDebitAction.ChangeToggleItem -> onToggle(prev, action)
             is AutoDebitAction.OnClickTnC -> prev to AutoDebitEffect.OnClickTnC(action.title, action.url)
             is AutoDebitAction.UpdateSkipLoading -> prev to AutoDebitEffect.OnSkipLoadingUpdate(action.skipLoading)
         }
     }
 
     private fun onConfigUpdate(
         prev: AutoDebitState, action: AutoDebitAction.UpdateData
     ): Pair<AutoDebitState, AutoDebitEffect?> {
         val serviceInfo =
             action.items?.filter { it.component.toggle && !it.itemId.isNullOrBlank() }?.associate {
                 if (it.subItemId.isNullOrBlank()) "${it.itemId}" to it.serviceInfo
                 else "${it.itemId}_${it.subItemId}" to it.serviceInfo
             }?.toJsonObject()
 
         return prev.copy(
             items = action.items ?: emptyList(), serviceInfo = serviceInfo
         ) to AutoDebitEffect.OnConfigUpdate(serviceInfo)
     }
 
     private fun onToggle(
         prev: AutoDebitState, action: AutoDebitAction.ChangeToggleItem
     ): Pair<AutoDebitState, AutoDebitEffect?> {
         val items = prev.items.mapIndexed { index, prevItem ->
             if (index == action.index) prevItem.copy(
                 component = prevItem.component.copy(toggle = !prev.items[index].component.toggle)
             ) else prevItem
         }
 
         val serviceInfo =
             items.filter { it.component.toggle && !it.itemId.isNullOrBlank() }.associate {
                 if (it.subItemId.isNullOrBlank()) "${it.itemId}" to it.serviceInfo
                 else "${it.itemId}_${it.subItemId}" to it.serviceInfo
             }.toJsonObject()
 
         return prev.copy(
             items = items, serviceInfo = serviceInfo
         ) to AutoDebitEffect.OnConfigUpdate(serviceInfo)
     }
 }
