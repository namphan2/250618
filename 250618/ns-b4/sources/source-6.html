 fun formatToDDMMYYYY(time: Double): String {
     val date = GMTDate(time.tryForceToLong())
     val month = ("0" + (date.month.ordinal + 1)).takeLast(2)
     val day = ("0" + date.dayOfMonth).takeLast(2)
     val year = date.year
     return "${day}/${month}/${year}"
 }
