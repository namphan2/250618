 @OptIn(ExperimentalMaterial3Api::class)
 @Composable
 fun PromotionBottomSheet(
     isVisible: Boolean = false,
     onDismiss: () -> Unit = {},
     sheetState: SheetState,
     headerTitle: String? = "",
     headerTransparent: Boolean? = false,
     contentWindowInsets: @Composable () -> WindowInsets = { WindowInsets(bottom = 0.dp) },
     content: @Composable () -> Unit = {},
 ) {
     val headerColor = if (headerTransparent == true) {
         AppTheme.current.colors.background.default
     } else {
         AppTheme.current.colors.background.surface
     }
 
     if (isVisible) {
         ModalBottomSheet(
             onDismissRequest = onDismiss,
             sheetState = sheetState,
             dragHandle = {
                 Box(
                     modifier = Modifier
                         .fillMaxWidth()
                         .background(color = headerColor)
                         .padding(top = Spacing.S, bottom = Spacing.XS),
                     contentAlignment = Alignment.Center,
                 ) {
                     Box(
                         modifier = Modifier
                             .height(Spacing.XS)
                             .width(40.dp)
                             .background(
                                 color = Colors.black_06,
                                 shape = RoundedCornerShape(Spacing.XS)
                             )
                     )
                 }
             },
             properties = ModalBottomSheetProperties(shouldDismissOnBackPress = true),
             shape = RoundedCornerShape(topStart = Radius.M, topEnd = Radius.M),
             contentWindowInsets = contentWindowInsets,
         ) {
             Column(
                 modifier = Modifier
                     .fillMaxWidth()
                     .background(color = headerColor)
             ) {
                 Row(
                     modifier = Modifier
                         .fillMaxWidth()
                         .padding(vertical = Spacing.L, horizontal = Spacing.M),
                     verticalAlignment = Alignment.CenterVertically,
                     horizontalArrangement = Arrangement.spacedBy(Spacing.S),
                 ) {
                     Spacer(modifier = Modifier.size(Spacing.L))
                     Text(
                         text = headerTitle ?: "",
                         style = Typography.headerDefaultBold,
                         color = Colors.black_17,
                         modifier = Modifier.weight(1f),
                         textAlign = TextAlign.Center,
                     )
                     Icon(
                         source = "navigation_close",
                         size = Spacing.XL,
                         modifier = Modifier
                             .clickable(
                                 enabled = true,
                                 onClick = { onDismiss() },
                                 interactionSource = remember { MutableInteractionSource() },
                                 indication = null,
                             )
                     )
                 }
                 if (headerTransparent != true) {
                     Divider()
                 }
                 Column(
                     modifier = Modifier
                         .fillMaxWidth()
                         .background(color = AppTheme.current.colors.background.default)
                 ) {
                     content()
                 }
             }
         }
