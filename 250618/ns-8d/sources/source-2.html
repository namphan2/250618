 @Composable
 fun SofListView(
     sofListState: PayXSofReducer.State,
     sofList: PersistentList<Sof>,
     isNoSof: Boolean,
     number: Int = 2,
     onClickShowAll: () -> Unit,
     onPressSof: (SOFItem) -> Unit,
     onPressCta: (SOFItem) -> Unit,
     isAllowAddSOF: Boolean = true,
     onAddSof: () -> Unit = {},
     onLoaded: () -> Unit,
     onToggleBalance: () -> Unit = {},
     isAllowAddCard: Boolean = true,
     onAddCreditCard: () -> Unit = {},
     ctaList: PersistentList<SofCTAItem>,
     onPressNotice: (SofCTAItem) -> Unit,
     isCollapsed: Boolean,
     onToggleExpand: () -> Unit
 ) {
     LaunchedEffect(Unit) {
         onLoaded()
     }
     val canShowIcon by remember(sofListState.list) {
         derivedStateOf { sofListState.list.any { it.shortDesc.isCurrency() } }
     }
     val actualSize = sofList.size + (if (isAllowAddSOF) 1 else 0) + (if (isAllowAddCard) 1 else 0)
 
     Column {
         ContentInformationCard(
             Modifier.background(
                 AppTheme.current.colors.background.surface, shape = RoundedCornerShape(12.dp)
             ).scaledPadding(12f), shape = RectangleShape, header = {
                 BasicCardHeader(
                     title = stringResource(Res.string.debit), titleExtra = {
                         if (canShowIcon) {
                             Icon(
                                 source = if (sofListState.isHideBalance) Images.ic_eye_close else Images.ic_eye_open,
                                 size = 24.dp,
                                 modifier = Modifier.semantics {
                                     testTag = "icon_eye"
                                     contentDescription = "icon_eye"
                                 }.clickable(onClick = onToggleBalance)
                             )
                         }
                     }
                 )
             }) {
             UserSofList(
                 list = sofList,
                 isNoSof = isNoSof,
                 selected = sofListState.selected,
                 isHideBalance = sofListState.isHideBalance,
                 number = number,
                 expanded = !isCollapsed,
                 onPressSof = onPressSof,
                 onPressCta = onPressCta,
                 isAllowAddSOF = isAllowAddSOF,
                 isAllowAddCard = isAllowAddCard,
                 onAddSof = onAddSof,
                 onAddCreditCard = onAddCreditCard,
                 iconBanks = sofListState.iconBankList,
                 ctaList = ctaList,
                 onPressNotice = onPressNotice
             )
         }
         if (actualSize > number) {
             CtaShowAllButton(
                 count = sofList.size - number, expanded = !isCollapsed
             ) {
                 onToggleExpand()
             }
         }
     }

 @Composable
 private fun UserSofList(
     list: PersistentList<Sof>,
     isNoSof: Boolean,
     isHideBalance: Boolean,
     selected: SOFItem?,
     number: Int = 2,
     expanded: Boolean = false,
     onPressSof: (SOFItem) -> Unit,
     onPressCta: (SOFItem) -> Unit,
     isAllowAddSOF: Boolean = true,
     isAllowAddCard: Boolean = true,
     onAddSof: () -> Unit = {},
     onAddCreditCard: () -> Unit = {},
     iconBanks: PersistentList<String>,
     ctaList: PersistentList<SofCTAItem>,
     onPressNotice: (SofCTAItem) -> Unit
 ) {
     val displayedItems = list.take(number)
     val hiddenItems = list.drop(number)
     Column(verticalArrangement = scaledSpacedBy(8f)) {
         if (isNoSof) {
             Surface(
                 Modifier.fillMaxWidth(),
                 shape = RoundedCornerShape(Radius.S),
                 color = Colors.blue_10,
                 border = BorderStroke(width = 0.5.dp, color = Colors.blue_07)
             ) {
                 Text(
                     text = stringResource(Res.string.noSof),
                     style = Typography.descriptionDefaultRegular,
                     modifier = Modifier.padding(12.dp)
                 )
             }
         }
         ctaList.forEach { cta ->
             SofCard.CtaItem(item = cta.action, onPress = { onPressNotice(cta) })
         }
         displayedItems.forEach { sof ->
             SofItemView(
                 sof,
                 isHideBalance = isHideBalance,
                 selected = selected,
                 onPress = onPressSof,
                 onPressCta = onPressCta
             )
         }
         if (list.size <= number) {
             key("addBank") {
                 SofCard.AddBank(onPress = onAddSof, iconBanks = iconBanks)
             }
         }
 
         PayXAnimatedVisibility(visible = expanded) {
             Column(verticalArrangement = scaledSpacedBy(8f)) {
                 hiddenItems.forEach { sof ->
                     SofItemView(
                         sof,
                         isHideBalance = isHideBalance,
                         selected = selected,
                         onPress = onPressSof,
                         onPressCta = onPressCta
                     )
                 }
                 if (list.size > number && isAllowAddSOF) {
                     key("addBank") {
                         SofCard.AddBank(onPress = onAddSof, iconBanks = iconBanks)
                     }
                 }
                 if (isAllowAddCard) {
                     key("addCard") {
                         SofCard.AddCard(onPress = onAddCreditCard)
                     }
                 }
             }
         }
 
     }
 }
 
 @Composable
 private fun SofItemView(
     sof: Sof,
     isHideBalance: Boolean,
     selected: SOFItem?,
     onPress: (SOFItem) -> Unit,
     onPressCta: (SOFItem) -> Unit
 ) {
     when (sof) {
         is SOFItem -> key(sof.uId) {
             SofCard.Default(
                 sof = sof,
                 isHideBalance = isHideBalance,
                 isSelected = sof.isMatchSource(
                     selected?.sourceToken, selected?.ID
                 ),
                 onPress = { onPress(sof) },
                 onPressCTA = { onPressCta(sof) })
         }
 
         is SOFGroup -> key(sof.parent.uId) {
             SofGroupView(
                 group = sof,
                 selected,
                 onSelect = onPress,
                 isHideBalance = isHideBalance
             )
         }
     }
 }
 
 @Composable
 fun PayLaterBlock(
     sofListState: PayXSofReducer.State,
     payLaters: PersistentList<SOFItem>,
     promoted: PromotedSof?,
     onPressSof: (SOFItem) -> Unit,
     onPressUnlock: (SOFItem) -> Unit,
     onSelectPromoted: (SOFItem?) -> Unit,
     onShowUnlock: () -> Unit,
     payLaterMessages: PersistentList<PayLaterConfigMessage>?
 ) {
     Column(
         Modifier.fillMaxWidth().padding(Spacing.M).animateContentSize(linearTween())
     ) {
         ContentInformationCard(color = Color.Transparent, shape = RectangleShape, header = {
             BasicCardHeader(
                 title = stringResource(Res.string.payLater),
             )
         }) {
             Column(
                 modifier = Modifier.fillMaxWidth().background(
                     Colors.orange_11, shape = RoundedCornerShape(
                         topStart = Spacing.M,
                         topEnd = Spacing.M,
                         bottomStart = Spacing.S,
                         bottomEnd = Spacing.S
                     )
                 )
             ) {
                 Column(
                     verticalArrangement = Arrangement.spacedBy(8.dp),
                     modifier = Modifier.fillMaxWidth()
                 ) {
                     payLaters.onEach { payLater ->
                         if (payLater.lockPayment != null) {
                             LaunchedEffect(Unit) { onShowUnlock() }
                             SofCard.LockedSof(
                                 sof = payLater,
                                 onPressCTA = { onPressUnlock(payLater) },
                                 modifier = Modifier.border(
                                     BorderStroke(0.5.dp, Colors.black_07),
                                     shape = RoundedCornerShape(Spacing.M)
                                 ).background(
                                     AppTheme.current.colors.background.surface,
                                     shape = RoundedCornerShape(Spacing.M)
                                 ).scaledPadding(8f),
                                 blur = false
                             )
                         } else {
                             SofCard.Default(
                                 sof = payLater,
                                 isHideBalance = sofListState.isHideBalance,
                                 isSelected = payLater.isMatchSource(
                                     sofListState.selected?.sourceToken, sofListState.selected?.ID
                                 ),
                                 onPress = { onPressSof(payLater) })
                         }
                     }
 
                     when (promoted) {
                         is PromotedSof.PromoteInstallment -> {
                             val installmentGroup = promoted.group.copy(
                                 parent = promoted.group.parent.copy(
                                     fullName = stringResource(
                                         Res.string.flexInstallment
                                     )
                                 )
                             )
                             SofGroupView(
                                 installmentGroup,
                                 selected = sofListState.selected,
                                 onSelect = onSelectPromoted,
                                 isHideBalance = false,
                                 modifier = Modifier.background(Colors.orange_11)
                             )
                         }
 
                         is PromotedSof.QuickActive -> {
                             QuickActiveBanner(
                                 promoted.group,
                                 showGroupChildren = payLaterMessages.isNullOrEmpty(),
                                 onActive = { onSelectPromoted(null) })
                         }
 
                         null -> {}
                     }
                 }
                 if (!payLaterMessages.isNullOrEmpty()) {
                     Column(modifier = Modifier.padding(Spacing.S)) {
                         payLaterMessages.forEach {
                             Row(
                                 modifier = Modifier.fillMaxWidth()
                                     .padding(vertical = Spacing.XXS),
                                 verticalAlignment = Alignment.CenterVertically
                             ) {
                                 if (!it.icon.isNullOrBlank()) {
                                     Image(it.icon, modifier = Modifier.size(Spacing.L))
                                     Spacer(modifier = Modifier.width(Spacing.XS))
                                 }
                                 Text(
                                     it.label,
                                     style = Typography.descriptionDefaultRegular,
                                     color = Colors.black_17,
                                     maxLines = 2,
                                     overflow = TextOverflow.Ellipsis,
                                     modifier = Modifier.weight(1f),
                                     textAlign = TextAlign.Start
                                 )
                             }
                         }
                     }
                 }
             }
         }
     }
