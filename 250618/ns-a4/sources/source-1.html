 fun getTrackingStatus(
     localType: VoucherLocalType?,
     selected: Boolean,
 ): String {
     return when {
         localType == VoucherLocalType.InvalidGiftMask -> "4" // collect_voucher_invalid
         localType == VoucherLocalType.Invalid && selected -> "5" // picked_invalid
         localType == VoucherLocalType.Invalid -> "2" // invalid
         localType == VoucherLocalType.GiftMask -> "3" // collect_voucher_valid
         localType == VoucherLocalType.VoucherCoin -> "6" // voucher_coin
         localType == VoucherLocalType.InvalidVoucherCoin -> "7" // invalid_voucher_coin
         else -> "1" // recommend
     }
 }
 
 fun IVoucherBase.toTrackingParams(selected: Boolean, index: Int): VoucherTrackingParams {
     return VoucherTrackingParams(
         giftType = typeId ?: "",
         campaignId = campaign ?: extra?.campaignId,
         slot = index,
         status = getTrackingStatus(localType, selected),
     )
 }

 fun VoucherDetailParams.toTrackingParams(): VoucherDetailTrackingParams {
     val voucherBase = voucher?.toVoucherBase()
 
     return VoucherDetailTrackingParams(
         giftType = giftType ?: voucherBase?.typeId ?: "",
         campaignId = campaignId ?: voucherBase?.campaign ?: voucherBase?.extra?.campaignId,
         status = getTrackingStatus(
             localType = voucherBase?.localType,
             selected = voucher?.selected ?: false,
         ),
         source = source,
     )
