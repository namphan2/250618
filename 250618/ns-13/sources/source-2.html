 fun interface IFirebaseTracer {
     fun track(eventName: String, eventParams: JsonObject?)
 }
 
 class PayXFirebaseTracer(private val api: IComposeApi) : IFirebaseTracer {
     override fun track(eventName: String, eventParams: JsonObject?) {
         api.requestWithFlow(
             APIs.trackEvent.name,
             listOf(eventName, eventParams ?: emptyJsonObject())
         )
             .take(1)
             .launchIn(CoroutineScope(Dispatchers.IO))
     }
