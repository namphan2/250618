 class SofReducerSingle : SofReducer<SofReducerSingle.State>() {
     @Immutable
     data class State(
         override val list: PersistentList<SOFItem> = persistentListOf(),
         override val selected: SOFItem? = null,
         override val isHideBalance: Boolean = false,
         override val groupOfSof: GroupOfSof = GroupOfSof(),
         override val version: UiVersion = UiVersion(0, "control"),
         override val moneyPayCTA: RecommendMoneyPayCTA? = null,
         override val iconBankList: PersistentList<String> = persistentListOf(),
         override val cta: PersistentList<SofCTAItem> = persistentListOf(),
         override val isCollapsed: Boolean = true
     ) : SofState
 
     override fun reduce(
         prev: State,
         action: Action
     ): Pair<State, Effect?> {
         return when (action) {
             is Action.UpdateShowBalance -> prev.copy(
                 isHideBalance = action.value,
             ) to null
 
             is Action.SetSofList -> prev.copy(
                 list = action.list.toPersistentList(),
                 selected = action.selected,
                 groupOfSof = syncGroupSof(
                     prev.groupOfSof,
                     originalList = action.list
                 )
             ) to null
 
             is Action.CashIn -> action.newSof.let { newSof ->
                 val newList =
                     prev.list.update(newSof) { current -> current.sourceToken == newSof.sourceToken }
                 prev.copy(
                     list = newList.toPersistentList(),
                     groupOfSof = syncGroupSof(
                         prev.groupOfSof,
                         originalList = newList
                     )
                 ) to if (newSof.balance >= 0) Effect.InitSofTransaction(newSof) else null
             }
 
             is Action.AddSofToCarousel -> prev.copy(
                 list = (prev.list + action.sof).toPersistentList(),
                 groupOfSof = syncGroupSof(
                     prev.groupOfSof,
                     originalList = prev.list + action.sof
                 )
             ) to null
 
             is Action.MoveOrAddSofToFirst -> {
                 val newSof = action.sof
                 val newListSof = when {
                     newSof.isInstallment -> prev.list.update(
                         newSof, 0
                     ) { sof -> sof.moneySource == newSof.moneySource && sof.isInstallment }
 
                     newSof.sourceToken.isEmpty() && !newSof.ID.isNullOrEmpty() -> {
                         prev.list.update(newSof, 0) { sof -> sof.ID == newSof.ID }
                     }
 
                     else -> prev.list.add(newSof, 0)
                 }
                 prev.copy(
                     list = newListSof.toPersistentList(),
                     groupOfSof = syncGroupSof(prev.groupOfSof, newListSof)
                 ) to Effect.InitSofTransaction(newSof)
             }
 
             is Action.ChangeSof -> {
                 val newSof = action.sof
                 prev.copy(
                     selected = newSof,
                     groupOfSof = if (prev.isCollapsed) rearrangeList(
                         prev.groupOfSof,
                         newSof,
                         if (prev.list.any { it.isSource(EnumMoneySource.Investment) }) 3 else 2
                     ) else prev.groupOfSof
                 ) to Effect.InitSofTransaction(
                     newSof, actor = action.actor,
                 )
             }
 
             Action.RefreshSof -> prev to prev.selected?.let { Effect.InitSofTransaction(it) }
 
             is Action.ActiveAndSelect -> {
                 val newList = prev.list.update(
                     action.sof,
                     0
                 ) { current -> current.sourceToken == action.sof.sourceToken }
                 prev.copy(
                     list = newList.toPersistentList(),
                     groupOfSof = syncGroupSof(prev.groupOfSof, newList)
                 ) to Effect.InitSofTransaction(
                     sof = action.sof,
                     actor = ChangeSofActor.ActiveSof
                 )
             }
 
             is Action.UpdateVersion -> prev.copy(version = action.ver) to null
 
             is Action.SetMoneyPayCTA -> prev.copy(moneyPayCTA = action.moneyPayCTA) to null
 
             Action.ResetPromoteSof -> prev.copy(
                 moneyPayCTA = null,
             ) to null
 
             is Action.UpdateMoneyPayCTA -> prev.copy(
                 moneyPayCTA = prev.moneyPayCTA?.copy(
                     btnTitle = action.title,
                     icon = if (action.isSelected) "ic_notifications_check_double.webp" else null,
                     isDisabled = action.isSelected,
                 )
             ) to null
 
             is Action.ToggleExpandSof -> {
                 val newState = !prev.isCollapsed
                 prev.copy(
                     isCollapsed = newState,
                     groupOfSof = if (newState) rearrangeList(
                         prev.groupOfSof,
                         prev.selected,
                         action.windowSize
                     ) else prev.groupOfSof
                 ) to null
             }
 
             is Action.SetIconList -> prev.copy(
                 iconBankList = action.icons.toPersistentList()
             ) to null
 
             else -> prev to null
         }
     }
 
     private fun syncGroupSof(prev: GroupOfSof, originalList: List<SOFItem>): GroupOfSof {
         data class Acc(
             val payLaterSofs: List<SOFItem> = emptyList(),
             val fundSofs: List<SOFItem> = emptyList(),
             val otherSofs: List<SOFItem> = emptyList(),
             val pocketSof: SOFItem? = null,
             val fundInsertIndex: Int = -1,
             val pocketSofIndex: Int = -1
         )
 
         val acc = originalList.foldIndexed(Acc()) { idx, acc, sof ->
             when (sof.moneySource) {
                 EnumMoneySource.PayLater ->
                     if (!sof.isInstallment) acc.copy(payLaterSofs = acc.payLaterSofs + sof) else acc
 
                 EnumMoneySource.Fund ->
                     acc.copy(
                         fundSofs = acc.fundSofs + sof,
                         fundInsertIndex = if (acc.fundInsertIndex == -1) idx else acc.fundInsertIndex
                     )
 
                 EnumMoneySource.Investment ->
                     if (acc.pocketSof == null)
                         acc.copy(pocketSof = sof, pocketSofIndex = idx)
                     else acc
 
                 else -> acc.copy(otherSofs = acc.otherSofs + sof)
             }
         }
 
         val prevOrderMap = prev.debit.mapIndexed { idx, sof ->
             when (sof) {
                 is SOFItem -> sof.uId to idx
                 is SOFGroup -> sof.parent.uId to idx
             }
         }.toMap()
 
         val fundGroup = acc.fundSofs.takeIf { it.isNotEmpty() }?.let { createFundGroup(it) }
         val debitWithoutPocket = buildList<Sof> {
             addAll(acc.otherSofs)
             fundGroup?.let {
                 val insertAt =
                     if (acc.fundInsertIndex in acc.otherSofs.indices) acc.fundInsertIndex else size
                 add(insertAt, it)
             }
         }
 
         val sortedDebit = debitWithoutPocket.sortedWith(
             compareByDescending<Sof> {
                 when (it) {
                     is SOFItem -> it.disableType == DisableType.Show
                     is SOFGroup -> it.parent.disableType == DisableType.Show
                 }
             }.thenBy {
                 when (it) {
                     is SOFItem -> prevOrderMap[it.uId] ?: Int.MAX_VALUE
                     is SOFGroup -> prevOrderMap[it.parent.uId] ?: Int.MAX_VALUE
                 }
             }
         )
 
         val finalDebit = acc.pocketSof?.let { pocket ->
             buildList<Sof> {
                 addAll(sortedDebit)
                 val insertIdx =
                     if (acc.pocketSofIndex in 0..2) acc.pocketSofIndex.coerceIn(0, size) else min(
                         2,
                         size
                     )
                 add(insertIdx, pocket)
             }
         } ?: sortedDebit
 
         return GroupOfSof(
             debit = finalDebit.toPersistentList(),
             payLater = acc.payLaterSofs.toPersistentList()
         )
     }
 
     private fun rearrangeList(
         group: GroupOfSof,
         selected: SOFItem?,
         windowSize: Int
     ): GroupOfSof {
         if (selected == null || selected.moneySource == EnumMoneySource.PayLater) {
             return group
         }
         val debitList = group.debit
         val selectedIdx = debitList.indexOfFirst {
             when (it) {
                 is SOFItem -> it.uId == selected.uId
                 is SOFGroup -> it.isSelected(selected.uId)
             }
         }
         val pocketIdx =
             debitList.indexOfFirst { it is SOFItem && it.isSource(EnumMoneySource.Investment) }
         if (selectedIdx < windowSize && pocketIdx < windowSize) {
             return group
         }
         val otherSofs = debitList.filterIndexed { idx, _ -> idx != selectedIdx && idx != pocketIdx }
         val newDebitList = buildList {
             addAll(otherSofs)
             val newSelected =
                 if (selectedIdx == -1 || selectedIdx < windowSize) selectedIdx.coerceIn(
                     0,
                     size
                 ) else 0
 
             if (newSelected != -1) {
                 add(newSelected, debitList[selectedIdx])
             }
             if (selectedIdx != pocketIdx) {
                 val newPocket =
                     (if (pocketIdx == -1 || pocketIdx < windowSize) pocketIdx else (windowSize - 1)).coerceIn(
                         0,
                         size
                     )
                 if (newPocket != -1) {
                     add(newPocket, debitList[pocketIdx])
                 }
             }
         }
 
         return group.copy(debit = newDebitList.toPersistentList())
     }
