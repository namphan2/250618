 import vn.momo.kits.const.scaleSize
 
 @Composable
 fun PaymentInfoCarousel(
     items: List<InvoiceItem>,
     shouldShowTid: Boolean,
     hasAppliedPromotion: Boolean,
     hideAmount: Boolean = false,
     screenWidth: Int,
 ) {
     val pagerState = rememberPagerState(pageCount = { items.size })
     var maxHeightPx by remember { mutableIntStateOf(0) }
     val density = LocalDensity.current
 
     val parentWidth = screenWidth - scaleSize(12f) * 2
     val itemWidth = if (items.size > 1) (parentWidth * 0.936).dp else parentWidth.dp
 
     HorizontalPager(
         userScrollEnabled = items.size > 1,
         state = pagerState,
         modifier = Modifier.fillMaxWidth(),
         pageSize = PageSize.Fixed(itemWidth),
         pageSpacing = scaleSize(8f).dp
     ) { page ->
         val item = items[page]
         Box(
             modifier = Modifier.onGloballyPositioned { coordinates ->
                 val h = coordinates.size.height
                 if (h > maxHeightPx) {
                     maxHeightPx = h
                 }
             }.animateContentSize(animationSpec = linearTween()).fillMaxWidth()
                 .then(if (maxHeightPx > 0) Modifier.height(with(density) { maxHeightPx.toDp() }) else Modifier),
             contentAlignment = Alignment.Center
         ) {
             PaymentInfoItem(
                 title = item.serviceName,
                 icon = item.icon,
                 invoice = item,
                 index = page,
                 length = items.size,
                 shouldShowTid = shouldShowTid,
                 hasAppliedPromotion = hasAppliedPromotion,
                 hideAmount = hideAmount,
             )
         }
     }
 }
