 @Composable
 fun PaymentInfoItem(
     title: String,
     icon: String?,
     invoice: InvoiceItem,
     index: Int = 0,
     length: Int = 1,
     shouldShowTid: Boolean,
     hasAppliedPromotion: Boolean,
     hideAmount: Boolean = false,
 ) {
     val backgroundColor = AppTheme.current.colors.background.surface
     val showBottomSheet = remember { mutableStateOf(false) }
 
     val invoiceInfos = buildList {
         addAll(invoice.info)
         if (shouldShowTid && invoice.visibleTid) {
             add(
                 InvoiceInfo(
                     title = stringResource(Res.string.transaction_id),
                     value = invoice.paymentTid.toString(),
                     key = "tid"
                 )
             )
         }
     }
 
     val billInvoiceInfos = buildList {
         addAll(invoiceInfos.take(2))
         if (invoice.visibleFee == true && !hideAmount && invoice.fee > 0) {
             add(
                 InvoiceInfo(
                     title = stringResource(Res.string.invoiceFee),
                     value = invoice.fee.toDouble().formatCurrency(),
                     key = "fee"
                 )
             )
         }
     }
 
     val btsInvoiceInfos = buildList {
         addAll(invoiceInfos)
         if (invoice.visibleFee == true && !hideAmount && invoice.fee > 0) {
             add(
                 InvoiceInfo(
                     title = stringResource(Res.string.invoiceFee),
                     value = invoice.fee.toDouble().formatCurrency(),
                     key = "fee"
                 )
             )
         }
     }
 
     val showCTA = invoiceInfos.size > 2
     val showQuantityChip = length > 1
 
     Column(
         modifier = Modifier.fillMaxSize().clip(
             shape = RoundedCornerShape(
                 bottomStart = Spacing.M,
                 bottomEnd = Spacing.M,
                 topStart = Spacing.XXS,
                 topEnd = Spacing.XXS
             )
         )
     ) {
         ScallopedTop(
             modifier = Modifier.fillMaxWidth(),
         )
         Column(
             modifier = Modifier.fillMaxSize().background(
                 backgroundColor, shape = RoundedCornerShape(
                     bottomStart = Spacing.M,
                     bottomEnd = Spacing.M,
                 )
             ).scaledPadding(horizontal = 12f),
         ) {
             PaymentInfoHeader(title = title,
                 icon = icon,
                 showCTA = showCTA,
                 index = index,
                 length = length,
                 showQuantityChip = showQuantityChip,
                 onClickDetail = {
                     if (billInvoiceInfos.isNotEmpty()) {
                         showBottomSheet.value = true
                     }
                 })
             DashedDivider()
             Spacer(modifier = Modifier.height(Spacing.S))
             billInvoiceInfos.forEachIndexed { _, item ->
                 PaymentInfoRow(
                     title = item.title,
                     value = item.value,
                     titleTextStyle = item.titleTextStyle,
                     valueTextStyle = item.valueTextStyle,
                 )
             }
             if (!hideAmount) {
                 PaymentInfoTotalAmount(
                     totalAmountInvoice = invoice.totalAmountInvoice,
                     subTotalAmountInvoice = invoice.subTotalAmountInvoice,
                     isShowSubTotal = invoice.totalAmountInvoice != invoice.subTotalAmountInvoice && hasAppliedPromotion
                 )
             }
             Spacer(modifier = Modifier.height(Spacing.XS))
         }
         PaymentInfoBottomSheet(state = showBottomSheet,
             invoice = invoice,
             invoiceInfos = btsInvoiceInfos,
             hasAppliedPromotion = hasAppliedPromotion,
             hideAmount = hideAmount,
             onClose = {})
     }
 }
