 import kotlinx.coroutines.CoroutineScope
 import kotlinx.coroutines.flow.MutableStateFlow
 import kotlinx.coroutines.flow.SharingStarted
 import kotlinx.coroutines.flow.stateIn
 import kotlinx.coroutines.flow.update
 
 class PState<Data>(
     scope: CoroutineScope,
     private val initialState: Data,
 ) {
     val _state = MutableStateFlow(initialState)
 
     val state =
         _state.stateIn(scope, SharingStarted.WhileSubscribed(stopTimeoutMillis = 200), initialState)
 
     val value
         get() = state.value
 
     val _value
         get() = _state.value
 
     fun update(function: (Data) -> Data) {
         _state.update(function)
     }
 
     fun update(value: Data) {
         _state.update { value }
     }
 
     fun reset() {
         _state.update { initialState }
     }
 }
