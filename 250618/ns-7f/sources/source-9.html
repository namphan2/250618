 @Composable
 fun QuickActiveBanner(group: SOFGroup, showGroupChildren: Boolean = true, onActive: () -> Unit) {
     val screenWidth = getScreenDimensions().screenWidth
     val scrollState = rememberScrollState()
     val parent = group.parent
     Column(
         Modifier.noFeedbackClickable(onClick = onActive), verticalArrangement = scaledSpacedBy(8f)
     ) {
         InfoCard(
             title = stringResource(Res.string.payLaterSof),
             titleExtra = {
                 Surface(shape = RoundedCornerShape(4.dp), color = Colors.blue_11) {
                     Text(
                         text = stringResource(Res.string.notUsed),
                         style = Typography.descriptionXsRegular,
                         color = Colors.blue_04,
                         modifier = Modifier.padding(horizontal = 8.dp, vertical = 2.dp)
                     )
                 }
             },
             titleStyle = InfoTextStyle(
                 font = Typography.descriptionDefaultRegular,
                 color = AppTheme.current.colors.text.hint
             ),
             description = stringResource(Res.string.quickActiveDesc),
             descriptionStyle = InfoTextStyle(
                 font = Typography.headerXsSemibold, color = AppTheme.current.colors.text.default
             ),
             leading = {
                 Image(source = parent.logo ?: "", modifier = Modifier.size(24.dp))
             },
             extra = {},
             trailing = {
                 Row(verticalAlignment = Alignment.CenterVertically) {
                     RewardText(
                         coin = parent.reward ?: 0,
                         visible = parent.reward != null && parent.reward > 0
                     )
                     Spacer(modifier = Modifier.scaledWidth(8f))
                     RadioButton(selected = false)
                 }
             },
             onPress = onActive,
             modifier = Modifier.border(
                 BorderStroke(0.5.dp, Colors.black_07), shape = RoundedCornerShape(Spacing.M)
             ).background(
                 AppTheme.current.colors.background.surface, shape = RoundedCornerShape(
                     Spacing.M
                 )
             ).scaledPadding(8f)
         )
         if (showGroupChildren) {
             Row(
                 Modifier.padding(top = Spacing.XS, bottom = Spacing.S)
                     .horizontalScroll(scrollState),
                 horizontalArrangement = scaledSpacedBy(8f),
                 verticalAlignment = Alignment.CenterVertically
             ) {
                 group.children.onEachIndexed { index, installment ->
                     SofCard.MiniSof(
                         modifier = Modifier.widthIn(max = (screenWidth / 3).dp).padding(
                             start = if (index == 0) Spacing.S else 0.dp,
                             end = if (index == group.children.size - 1) Spacing.S else 0.dp
                         ), sof = installment, isSelected = false, isHideBalance = false
                     )
                 }
             }
         }
     }
 }
