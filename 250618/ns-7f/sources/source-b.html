 @Composable
 fun TransactionDetailV2(
     invoice: InvoiceItem,
     initialExpanded: Boolean = true,
     shouldShowTid: Boolean,
     hasAppliedPromotion: Boolean,
     hideAmount: Boolean = false,
     onChangeExpandState: (String) -> Unit = {},
 ) {
     val isPayForService = invoice.serviceName.isNotBlank()
     val invoiceInfos = buildList {
         addAll(invoice.info)
         if (shouldShowTid && invoice.visibleTid) {
             add(
                 InvoiceInfo(
                     title = stringResource(Res.string.transaction_id),
                     value = invoice.paymentTid.toString(),
                     key = "tid"
                 )
             )
         }
         if (invoice.visibleFee == true && !hideAmount && invoice.fee > 0) {
             add(
                 InvoiceInfo(
                     title = stringResource(Res.string.invoiceFee),
                     value = invoice.fee.toDouble().formatCurrency(),
                     key = "fee"
                 )
             )
         }
     }
     val headerInfos = invoiceInfos.take(2)
     val contentInfos = invoiceInfos.drop(2)
     ExpandableCard(
         modifier = Modifier.clip(shape = RoundedCornerShape(12.dp)),
         title = if (isPayForService) invoice.serviceName else stringResource(Res.string.transactionDetail),
         defaultState = initialExpanded,
         cardTitleColor = Colors.pink_09,
         icon = {
             Icon(
                 source = "arrow_chevron_down_small",
                 size = 24.dp,
                 color = AppTheme.current.colors.text.default,
             )
         },
         header = {
             headerInfos.forEachIndexed { index, item ->
                 InvoiceItemUiView(
                     modifier = Modifier.then(
                         if (index % 2 != 0) Modifier.background(Colors.black_02)
                         else Modifier.background(AppTheme.current.colors.background.surface)
                     )
                         .scaledPadding(horizontal = 12f, vertical = 8f),
                     key = item.title,
                     value = item.value,
                     titleTextStyle = item.titleTextStyle,
                     valueTextStyle = item.valueTextStyle,
                     isShowBorderTop = false,
                     isShowBorderBottom = false,
                 )
             }
         },
         content = {
             contentInfos.forEachIndexed { index, item ->
                 val absoluteIndex = index + headerInfos.size
                 InvoiceItemUiView(
                     modifier = Modifier.then(
                         if (absoluteIndex % 2 != 0) Modifier.background(Colors.black_02)
                         else Modifier.background(AppTheme.current.colors.background.surface)
                     )
                         .scaledPadding(horizontal = 12f, vertical = 8f),
                     key = item.title,
                     value = item.value,
                     titleTextStyle = item.titleTextStyle,
                     valueTextStyle = item.valueTextStyle,
                     isShowBorderTop = false,
                     isShowBorderBottom = false,
                 )
             }
         },
         footer = { isExpanded ->
             val footerIndex =
                 if (isExpanded) invoiceInfos.size else headerInfos.size
             if (!hideAmount) {
                 InvoiceTotalAmount(
                     modifier = Modifier
                         .then(
                             if (footerIndex % 2 != 0) Modifier.background(Colors.black_02)
                             else Modifier.background(AppTheme.current.colors.background.surface)
                         )
                         .scaledPadding(horizontal = 12f, vertical = 8f),
                     totalAmountInvoice = invoice.totalAmountInvoice,
                     subTotalAmountInvoice = invoice.subTotalAmountInvoice,
                     isShowSubTotal = invoice.totalAmountInvoice != invoice.subTotalAmountInvoice && hasAppliedPromotion
                 )
             }
         },
         onChange = { newState ->
             onChangeExpandState(if (newState) "expand" else "collapse")
         })
 }
