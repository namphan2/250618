 @Composable
 fun InvoiceFeeSummary(
     fee: Int = 0,
     feeInfo: List<FeeDetail> = emptyList(),
     feeInfoHighlight: String
 ) {
     var expandedState by remember { mutableStateOf(false) }
     Column(
         modifier = Modifier.fillMaxWidth().padding(vertical = Spacing12)
     ) {
         Row(
             modifier = Modifier.fillMaxWidth(),
             horizontalArrangement = Arrangement.SpaceBetween
         ) {
             Text(
                 text = stringResource(Res.string.transactionFee),
                 style = Typography.bodyDefaultRegular,
                 color = Colors.black_12
             )
             Row(
                 verticalAlignment = Alignment.CenterVertically,
                 horizontalArrangement = Arrangement.spacedBy(Spacing.S),
                 modifier = Modifier.noFeedbackClickable {
                     if (feeInfo.isNotEmpty()) {
                         expandedState = !expandedState
                     }
                 }
             ) {
                 Text(
                     text = fee.toDouble()
                         .formatCurrencyString(fallback = stringResource(Res.string.free)),
                     style = Typography.headerSSemibold,
                     color = Colors.black_17,
                     textAlign = TextAlign.End,
                 )
 
                 if (feeInfo.isNotEmpty()) {
                     val rotationState by animateFloatAsState(
                         targetValue = if (expandedState) 180f else 0f, label = "FeeRotate",
                         animationSpec = linearTween()
                     )
                     Icon(
                         source = "arrow_chevron_down",
                         size = 24.dp,
                         modifier = Modifier
                             .rotate(rotationState),
                     )
                 }
             }
         }
         feeInfoHighlight.takeIf { it.isNotEmpty() }?.let {
             Information(
                 description = it
             )
             Spacer(modifier = Modifier.height(Spacing.S))
         }
         PayXAnimatedVisibility(visible = expandedState) {
             Column(
                 modifier = Modifier.fillMaxWidth(),
                 verticalArrangement = Arrangement.spacedBy(Spacing.M)
             ) {
                 feeInfo.forEach {
                     InvoiceItemUiView(
                         key = it.title,
                         value = it.value,
                         isShowBorderBottom = false,
                         isShowBorderTop = true
                     )
                 }
             }
         }
     }
 }
