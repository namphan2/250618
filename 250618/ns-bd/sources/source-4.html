 package vn.momo.compose.payment.promotion.presentation.viewModel.reducers
 
 import vn.momo.compose.payment.base.presentation.viewModel.reducer.Reducer
 import vn.momo.compose.payment.promotion.domain.entities.IVoucherBase
 import vn.momo.compose.payment.promotion.domain.entities.VoucherCoinInfo
 import vn.momo.core.features.annotation.Immutable
 
 class VoucherCoinReducer :
     Reducer<VoucherCoinReducer.State, VoucherCoinReducer.Action, VoucherCoinReducer.Effect> {
     @Immutable
     data class State(
         val isVoucherCoinBottomSheetVisible: Boolean = false,
         val coinInfo: VoucherCoinInfo? = null,
         val selectedVoucherCoin: IVoucherBase? = null,
         val claimedVouchers: List<IVoucherBase> = emptyList(),
         val isClaiming: Boolean = false,
         val claimingId: String? = null,
         val claimError: String? = null,
     ) : Reducer.State
 
     @Immutable
     sealed class Action : Reducer.Action {
         data class SetVoucherCoinState(
             val isVoucherCoinBottomSheetVisible: Boolean,
             val coinInfo: VoucherCoinInfo?,
             val selectedVoucherCoin: IVoucherBase?,
             val claimedVouchers: List<IVoucherBase> = emptyList(),
         ) : Action()
 
         data class SetClaimingVoucherState(
             val isClaiming: Boolean,
             val claimingId: String? = null,
             val errorMessage: String?,
         ) : Action()
     }
 
     @Immutable
     class Effect() : Reducer.Effect
 
     override fun reduce(prev: State, action: Action): Pair<State, Effect?> {
         return when (action) {
 
             is Action.SetVoucherCoinState -> prev.copy(
                 isVoucherCoinBottomSheetVisible = action.isVoucherCoinBottomSheetVisible,
                 coinInfo = action.coinInfo,
                 selectedVoucherCoin = action.selectedVoucherCoin,
                 claimedVouchers = action.claimedVouchers,
             ) to null
 
             is Action.SetClaimingVoucherState -> prev.copy(
                 isClaiming = action.isClaiming,
                 claimingId = action.claimingId,
                 claimError = action.errorMessage,
             ) to null
 
         }
     }
 }
