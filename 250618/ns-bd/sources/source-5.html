 package vn.momo.compose.payment.promotion.presentation.viewModel.reducers
 
 import vn.momo.compose.payment.base.presentation.viewModel.reducer.Reducer
 import vn.momo.compose.payment.promotion.domain.entities.IVoucherBase
 import vn.momo.compose.payment.promotion.domain.entities.VoucherBase
 import vn.momo.compose.payment.promotion.domain.entities.VoucherExtra
 import vn.momo.core.features.annotation.Immutable
 
 class VoucherDetailReducer :
     Reducer<VoucherDetailReducer.State, VoucherDetailReducer.Action, VoucherDetailReducer.Effect> {
     @Immutable
     data class State(
         val voucherData: IVoucherBase? = null,
         val collectedVoucher: IVoucherBase? = null,
         val isLoading: Boolean = true,
     ) : Reducer.State
 
     @Immutable
     sealed class Action : Reducer.Action {
         data class SetVoucherData(val voucherData: IVoucherBase?) : Action()
         data class SetIsLoading(val isLoading: Boolean) : Action()
         data class SetCollectedVoucher(val collectedVoucher: IVoucherBase?) : Action()
     }
 
     @Immutable
     class Effect() : Reducer.Effect
 
     override fun reduce(prev: State, action: Action): Pair<State, Effect?> {
         return when (action) {
             is Action.SetIsLoading -> prev.copy(
                 isLoading = action.isLoading
             ) to null
 
             is Action.SetVoucherData -> {
                 /**
                  * Keep extra data from the default voucher
                  * since detail API cannot check invalid voucher.
                  */
                 val remapVoucher = (action.voucherData as VoucherBase?)?.copy(
                     extra = (prev.voucherData?.extra ?: action.voucherData?.extra) as VoucherExtra?,
                 )
                 return prev.copy(
                     voucherData = remapVoucher,
                     isLoading = false,
                 ) to null
             }
 
             is Action.SetCollectedVoucher -> prev.copy(
                 collectedVoucher = action.collectedVoucher,
             ) to null
         }
     }
 }
