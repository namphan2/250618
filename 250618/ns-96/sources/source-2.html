 import androidx.compose.runtime.setValue
 import androidx.compose.runtime.staticCompositionLocalOf
 import androidx.compose.ui.unit.Dp
 import org.koin.compose.koinInject
 import vn.momo.compose.container.KitComposeContainer
 import vn.momo.compose.payment.di.PaymentKoinScope
 import vn.momo.compose.payment.payx.localization.PaymentLocalize
 import vn.momo.compose.payment.payx.localization.setComposeLanguage
 
 @Composable
 fun ComposeContainer(statusBarHeight: Dp? = null, content: @Composable () -> Unit) {
     val paymentLocalize =
         koinInject<PaymentLocalize>(scope = PaymentKoinScope.getScope(name = PaymentKoinScope.PAYMENT_BASE))
     var language by remember { mutableStateOf("") }
 
     LaunchedEffect(Unit) {
         paymentLocalize.lang.also {
             setComposeLanguage(it)
             language = it
         }
     }
 
     LocalizedScreen(language) {
         KitComposeContainer(statusBarHeight = statusBarHeight) {
             content()
         }
     }
 }
 
 val LocalLocalization = staticCompositionLocalOf { "en" }
 
 @Composable
 fun LocalizedScreen(language: String, content: @Composable () -> Unit) {
     CompositionLocalProvider(
         LocalLocalization provides language,
         content = content
     )
 }
