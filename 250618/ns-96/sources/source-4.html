 import androidx.compose.ui.Modifier
 import org.jetbrains.compose.resources.stringResource
 import vn.momo.compose.payment.payx.presentation.view.components.PayXWebView
 import vn.momo.compose.payment.payx.presentation.viewModel.NapasConfirmVM

 import vn.momo.compose.resources.Res

 @Composable
 fun NapasConfirmScreen(
     url: String,
     shouldHandleOnDone: Boolean,
     viewModel: NapasConfirmVM,
 ) {
 
     fun onGoBack() {
         viewModel.dispatch(NapasConfirmAction.OnGoBack(url))
     }
 
     fun onMessage(message: String?) {
         viewModel.dispatch(NapasConfirmAction.OnMessage(message, shouldHandleOnDone))
     }
 
 
     Screen(
         title = stringResource(Res.string.napasConfirmTitle),
         headerRight = null,
         goBack = { onGoBack() },
         scrollable = false,
         content = {
             PayXWebView(
                 modifier = Modifier.fillMaxSize(),
                 url,
                 onMessage = { msg -> onMessage(msg) }
             )
         }
     )
 }
