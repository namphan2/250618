 class SBNapasConfirmRemoteDataSource(
     private val remoteApi: IRemoteService,
     private val savedSingleBillData: SBSavedTransaction,
     private val clientIpAddress: IClientIpAddress,
 ) {
     @OptIn(ExperimentalCoroutinesApi::class)
     fun requestNapasConfirmUrlByToken(
         requestTokenHash: String,
         cardNumber: String,
         tranType: Int,
         ID: String?
     ): Flow<NapasConfirmUrlByTokenResponseDTO> {
         return buildGetUrlByTokenMessage(
             requestTokenHash,
             cardNumber,
             tranType,
             ID
         ).flatMapLatest { message ->
             message?.let {
                 remoteApi.post(it, NapasConfirmUrlByTokenResponseDTO.serializer())
                     .apiTrace(SBApiPerformanceEvent.GetNapasUrlByToken.event)
                     .map { it?.response }
                     .assertNonNullResponse("NAPAS_CASHIN_INIT_MSG no response")
             } ?: throw SingleBillError.DevError(
                 DevError.PurchaseIdNotFound.code,
                 "NAPAS_CASHIN_INIT_MSG message is null",
                 cause = Exception(this::class.simpleName)
             )
         }
     }
 
 
     @OptIn(ExperimentalCoroutinesApi::class)
     fun requestNapasConfirmUrlByCardInfo(napasInfo: NapasInfo): Flow<NapasConfirmUrlByCardInfoResponseDTO> {
         return buildGetUrlByCardInfoMessage(napasInfo).flatMapLatest { message ->
             message?.let {
                 remoteApi.post(it, NapasConfirmUrlByCardInfoResponseDTO.serializer())
                     .apiTrace(SBApiPerformanceEvent.GetNapasUrlByCardInfo.event)
                     .map { it?.response }
                     .assertNonNullResponse("NAPAS_CASHIN_INIT_MSG no response")
             } ?: throw SingleBillError.DevError(
                 DevError.PurchaseIdNotFound.code,
                 "NAPAS_CASHIN_INIT_MSG message is null",
                 cause = Exception(this::class.simpleName)
             )
         }
     }
 
     private fun buildGetUrlByTokenMessage(
         requestTokenHash: String,
         cardNumber: String,
         tranType: Int,
         ID: String?
     ): Flow<GetNapasConfirmUrlByTokenRequestDTO?> {
         val momoMsg = savedSingleBillData.data
         return momoMsg?.let {
             val extras = buildJsonObject {
                 put("paymentInfo", momoMsg.tranHisMsg?.extras?.rawPaymentInfo ?: emptyJsonObject())
                 put("paymentTranType", tranType)
                 put("ID", ID ?: momoMsg.tranHisMsg?.ID)
             }
             clientIpAddress.getIpAddress().map { clientIP ->
                 GetNapasConfirmUrlByTokenRequestDTO(
                     pinKeySessionID = savedSingleBillData.pinKeySessionId,
                     momoMsg = MoMoMsgGetUrlByTokenRequestDTO(
                         clientIP = clientIP,
                         amount = it.tranHisMsg?.amount,
                         requestTokenHash = requestTokenHash,
                         cardNumber = cardNumber,
                         serviceId = it.tranHisMsg?.serviceId,
                         merchantId = it.tranHisMsg?.merchantId,
                         extras = ModelConvertUtils.encodeToString(extras)
                     )
                 )
             }
         } ?: flowOf(null)
     }
 
     private fun buildGetUrlByCardInfoMessage(napasInfo: NapasInfo): Flow<GetNapasConfirmUrlByCardInfoRequestDTO?> {

                         amount = it.tranHisMsg?.amount,
                         serviceId = it.tranHisMsg?.serviceId,
                         merchantId = it.tranHisMsg?.merchantId,
                         cardNumber = napasInfo.cardNumber ?: "",
                         cardFullName = napasInfo.cardFullName ?: "",
                         cardIssueDate = napasInfo.cardIssueDate ?: "",
                         isSaveToken = napasInfo.isSaveToken,
                         partnerCode = napasInfo.partnerCode,
                         extras = it.tranHisMsg?.rawExtras ?: ""
                     )
                 )
             }
         } ?: flowOf(null)
     }
