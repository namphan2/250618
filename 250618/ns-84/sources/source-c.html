 class SuggestedSofBottomSheetController(
     private val params: Map<String, Any?>?,
     private val viewModel: PaymentSuggestedSofVM
 ) {
     fun onLoaded(props: SuggestedSofBottomSheetProps) {
         viewModel.onCreate()
         viewModel.setProps(props.shouldSuggestFirstSof)
     }
 
     fun generateProps(): SuggestedSofBottomSheetProps {
         val props = ModelConvertUtils.decode(
             SuggestedSofBottomSheetProps.serializer(),
             params?.tryGetMapNullable("props") ?: emptyMap()
         )
             ?: SuggestedSofBottomSheetProps(
                 isAllowAddSOF = true,
                 shouldSuggestFirstSof = false
             )
         return props
     }
 
     fun shouldClosePopup(props: SuggestedSofBottomSheetProps): Boolean {
         return props.shouldSuggestFirstSof && props.koinId.isNotEmpty() && !PaymentKoinScope.isExistScope(
             props.koinId
         )
     }
 
 }
