 package vn.momo.compose.payment.payx.presentation.viewModel.controller
 
 import kotlinx.serialization.json.JsonObject
 import kotlinx.serialization.json.jsonPrimitive
 import vn.momo.compose.payment.base.domain.entities.ErrorPopupCallback
 import vn.momo.compose.payment.payx.presentation.model.RefIdEvent
 
 fun shouldForceReInit(
     refId: String,
     refParams: JsonObject?,
     response: ErrorPopupCallback?,
 ): Boolean {
     return when (refId) {
         RefIdEvent.EKyc.id -> {
             verifyEKycResponse(
                 response,
                 refParams?.get("option")?.jsonPrimitive?.content
             )
         }
 
         RefIdEvent.EKycByFriend.id -> true
         else -> response?.confirmToPayment == true
     }
 }
 
 private fun verifyEKycResponse(response: ErrorPopupCallback?, option: String?): Boolean {
     val isFaceMatchingSuccess =
         response?.faceData?.status == "SUCCESS"
     val isOcrSuccess =
         response?.ocrData?.kycStatus == "APPROVED"
     val isNfcSuccess = response?.ocrData?.resultCode == 0
     return when (option) {
         "FULL" -> isFaceMatchingSuccess && isOcrSuccess && isNfcSuccess
         "NFC_FACE" -> isNfcSuccess && isFaceMatchingSuccess
         else -> isFaceMatchingSuccess || isOcrSuccess || isNfcSuccess
     }
 }
