 class SBConfirmTwoStepUseCase(
     private val navigator: INavigator,
     private val initCashInTransactionRepository: InitCashInTransactionRepository,
     private val sbConfirmWithMoMoUseCase: SBConfirmWithMoMoUseCase,
     private val sbNapasLinkedConfirmUseCase: SBNapasLinkedConfirmUseCase,
     private val sbNapasNotLinkedConfirmUseCase: SBNapasNotLinkedConfirmUseCase,
     private val sbCreditConfirmRepository: SBCreditConfirmRepository,
     private val logger: ILogger
 ) : SBConfirmUseCase {
     @OptIn(ExperimentalCoroutinesApi::class)
     override fun confirm(params: SBConfirmParams): Flow<SBConfirmResponse> {
         return flow {
             val cashInFlow = withContext(Dispatchers.Default) {
                 handleCashIn(params)
             }
             emitAll(cashInFlow)
         }.flatMapLatest { cashInResponse ->
             when (cashInResponse) {
                 is SBConfirmResponse.ToPaymentResult -> {
                     logger.i("[PayX] SBConfirmTwoStepUseCase.cashInResponse::: $cashInResponse")
                     logger.i("[PayX] SBConfirmTwoStepUseCase.result::: ${cashInResponse.response["result"]?.jsonPrimitive?.booleanOrNull}")
                     if ((cashInResponse.response["result"]?.jsonPrimitive?.booleanOrNull == true)) {
                         sbConfirmWithMoMoUseCase(params)
                     } else {
                         flowOf(cashInResponse)
                     }
                 }
 
                 else -> flowOf(cashInResponse)
             }
         }
     }
 
     private suspend fun handleCashIn(params: SBConfirmParams): Flow<SBConfirmResponse> {
         return when (params.currentSof?.moneySource?.id) {
             MoneySource.Napas.id -> cashInNapasToMoMo(params)
             MoneySource.Card.id -> cashInCardToMoMo()
             else -> throw SingleBillErrorHelper.generateDefaultErrorPopup(
                 DevError.NoResponse.code,
                 ""
             )
         }
     }
 
     @OptIn(ExperimentalCoroutinesApi::class)
     private fun cashInNapasToMoMo(params: SBConfirmParams): Flow<SBConfirmResponse> {
         val execute =
             if (params.currentSof?.isNapasNotLinked == true) sbNapasNotLinkedConfirmUseCase else sbNapasLinkedConfirmUseCase
         return initCashInTransactionRepository.init(params.currentSof)
             .flatMapLatest { id -> execute.confirm(params.copy(ID = id)) }
     }
 
     @OptIn(ExperimentalCoroutinesApi::class)
     private fun cashInCardToMoMo(): Flow<SBConfirmResponse> {
         return sbCreditConfirmRepository.getSongbirdSetupScript().flatMapLatest {
             it.takeIf { it?.isNotEmpty() == true }?.let { script ->
                 val params = buildJsonObject {
                     put("songbirdSetupAction", script.toJsonElement())
                 }
                 navigator.navigateFeatureCode(
                     "payment_3ds",
                     params,
                     WebView3dsResponse.serializer()
                 ).map {
                     return@map SBConfirmResponse.Retry
                 }
             } ?: throw SingleBillErrorHelper.generateDefaultErrorPopup(
                 DevError.NoResponse.code,
                 ""
             )
         }
     }
 }
