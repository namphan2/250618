 @Composable
 fun PaymentBankOtpScreen(viewModel: PaymentBankOtpVM) {
     val state by viewModel.state.collectAsState()
 
     Screen(
         title = stringResource(Res.string.verifyBankOtpTitle),
         goBack = { viewModel.screenGoBack() },
         headerType = HeaderType.EXTENDED,
         headerRight = null,
         footer = {
             FooterBankOtp(
                 onClick = { viewModel.onConfirm() },
                 type = if (state.isValid) ButtonType.PRIMARY else ButtonType.DISABLED
             )
         },
         content = {
             Column(
                 modifier = Modifier.fillMaxSize().padding(Spacing.M),
                 verticalArrangement = Arrangement.spacedBy(Spacing.M)
             ) {
                 when (viewModel.screenType) {
                     EOtpType.SMS -> SmsOtp(viewModel)
                     EOtpType.DOTP -> DigitalOtp(viewModel)
                 }
             }
         }
     )
 }
 
 @Composable
 fun SmsOtp(vm: PaymentBankOtpVM) {
     val state by vm.state.collectAsState()
     val focusRequester = remember { FocusRequester() }
 
     LaunchedEffect(Unit) {
         focusRequester.requestFocus()
         state.otpInputProps.maxLength?.let {
             vm.listenSmsOtp(it)
         }
     }
 
     Column(
         modifier = Modifier
             .clip(RoundedCornerShape(Radius.M))
             .background(Colors.black_01)
             .padding(Spacing.M),
         verticalArrangement = Arrangement.spacedBy(Spacing.M)
     ) {
         Text(
             text = stringResource(Res.string.confirmOtp),
             style = Typography.headerDefaultBold,
         )
         InformationBanner(
             content = (state.otpInputProps.headerText ?: stringResource(Res.string.smsInform))
                 .replace("{bank}", vm.params.bankName),
             buttonTitle = vm.params?.guideOtpUrl?.let { stringResource(Res.string.guide) },
             onPressButton = {
                 vm.onViewGuideUrlDOtp()
             },
         )
         BankInputOTP(
             initialValue = state.otp,
             length = state.otpInputProps.maxLength,
             floatingValue = stringResource(Res.string.verifyBankOtpTitle),
             onChangeText = { vm.onChangeOtp(it) },
             dataType = state.otpInputProps.keyboardType ?: "default",
             onBlur = { vm.onBlurInput() },
             onFocus = { vm.onFocusInput() },
             errorMessage = state.errorMessage,
             modifier = Modifier.focusRequester(focusRequester),
         )
 
     }
 }
 
 @Composable
 fun DigitalOtp(vm: PaymentBankOtpVM) {
     val state by vm.state.collectAsState()
     val showPopup by vm.showPopup.collectAsState()
     val focusRequester = remember { FocusRequester() }
 
     LaunchedEffect(Unit) {
         vm.getBankData()
     }
 
     LaunchedEffect(state.bankData) {
         if (listOf(
                 state.bankData.bankName,
                 state.bankData.logo,
                 state.bankData.bankCode
             ).all { !it.isNullOrBlank() }
         ) {
             vm.showPopupGuideDOtp()
         }
     }
 
     if (showPopup) {
         GuideDOtpPopup(
             bankData = state.bankData,
             onClickGetDOTP = { vm.getDOtp() },
             onDismiss = {
                 vm.hidePopupGuide()
                 focusRequester.requestFocus()
             }
         )
     }
 
     Column(
         modifier = Modifier
             .clip(RoundedCornerShape(Radius.M))
             .background(Colors.black_01)
             .padding(Spacing.M),
         verticalArrangement = Arrangement.spacedBy(Spacing.M)
     ) {
         Text(
             text = stringResource(Res.string.confirmOtp),
             style = Typography.headerDefaultBold,
         )
         InformationBanner(
             content = (state.otpInputProps.headerText
                 ?: stringResource(Res.string.subtitleDOTP)).replace("{bank}", vm.params.bankName),
             buttonTitle = vm.params?.guideOtpUrl?.let { stringResource(Res.string.guide) },
             onPressButton = {
                 vm.onViewGuideUrlDOtp()
             },
         )
         BankInputOTP(
             initialValue = state.otp,
             length = state.otpInputProps.maxLength,
             floatingValue = stringResource(Res.string.verifyBankOtpTitle),
             onChangeText = { vm.onChangeOtp(it) },
             dataType = state.otpInputProps.keyboardType ?: "default",
             onBlur = { vm.onBlurInput() },
             onFocus = { vm.onFocusInput() },
             errorMessage = state.errorMessage,
             modifier = Modifier.focusRequester(focusRequester),
         )
         vm.params.otpURL.let {
             Button(
                 onClick = {
                     vm.getDOtp()
                 },
                 type = ButtonType.TEXT,
                 title = stringResource(Res.string.getDOtpBtn),
             )
         }
     }
