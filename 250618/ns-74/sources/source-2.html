 package vn.momo.compose.payment.payx.domain.useCases
 
 import vn.momo.compose.utils.formatCurrency
 
 class FormatCurrencyUseCase {
     operator fun invoke(
         value: Double?,
         fallback: String? = null,
         currency: String = "Ä‘",
         separator: String = "."
     ): String {
         if (value != null && value > 0) {
             return value.formatCurrency(currency, separator)
         }
         return if (fallback.isNullOrEmpty()) "0${currency}" else fallback
     }
 }
