 @Composable
 fun CtaShowAllButton(count: Int, expanded: Boolean, onClick: () -> Unit) {
     val rotationState by animateFloatAsState(
         targetValue = if (expanded) 180f else 0f, label = "ShowAllSofRotate",
         animationSpec = linearTween()
     )
     Box(
         modifier = Modifier
             .scaledPadding(horizontal = 14f)
             .fillMaxWidth()
             .drawBehind {
                 val width = size.width
                 val height = size.height
 
                 val path = Path().apply {
                     moveTo(0f, 0f)
                     lineTo(width, 0f)
                     lineTo(width * 0.982f, height * 0.603f)
                     cubicTo(
                         x1 = width * 0.974f,
                         y1 = height * 0.842f,
                         x2 = width * 0.956f,
                         y2 = height,
                         x3 = width * 0.935f,
                         y3 = height
                     )
                     lineTo(width * 0.065f, height)
                     cubicTo(
                         x1 = width * 0.044f,
                         y1 = height,
                         x2 = width * 0.026f,
                         y2 = height * 0.842f,
                         x3 = width * 0.018f,
                         y3 = height * 0.603f
                     )
                     lineTo(0f, 0f)
                     close()
                 }
 
                 drawPath(
                     path = path,
                     color = Colors.pink_11
                 )
             }
             .scaledPadding(vertical = 4f, horizontal = 16f)
             .noFeedbackClickable(onClick = onClick)
     ) {
         Row(
             modifier = Modifier.align(Alignment.Center),
             verticalAlignment = Alignment.CenterVertically,
             horizontalArrangement = Arrangement.Center
         ) {
             Text(
                 text = if (expanded) stringResource(Res.string.collapse) else stringResource(Res.string.viewAll) + if (count > 0) " ($count)" else "",
                 style = Typography.labelSMedium,
                 color = AppTheme.current.colors.primary
             )
             Icon(
                 modifier = Modifier.rotate(rotationState),
                 source = "arrow_chevrons_double_down",
                 size = 16.dp,
                 color = AppTheme.current.colors.primary
             )
         }
     }
 }
