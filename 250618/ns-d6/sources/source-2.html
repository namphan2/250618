 @Composable
 fun PaymentSingleBillScreen(viewModel: PaymentSingleBillVM) {
     val scrollState = rememberScrollState()
     var contentViewHeight by remember { mutableStateOf(0) }
     val screenWidth = getScreenDimensions().screenWidth
     val state by viewModel.state.collectAsState()
 
     fun goBack() {
         viewModel.goBack()
     }
 
     Screen(
         title = stringResource(Res.string.paymentTitle),
         goBack = ::goBack,
         headerRight = null,
         useAvoidKeyboard = false,
         scrollState = scrollState,
         onContentLayout = { contentViewHeight = it.size.height },
         content = {
             when {
                 state.isLoading -> ContentLoadingSkeleton(version = state.version)
                 state.version.code == 2 -> PaymentTransferContentV2(viewModel)
                 else -> PaymentSingleBillContentV1(viewModel)
             }
         },
         layoutOffset = if (state.version.code == 2) 0.dp else 56.dp,
         animatedHeader = if (state.version.code == 2) AnimatedHeader(composable = {
             Image(
                 source = Images.bg_ttat_revamp,
                 modifier = Modifier.fillMaxWidth().height((screenWidth * 9 / 16).dp),
                 options = Options(contentScale = ContentScale.Crop)
             )
         }) else null,
         footer = {
             when  {
                 state.isLoading -> FooterLoading()
                 state.version.code == 2 -> TransferFooter(
                     amount = state.amount,
                     enableConfirm = state.enableConfirm,
                     onConfirm = viewModel::processConfirm,
                     invoice = state.invoice
                 )
 
                 else -> SBFooter(
                     enableConfirm = state.enableConfirm,
                     amount = state.amount,
                     warningMsg = state.warningMsg,
                     onConfirm = viewModel::processConfirm
                 )
             }
         })
 }
