 enum class ButtonDirection {
     COL, ROW
 }
 
 @Composable
 fun PopupNotify(
     show: Boolean,
     hasBackground: Boolean = true,
     image: String? = null,
     title: String? = null,
     content: String? = null,
     primaryButton: String? = null,
     secondaryButton: String? = null,
     buttonDirection: ButtonDirection = ButtonDirection.ROW,
     onPressPrimary: () -> Unit = {},
     onPressSecondary: () -> Unit = {},
     onDismissRequest: () -> Unit = {}
 ) {
     if (!show) return
 
     val contentComponent: @Composable () -> Unit = {
         Box(
             contentAlignment = Alignment.Center,
             modifier = Modifier.fillMaxSize().padding(horizontal = Spacing.L)
                 .background(Color.Transparent).clickable(onClick = {},
                     indication = null,
                     interactionSource = remember { MutableInteractionSource() }).zIndex(10f)
         ) {
             Box(
                 modifier = Modifier.clip(RoundedCornerShape(Spacing.L)).background(
                     Colors.black_01, shape = RoundedCornerShape(
                         Spacing.M
                     )
                 )
             ) {
                 Column(modifier = Modifier.fillMaxWidth()) {
                     image?.let {
                         MMImage(
                             it,
                             modifier = Modifier.fillMaxWidth()
                                 .clip(RoundedCornerShape(topStart = Spacing.L, topEnd = Spacing.L)),
                             contentDescription = "Image",
                             contentScale = ContentScale.FillWidth
                         )
                     }
                     Column(
                         horizontalAlignment = Alignment.Start,
                         modifier = Modifier.fillMaxWidth()
                             .padding(horizontal = Spacing.XL, vertical = Spacing.XL)
                     ) {
                         title?.let {
                             Text(
                                 text = it,
                                 style = Typography.headerSSemibold,
                                 textAlign = TextAlign.Center
                             )
                         }
                         content?.let {
                             Text(
                                 text = it,
                                 style = Typography.bodyDefaultRegular,
                                 modifier = Modifier.padding(top = Spacing.S),
                                 textAlign = TextAlign.Start
                             )
                         }
                         ButtonContainer(buttonDirection = buttonDirection, primary = {
                             primaryButton?.let {
                                 Button(
                                     title = it, onClick = onPressPrimary, size = Size.MEDIUM
                                 )
                             }
                         }, secondary = {
                             secondaryButton?.let {
                                 Button(
                                     title = it,
                                     onClick = onPressSecondary,
                                     type = ButtonType.TEXT,
                                     size = Size.MEDIUM
                                 )
                             }
                         })
                     }
                 }
             }
         }
     }
 
     if (hasBackground) {
         Dialog(
             onDismissRequest = onDismissRequest, properties = DialogProperties(
                 dismissOnClickOutside = false,
                 dismissOnBackPress = false,
                 usePlatformDefaultWidth = false,
             )
         ) {
             contentComponent()
         }
     } else {
         contentComponent()
     }
 }
 
 @Composable
 fun ButtonContainer(
     buttonDirection: ButtonDirection,
     primary: @Composable () -> Unit,
     secondary: @Composable () -> Unit
 
 ) {
     if (buttonDirection == ButtonDirection.COL) {
         Column(modifier = Modifier.fillMaxWidth().padding(top = Spacing.XL)) {
             primary()
             Spacer(modifier = Modifier.height(Spacing.S))
             secondary()
         }
     } else {
         Row(modifier = Modifier.fillMaxWidth().padding(top = Spacing.XL)) {
             Box(modifier = Modifier.weight(1f)) {
                 primary()
             }
             Spacer(modifier = Modifier.width(Spacing.S))
             Box(modifier = Modifier.weight(1f)) {
                 secondary()
             }
         }
     }
 }
