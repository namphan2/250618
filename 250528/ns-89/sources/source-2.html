 @Composable
 private fun UserSofList(
     list: List<SOFItem>,
     isNoSof: Boolean,
     isHideBalance: Boolean,
     selected: SOFItem?,
     number: Int = 2,
     onPressSof: (SOFItem) -> Unit,
     onPressCta: (SOFItem) -> Unit,
     isAllowAddSOF: Boolean = true,
     onAddSof: () -> Unit = {},
 ) {
     Column {
         if (isNoSof) {
             Surface(
                 Modifier.fillMaxWidth().padding(bottom = Spacing.S),
                 shape = RoundedCornerShape(Radius.S),
                 color = Colors.blue_10,
                 border = BorderStroke(width = 0.5.dp, color = Colors.blue_07)
             ) {
                 Text(
                     text = stringResource(Res.string.noSof),
                     style = Typography.descriptionDefaultRegular,
                     modifier = Modifier.padding(12.dp)
                 )
             }
         }
         val displayedItems = list.take(number)
         val remainingSlots = number - displayedItems.size
         displayedItems.forEachIndexed { idx, sof ->
             SofCard.Default(
                 modifier = Modifier.padding(vertical = Spacing.S),
                 sof = sof,
                 isHideBalance = isHideBalance,
                 isSelected = sof.isMatchSource(
                     selected?.sourceToken,
                     selected?.ID
                 ),
                 onPress = { onPressSof(sof) },
                 onPressCTA = { onPressCta(sof) }
             )
             if (idx < displayedItems.size - 1) {
                 Box(
                     modifier = Modifier
                         .fillMaxWidth()
                         .height(1.dp)
                         .padding(start = 16.dp)
                         .background(color = AppTheme.current.colors.border.default)
                         .padding(vertical = Spacing.XS)
                 )
             }
         }
 
         if (remainingSlots > 0 && isAllowAddSOF) {
             SofCard.AddBank(
                 modifier = Modifier.padding(vertical = Spacing.S),
                 onPress = onAddSof
             )
         }
     }
 }
 
 @Composable
 fun PayLaterBlock(
     sofListState: SofReducer.State,
     payLaters: List<SOFItem>,
     promoted: PromotedSof?,
     onPressSof: (SOFItem) -> Unit,
     onPressUnlock: (SOFItem) -> Unit,
     onSelectPromoted: (SOFItem?) -> Unit,
     onShowUnlock: () -> Unit,
 ) {
     Column(
         Modifier.padding(
             vertical = Spacing.S,
             horizontal = Spacing.M
         ).animateContentSize()
     ) {
         ContentInformationCard(
             shape = RectangleShape,
             header = {
                 BasicCardHeader(
                     title = stringResource(Res.string.payLater),
                 )
             }) {
             Column(
                 modifier = Modifier.padding(vertical = Spacing.S),
                 verticalArrangement = Arrangement.spacedBy(8.dp)
             ) {
                 payLaters.onEach { payLater ->
                     if (payLater.lockPayment != null) {
                         LaunchedEffect(Unit) { onShowUnlock() }
                         SofCard.LockedSof(sof = payLater, onPressCTA = { onPressUnlock(payLater) })
                     } else {
                         SofCard.Default(
                             sof = payLater,
                             isHideBalance = sofListState.isHideBalance,
                             isSelected = payLater.isMatchSource(
                                 sofListState.selected?.sourceToken,
                                 sofListState.selected?.ID
                             ),
                             onPress = { onPressSof(payLater) }
                         )
                     }
 
                     if (promoted != null) {
                         Divider()
                     }
                 }
 
                 when (promoted) {
                     is PromotedSof.PromoteInstallment -> {
                         val installmentGroup =
                             promoted.group.copy(
                                 parent = promoted.group.parent.copy(
                                     fullName = stringResource(
                                         Res.string.flexInstallment
                                     )
                                 )
                             )
                         SofGroupView(
                             installmentGroup,
                             selected = sofListState.selected,
                             onSelect = onSelectPromoted
                         )
                     }
 
                     is PromotedSof.QuickActive -> {
                         QuickActiveBanner(
                             promoted.group,
                             onActive = { onSelectPromoted(null) })
                     }
 
                     null -> {}
                 }
             }
 
         }
     }
