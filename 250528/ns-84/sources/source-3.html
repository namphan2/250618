 @Composable
 fun ActionCard(
     modifier: Modifier = Modifier,
     title: String,
     description: String = "",
     iconLeft: @Composable (() -> Unit)? = null,
     iconRight: @Composable (() -> Unit)? = null,
     cardStyle: ActionCardStyle = ActionCardStyle(),
     titleModifier: Modifier = Modifier,
     showIconNotificationInvestment: Boolean = false,
 ) {
     val navigator: INavigator = KoinHelper.getInjected()
     val viewModelScope = rememberCoroutineScope()
     Surface(
         modifier = Modifier
             .fillMaxWidth()
             .heightIn(min = 64.dp)
             .then(modifier),
         shape = cardStyle.shape,
         border = cardStyle.border,
         color = cardStyle.backgroundColor,
     ) {
         Row(
             modifier = Modifier
                 .fillMaxWidth()
                 .padding(
                     vertical = cardStyle.paddingVertical,
                     horizontal = cardStyle.paddingHorizontal
                 ),
             verticalAlignment = Alignment.CenterVertically,
             horizontalArrangement = Arrangement.spacedBy(Spacing.S)
         ) {
             Row(
                 modifier = Modifier
                     .alpha(cardStyle.alpha)
                     .weight(1f),
                 verticalAlignment = Alignment.CenterVertically,
                 horizontalArrangement = Arrangement.spacedBy(Spacing.S)
             ) {
                 iconLeft?.invoke()
                 Column(
                     modifier = Modifier.weight(1f),
                     verticalArrangement = Arrangement.spacedBy(Spacing.XXS)
                 ) {
                     Row(verticalAlignment = Alignment.CenterVertically) {
                         Text(
                             text = title,
                             style = Typography.headerSSemibold,
                             modifier = titleModifier,
                             maxLines = 1,
                             overflow = TextOverflow.Ellipsis
                         )
                         if (showIconNotificationInvestment) {
                             Spacer(modifier = Modifier.width(Spacing.S))
                             Icon(
                                 source = "notifications_info",
                                 size = Spacing.L,
                                 color = Colors.blue_04,
                                 modifier = Modifier.noFeedbackClickable(onClick = {
                                     viewModelScope.launch {
                                         navigator.showDialog(
                                             getString(Res.string.notification),
                                             getString(Res.string.infoSourceInvestment),
                                             listOf(getString(Res.string.textUnderstand)),
                                         ).launchIn(this)
                                     }
                                 })
                             )
                         }
                     }
 
                     Text(
                         text = description,
                         style = Typography.descriptionDefaultRegular,
                         color = Colors.black_12,
                         maxLines = 2,
                         overflow = TextOverflow.Ellipsis
                     )
                 }
             }
             iconRight?.invoke()
         }
     }
