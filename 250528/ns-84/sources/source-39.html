 @Composable
 fun TextRow(
     modifier: Modifier = Modifier, data: TextRowItem
 ) {
     val interactionSource = remember { MutableInteractionSource() }
     val tintColor = data.rightIconTintColor?.let { Color(it) } ?: Color.Unspecified
     val key = data.key
     val value = data.value
     val valueColor = Color(data.valueColor)
 
     Row(
         modifier = modifier.fillMaxWidth(),
         verticalAlignment = Alignment.Top,
         horizontalArrangement = Arrangement.SpaceBetween
     ) {
         Text(
             key,
             style = Typography.bodyDefaultRegular,
             color = Colors.black_12,
             modifier = Modifier.weight(0.45f),
             maxLines = 2,
             overflow = TextOverflow.Ellipsis
         )
         Row(verticalAlignment = Alignment.CenterVertically, modifier = Modifier.weight(0.50f),
             horizontalArrangement = Arrangement.End
         ) {
             Text(
                 value,
                 style = Typography.headerSSemibold,
                 color = valueColor,
                 modifier = Modifier
                     .setAutomationId(data.accessibilityId, value)
                     .clickable(interactionSource = interactionSource, indication = null) {
                         data.onPress.invoke()
                     },
 
                 textAlign = TextAlign.End,
                 maxLines = 2,
                 overflow = TextOverflow.Ellipsis
             )
             if (data.rightIcon.isNotBlank()) Image(
                 painter = getResource(data.rightIcon),
                 contentDescription = null,
                 modifier = Modifier.size(Spacing.L),
                 colorFilter = ColorFilter.tint(tintColor)
             )
         }
     }
 }
