 @Composable
 fun SofIconList(
     data: List<String>,
     maxItem: Int = 3,
     itemSize: Int = 32,
     extraItem: String? = null,
     extraStyle: TextStyle = Typography.descriptionDefaultRegular,
     iconOffset: Int = 8
 ) {
     Row(
         verticalAlignment = Alignment.CenterVertically
     ) {
         val dataDisplay = data.slice(0 until maxItem.coerceIn(0, data.size))
         dataDisplay.mapIndexed { index, icon ->
             Surface(
                 shape = CircleShape,
                 border = BorderStroke(1.dp, AppTheme.current.colors.border.default),
                 modifier = Modifier
                     .then(
                         if (extraItem != null) {
                             Modifier.scaledOffset(x = (iconOffset * (maxItem - index)).toFloat())
                         } else {
                             Modifier.scaledOffset(x = (iconOffset * (maxItem - index - 1)).toFloat())
                         }
                     )
             ) {
                 Image(
                     source = icon,
                     modifier = Modifier
                         .scaledSize(itemSize.toFloat())
                         .background(AppTheme.current.colors.background.surface)
                 )
             }
         }
         if (dataDisplay.isNotEmpty() && extraItem != null) {
             Surface(
                 shape = CircleShape,
                 border = BorderStroke(1.dp, AppTheme.current.colors.border.default),
                 modifier = Modifier.scaledSize(itemSize.toFloat())
             ) {
                 Box(
                     contentAlignment = Alignment.Center,
                     modifier = Modifier
                         .fillMaxSize()
                         .background(Colors.black_03)
                 ) {
                     Text(
                         text = "+${extraItem}",
                         color = AppTheme.current.colors.text.hint,
                         style = extraStyle
                     )
                 }
             }
         }
     }
 }
