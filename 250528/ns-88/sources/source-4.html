 data class AsyncIconProp(
     val source: String,
     val onPress: () -> Unit = {},
     val size: Dp = 24.dp,
     val color: Color = Colors.black_17
 )
 
 @Composable
 fun AsyncIcon(
     modifier: Modifier = Modifier,
     prop: AsyncIconProp
 ) {
     IconButton(
         onClick = prop.onPress,
         modifier = Modifier
             .size(prop.size)
             .then(modifier),
         interactionSource = NoFeedbackInteractionSource()
     ) {
         Image(
             source = prop.source,
             options = Options(colorFilter = ColorFilter.tint(prop.color)),
             modifier = Modifier
                 .fillMaxSize()
         )
     }
 }
 
 class NoFeedbackInteractionSource : MutableInteractionSource {

         //empty
     }
 
     override fun tryEmit(interaction: Interaction): Boolean {
         return true
     }
