 @Composable
 fun TransactionTicket(
     data: TransactionTicketItem,
     index: Number,
     state: TransactionResultState,
     dispatcher: (TransactionResultAction) -> Unit,
 ) {
     val logger by lazy { KoinHelper.getInjected<ILogger>() }
 
     var separatorOffsetY by remember { mutableStateOf<Float?>(null) }
     var separatorOffsetX by remember { mutableStateOf<Float?>(null) }
     var expanded by remember { mutableStateOf(false) }
     var widgetLoadDone by remember { mutableStateOf(false) }
 
     var size by remember { mutableStateOf(IntSize.Zero) }
     val animRotation = remember { Animatable(180f) }
     val coroutineScope = rememberCoroutineScope()
     val interactionSource = remember { MutableInteractionSource() }
     var basicInfo by remember { mutableStateOf(emptyList<TextRowItem>()) }
 
     val transaction = data.transaction ?: TransactionItem()
 
     val transactionStatus = transaction.transactionStatus
     val statusTxt = stringResource(
         when (transaction.transactionStatus) {
             TransactionStatus.FAILED -> Res.string.failedTransaction
             TransactionStatus.SUCCESS -> Res.string.successfullyTransaction
             TransactionStatus.PROCESSING, TransactionStatus.PENDING -> Res.string.processingTransaction
         }
     )
     val amountTxt = transaction.amountText
     val messageInfo = data.messageInfo?.takeIf { it.isNotBlank() } ?: transaction.desc
     val messageInfoErrorCode = buildString {
         messageInfo?.takeIf { it.isNotBlank() }?.let {
             append(it)
             if (transaction.error != null && !transaction.isSuccess) {
                 append(" (${stringResource(Res.string.errorCode)}: ${transaction.error})")
             }
         }
     }
     val showMessageInfo = messageInfoErrorCode.isNotBlank()
     val originalAmountTxt = transaction.originalAmountText
 
     LaunchedEffect(state) {
         val disableTid = state.disableTID || data.transaction?.disableTid == true
         basicInfo = transaction.getBasicInfo(disableTid).map {
             if (it.value == transaction.transactionId && data.transaction?.isVirtualTID == false && !disableTid) {
                 it.onPress = {
                     dispatcher.invoke(
                         TransactionResultAction.OpenTransactionDetailScreen(
                             transaction
                         )
                     )
                 }
             }
             it
         } + (data.extraInfo ?: emptyList())
     }
 
     val imageBitmap = getImageBimap("img_transaction_watermark.webp")
 
     val widgetVisible by remember {
         derivedStateOf { expanded }
     }
     val ticketShape = remember(separatorOffsetY, separatorOffsetY) {
         TransactionTicketShape(
             separatorOffsetY,
             separatorOffsetX,
             offsetBillDetail,
             cornerRadius,
             dotSize,
             statusSize,
             buttonWidth,
             buttonHeight
         )
     }
 
     LaunchedEffect(index) {
         expanded = index == 0
     }
 
     LaunchedEffect(data.isWidgetLoadDone) {
         if (data.isWidgetLoadDone) {
             widgetLoadDone = true
         }
     }
 
     ComposeCaptureWrapper(onIdReady = {
         dispatcher.invoke(TransactionResultAction.OnComposeIdReady(it, transaction))
     }) {
         Column(Modifier.fillMaxWidth()
             .padding(top = 0.dp, bottom = Spacing.M, start = Spacing.M, end = Spacing.M)
             .shadow(1.dp, ticketShape, clip = false).clip(ticketShape).background(Color.White)
             .drawBehind {
                 imageBitmap.let { it ->
                     val aspectRatio = it.width.toFloat() / it.height.toFloat()
                     val targetWidth = size.width * 0.95
                     val targetHeight = (targetWidth / aspectRatio)
                     val offsetX = (size.width - targetWidth) / 2
                     val offsetY =
                         separatorOffsetY?.let { it - (targetHeight / 2) } ?: -(targetHeight / 2)
 
                     val gradientRectHeight = (size.height / 4).toFloat()
 
                     // draw watermark
                     drawImage(
                         image = imageBitmap,
                         dstSize = IntSize(targetWidth.toInt(), targetHeight.toInt()),
                         dstOffset = IntOffset(offsetX.toInt(), offsetY.toInt()),
                     )
 
                     //draw gradient top
                     drawRect(
                         brush = Brush.verticalGradient(
                             0.0f to Color.White, 1f to Color.Transparent,
 
                             startY = offsetBillDetail.toPx(), endY = gradientRectHeight
                         ), size = Size(size.width.toFloat(), gradientRectHeight)
                     )
 
                     //draw gradient bottom
                     drawRect(
                         brush = Brush.verticalGradient(
                             0.0f to Color.Transparent,
                             1f to Color.White,
                             startY = size.height.toFloat() - gradientRectHeight - buttonHeight.toPx(),
                             endY = size.height.toFloat()
                         ), size = Size(size.width.toFloat(), size.height.toFloat())
                     )
                 }
             }.onSizeChanged {
                 size = it
             }) {
             Column(
                 modifier = Modifier.fillMaxWidth()
             ) {
                 StatusIcon(
                     modifier = Modifier.align(Alignment.CenterHorizontally).size(64.dp)
                         .shadow(2.dp, CircleShape).background(Color.White),
                     transaction.transactionStatus
                 )
 
                 Spacer(Modifier.height(Spacing.S))
 
                 Text(
                     statusTxt,
                     style = Typography.headerSSemibold,
                     color = textColor,
                     modifier = Modifier.align(Alignment.CenterHorizontally)
                 )
 
                 Spacer(Modifier.height(Spacing.XXS))
                 if (transaction.isValidAmount) {
                     Text(
                         amountTxt,
                         style = Typography.headerMBold,
                         color = textColor,
                         modifier = Modifier.align(Alignment.CenterHorizontally)
                     )
                 }
                 if (transaction.isPaymentSDK && transaction.isValidOriginalAmount) {
                     Spacer(Modifier.height(Spacing.XXS))
                     Text(
                         originalAmountTxt,
                         style = Typography.bodyDefaultRegular,
                         color = captionTextColor,
                         modifier = Modifier.align(Alignment.CenterHorizontally),
                         textDecoration = TextDecoration.LineThrough
                     )
                 }
             }
 
             Box(Modifier.padding(horizontal = dotSize, vertical = 7.dp).height(2.dp).fillMaxWidth()
                 .background(dotColor, shape = DottedShape(step = Spacing.S)).onGloballyPositioned {
                     separatorOffsetY = it.boundsInParent().center.y
                     separatorOffsetX = it.boundsInParent().center.x
                 })
             Column(
                 Modifier.fillMaxWidth().heightIn(min = cornerRadius).padding(horizontal = Spacing.M)
                     .padding(bottom = buttonHeight)
             ) {
 
                 Spacer(modifier = Modifier.height(Spacing.S))
                 if (showMessageInfo) {
                     MessageInfo(messageInfoErrorCode, transactionStatus)
                     Spacer(modifier = Modifier.height(Spacing.M))
                 }
 
                 Column(
                     Modifier.padding(bottom = Spacing.M),
                 ) {
                     TextRowList(Modifier,
                         data = data.serviceName?.takeIf { it.isNotBlank() }?.let {
                             listOf(TextRowItem(stringResource(Res.string.paymentService), it))
                         } ?: emptyList(),
                         divider = false,
                         verticalArrangement = Arrangement.spacedBy(Spacing.M))
                 }
                 Column {
                     PayXAnimatedVisibility(visible = expanded) {
                         Column {
                             TextRowList(
                                 Modifier,
                                 data = basicInfo,
                                 divider = false,
                                 verticalArrangement = Arrangement.spacedBy(Spacing.M)
                             )
                             UnloadedWidgetView(visible = !widgetLoadDone)
                         }
                     }
                     TransactionResultWidget(Modifier.padding(top = Spacing.S),
                         widgetVisible,
                         data,
                         {
                             widgetLoadDone = true
                             dispatcher.invoke(TransactionResultAction.OnWidgetLoadSuccess())
                             logger.i("[TR] Loading Widget Successful ${data.widget.appId}")
                         },
                         {
                             widgetLoadDone = true
                             dispatcher.invoke(TransactionResultAction.OnWidgetLoadFailed())
                             logger.i("[TR] Loading Widget Failed ${data.widget.appId}")
                         },
                         { width: Int, height: Int ->
                             widgetLoadDone = true
                             dispatcher.invoke(TransactionResultAction.OnWidgetLoadSuccess())
                             logger.i("[TR] Layout Changed Widget ${data.widget.appId} : width : $width - height : $height")
                         })
                 }
                 Spacer(Modifier.height(Spacing.M))
                 ActionButtonGroup(
                     Modifier,
                     primaryButton = data.primaryActionButton,
                     secondaryButton = data.secondaryActionButton,
                     dispatcher
                 )
             }
             Column(
                 Modifier.height(buttonHeight).align(Alignment.CenterHorizontally),
                 verticalArrangement = Arrangement.Center,
                 horizontalAlignment = Alignment.CenterHorizontally
             ) {
                 Image(painter = getResource("ic_arrow_chevron_down.webp"),
                     contentDescription = null,
                     modifier = Modifier.rotate(animRotation.value).clickable(
                         interactionSource = interactionSource, indication = null
                     ) {
                         coroutineScope.launch {
                             val targetValue = if (expanded) 180f else 0f
                             animRotation.animateTo(targetValue)
                         }
                         expanded = !expanded
                     })
             }
         }
     }
 }
