 val APP_IDS_BLACKLIST = arrayOf("vn.momo.nativepayment")
 
 @Composable
 fun TransactionResultWidget(
     modifier: Modifier,
     visible: Boolean = true,
     data: TransactionTicketItem,
     onLoadSuccess: (() -> Unit)?,
     onLoadFailed: (() -> Unit)?,
     onLayoutChanged: ((width: Int, height: Int) -> Unit)?,
 ) {
     var widgetHeight by remember { mutableIntStateOf(1) }
     var dataWidget by remember { mutableStateOf(emptyMap<String, Any?>()) }
 
     LaunchedEffect(data.widget) {
         dataWidget = data.widget.toMap()
     }
 
     val appId = remember(dataWidget) { dataWidget.tryGetString("appId") }
 
     val shouldShow =
         visible && dataWidget.isNotEmpty() && appId.isNotEmpty() && !APP_IDS_BLACKLIST.contains(
             appId
         )
 
     PayXAnimatedVisibility(
         visible = shouldShow
     ) {
         val heightDp = if (!visible) 1.dp else widgetHeight.dp
         val alpha = if (!visible) 0f else 1f
 
         Column(
             modifier = modifier.height(heightDp).heightIn(max = 108.dp).alpha(alpha)
         ) {
             RNWidgetComposeContainer(
                 ComposeApplication.initWithMap(dataWidget), object : AppContainerViewDelegate {
                     override fun onLoadSuccess() {
                         super.onLoadSuccess()
                         onLoadSuccess?.invoke()
                     }
 
                     override fun onLoadFailed() {
                         super.onLoadFailed()
                         onLoadFailed?.invoke()
                     }
 
                     override fun onLayoutChanged(
                         event: LayoutEvent
                     ) {
                         super.onLayoutChanged(event)
                         onLayoutChanged?.invoke(event.width, event.height)
                         widgetHeight = if (event.height > 0) event.height else 1
                     }
                 }, false
             )
         }
     }
 }
