 package vn.momo.compose.payment.payx.presentation.view.components
 
 import androidx.compose.foundation.layout.Box
 import androidx.compose.foundation.layout.BoxScope
 import androidx.compose.runtime.Composable
 import androidx.compose.ui.Alignment
 import androidx.compose.ui.Modifier
 import androidx.compose.ui.layout.FirstBaseline
 import androidx.compose.ui.layout.LastBaseline
 import androidx.compose.ui.layout.Layout
 import androidx.compose.ui.layout.Placeable
 import androidx.compose.ui.layout.layoutId
 import androidx.compose.ui.unit.IntOffset
 
 @Composable
 fun PositionedBadgedBox(
     modifier: Modifier = Modifier,
     badge: @Composable BoxScope.() -> Unit,
     offset: (contentSize: Placeable, badgeSize: Placeable) -> IntOffset = { contentSize, badgeSize ->
         IntOffset(
             x = contentSize.width,
             y = badgeSize.height / 2
         )
     },
     content: @Composable BoxScope.() -> Unit,
 ) {
     Layout(
         {
             Box(
                 modifier = Modifier.layoutId("anchor"),
                 contentAlignment = Alignment.Center,
                 content = content
             )
             Box(
                 modifier = Modifier.layoutId("badge"),
                 content = badge
             )
         },
         modifier = modifier
     ) { measurables, constraints ->
         
         val badgePlaceable = measurables.first { it.layoutId == "badge" }.measure(
             constraints.copy(minHeight = 0)
         )
         val anchorPlaceable = measurables.first { it.layoutId == "anchor" }.measure(constraints)
         
         val firstBaseline = anchorPlaceable[FirstBaseline]
         val lastBaseline = anchorPlaceable[LastBaseline]
         val badgeOffset = offset(anchorPlaceable, badgePlaceable)
         val totalWidth = maxOf(anchorPlaceable.width, anchorPlaceable.width - badgeOffset.x, badgeOffset.x + badgePlaceable.width)
         val totalHeight = maxOf(badgeOffset.y + badgePlaceable.height, anchorPlaceable.height, anchorPlaceable.height - badgeOffset.y)
         val newContentX = if(totalWidth == anchorPlaceable.width - badgeOffset.x) -badgeOffset.x else 0
         val newContentY = if(totalHeight == anchorPlaceable.height - badgeOffset.y) -badgeOffset.y else 0
         val newAnchorOffset = IntOffset(x = newContentX, y = newContentY)
         val newBadgeOffset = IntOffset(x = badgeOffset.x + newAnchorOffset.x, y = badgeOffset.y + newAnchorOffset.y)
         layout(
             totalWidth,
             totalHeight,
             mapOf(
                 FirstBaseline to firstBaseline,
                 LastBaseline to lastBaseline
             )
         ) {
             anchorPlaceable.placeRelative(newAnchorOffset)
             badgePlaceable.placeRelative(newBadgeOffset)
         }
     }
 }
