 @Composable
 fun InstallmentPlanItem(
     sof: SOFItem,
     isSelected: Boolean,
     onPress: () -> Unit = {},
 ) {
     val backgroundColor = when {
         isSelected -> Colors.pink_11
         sof.disableType == DisableType.DisableAndShow -> Colors.black_02
         else -> Colors.black_01
     }
     val pressable = listOf(DisableType.Show, DisableType.DisableAndShow).contains(sof.disableType)
 
 
     Column {
         ActionCard(
             title = sof.fullName,
             description = sof.action?.description ?: "",
             cardStyle = ActionCardStyle(
                 backgroundColor = backgroundColor,
                 alpha = if (sof.pressable) 1f else 0.5f,
                 shape = RectangleShape,
                 border = null,
                 paddingHorizontal = Spacing.M,
             ),
             iconRight = {
                 Image(
                     source = if (isSelected)
                         Images.ic_select_bank
                     else Images.ic_non_selected,
                     modifier = Modifier.size(Spacing.XL)
                 )
             },
             modifier = Modifier.noFeedbackClickable(enabled = pressable, onClick = onPress)
         )
         Divider(
             color = Colors.black_04,
             thickness = 1.dp,
             modifier = Modifier.padding(horizontal = Spacing.M)
         )
     }
 }
