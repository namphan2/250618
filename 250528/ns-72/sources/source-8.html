 class PayX3dsStrategy(
     private val navigator: INavigator,
     private val confirmWithOtpUseCase: ConfirmWithOtpUseCase,
     private val handlePaymentResultUseCase: HandlePaymentResultUseCase,
 ) : Payment3dsStrategy {
     override fun onSetupComplete(sessionId: String): Flow<Payment3dsAction> = flow {
         try {
             confirmWithOtpUseCase(
                 ConfirmWithOtpRequest(null, "PAYMENT_OTP_CASH_IN", referenceKey = sessionId),
                 ConfirmType.ConfirmOtpCyberSource
             ).collect { result ->
                 if (result.isFailure) throw result.exception
                 handlePayment3dsResult(result.data)
                 emit(Payment3dsAction.Suspend)
             }
         } catch (error: Throwable) {
             when (error) {
                 is PaymentError.SofOTPError -> {
                     val response = ModelConvertUtils.decodeFromJson(
                         ConfirmResponse.serializer(),
                         error.data
                     )
                     emit(
                         Payment3dsAction.InjectContinue(
                             response?.momoMsg?.songbirdContinueAction ?: "",
                             response?.momoMsg?.referenceKey ?: ""
                         )
                     )
                 }
 
                 else -> {
                     GrafanaTrace.fail(PayXUserEvent.CyberSource.event)
                     navigator.dismiss(buildJsonObject { "actor" to ScreenDismissActor.SYSTEM.id })
                     emit(Payment3dsAction.Suspend)
                     throw error
                 }
             }
         }
     }.handlePaymentConfirmError()
 
 
     override fun onPayment(referenceKey: String) {
         confirmWithOtpUseCase(
             ConfirmWithOtpRequest(null, "PAYMENT_OTP_CONFIRM", referenceKey),
             ConfirmType.ConfirmOtpCyberSource
         ).handleConfirmOverlay().onEach {
             if (it.isFailure) {
                 throw it.exception
             }
             handlePayment3dsResult(it.data)
         }.catch {
             GrafanaTrace.fail(PayXUserEvent.CyberSource.event)
             throw it
         }.handlePaymentConfirmError().launchIn(CoroutineScope(Dispatchers.IO))
     }
 
     private fun handlePayment3dsResult(data: ConfirmResponse?) {
         GrafanaTrace.success(PayXUserEvent.CyberSource.event)
         handlePaymentResultUseCase(data)
         FirebaseTrace.track(
             WebViewDismissed(
                 webView = WebViewScreen.TDS_OTP,
                 actor = ScreenDismissActor.SYSTEM.id
             )
         )
     }
 }
