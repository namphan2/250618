 @Composable
 fun ConfirmContentV2(
     state: TransactionReducer.State,
     hasAppliedPromotion: Boolean,
     viewModel: PaymentConfirmVM,
     sofListState: SofReducer.State,
     rewardState: RewardReducer.RewardState,
 ) {
     val isNoSofAvailable by remember(sofListState.list) {
         derivedStateOf { sofListState.list.none { it.available } }
     }
     val groupSof = sofListState.groupOfSof
     val sofList = remember(groupSof.debit, state.amount.total.value) {
         prepareDebitSof(groupSof.debit, state.amount.total.value)
     }
     val payLater = groupSof.payLater
     val promoted = sofListState.promoteSof
     val screenWidth = getScreenDimensions().screenWidth
     val rewardInline = rewardState.missions
 
     Body(
         modifier = Modifier
             .background(AppTheme.current.colors.background.default)
             .padding(Spacing.M)
             .animateContentSize(
                 animationSpec = linearTween(delayMillis = 0)
             )
     ) {
         state.invoices.forEach { item ->
             blockKey(item.itemId) {
                 TransactionDetailV2(
                     initialExpanded = false,
                     shouldShowTid = state.shouldShowTid,
                     hideAmount = state.hideAmount,
                     onChangeExpandState = viewModel::onToggleCartDetail,
                     hasAppliedPromotion = hasAppliedPromotion,
                     invoice = item
                 )
             }
         }
 
         if ((state.promotion.isDisplay && !state.hideAmount) || !state.inlineMessage.isNullOrBlank()) {
             blockKey("promotion") {
                 PickPromotionBlock(
                     promotion = state.promotion,
                     inlineMessage = state.inlineMessage,
                     onSelectPromotion = viewModel::navigateChoosePromotion,
                     hideAmount = state.hideAmount,
                     rewardMission = rewardInline,
                     onMissionShowed = {
                         viewModel.rewardSlice.trackMissionShow(
                             moneySource = sofListState.selected?.moneySource,
                             promotions = state.promotion.promoDisplay
                         )
                     }
                 )
             }
         }
 
         if (viewModel.props.isAllowPickSource) {
             blockKey("sof") {
                 Box {
                     SofListView(
                         modifier = Modifier.scaledPadding(horizontal = 12f, vertical = 8f),
                         sofListState = sofListState,
                         sofList = sofList,
                         isNoSof = isNoSofAvailable,
                         number = 2,
                         onPressSof = viewModel.sofSlice::changeSofIfNotSelected,
                         onPressCta = viewModel.sofSlice::onPressCta,
                         onClickShowAll = {
                             viewModel.onClickShowAll()
                             hideTooltip(viewModel)
                         },
                         isAllowAddSOF = viewModel.props.isAllowAddSource,
                         onLoaded = {
                             viewModel.sofSlice.trackSofDisplayed(
                                 "list_sof",
                                 sofListState.list,
                                 sofListState.version
                             )
                         },
                         onAddSof = viewModel::addSof
 //                            onToggleBalance = sofSlice::toggleHideBalance,
                     )
                     GuideTooltip(
                         Modifier.widthIn(max = (screenWidth * 0.65).dp).align(Alignment.CenterEnd),
                         isShow = sofListState.toolTip,
                         text = stringResource(Res.string.showAllTooltip),
                         onClick = { hideTooltip(viewModel) },
                         requestHide = { hideTooltip(viewModel) }
                     )
                 }
             }
             if (payLater.isNotEmpty() || promoted != null) {
                 blockKey("payLater") {
                     PayLaterBlock(
                         sofListState = sofListState,
                         payLaters = payLater,
                         promoted = promoted,
                         onPressSof = viewModel.sofSlice::changeSofIfNotSelected,
                         onPressUnlock = viewModel.sofSlice::onUnlock,
                         onSelectPromoted = {
                             viewModel.sofSlice.trackPressInstallment(sofListState.version)
                             viewModel.sofSlice.dispatch(
                                 SofReducer.Action.SelectPromotedSof(it)
                             )
                         },
                         onShowUnlock = { viewModel.sofSlice.trackShowUnlock(sofListState.version) },
                     )
                 }
             }
         }
         blockKey("smartWidget") {
             SmartWidget(viewModel.props.smartWidget)
         }
     }
 }
 
 private fun prepareDebitSof(debits: List<SOFItem>, amount: Long): List<SOFItem> {
     return debits.map { sof ->
         if (sof.moneySource == EnumMoneySource.MoMo && sof.disableType != DisableType.Show && sof.balance < amount) {
             sof.copy(
                 action = SofAction(
                     action = SofActionType.CashIn(amount = amount - sof.balance),
                     refId = ServiceFeatureCode.CashIn.code,
                     title = "Nạp tiền",
                     description = "",
                     isShow = true
                 )
             )
         } else sof
     }
 }
 
 private fun hideTooltip(viewModel: PaymentConfirmVM) {
     viewModel.sofSlice.dispatch(
         SofReducer.Action.UpdateTooltip(
             false
         )
     )
 }
