 @Composable
 fun PromoCodeInput(
     modifier: Modifier = Modifier,
     promoCode: String = "",
     onChange: (String) -> Unit = {},
     onSubmit: (String) -> Unit = {},
     isChecking: Boolean = false,
     errorMessage: String? = null,
     onValueCleared: () -> Unit = {},
     isApplied: Boolean = false,
     promoCodeApplied: String? = "",
     withInlineButton: Boolean? = true,
     focusRequester: FocusRequester? = null,
 ) {
     val textFieldValue = remember {
         mutableStateOf(
             TextFieldValue(
                 text = promoCode,
                 selection = TextRange(promoCode.length)
             )
         )
     }
     val defaultDevErrorMessage = stringResource(Res.string.devErrorDesc)
     val isCancelAction = isApplied && promoCode == promoCodeApplied
 
     LaunchedEffect(promoCode) {
         if (textFieldValue.value.text != promoCode) {
             textFieldValue.value = TextFieldValue(
                 text = promoCode,
                 selection = TextRange(promoCode.length)
             )
         }
     }
 
     Surface(
         modifier = modifier
             .fillMaxWidth(),
         shape = RoundedCornerShape(Spacing.M),
     ) {
         Row(
             modifier = Modifier
                 .fillMaxWidth()
                 .background(color = Colors.black_01)
                 .padding(Spacing.M),
             horizontalArrangement = Arrangement.spacedBy(Spacing.M),
         ) {
             Box(Modifier.weight(1f).fillMaxWidth().padding(top = Spacing.S)) {
                 PromotionInput(
                     textFieldValue = textFieldValue.value,
                     placeholder = stringResource(Res.string.promoCodeInputPlaceholder),
                     floatingValue = stringResource(Res.string.promoCodeInputLabel),
                     onChangeText = {
                         textFieldValue.value = TextFieldValue(
                             text = it.text,
                             selection = TextRange(it.text.length)
                         )
                         onChange(it.text)
                     },
                     disabled = isChecking,
                     error = errorMessage?.ifBlank {
                         defaultDevErrorMessage
                     } ?: "",
                     modifier = Modifier.focusRequester(focusRequester ?: FocusRequester()),
                 )
             }
 
             if (withInlineButton == true) {
                 Box(Modifier.padding(top = Spacing.S)) {
                     Button(
                         title = stringResource(
                             if (isCancelAction) {
                                 Res.string.cancelLabel
                             } else Res.string.checkPromoCode
                         ),
                         onClick = {
                             if (isCancelAction) {
                                 onChange("")
                                 onValueCleared()
                             } else {
                                 onSubmit(textFieldValue.value.text)
                             }
                         },
                         isFull = false,
                         modifier = Modifier.widthIn(96.dp, 124.dp),
                         type = if (isChecking || textFieldValue.value.text.isBlank()) {
                             ButtonType.DISABLED
                         } else {
                             ButtonType.PRIMARY
                         },
                     )
                 }
             }
         }
     }
 }
