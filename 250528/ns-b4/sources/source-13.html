 @OptIn(ExperimentalMaterial3Api::class)
 @Composable
 fun VoucherCoinBottomSheet(
     isVisible: Boolean = false,
     onDismiss: () -> Unit = {},
     sheetState: SheetState,
     vouchers: List<IVoucherBase> = emptyList(),
     selectedVoucher: IVoucherBase? = null,
     coinInfo: VoucherCoinInfo? = null,
     isFetching: Boolean? = false,
     onFetchNextPage: () -> Unit? = {},
     onToggleVoucherSelect: (IVoucherBase) -> Unit? = {},
     onClaimVoucher: (voucher: IVoucherBase, index: Int) -> Unit? = { _, _ -> },
     onVoucherDisplayed: (voucher: IVoucherBase) -> Unit? = {},
     isClaiming: Boolean = false,
     scrollState: LazyListState = rememberLazyListState(),
 ) {
     val screenDimension = getScreenDimensions()
     val maxHeight = screenDimension.screenHeight * 0.5f
 
     PromotionBottomSheet(
         isVisible = isVisible,
         onDismiss = onDismiss,
         sheetState = sheetState,
         headerTitle = stringResource(Res.string.voucherCoinSheetTitle),
         contentWindowInsets = { BottomSheetDefaults.windowInsets }
     ) {
         Column(
             Modifier
                 .heightIn(max = maxHeight.dp)
                 .background(color = AppTheme.current.colors.background.default)
         ) {
             TotalCoins(
                 value = coinInfo?.numberCoins ?: 0,
                 modifier = Modifier
                     .fillMaxWidth()
                     .background(AppTheme.current.colors.background.default)
                     .padding(Spacing.M),
             )
             LazyColumn(
                 modifier = Modifier
                     .weight(1f),
                 state = scrollState,
                 verticalArrangement = Arrangement.spacedBy(Spacing.S),
                 contentPadding = PaddingValues(top = Spacing.XXS, bottom = Spacing.M)
             ) {
                 if (vouchers.isNotEmpty()) {
                     items(
                         items = vouchers,
                         key = { it.extra?.rewardId ?: Random.nextInt() },
                     ) { voucher ->
                         Row(
                             Modifier.animateItem(
                                 fadeInSpec = tween(durationMillis = 250),
                                 fadeOutSpec = tween(durationMillis = 100),
                                 placementSpec = spring(
                                     stiffness = Spring.StiffnessLow,
                                     dampingRatio = Spring.DampingRatioMediumBouncy
                                 )
                             ).padding(horizontal = Spacing.M)
                         ) {
                             VoucherCoinItem(
                                 voucher = voucher,
                                 onToggleSelect = {
                                     if (!isClaiming) {
                                         onToggleVoucherSelect(voucher)
                                     }
                                 },
                                 selected = selectedVoucher?.extra?.rewardId == voucher.extra?.rewardId,
                                 onDisplayed = { onVoucherDisplayed(voucher) },
                             )
                         }
                     }
                 }
 
                 if (isFetching == true) {
                     item {
                         Column(
                             verticalArrangement = Arrangement.spacedBy(Spacing.S),
                             modifier = Modifier
                                 .fillMaxSize()
                                 .padding(horizontal = Spacing.M)
                         ) {
                             repeat(5) {
                                 VoucherSkeleton()
                             }
                         }
                     }
                 }
             }
 
             Footer(
                 withBottomPadding = false
             )
             {
                 Button(
                     title = stringResource(Res.string.voucherCoinSheetAction),
                     onClick = {
                         if (selectedVoucher != null) {
                             onClaimVoucher(
                                 selectedVoucher,
                                 vouchers.indexOfFirst { it.extra?.rewardId == selectedVoucher.extra?.rewardId }
                             )
                         }
                     },
                     type = if (isClaiming || selectedVoucher == null) {
                         ButtonType.DISABLED
                     } else {
                         ButtonType.PRIMARY
                     }
                 )
             }
         }
     }
 
     LaunchedEffect(scrollState) {
         snapshotFlow {
             val visibleItems = scrollState.layoutInfo.visibleItemsInfo
             val totalItems = scrollState.layoutInfo.totalItemsCount
             visibleItems.lastOrNull()?.index to totalItems
         }.collect { (lastVisibleItemIndex, totalItems) ->
             if (lastVisibleItemIndex != null && totalItems > 0 && lastVisibleItemIndex >= totalItems - 2) {
                 onFetchNextPage()
             }
         }
     }
 }
 
 @Composable
 fun TotalCoins(
     value: Long = 0,
     modifier: Modifier = Modifier,
 ) {
     Row(
         modifier = modifier,
         verticalAlignment = Alignment.CenterVertically,
         horizontalArrangement = Arrangement.spacedBy(Spacing.XS)
     ) {
         if (value < 0) {
             Text(
                 text = stringResource(Res.string.voucherCoinSheetInsufficientCoinLabelPrefix),
                 color = AppTheme.current.colors.text.secondary,
                 style = Typography.descriptionDefaultRegular
             )
             Image(
                 source = rememberAsyncImagePainter(Images.ic_rewards_coin),
                 modifier = Modifier.size(Spacing.L),
             )
             Text(
                 text = stringResource(Res.string.voucherCoinSheetInsufficientCoinLabel)
                     .replace(
                         "{userCoin}",
                         abs(value).toDouble().formatCurrency(currency = "", separator = ".")
                     ),
                 color = AppTheme.current.colors.text.secondary,
                 style = Typography.actionXsBold
             )
             Text(
                 text = stringResource(Res.string.voucherCoinSheetInsufficientCoinLabelSuffix),
                 color = AppTheme.current.colors.text.secondary,
                 style = Typography.descriptionDefaultRegular
             )
         } else {
             Text(
                 text = stringResource(Res.string.voucherCoinSheetCoinLabelPrefix),
                 color = AppTheme.current.colors.text.secondary,
                 style = Typography.descriptionDefaultRegular
             )
             Image(
                 source = rememberAsyncImagePainter(Images.ic_rewards_coin),
                 modifier = Modifier.size(Spacing.L),
             )
             Text(
                 text = stringResource(Res.string.voucherCoinSheetCoinLabelSuffix)
                     .replace(
                         "{userCoin}",
                         abs(value).toDouble().formatCurrency(currency = "", separator = ".")
                     ),
                 color = AppTheme.current.colors.text.secondary,
                 style = Typography.actionXsBold
             )
         }
     }
