 class CollectVoucherSlice(
     private val scope: CoroutineScope,
     private val repository: IVoucherRepository,
     private val navigator: INavigator,
     private val logger: ILogger,
 ) : ViewModelSlice<CollectVoucherReducer.State, CollectVoucherReducer.Action, CollectVoucherReducer.Effect>(
     initialState = CollectVoucherReducer.State(),
     reducer = CollectVoucherReducer(),
     scope,
 ) {
     fun onCollect(
         voucher: VoucherBase,
         onSuccess: ((collectedVoucher: IVoucherBase) -> Unit)? = null,
         onError: ((errorMessage: String?) -> Unit)? = null,
     ) {
         if (!voucher.extra?.collectId.isNullOrEmpty()) {
             scope.launch {
                 repository
                     .collectVoucher(voucher.extra?.collectId!!, voucher.extra.campaignId)
                     .take(1)
                     .onStart {
                         logger.i("onCollectVoucher start")
                         navigator.showLoading()
                         dispatch(
                             CollectVoucherReducer.Action.SetCollectVoucherState(
                                 collectingVoucherId = voucher.extra.collectId,
                                 collectError = null,
                             )
                         )
                     }
                     .onEach {
                         logger.i("onCollectVoucher result $it")
                         navigator.hideLoading()
                         dispatch(
                             CollectVoucherReducer.Action.SetCollectVoucherState(
                                 collectingVoucherId = null,
                                 collectError = null,
                             )
                         )
                         if (it?.result?.collectStatus == true) {
                             val collectedVoucher = voucher.copy(
                                 ID = it.result.giftId,
                                 localType = if (voucher.localType == VoucherLocalType.GiftMask) {
                                     VoucherLocalType.Recommend
                                 } else {
                                     VoucherLocalType.Invalid
                                 },
                                 status = 1,
                             )
                             navigator.showToast(
                                 buildJsonObject {
                                     put("title", getString(Res.string.voucherCollected))
                                     put("type", "success")
                                     put("duration", 2000)
                                 }, String.serializer()
                             )
                             onSuccess?.invoke(collectedVoucher)
                         } else {
                             if (it?.result?.collectCode == 5) {
                                 navigator.showToast(
                                     buildJsonObject {
                                         put("title", getString(Res.string.voucherExpire))
                                         put("type", "failure")
                                         put("duration", 2000)
                                     }, String.serializer()
                                 ).launchIn(scope)
                                 throw Exception("Localization.voucherExpire")
                             }
 
                             if (listOf(1001, 1002, 1003, 1004).contains(
                                     it?.result?.collectCode ?: 0
                                 )
                             ) {
                                 if (it?.result?.popup?.body == null) {
                                     navigator.showToast(
                                         buildJsonObject {
                                             put("title", getString(Res.string.dontSaveVoucher))
                                             put("type", "warning")
                                             put("duration", 2000)
                                         }, String.serializer()
                                     ).launchIn(scope)
                                     return@onEach
                                 }
 
                                 val popup = it.result.popup
                                 navigator.showDialog(
                                     title = getString(Res.string.noticeCollectVoucher),
                                     body = popup.body ?: "",
                                     buttons = listOf(
                                         popup.cta ?: "",
                                         getString(Res.string.close)
                                     )
                                 ).onEach { index ->
                                     if (index == 0) {
                                         navigator.navigateFeatureCode(
                                             popup.refId ?: "",
                                             popup.params ?: emptyJsonObject()
                                         )
                                     }
                                 }
                                     .launchIn(scope)
                                 return@onEach
                             }
                             navigator.showToast(
                                 buildJsonObject {
                                     put("title", getString(Res.string.collectVoucherMessageFailed))
                                     put("type", "warning")
                                     put("duration", 2000)
                                 }, String.serializer()
                             ).launchIn(scope)
                         }
                     }
                     .catch {
                         navigator.hideLoading()
                         logger.e("onCollectVoucher error $it")
                         onError?.invoke(it.message)
                         dispatch(
                             CollectVoucherReducer.Action.SetCollectVoucherState(
                                 collectingVoucherId = null,
                                 collectError = it.message,
                             )
                         )
                     }
                     .launchIn(this)
             }
         }
     }
 
 }
