 class VoucherCoinSlice(
     private val scope: CoroutineScope,
     private val repository: IVoucherRepository,
     private val navigator: INavigator,
     private val params: ChoosePromotionScreenProps,
     private val logger: ILogger,
     private val promotionTracer: IPromotionVMTracking,
 ) : ViewModelSlice<VoucherCoinReducer.State, VoucherCoinReducer.Action, VoucherCoinReducer.Effect>(
     initialState = VoucherCoinReducer.State(),
     reducer = VoucherCoinReducer(),
     scope,
 ), IPromotionVMTracking by promotionTracer {
 
     val voucherCoinQuery: InfiniteQuery<IVoucherCoin> =
         scope.createInfiniteQuery(
             queryKey = "RecommendVoucherCoin",
             queryFn = { page ->
                 repository.getListVoucherCoin(
                     page,
                     params
                 )
             },
             onEachPage = {
                 val page = it as? VoucherCoinListPage
                 if (page?.coinInfo != null) {
                     dispatch(
                         VoucherCoinReducer.Action.SetVoucherCoinState(
                             coinInfo = page.coinInfo,
                             isVoucherCoinBottomSheetVisible = state.value.isVoucherCoinBottomSheetVisible,
                             selectedVoucherCoin = state.value.selectedVoucherCoin,
                             claimedVouchers = state.value.claimedVouchers,
                         )
                     )
                 }
             },
             enabled = params.isSingleBillFlow != true
         )
 
     fun toggleVoucherCoinSelect(voucher: IVoucherCoin) {
         if (voucher.extra?.isInsufficientCoin == true) {
             trackVoucherCoinItemClicked(
                 giftType = voucher.typeId ?: "",
                 campaignId = voucher.extra?.campaignId ?: voucher.campaign,
                 status = "invalid",
             )
             val params = VoucherDetailParams(
                 giftType = when {
                     !voucher.giftType.isNullOrBlank() -> voucher.giftType
                     !voucher.typeId.isNullOrBlank() -> voucher.typeId
                     else -> null
                 },
                 campaignId = voucher.extra?.campaignId,
                 widgetId = "REWARD_DETAIL",
                 source = ChoosePromotionSource.ttat.toString(),
                 isInsufficientCoin = true,
             )
             navigator.navigateFeatureCode(
                 "instant_voucher_detail",
                 ModelConvertUtils.encodeToJson(serializer(), params)?.jsonObject
                     ?: emptyJsonObject(),
             )
         } else {
             val isSelected = state.value.selectedVoucherCoin?.extra?.rewardId == voucher.extra?.rewardId
             if (!isSelected) {
                 trackVoucherCoinItemSelected(
                     giftType = voucher.typeId ?: "",
                     campaignId = voucher.extra?.campaignId ?: voucher.campaign
                 )
             }
             dispatch(
                 VoucherCoinReducer.Action.SetVoucherCoinState(
                     selectedVoucherCoin = if (isSelected) {
                         null
                     } else {
                         voucher
                     },
                     isVoucherCoinBottomSheetVisible = state.value.isVoucherCoinBottomSheetVisible,
                     coinInfo = state.value.coinInfo,
                     claimedVouchers = state.value.claimedVouchers,
                 )
             )
         }
     }
 
     fun claimVoucherCoin(
         voucher: IVoucherCoin,
         onSuccess: ((claimedVouchers: IVoucherBase) -> Unit)? = null,
         onError: ((errorMessage: String?) -> Unit)? = null,
     ) {
         if (!voucher.extra?.rewardId.isNullOrEmpty()) {
             trackVoucherCoinExchangeNowClicked(
                 giftType = voucher.typeId ?: "",
                 campaignId = voucher.extra?.campaignId ?: voucher.campaign,
             )
             scope.launch {
                 repository
                     .claimVoucherCoin(voucher.extra?.rewardId!!)
                     .take(1)
                     .onStart {
                         logger.i("onClaimVoucherCoin start")
                         navigator.showLoading()
                         dispatch(
                             VoucherCoinReducer.Action.SetClaimingVoucherState(
                                 isClaiming = true,
                                 errorMessage = null,
                             )
                         )
                     }
                     .onEach {
                         logger.i("onClaimVoucherCoin result $it")
                         dispatch(
                             VoucherCoinReducer.Action.SetClaimingVoucherState(
                                 isClaiming = false,
                                 errorMessage = null,
                             )
                         )
                         if (!it?.giftId.isNullOrEmpty()) {
                             val giftId = it?.giftId!!
                             val prefixName = when {
                                 !voucher.prefixName.isNullOrBlank() -> voucher.prefixName
                                 !voucher.extra?.prefixName.isNullOrBlank() -> voucher.extra?.prefixName
                                 !voucher.extraGiftType?.prefixName.isNullOrBlank() -> voucher.extraGiftType?.prefixName
                                 else -> null
                             }
                             val suffixName = when {
                                 !voucher.suffixName.isNullOrBlank() -> voucher.suffixName
                                 !voucher.extra?.suffixName.isNullOrBlank() -> voucher.extra?.suffixName
                                 !voucher.extraGiftType?.suffixName.isNullOrBlank() -> voucher.extraGiftType?.suffixName
                                 else -> null
                             }
                             val virtualVoucher: IVoucherBase = VoucherBase(
                                 ID = giftId,
                                 id = giftId,
                                 modelId = giftId,
                                 prefixName = prefixName,
                                 suffixName = suffixName,
                                 icon = voucher.icon,
                                 extra = VoucherExtra(
                                     voucherServiceName = voucher.extra?.voucherServiceName,
                                     failGiftDesc = voucher.extra?.failGiftDesc,
                                     campaign = voucher.campaign,
                                 ),
                                 groups = voucher.groups,
                                 type_ = voucher.type_,
 
                                 localType = if (voucher.extra?.isInvalid == true) {
                                     VoucherLocalType.Invalid
                                 } else {
                                     VoucherLocalType.Recommend
                                 },
                                 generatedAt = TimeUtils.getCurrentTimeMillis(),
                             )
                             dispatch(
                                 VoucherCoinReducer.Action.SetVoucherCoinState(
                                     isVoucherCoinBottomSheetVisible = false,
                                     coinInfo = state.value.coinInfo,
                                     selectedVoucherCoin = null,
                                     claimedVouchers = state.value.claimedVouchers + virtualVoucher,
                                 )
                             )
 
                             onSuccess?.invoke(virtualVoucher)
 
                             voucherCoinQuery.onRefetch.invoke()
                         }
 
                         navigator.hideLoading()
                     }
                     .catch {
                         logger.i("onClaimVoucherCoin error $it")
                         dispatch(
                             VoucherCoinReducer.Action.SetClaimingVoucherState(
                                 isClaiming = false,
                                 errorMessage = it.message,
                             )
                         )
                         navigator.hideLoading()
                         navigator.showToast(
                             buildJsonObject {
                                 put("title", it.message ?: getString(Res.string.devErrorDesc))
                                 put("type", "failure")
                                 put("duration", 2000)
                             }, String.serializer()
                         ).launchIn(scope)
                         onError?.invoke(it.message)
                     }
                     .launchIn(this)
             }
         }
     }
 
     fun toggleVoucherCoinBottomSheet(isVisible: Boolean) {
         dispatch(
             VoucherCoinReducer.Action.SetVoucherCoinState(
                 isVoucherCoinBottomSheetVisible = isVisible,
                 coinInfo = state.value.coinInfo,
                 selectedVoucherCoin = state.value.selectedVoucherCoin,
                 claimedVouchers = state.value.claimedVouchers,
             )
         )
     }
