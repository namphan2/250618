 package vn.momo.compose.payment.napas.presentation.utils.extension
 
 sealed class ValidateType {
     data class Required(val message: String) : ValidateType()
     data class Length(val minLength: Int, val message: String) : ValidateType()
     data class MatchRegex(val regex: Regex, val message: String) : ValidateType()
 }
 
 class FormValidator(private val rules: List<ValidateType>) {
 
     fun validate(input: String?): ValidateType? {
         val ruleFound =  rules.firstOrNull { rule ->
             when (rule) {
                 is ValidateType.Required -> input.isNullOrBlank()
                 is ValidateType.Length -> (input?.length ?: 0) < rule.minLength
                 is ValidateType.MatchRegex -> input?.matches(rule.regex) == false
             }
         }
         return ruleFound
     }
 
     fun getErrorMessage(input: String?): String? {
         val rule = validate(input)
         return rule?.run {
             when (this) {
                 is ValidateType.Required -> message
                 is ValidateType.Length -> message
                 is ValidateType.MatchRegex -> message
             }
         }
     }
 }
