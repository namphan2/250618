 import kotlinx.coroutines.flow.Flow
 import kotlinx.serialization.json.JsonObject
 import kotlinx.serialization.json.buildJsonObject
 import kotlinx.serialization.json.put
 import vn.momo.compose.payment.base.core.composeApi.INavigator
 import vn.momo.compose.payment.payx.domain.entities.ConfirmResponse
 
 class OpenFraudOtpUseCase(
     private val navigator: INavigator
 ) {
     operator fun invoke(data: ConfirmResponse?): Flow<JsonObject> {
         val referenceKey = data?.momoMsg?.referenceKey ?: ""
         return navigator.showBottomSheet(
             params = buildJsonObject {
                 put("type", "otp")
                 put("referenceKey", referenceKey)
                 put("flow", if (data == null) "quickFraud" else "fraud")
                 put("paymentType", "multi")
             }
         )
     }
 }
