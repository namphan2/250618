 package vn.momo.compose.payment.payx.presentation.viewModel.reducers
 
 import vn.momo.compose.payment.base.presentation.viewModel.reducer.Reducer
 import vn.momo.core.features.annotation.Immutable
 
 class PaymentQuickPayReducer :
     Reducer<PaymentQuickPayReducer.QuickPayState, PaymentQuickPayReducer.QuickPayAction, PaymentQuickPayReducer.QuickPayEffect> {
 
     @Immutable
     data class QuickPayState(
         val enableConfirm: Boolean = false,
         val statusDesc: String = "",
     ) : Reducer.State
 
     @Immutable
     sealed class QuickPayAction : Reducer.Action {
         object OnPressConfirm : QuickPayAction()
         object RequestClose : QuickPayAction()
         object BeginCheckout : QuickPayAction()
         object ErrorCheckout : QuickPayAction()
         object ErrorAddCart : QuickPayAction()
         object OnCheckoutSuccess: QuickPayAction()
         object OnRedirectResult: QuickPayAction()
         object DismissConfirmScreen: QuickPayAction()
     }
 
     @Immutable
     sealed class QuickPayEffect : Reducer.Effect {
         object RequestConfirmPayment : QuickPayEffect() //share action for PaymentConfirmVM
 //        data class CheckShowConfirmScreen(val isShow: Boolean): QuickPayEffect()
         object ShowDismissConfirmScreenPopup : QuickPayEffect()
         object DismissConfirmScreen: QuickPayEffect()
     }
 
     override fun reduce(
         prev: QuickPayState,
         action: QuickPayAction
     ): Pair<QuickPayState, QuickPayEffect?> {
         return when (action) {
             is QuickPayAction.OnPressConfirm -> onPressConfirm(prev)
             is QuickPayAction.BeginCheckout -> disableConfirm(prev) to null
             is QuickPayAction.OnCheckoutSuccess -> onCheckoutSuccess(prev)
             is QuickPayAction.ErrorAddCart -> onAddCartError(prev)
             is QuickPayAction.ErrorCheckout -> onCheckoutError(prev)
             is QuickPayAction.RequestClose -> requestClose(prev)
             is QuickPayAction.OnRedirectResult -> onRedirectResult(prev)
             is QuickPayAction.DismissConfirmScreen -> prev to QuickPayEffect.DismissConfirmScreen
         }
     }
 
     private fun onRedirectResult(prev: QuickPayState): Pair<QuickPayState, QuickPayEffect?> {
         return prev to QuickPayEffect.DismissConfirmScreen
     }
 
     private fun onPressConfirm(prev: QuickPayState): Pair<QuickPayState, QuickPayEffect?> {
         return disableConfirm(prev) to QuickPayEffect.RequestConfirmPayment
     }
 
     private fun onAddCartError(prev: QuickPayState): Pair<QuickPayState, QuickPayEffect?> {
         //Confirm VM handle self it
         return disableConfirm(prev) to null
     }
 
     private fun onCheckoutError(prev: QuickPayState): Pair<QuickPayState, QuickPayEffect?> {
         //Confirm VM handle self it
         return disableConfirm(prev) to QuickPayEffect.ShowDismissConfirmScreenPopup
     }
 
     private fun requestClose(prev: QuickPayState): Pair<QuickPayState, QuickPayEffect?> {
         return prev to QuickPayEffect.DismissConfirmScreen
     }
 
     private fun onCheckoutSuccess(prev: QuickPayState): Pair<QuickPayState, QuickPayEffect?> {
         return enableConfirm(prev) to null
     }
 
     private fun disableConfirm(prev: QuickPayState): QuickPayState {
         return prev.copy(enableConfirm = false, statusDesc = "Checkout processing")
     }
 
     private fun enableConfirm(prev: QuickPayState): QuickPayState {
         return prev.copy(enableConfirm = true, statusDesc = "Checkout ready")
     }
 
 
 
 }
