 @Composable
 fun NapasInputScreen(
     vm: NapasInputVM,
 ) {
     val state by vm.state.collectAsState()
     val floatingIcon =
         "https://img.mservice.io/momo_app_v2/new_version/img/appx_icon/24_notifications_icon24_info.png"
     val focusManager = LocalFocusManager.current
 
     @Composable
     fun InputByField(
         field: FormField,
         placeHolder: String,
         floatingValue: String,
         keyboardType: KeyboardType,
         maxLength: Int? = null,
         visualTransformation: VisualTransformation? = null
     ) {
         val fieldState = state.formState[field]
         val fieldData = vm.formFieldData[field]
         var fieldValue by remember {
             mutableStateOf(fieldState?.value ?: "")
         }
 
         LaunchedEffect(fieldState?.value) {
             if (fieldValue != fieldState?.value) {
                 fieldValue = fieldState?.value ?: ""
             }
         }
 
         if (fieldData?.isHide == false) {
             PaymentInput(
                 text = fieldValue,
                 placeholder = placeHolder,
                 floatingValue = floatingValue,
                 onChangeText = {
                     if (maxLength == null || it.length <= maxLength) {
                         fieldValue = it
                         vm.onChangeForm(field, it)
                     }
                 },
                 error = fieldState?.errorMessage ?: "",
                 errorSpacing = true,
                 onBlur = { vm.onBlurForm(field) },
                 floatingIcon = fieldData.popupImage?.let {
                     floatingIcon
                 } ?: "",
                 floatingIconColor = Colors.blue_04,
                 onClickFloating = {
                     vm.showInstruction(field)
                 },
                 disabled = fieldData.isDisable ?: false,
                 keyboardType = keyboardType,
                 visualTransformation = visualTransformation
             )
         }
     }
 
     @Composable
     fun InputDateByField(
         field: FormField,
         placeHolder: String,
         floatingValue: String,
     ) {
         val fieldState = state.formState[field]
         val fieldData = vm.formFieldData[field]
         if (fieldData?.isHide == false) {
             Box(
                 modifier = Modifier
                     .pointerInput(Unit) {
                         awaitPointerEventScope {
                             while (true) {
                                 val event = awaitPointerEvent()
                                 if (event.type == PointerEventType.Press) {
                                     event.changes.forEach { it.consume() }
                                 }
                             }
                         }
                     }.clickable {
                         vm.showDateSelection(field)
                     }
             ) {
                 PaymentInput(
                     text = fieldState?.value ?: "",
                     onChangeText = {
                         vm.onChangeForm(field, it)
                     },
                     placeholder = placeHolder,
                     floatingValue = floatingValue,
                     error = fieldState?.errorMessage ?: "",
                     errorSpacing = true,
                     floatingIcon = fieldData?.popupImage?.let { floatingIcon } ?: "",
                     onClickFloating = {
                         vm.showInstruction(field)
                     },
                     floatingIconColor = Colors.blue_04,
                     readOnly = true,
                     icon = "24_time_calendar",
                     iconColor = Colors.blue_04,
                     onRightIconPressed = {
                         vm.showDateSelection(field)
                     },
                     keyboardType = KeyboardType.Text
                 )
             }
         }
     }
 
     @Composable
     fun FormInput() {
         Column(
             verticalArrangement = Arrangement.spacedBy(Spacing.S)
         ) {
             Text(
                 stringResource(Res.string.paymentNapasTitle),
                 style = Typography.headerMBold,
                 color = Colors.black_17,
                 modifier = Modifier.padding(bottom = Spacing.L)
             )
             InputByField(
                 FormField.CARD_NUMBER,
                 stringResource(Res.string.placeHolderCardNumber),
                 stringResource(Res.string.titleCardNumber),
                 keyboardType = KeyboardType.Number,
                 maxLength = 19,
                 visualTransformation = CardVisualTransformation()
             )
             InputByField(
                 FormField.CARD_FULL_NAME,
                 stringResource(Res.string.holderNamePlaceHolder),
                 stringResource(Res.string.holderNameTitle),
                 keyboardType = KeyboardType.Text,
                 maxLength = 100,
             )
             InputDateByField(
                 FormField.CARD_EXPIRE_DATE,
                 stringResource(Res.string.cardExpireDateTitlePlaceHolder),
                 stringResource(Res.string.cardExpireDateTitle)
             )
             InputDateByField(
                 FormField.CARD_ISSUE_DATE,
                 stringResource(Res.string.cardIssueDateTitlePlaceHolder),
                 stringResource(Res.string.cardIssueDateTitle)
             )
             InputByField(
                 FormField.PERSONAL_ID,
                 stringResource(Res.string.cardIssueDateTitlePlaceHolder),
                 stringResource(Res.string.cardIssueDateTitle),
                 keyboardType = KeyboardType.Text
             )
             if (vm.props.isMapNapas == false) CheckBox(
                 title = stringResource(Res.string.saveCardNextTime),
                 checked = state.checkedCardNextTime,
                 onCheckedChange = {
                     vm.onToggleSaveCardNextTime(it)
                 }
             )
         }
     }
 
     Screen(
         backgroundColor = Colors.black_01,
         goBack = {
             vm.onBackPress()
         },
         title = stringResource(Res.string.paymentNapasScreenTitle),
         content = {
             Column(
                 modifier = Modifier.fillMaxSize().padding(Spacing.M).pointerInput(Unit) {
                     detectTapGestures(onTap = {
                         focusManager.clearFocus()
                     })
                 },
                 verticalArrangement = Arrangement.spacedBy(Spacing.M)
             ) {
                 BannerHeader(
                     if (vm.props.isClaimCodeNapas == true) stringResource(Res.string.hasPromotion) else stringResource(
                         Res.string.promotionInfo
                     )
                 )
                 TransactionInfo()
                 FormInput()
                 SecurityBlock()
             }
         },
         footer = {
             Button(
                 title = stringResource(Res.string.btnContinue),
                 type = if (state.formState.values.all { it.isValid() }) ButtonType.PRIMARY else ButtonType.DISABLED,
                 onClick = {
                     vm.onSubmit()
                 }
             )
         }
     )
