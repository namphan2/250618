 class SBPaymentStatusRepository(
     private val remoteDataSource: SBGetPaymentStatusP2PDataSource,
     private val savedSingleBillData: SBSavedTransaction,
     private val subject: IObserverService,
     private val logger: ILogger
 ) : PaymentStatusRepository {
     override fun getPaymentStatus(): Flow<JsonObject> {
         return getPaymentStatusP2P().filter { it.isNotEmpty() }
     }
 
     private fun getPaymentStatusP2P(): Flow<JsonObject> {
         return remoteDataSource.request().map {
             if (it.success == true && it.data?.isNotEmpty() == true) {
                 return@map it.data
             }
             throw SingleBillError.CustomPopupError(
                 code = it.resultCode ?: DevError.NoResponse.code,
                 message = it.errorDesc ?: "",
                 cause = Exception(this::class.simpleName),
                 dialog = PayXDialog(
                     errorCode = it.resultCode ?: DevError.NoResponse.code,
                     title = getString(Res.string.serviceInterrupted),
                     content = getString(Res.string.serviceInterruptedDesc),
                     buttons = listOf(
                         PayXButton(
                             title = getString(Res.string.navigateToTransHis),
                             refId = "tabbar_transaction"
                         )
                     )
                 )
             )
         }
     }
 
     override fun listenMqtt(): Flow<JsonObject> {
         return subject.observe(MQTT_UPDATE_PAYMENT_STATUS, JsonObject.serializer()).filter {
             logger.i("[PayX listenMqtt: $it]")
             val momoMsg =
                 ModelConvertUtils.decode(MqttData.serializer(), it ?: emptyJsonObject())?.momoMsg
             val isMatchTransaction =
                 momoMsg?.purchaseInfo?.purchaseId?.content == savedSingleBillData.data?.tranHisMsg?.tranHisMsgs?.firstOrNull()?.tranId?.toString()
             val isP2P = momoMsg?.flowType == "P2P"
             return@filter isMatchTransaction && isP2P
         }.map {
             return@map it ?: emptyJsonObject()
         }.filter { it.isNotEmpty() }
     }
 }
