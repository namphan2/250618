 @Composable
 fun GuideTooltip(
     modifier: Modifier = Modifier,
     isShow: Boolean,
     text: String,
     onClick: () -> Unit = {},
     requestHide: () -> Unit
 ) {
     val localDensity = LocalDensity.current
     var tooltipHeightPx by remember { mutableStateOf(0) }
     val tooltipOffsetY = with(localDensity) { -tooltipHeightPx.toDp() }
 
     LaunchedEffect(isShow) {
         if(isShow) {
             delay(5.seconds)
             requestHide()
         }
     }
 
     AnimatedVisibility(
         visible = isShow,
         enter = fadeIn(tween(delayMillis = 500)),
         exit = fadeOut(),
         modifier = modifier.onGloballyPositioned { coordinates ->
             tooltipHeightPx = coordinates.size.height
         }
             .offset(y = tooltipOffsetY - 12.dp)
             .noFeedbackClickable(onClick = onClick)
     ) {
         Column(horizontalAlignment = Alignment.End) {
             Box(
                 Modifier.clip(RoundedCornerShape(Radius.S))
                     .background(AppTheme.current.colors.interactive.primary)
                     .padding(12.dp)
             ) {
                 Text(text, style = Typography.bodyDefaultRegular, color = Colors.black_01)
             }
             Box(
                 modifier = Modifier
                     .padding(end = 12.dp)
                     .background(
                         color = AppTheme.current.colors.interactive.primary,
                         shape = TriangleShape()
                     )
                     .width(12.dp)
                     .height(6.dp)
             )
         }
     }
 }
 
 private class TriangleShape() : Shape {
     override fun createOutline(
         size: Size,
         layoutDirection: LayoutDirection,
         density: Density
     ): Outline {
         val path = Path().apply {
             moveTo(x = size.width / 2, y = size.height)
             lineTo(x = 0f, y = 0f)
             lineTo(x = size.width, y = 0f)
         }
         return Outline.Generic(path)
     }
