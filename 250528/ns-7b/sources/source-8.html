 @Composable
 fun SofGroupView(group: SOFGroup, selected: SOFItem?, onSelect: (SOFItem) -> Unit = {}) {
     val density = LocalDensity.current
     val screenWidth = getScreenDimensions().screenWidth
     val scrollState = rememberScrollState()
 
     val index by remember(group.children, selected) {
         derivedStateOf {
             group.children.indexOfFirst {
                 it.isMatchSource(selected?.sourceToken, selected?.ID)
             }
         }
     }
 
     LaunchedEffect(index) {
         if (index >= 0) {
             scrollState.animateScrollTo(with(density) { ((screenWidth / 3 + 8).dp * index).toPx() }.toInt())
         }
     }
 
     Column(verticalArrangement = scaledSpacedBy(8f)) {
         SofCard.Default(
             sof = group.parent,
             isSelected = group.children.any { it.sourceToken == selected?.sourceToken })
         Row(
             Modifier.padding(start = 32.dp).horizontalScroll(scrollState),
             horizontalArrangement = scaledSpacedBy(8f),
             verticalAlignment = Alignment.CenterVertically
         ) {
             group.children.onEach { installment ->
                 key(installment.uId) {
                     val isSelected = installment.isMatchSource(selected?.sourceToken, selected?.ID)
                     SofCard.MiniSof(
                         modifier = Modifier.widthIn(max = (screenWidth / 3).dp),
                         sof = installment,
                         isSelected = isSelected,
                         onPress = {
                             onSelect(installment)
                         })
                 }
             }
         }
     }
 }
