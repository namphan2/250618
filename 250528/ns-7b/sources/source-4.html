 @OptIn(ExperimentalMaterial3Api::class)
 @Composable
 fun FooterV2(
     amount: TotalAmount,
     hideAmount: Boolean,
     hasAppliedPromotion: Boolean,
     enableConfirm: Boolean,
     onConfirm: () -> Unit,
     amountInfo: List<InfoRowContent> = emptyList(),
     onToggleShowAmount: (state: String) -> Unit
 ) {
     val confirmButtonType: ButtonType =
         if (enableConfirm) ButtonType.PRIMARY else ButtonType.DISABLED
     var showBottomSheet = remember { mutableStateOf(false) }
 
     LaunchedEffect(showBottomSheet.value) {
         onToggleShowAmount(if (showBottomSheet.value == true) "expand" else "collapse")
     }
 
     if (hideAmount) {
         ConfirmButton(hideAmount, confirmButtonType, onConfirm)
     } else {
         Row(
             modifier = Modifier.fillMaxWidth(),
             verticalAlignment = Alignment.CenterVertically,
             horizontalArrangement = scaledSpacedBy(12f)
         ) {
             Row(
                 modifier = Modifier.weight(1f),
                 verticalAlignment = Alignment.CenterVertically,
                 horizontalArrangement = scaledSpacedBy(8f)
             ) {
                 Row(
                     verticalAlignment = Alignment.CenterVertically,
                     modifier = Modifier.noFeedbackClickable(onClick = debounceCompose {
                         if (amountInfo.isNotEmpty()) {
                             showBottomSheet.value = true
                         }
                     })
                 ) {
                     if (amountInfo.isNotEmpty()) {
                         Icon(
                             source = "arrow_chevron_up_small",
                             size = 24.dp,
                             color = AppTheme.current.colors.text.default,
                         )
                     }
                     Text(
                         text = stringResource(Res.string.totalPayment),
                         style = Typography.descriptionDefaultRegular,
                         color = AppTheme.current.colors.text.default,
                     )
                 }
                 Column(modifier = Modifier.weight(1f), horizontalAlignment = Alignment.End) {
                     val isPromotionApplied =
                         amount.totalBeforePromotion.isShow && hasAppliedPromotion
                     if (amount.total.isShow) {
                         AutoResizedText(
                             text = amount.total.format,
                             style = Typography.headerMBold,
                             maxLines = 1,
                             textAlign = TextAlign.End,
                             color = if (isPromotionApplied) AppTheme.current.colors.success.primary else AppTheme.current.colors.text.default,
                         )
                     }
                     if (isPromotionApplied) {
                         AutoResizedText(
                             text = amount.totalBeforePromotion.format,
                             style = Typography.bodyDefaultRegular,
                             textDecoration = TextDecoration.LineThrough,
                             maxLines = 1,
                             textAlign = TextAlign.End,
                         )
                     }
                 }
             }
             ConfirmButton(hideAmount, confirmButtonType, onConfirm)
         }
     }
     AmountBts(showBottomSheet, amountInfo)
 }
 
 @Composable
 private fun ConfirmButton(
     hideAmount: Boolean,
     confirmButtonType: ButtonType,
     onConfirm: () -> Unit
 ) {
     val buttonModifier = Modifier.semantics {
         testTag = "btn_primary_confirm"
         contentDescription = "btn_primary_confirm"
     }
 
     Button(
         title = stringResource(Res.string.confirm),
         onClick = onConfirm,
         type = confirmButtonType,
         modifier = buttonModifier,
         isFull = hideAmount
     )
 }
 
 @OptIn(ExperimentalMaterial3Api::class)
 @Composable
 private fun AmountBts(
     state: MutableState<Boolean>,
     amountInfo: List<InfoRowContent>
 ) {
     val screenHeight = getScreenDimensions().screenHeight
     val sheetState = rememberModalBottomSheetState()
     val coroutineScope = rememberCoroutineScope()
 
     fun requestClose() {
         coroutineScope.launch { sheetState.hide() }
             .invokeOnCompletion {
                 if (!sheetState.isVisible) {
                     state.value = false
                 }
             }
     }
 
     if (state.value) {
         ModalBottomSheet(
             onDismissRequest = {
                 state.value = false
             },
             sheetState = sheetState,
             dragHandle = {
                 BottomSheetDefaults.DragHandle(
                     color = Colors.black_06,
                     width = 40.dp
                 )
             },
             shape = RoundedCornerShape(topStart = 8.dp, topEnd = 8.dp),
         ) {
             Column(
                 modifier = Modifier.fillMaxWidth()
                     .heightIn(max = (screenHeight * 0.65).dp)
                     .animateContentSize()
             ) {
                 BottomSheetHeader(
                     modifier = Modifier.padding(horizontal = Spacing.L),
                     title = stringResource(Res.string.paymentInformation),
                     iconRight = AsyncIconProp(
                         Images.ic_close_fat,
                         size = 14.dp,
                         onPress = { requestClose() })
                 )
                 Spacer(modifier = Modifier.height(28.dp))
                 Box(
                     modifier = Modifier
                         .padding(horizontal = 12.dp)
                         .border(
                             1.dp,
                             AppTheme.current.colors.border.default,
                             shape = RoundedCornerShape(8.dp)
                         )
 
                 ) {
                     Column {
                         amountInfo.forEach { item ->
                             InfoRowView(
                                 modifier = Modifier.scaledPadding(horizontal = 12f, vertical = 8f)
                             ) {
                                 Left {
                                     Text(
                                         text = item.key.data,
                                         style = Typography.bodyDefaultRegular,
                                         color = item.key.color
                                             ?: AppTheme.current.colors.text.secondary,
                                     )
                                 }
                                 Right {
                                     Text(
                                         text = item.value.data,
                                         style = Typography.bodyDefaultRegular,
                                         color = item.value.color
                                             ?: AppTheme.current.colors.text.default
                                     )
                                 }
                             }
                         }
                     }
                 }
                 Spacer(modifier = Modifier.height(24.dp))
                 Button(
                     modifier = Modifier.padding(horizontal = Spacing.M),
                     title = stringResource(Res.string.close), onClick = {
                         requestClose()
                     })
             }
         }
     }
 }
