 @Composable
 fun QuickActiveBanner(group: SOFGroup, onActive: () -> Unit) {
     val screenWidth = getScreenDimensions().screenWidth
     val scrollState = rememberScrollState()
     val parent = group.parent
     Column(
         Modifier.noFeedbackClickable(onClick = onActive),
         verticalArrangement = scaledSpacedBy(8f)
     ) {
         InfoCard(
             title = stringResource(Res.string.payLaterSof),
             titleExtra = {
                 Surface(shape = RoundedCornerShape(4.dp), color = Colors.blue_11) {
                     Text(
                         text = stringResource(Res.string.notUsed),
                         style = Typography.descriptionXsRegular,
                         color = Colors.blue_04,
                         modifier = Modifier.padding(horizontal = 8.dp, vertical = 2.dp)
                     )
                 }
             },
             description = stringResource(Res.string.quickActiveDesc),
             leading = {
                 Image(source = parent.logo ?: "", modifier = Modifier.size(24.dp))
             },
             extra = {},
             trailing = { RadioButton(selected = false) },
             onPress = onActive
         )
         Row(
             Modifier.padding(start = 32.dp).horizontalScroll(scrollState),
             horizontalArrangement = scaledSpacedBy(8f),
             verticalAlignment = Alignment.CenterVertically
         ) {
             group.children.onEach { installment ->
                 SofCard.MiniSof(
                     modifier = Modifier.widthIn(max = (screenWidth / 3).dp),
                     sof = installment,
                     isSelected = false,
                 )
             }
         }
     }
 }
