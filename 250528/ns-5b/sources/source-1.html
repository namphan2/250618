 class ConfirmRepository(
     private val remoteDataSource: ConfirmRemoteDataSource,
     private val savedPaymentData: SavedPaymentData,
     private val pinKeyService: IRemoteService,
     private val logger: ILogger,
     private val paymentSession: PaymentSession
 ) : IConfirmRepository {
     override fun confirm(
         otp: String?, napasInfo: NapasInfo?, confirmType: ConfirmType
     ): Flow<Response<ConfirmResponse>> {
         val pinKeySessionId = savedPaymentData.pinKeySessionId.takeIf { it.isNotEmpty() }
             ?: savedPaymentData.generatePinKey()
         val currentSof =
             savedPaymentData.cloneTransaction()?.purchaseInfo?.bundleSof?.getOrNull(0)?.moneySource
         FirebaseTrace.track(PaymentRequested(moneySource = currentSof))
         val confirmMessage = generateConfirmMessage(
             pinKeySessionId,
             otp,
             napasInfo,
             confirmType = confirmType
         )
         if (confirmMessage != null) {
             return remoteDataSource.confirm(confirmMessage, confirmType, currentSof)
                 .verifyResponse()
                 .map { response ->
                     savedPaymentData.setConfirmTransaction(response.data.momoMsg)
                     if (response.isSuccess) {
                         return@map Response.success(response.data.toModel())
                     }
                     return@map Response.fail(response.data.toModel(), response.exception)
                 }.flowOn(Dispatchers.IO)
         }
         return flowOf(
             Response.fail(
                 ConfirmResponse(),
                 PaymentError.DevError(
                     PaymentErrorGenerator.DevError.PurchaseIdNotFound.code,
                     "Purchase ID is empty at confirm api",
                     cause = Exception(this::class.simpleName)
                 )
             )
         )
     }
 
     override fun retryConfirm(
         otp: String?, napasInfo: NapasInfo?, confirmType: ConfirmType
     ): Flow<Response<ConfirmResponse>> {
         val pinKeySessionId = savedPaymentData.pinKeySessionId
         val confirmMessage =
             generateConfirmMessage(
                 pinKeySessionId,
                 otp,
                 napasInfo,
                 "RE_PAYMENT",
                 confirmType
             )
         if (confirmMessage != null) {
             val currentSof =
                 savedPaymentData.cloneTransaction()?.purchaseInfo?.bundleSof?.getOrNull(0)?.moneySource
             return remoteDataSource.confirm(confirmMessage, confirmType, currentSof)
                 .map { response ->
                     return@map Response.success(response.toModel())
                 }.flowOn(Dispatchers.IO)
         }
         return flowOf(
             Response.fail(
                 ConfirmResponse(),
                 PaymentError.DevError(
                     PaymentErrorGenerator.DevError.PurchaseIdNotFound.code,
                     "Purchase ID not found at retry confirm",
                     cause = Exception(this::class.simpleName)
                 )
             )
         )
     }
 
     override fun requestPinKey(pinKey: String?): Flow<PinKeySessionResponse> {
         return pinKeyService.requestPinKey(pinKey, buildJsonObject {
             put("source", "request_payment")
             put(
                 "service_code",
                 savedPaymentData.cloneTransaction()?.purchaseInfo?.invoiceInfo?.firstOrNull()?.items?.firstOrNull()?.serviceId
             )
         }).map {
             if (it?.status == "SUCCESS") {
                 savedPaymentData.setPinKeySession(it.pinKeySessionID ?: "")
             }
             it ?: PinKeySessionResponse(status = "CLOSE")
         }
     }
 
     override fun convertInitToResponseData(): ConfirmResponse {
         val momoMsgLocal = savedPaymentData.cloneTransaction()
         if (momoMsgLocal?.purchaseId != null) {
             val extra = savedPaymentData.getTransactionExtra()
             val moneySource = momoMsgLocal.purchaseInfo?.bundleSof?.get(0)?.moneySource
             val response = momoMsgLocal.toSkipResponse(extra, moneySource)
             return response.toModel()
         }
         throw PaymentError.DevError(
             DevError.PurchaseIdNotFound.code,
             "Purchase ID not found at convert init to response data",
             cause = Exception(this::class.simpleName)
         )
     }
 
     private fun generateConfirmMessage(
         pinKeySessionId: String,
         otp: String?,
         napasInfo: NapasInfo?,
         msgType: String = "PAYMENT",
         confirmType: ConfirmType
     ): ConfirmRequestDTO? {
         val momoMsgSaved = savedPaymentData.cloneTransaction()
         GrafanaTrace.start(PayXUserEvent.GenerateRequestMessage.event)
         logger.i("[PayX] ConfirmMessage.purchaseId: ${momoMsgSaved?.purchaseId}")
         return momoMsgSaved?.takeIf { !it.purchaseId.isNullOrEmpty() }?.let {
             GrafanaTrace.success(
                 PayXUserEvent.GenerateRequestMessage.event,
                 TraceParameter(errorCode = confirmType.traceError)
             )
             val extra = savedPaymentData.getTransactionExtra().toMutableMap()
             extra.remove("sofInfo")
             val moneySource = it.purchaseInfo?.bundleSof?.get(0)?.moneySource
             val paymentType = it.paymentType ?: "IN_APP"
 
             val sourceInit =
                 if (napasInfo != null) SourceInitConfirmDTO(
                     moneySource = napasInfo.moneySource?.id,
                     partnerCode = napasInfo.partnerCode,
                     moneySourceInfo = MoneySourceInfoConfirmDTO(
                         cardNumber = napasInfo.cardNumber,
                         cardFullName = napasInfo.cardFullName,
                         cardIssueDate = napasInfo.cardIssueDate,
                         isSaveToken = napasInfo.isSaveToken,
                         moneySource = napasInfo.moneySource?.id,
                         partnerCode = napasInfo.partnerCode,
                         sofState = napasInfo.sofState,
                         bankName = napasInfo.bankName,
                     )
                 ) else null
 
             val headers = mapOf(
                 "money-source" to moneySource,
                 "payment_type" to paymentType,
                 "mini-app-timestamp" to TimeUtils.getCurrentTimeMillis(),
             )
 
             val cartInfo = it.purchaseInfo?.invoiceInfo?.mapNotNull { invoice ->
                 val item = invoice.items?.firstOrNull()
                 val serviceInfo = savedPaymentData.getServiceInfo()?.get(item?.id) as? JsonObject? // logic CVS, keep old
                 val extraServiceInfo = savedPaymentData.getExtraServiceInfo()
                 val mergedServiceInfo = buildFullServiceInfo(serviceInfo, extraServiceInfo, item?.itemId, item?.subItemId)
                 logger.i(
                     "ConfirmRepository-serviceInfo: ${
                         mapOf(
                             "extraServiceInfo" to savedPaymentData.getExtraServiceInfo(),
                             "mergedServiceInfo" to mergedServiceInfo,
                             "serviceInfo" to serviceInfo
                         )
                     }"
                 )
                 mergedServiceInfo?.let {
                     CartInfoConfirmDTO(
                         itemId = invoice.items?.firstOrNull()?.itemId.orEmpty(),
                         subItemId = invoice.items?.firstOrNull()?.subItemId.orEmpty(),
                         serviceInfo = it
                     )
                 }
             }
 
             val loading =
                 confirmType !in listOf(ConfirmType.ConfirmSkipLoading, ConfirmType.RetryConfirm)
             val timeout: Long = when (confirmType) {
                 ConfirmType.ConfirmSkipLoading -> 5000
                 ConfirmType.RetryConfirm -> 3000
                 else -> 20000
             }
             ConfirmRequestDTO(
                 momoMsg = MoMoMsgConfirmDTO(
                     purchaseId = it.purchaseId!!,
                     otpValue = otp,
                     sourceInit = sourceInit,
                     purchaseToken = it.purchaseToken ?: "",
                     cartInfo = cartInfo
                 ),
                 extra = extra.toJsonObject(),
                 pinKeySessionID = pinKeySessionId,
                 isManualPass = true,
                 msgType = msgType,
                 headers = headers.toJsonObject(),
                 options = RemoteAPIOption(
                     logging = true,
                     loading = loading,
                     bypassErrors = listOf(40000),
                     timeout = timeout,
                     retry = if(confirmType == ConfirmType.RetryConfirm) 3 else 0
                 ),
                 subDomain = paymentSession.requestPayXData.apiPaymentConfig.subDomain,
                 subPath = paymentSession.requestPayXData.apiPaymentConfig.subPath
             )
         } ?: run {
             GrafanaTrace.fail(
                 PayXUserEvent.GenerateRequestMessage.event,
                 TraceParameter(errorCode = confirmType.traceError)
             )
             null
         }
     }
 
     private fun buildFullServiceInfo(
         serviceInfo: JsonObject?, extraServiceInfo: JsonObject?, itemId: String?, subItemId: String?
     ): JsonObject? {
         val key = when {
             itemId.isNullOrBlank() && subItemId.isNullOrBlank() -> null
             subItemId.isNullOrBlank() -> "${itemId}"
             else -> "${itemId}_${subItemId}"
         }
 
         return when {
             serviceInfo.isNullOrEmpty() && extraServiceInfo.isNullOrEmpty() -> null
             extraServiceInfo.isNullOrEmpty() || key.isNullOrBlank() -> serviceInfo
             else -> {
                 val current = serviceInfo ?: JsonObject(emptyMap())
                 val matched = extraServiceInfo[key]?.jsonObject
                 val extra = matched ?: JsonObject(emptyMap())
                 return JsonObject(current.toMutableMap() + extra.toMutableMap())
             }
         }
     }
