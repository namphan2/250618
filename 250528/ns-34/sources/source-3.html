 package vn.momo.compose.payment.base.domain.useCases
 
 import kotlinx.coroutines.flow.Flow
 import kotlinx.serialization.json.JsonObject
 import kotlinx.serialization.json.buildJsonObject
 import kotlinx.serialization.json.put
 import kotlinx.serialization.json.putJsonObject
 import vn.momo.compose.payment.base.core.composeApi.INavigator
 import vn.momo.compose.payment.base.domain.entities.BankOtpCallbackData
 import vn.momo.compose.payment.base.domain.entities.SOFItem
 
 class StartBankOtpNativeUseCase(
     private val navigator: INavigator,
 ) {
     operator fun invoke(
         bankAuthType: String,
         confirmExtras: JsonObject,
         currentSof: SOFItem?
     ): Flow<BankOtpCallbackData?> {
         val bankName = currentSof?.fullName
         val bankCode = currentSof?.info?.bankCode
         val params = buildJsonObject {
             put("shouldHandleBackPress", true)
             put("typeOTP", bankAuthType)
             put("paymentType", "single")
             put("confirmExtras", confirmExtras)
             put("bankName", bankName)
             putJsonObject("initSource") {
                 put("bankName", bankName)
                 put("bankCode", bankCode)
             }
         }
         return navigator.navigateFeatureCode(
             "payx_bank_otp", params, BankOtpCallbackData.serializer()
         )
     }
 }
