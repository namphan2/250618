 package vn.momo.compose.payment.bankOtp.presentation.viewModel
 
 import kotlinx.coroutines.CoroutineScope
 import kotlinx.coroutines.Dispatchers

 import kotlinx.coroutines.flow.MutableStateFlow
 import kotlinx.coroutines.flow.asStateFlow

 import kotlinx.serialization.json.put

 class PaymentBankOtpVM(
     val navigator: INavigator,
     val logger: ILogger,
     val paymentBankOtpRepository: IPaymentBankOtpRepository,
     val routeApi: IRouteApi,
     val smsApi: ISmsApi,
     val bankOtpTracker: PaymentBankOtpTracker,
 ) :
     ViewModel<PaymentBankOtpReducer.State, PaymentBankOtpReducer.Action, PaymentBankOtpReducer.Effect>(
         PaymentBankOtpReducer.State(),
         PaymentBankOtpReducer()
     ) {
 
     private val LOG_NAME = "[PAYX_BANK_OTP]"
     lateinit var params: PaymentBankOtpParams
     lateinit var screenType: EOtpType
     private val _showPopup = MutableStateFlow(false)
     val showPopup = _showPopup.asStateFlow()
 
     fun onCreate(params: Map<String, Any?>) {
         val paramsFormated = ModelConvertUtils.decode(PaymentBankOtpParams.serializer(), params)
             ?: PaymentBankOtpParams()
         logger.i("$LOG_NAME - paramsFormated: $paramsFormated")
         paramsFormated?.let {
             this.params = it
             val bankOtpConfig = paymentBankOtpRepository.getConfig(it.bankCode, it.confirmExtras)
             screenType = it.otpType
             dispatch(PaymentBankOtpReducer.Action.SyncConfig(bankOtpConfig))
             bankOtpTracker.trackScreenLoad(it.bankCode?: "")
             logger.i("$LOG_NAME - bankOtpConfig, screenType: $bankOtpConfig $screenType")
         }
     }
 
     fun listenSmsOtp(maxLength: Int) {
         logger.i("$LOG_NAME -  start listenOtp")
         CoroutineScope(Dispatchers.IO).launch {
             smsApi.listenOTP(maxLength)
                 .catch { e -> logger.i("$LOG_NAME listen otp error: ${e.message}") }
                 .collect {
                     logger.i("$LOG_NAME otp listen response $it")
                     if (it.length == maxLength) {
                         logger.i("$LOG_NAME set otp to input $it")
                         dispatch(PaymentBankOtpReducer.Action.SetOtp(it))
                     }
                 }
         }
     }
 
     fun getBankData() {
         params.bankCode?.let {
             viewModelScope.launch(Dispatchers.IO) {
                 val paymentData = paymentBankOtpRepository.getBankDataByBankCode(it)
                 dispatch(PaymentBankOtpReducer.Action.SetBankData(paymentData))
                 logger.i("$LOG_NAME - bank data $paymentData")
             }
         }
     }
 
     fun onChangeOtp(value: String) {
         logger.i("$LOG_NAME - $value")
         if (value.isNotEmpty()) {
             bankOtpTracker.trackOnInputOtp(params?.bankCode ?: "")
         }
         dispatch(PaymentBankOtpReducer.Action.SetOtp(value))
         validateOtp(value)
 
     }
 
     fun onFocusInput() {
         dispatch(PaymentBankOtpReducer.Action.SetError(""))
     }
 
     fun onBlurInput() {
         val otp = state.value.otp
         validateOtp(otp)
         val errorMessage = getErrorMessageByValue(otp)
         dispatch(PaymentBankOtpReducer.Action.SetError(errorMessage))
     }
 
     private fun validateOtp(value: String) {
         val errorMessage = getErrorMessageByValue(value)
         dispatch(PaymentBankOtpReducer.Action.SetIsValidInput(errorMessage.isEmpty()))
     }
 
 
     private fun getErrorMessageByValue(value: String): String {
         val regexList = state.value.otpInputProps?.otpRegex ?: return "";
         val sortedRegexList = regexList.sortedBy { it.order ?: 0 }
         for (regexObj in sortedRegexList) {
             val pattern = regexObj.regex?.toRegex() ?: continue
             if (!pattern.matches(value)) {
                 return regexObj.error ?: ""
 
             }
         }
         return ""
     }
 
     fun getDOtp() {
         logger.i("$LOG_NAME - getDOTP with link ${params.otpURL}")
         viewModelScope.launch {
             val responseOpenUrl = routeApi.openUrl(params.otpURL)
             logger.i("$LOG_NAME response open url - $responseOpenUrl")
         }
     }
 
     fun showPopupGuideDOtp() {
         _showPopup.value = true
     }
 
     fun hidePopupGuide() {
         _showPopup.value = false
     }
 
     fun onViewGuideUrlDOtp() {
         logger.i("$LOG_NAME - view guide url otp ${params.guideOtpUrl}")
         params?.guideOtpUrl?.let {
             viewModelScope.launch {
                 routeApi.openWeb(it, getString(Res.string.verifyInstructionTitle));
             }
         }
 
     }
 
     fun onConfirm() {
         logger.i("$LOG_NAME - confirm otp ${state.value.otp}")
         if (state.value.otp.isEmpty()) return;
         val dismissData = buildJsonObject {
             put("otp", state.value.otp)
             put("onInputOTP", true)
         }
         bankOtpTracker.trackSubmitOtp(params?.bankCode ?: "")
         navigator.dismiss(dismissData)
     }
 
     fun screenGoBack() {
         bankOtpTracker.trackScreenGoBack(
             bankCode = params?.bankCode ?: "",
             type = when(screenType) {
                 EOtpType.SMS -> "sms_otp"
                 EOtpType.DOTP -> "digital_otp"
             },
         )
         navigator.dismiss()
     }
 }
