 @Composable
 fun TransactionResultScreen(viewModel: TransactionResultVM) {
     val transactionResultState by viewModel.state.state.collectAsState()
     val ticketListState = transactionResultState.ticketList
     val firstTransaction = remember(ticketListState) {
         ticketListState.takeIf { it.isNotEmpty() }?.first()
     }
 
     BackHandler(true, null)
     Screen(title = stringResource(Res.string.transactionResult),
         titlePosition = TitlePosition.CENTER,
         headerType = HeaderType.EXTENDED,
         scrollable = true,
         headerRight = {
             HeaderIcon("basic_home") {
                 viewModel.dispatch(TransactionResultAction.GoHome)
             }
         },
         content = {
             Column(
                 modifier = Modifier.fillMaxWidth(),
             ) {
                 FeeBlocks(transactionResultState.fees)
 
                 ticketListState.fastForEachIndexed { index, item ->
                     TransactionTicket(
                         item, index, transactionResultState, dispatcher = viewModel::dispatch
                     )
                     if (index < ticketListState.lastIndex) {
                         Spacer(
                             modifier = Modifier.height(20.dp).fillMaxWidth()
                                 .background(Color.Transparent)
                         )
                     }
                 }
 
                 GamificationBlock(
                     dispatcher = viewModel::dispatch,
                     firstTransaction,
                     transactionResultState.gamificationData,
                     show = transactionResultState.showOtherAddons
                 )
                 ExpenseWidget(firstTransaction, show = transactionResultState.widgetServiceLoaded && transactionResultState.showOtherAddons)
                 CornerstoneWidget(
                     firstTransaction, 0, show = transactionResultState.widgetServiceLoaded && transactionResultState.showOtherAddons
                 )
                 CornerstoneWidget(
                     firstTransaction, 1, show = transactionResultState.widgetServiceLoaded && transactionResultState.showOtherAddons
                 )
                 Spacer(Modifier.height(24.dp))
             }
         })
 }
