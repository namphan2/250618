 @Composable
 fun TransactionResultLoadingScreen(viewModel: TransactionResultLoadingVM) {
     val trLoadingState by viewModel.state.state.collectAsState()
     val (title, icon, desc) = remember(trLoadingState.mode) {
         val mode = trLoadingState.mode
         val _title =
             if (mode == TRLoadingMode.FAIL) Res.string.failToShowTransaction else if (mode == TRLoadingMode.TIMEOUT) Res.string.timeoutProcessingTransaction else Res.string.processingTransaction2
         val _icon =
             if (mode == TRLoadingMode.FAIL) IC_ARROW_REFRESH else if (mode == TRLoadingMode.TIMEOUT) IC_STOP_WATCH else null
         val _desc = if (arrayOf(
                 TRLoadingMode.FAIL, TRLoadingMode.TIMEOUT
             ).contains(mode)
         ) Res.string.endProcessingTransactionDesc else Res.string.processingTransactionDesc
 
         Triple(_title, _icon, _desc)
     }
 
     val (showIndicator, showButton) = remember(trLoadingState.mode) {
         val mode = trLoadingState.mode
         val _showIndicator = arrayOf(TRLoadingMode.LOADING, TRLoadingMode.UNKNOWN).contains(mode)
         val _showButton = arrayOf(TRLoadingMode.FAIL).contains(mode)
         Pair(_showIndicator, _showButton)
     }
 
     LaunchedEffect(trLoadingState.mode) {
         if (trLoadingState.mode == TRLoadingMode.TIMEOUT) {
             delay(100)
             viewModel.dispatch(TransactionResultLoadingAction.ShowPopup)
         }
     }
 
     Box(modifier = Modifier.fillMaxSize()) {
         BackHandler(false, null)
         Screen(title = stringResource(Res.string.paymentResultLoading),
             titlePosition = TitlePosition.CENTER,
             headerType = HeaderType.EXTENDED,
             scrollable = true,
 //            headerRight = {
 //                HeaderIcon("navigation_close") {
 //                    viewModel.dispatch(TransactionResultLoadingAction.Close)
 //                }
 //            },
             footer = {
                 if (showButton) {
                     Button(
                         title = stringResource(Res.string.navigateToTransHis),
                         onClick = {
                             viewModel.dispatch(TransactionResultLoadingAction.NavigateToTransHis)
                         },
                     )
                 }
             },
             content = {
                 Box(
                     modifier = Modifier.fillMaxSize().padding(horizontal = Spacing.M),
                 ) {
                     Row(
                         modifier = Modifier.background(
                             Colors.black_01, shape = RoundedCornerShape(Spacing.L)
                         ).padding(Spacing.S).zIndex(10f).align(Alignment.TopCenter),
                         horizontalArrangement = Arrangement.Center,
                         verticalAlignment = Alignment.CenterVertically
                     ) {
                         MMImage(
                             IC_MOMO,
                             modifier = Modifier.size(52.dp),
                             contentDescription = "MoMo Logo",
                             contentScale = ContentScale.Fit
                         )
                         Spacer(modifier = Modifier.width(Spacing.S))
                         MMImage(
                             IC_CONNECT,
                             modifier = Modifier.size(Spacing.XL),
                             contentDescription = "Connection Icon",
                             contentScale = ContentScale.Fit
                         )
                         Spacer(modifier = Modifier.width(Spacing.S))
                         trLoadingState.bankData.icon?.let {
                             MMImage(
                                 it,
                                 modifier = Modifier.size(52.dp),
                                 contentDescription = "Bank Logo",
                                 contentScale = ContentScale.Fit
                             )
                         }
                     }
                     Column {
                         Spacer(modifier = Modifier.height(Spacing.XL))
                         Column(
                             modifier = Modifier.background(
                                 Colors.black_01, shape = RoundedCornerShape(Spacing.L)
                             ).padding(vertical = Spacing.XL).fillMaxWidth(),
                             horizontalAlignment = Alignment.CenterHorizontally
                         ) {
                             Spacer(modifier = Modifier.height(Spacing.XXL))
                             // renderTitle
                             Row(
                                 verticalAlignment = Alignment.CenterVertically,
                                 horizontalArrangement = Arrangement.SpaceBetween
                             ) {
                                 icon?.let {
                                     MMImage(
                                         it,
                                         modifier = Modifier.size(Spacing.L)
                                             .padding(end = Spacing.XS),
                                         contentDescription = null,
                                     )
                                 }
                                 Text(
                                     text = stringResource(title),
                                     style = Typography.headerSSemibold,
                                     textAlign = TextAlign.Center
                                 )
                             }
                             // renderDescription
                             Text(
                                 text = stringResource(desc),
                                 style = Typography.bodyDefaultRegular,
                                 modifier = Modifier.padding(top = Spacing.S, bottom = Spacing.L),
                                 textAlign = TextAlign.Center
                             )
                             if (showIndicator) {
                                 Spacer(modifier = Modifier.height(Spacing.XS))
                                 DotWave(dotSize = 12.dp, dotSpacing = 6.dp)
                             }
                         }
                     }
                 }
             })
 
         PopupNotify(show = trLoadingState.showPopup,
             title = stringResource(Res.string.timeoutTransactionTitle),
             content = stringResource(Res.string.timeoutTransactionContent),
             primaryButton = stringResource(Res.string.transactionHistory),
             secondaryButton = stringResource((Res.string.keepWaiting)),
             image = POPUP_IMAGE,
             buttonDirection = ButtonDirection.COL,
             onPressPrimary = { viewModel.dispatch(TransactionResultLoadingAction.NavigateToTransHis) },
             onPressSecondary = { viewModel.dispatch(TransactionResultLoadingAction.HidePopup) },
             onDismissRequest = { viewModel.dispatch(TransactionResultLoadingAction.HidePopup) })
     }
