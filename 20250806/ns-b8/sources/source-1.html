 public fun htmlToAnnotatedString(
     parser: HtmlParser,
     compactMode: Boolean = false,
     style: HtmlStyle = HtmlStyle.DEFAULT,
     linkInteractionListener: LinkInteractionListener? = null
 ): AnnotatedString {
