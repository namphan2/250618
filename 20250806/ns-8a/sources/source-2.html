 inline fun String.toColor(): Color {
     val cleanedHex = this.removePrefix("#")
     val alphaHex = if (cleanedHex.length == 6) "FF" else ""
     val fullHex = "$alphaHex$cleanedHex"
     val colorInt = fullHex.toLong(16).toInt()
     return Color(colorInt)
 }
 
 inline fun String.toTypography(): TextStyle? {

     }
