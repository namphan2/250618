 class TransactionResultLoadingRepository(

 ) : ITransactionResultLoadingRepository {

     override fun getBankData(serviceCode: String): Flow<TRLoadingBankData> {
         return appFeatureConfig.getFeaturesWittType(listOf("BANK")).map { features ->

         }.flowOn(Dispatchers.IO)
     }
