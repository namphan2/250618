 class PaymentInstallmentVM(
     private val sofSlice: SofSlice<out SofState>,
     private val installmentRepository: IInstallmentRepository,
     private val navigator: INavigator
 ) : ViewModel(), IPaymentVM<InstallmentAction> {
     val installmentList = PState(
         viewModelScope, InstallmentListState(
             list = emptyList(),
             loading = true,
             showRetry = true,
             errorMessage = ""
         )
     )
     val questions = PState<List<Pair<String, String>>>(viewModelScope, emptyList())
 
     fun onCreate(props: InstallmentPlanScreenProps?) {
         FirebaseTrace.track(PaymentInstallmentDisplayed())
         dispatch(InstallmentAction.GetInstallmentPlan(container = props?.fakeSofContainer))
         dispatch(InstallmentAction.GetQuestions)
     }
 
     override fun dispatch(action: InstallmentAction) {
         when (action) {
             is InstallmentAction.GetInstallmentPlan -> onGetInstallmentPlan(action.container)
             is InstallmentAction.GetQuestions -> onGetQuestions()
             is InstallmentAction.SelectInstallmentPlan -> onSelectPlan(action.sof)
             is InstallmentAction.OpenInstruction -> onOpenInstruction(action.url)
         }
     }
 
     private fun onGetInstallmentPlan(container: InstallmentPlanContainer?) {
         installmentRepository.getInstallmentPlan().map { it.filterNotNull() }
             .catch { error ->
                 when (error) {
                     is PaymentError.InstallmentError -> installmentList.update {
                         it.copy(
                             loading = false,
                             showRetry = false,
                             errorMessage = error.message
                         )
                     }
 
                     is PaymentError.UnknownError, is PaymentError.DevError -> {
                         val message = getString(Res.string.devErrorDesc)
                         installmentList.update {
                             it.copy(
                                 loading = false,
                                 showRetry = true,
                                 errorMessage = message
                             )
                         }
                     }
 
                     else -> installmentList.update {
                         it.copy(
                             loading = false,
                             showRetry = true,
                             errorMessage = error.message ?: ""
                         )
                     }
                 }
             }
             .onEach { sofItems ->
                 installmentList.update {
                     InstallmentListState(
                         list = sofItems,
                         loading = false,
                         showRetry = sofItems.isEmpty(),
                         errorMessage = ""
                     )
                 }
                 sofSlice.dispatch(
                     SofReducer.Action.SyncInstallmentPackage(
                         group = SOFGroup(
                             parent = SOFItem(
                                 ID = container?.ID,
                                 logo = container?.logo,
                                 shortName = container?.shortName ?: container?.fullName ?: "",
                                 shortDesc = container?.shortDesc ?: container?.fullDesc ?: "",
                                 sourceToken = container?.sourceToken ?: "",
                                 moneySource = EnumMoneySource.PayLater,
                                 disableType = DisableType.Show,
                                 info = PayXSofInfo(),
                                 balance = -1
                             ),
                             children = sofItems
                         )
                     )
                 )
             }
             .launchIn(viewModelScope)
     }
 
     private fun onGetQuestions() {
         installmentRepository.getQuestions().onEach {
             questions.update(it ?: emptyList())
         }.launchIn(viewModelScope)
     }
 
     private fun onSelectPlan(sof: SOFItem) {
         FirebaseTrace.track(
             InstallmentOptionClicked(
                 listValue = sof.sourceToken,
                 listValueDetail = "1"
             )
         )
         if (sof.disableType == DisableType.Show) {
             navigator.dismiss()
             sofSlice.dispatch(
                 SofReducer.Action.MoveOrAddSofToFirst(
                     sof.copy(
                         isInstallment = true
                     )
                 )
             )
         } else {
             showPopupPayOff()
         }
 
     }
 
     private fun showPopupPayOff() {
         viewModelScope.launch {
             navigator.showDialog(
                 getString(Res.string.errorDefaultTitle),
                 getString(Res.string.installmentDescPopup),
                 listOf(getString(Res.string.textUnderstand))
             ).collect {}
         }
     }
 
     private fun onOpenInstruction(url: String) {
         FirebaseTrace.track(InstallmentInstructionClicked())
         viewModelScope.launch {
             val feature = buildJsonObject {
                 put("config", buildJsonObject {
                     put("useNavigator", true)
                     put("useRightHeader", true)
                     put("useParameter", false)
                     put("title", getString(Res.string.info))
                     put("url", url)
                 })
                 put("appId", "vn.momo.web")
             }
             val params = buildJsonObject {
                 put("title", getString(Res.string.info))
                 put("url", url)
             }
 
             navigator.navigateFeature(feature, params).launchIn(viewModelScope)
         }
     }
 }
