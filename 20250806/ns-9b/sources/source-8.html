 class PaymentSuggestedSofVM(
     private val sofSlice: SofSlice<out SofState>,
     private val newSofRepository: NewSofRepository,
     private val suggestedSofRepository: ISuggestedSofRepository,
     private val openMiniAppUseCase: OpenMiniAppUseCase,
     private val navigator: INavigator,
     private val tracker: SuggestedSofTracker
 ) : ViewModel(), IPaymentVM<SuggestedSofAction>, SuggestedSofTracker by tracker {
     private var shouldSuggestFirstSof = false
 
     val suggestedSofState = PState(
         viewModelScope,
         SuggestedSofDataState(
             sofList = sofSlice.state.value.list,
             isHideBalance = sofSlice.state.value.isHideBalance
         )
     )
 
     val selectedSofState = PState(viewModelScope, sofSlice.state.value.selected)
     val shouldCloseSheet = PState(viewModelScope, false)
 
     fun onCreate() {
         trackUserInBottomSheet()
         getListIconSuggest()
         onGetSuggestedSof()
         sofSlice.state.onEach { sofCarousel ->
             suggestedSofState.update { it.copy(isHideBalance = sofCarousel.isHideBalance) }
         }.launchIn(viewModelScope)
     }
 
     fun setProps(shouldPickFirst: Boolean) {
         shouldSuggestFirstSof = shouldPickFirst
         if (shouldPickFirst) {
             selectedSofState.update { prev ->
                 suggestedSofState.value.sofList.firstOrNull()?.takeIf { it.available } ?: prev
             }
         }
     }
 
     override fun onCleared() {
         trackerUserOutBottomSheet()
         if (shouldSuggestFirstSof && selectedSofState.value?.available == true) {
             changeSof(selectedSofState.value!!)
         }
         super.onCleared()
     }
 
     override fun dispatch(action: SuggestedSofAction) {
         when (action) {
             is SuggestedSofAction.SelectSof -> onSelectSof(action.sof)
             is SuggestedSofAction.TriggerSofAction -> {
                 onPressSofAction(action.sof)
             }
 
         }
     }
 
     private fun onCloseSheet() {
         shouldCloseSheet.update(true)
     }
 
     fun onAddBank() {
         onCloseSheet()
         sofSlice.addBankToSof()
     }
 
     fun onAddCard() {
         onCloseSheet()
         sofSlice.navigateAddInternationalCard()
     }
 
     fun toggleHideBalance() {
         sofSlice.toggleHideBalance()
     }
 
     private fun getListIconSuggest() {
         viewModelScope.launch {
             val bankSuggestIconList = suggestedSofRepository.getListIconSuggest()
             suggestedSofState.update { it.copy(bankSuggestIconList = bankSuggestIconList) }
         }
     }
 
     private fun onGetSuggestedSof() {
         suggestedSofRepository.getSuggestedSof().onEach { suggestedSof ->
             sofSlice.dispatch(Action.SyncSuggestedSof(data = suggestedSof))
             suggestedSofState.update {
                 it.copy(
                     sofList = suggestedSof.sofList, sofCtaList = suggestedSof.ctaList
                 )
             }
             if (shouldSuggestFirstSof) {
                 suggestedSof.sofList.firstOrNull()?.let {
                     selectedSofState.update(it)
                 }
             }
             FirebaseTrace.track(
                 PopupMoneySourceDisplayed(
                     validOptions = "", //TODO
                     defaultOptions = sofSlice.state.value.selected?.moneySource?.id.toString()
                 )
             )
         }.catch { error ->
             val toastParams = buildJsonObject {
                 put("type", "ERROR")
                 put("title", "Get Recommend Sof API")
                 put("message", error.message ?: "")
             }
             navigator.showToastDebugger(toastParams)
         }.launchIn(viewModelScope)
     }
 
     private fun onSelectSof(sof: SOFItem) {
         onCloseSheet()
         if (sof.isInstallment) {
             navigateToInstallmentPlan(sof)
         } else {
             FirebaseTrace.track(
                 ChangeMoneySourceFromPopup(
                     listValue = sof.moneySource.id.toString(), listValueDetail = "0"
                 )
             )
             shouldSuggestFirstSof = false
             changeSof(sof)
         }
     }
 
     private fun onPressSofAction(sof: SOFItem) {
         if (sof.isInstallment) {
             onCloseSheet()
             navigateToInstallmentPlan(sof)
             return
         }
         when (val action = sof.action?.action) {
             is SofActionType.CashIn -> onCashIn(action.amount, sof.action.refId, sof)
             is SofActionType.SofTransfer -> onTransferSof(sof.action.refId)
             is SofActionType.Activate -> onActivate(params = action.info)
             is SofActionType.HaveNextStep -> onDefaultSofAction(sof.action.refId, sof.action.params)
             is SofActionType.Select -> onSelectSof(sof)
             else -> {}
         }
     }
 
     private fun navigateToInstallmentPlan(sof: SOFItem) {
         navigator.navigateFeatureCode(
             ServiceFeatureCode.InstallmentPlan.code, buildJsonObject {
                 put(
                     "props", ModelConvertUtils.encodeToJson(
                         InstallmentPlanScreenProps.serializer(),
                         InstallmentPlanScreenProps(
                             selected = sofSlice.state.value.selected?.sourceToken,
                             fakeSofContainer = InstallmentPlanContainer(
                                 ID = sof.ID,
                                 logo = sof.logo,
                                 shortName = sof.shortName,
                                 fullName = sof.fullName,
                                 sourceToken = sof.sourceToken,
                                 shortDesc = sof.shortDesc,
                                 fullDesc = sof.fullDesc,
                                 isInstallment = sof.isInstallment
                             )
                         )
                     ) ?: emptyJsonObject()
                 )
             }, String.serializer()
         ).launchIn(viewModelScope)
     }
 
     private fun onCashIn(amount: Long, refId: String?, sof: SOFItem) {
         FirebaseTrace.track(
             ChangeMoneySourceFromPopup(
                 listValue = sof.moneySource.id.toString(), listValueDetail = "1"
             )
         )
         onCloseSheet()
         sofSlice.onCashIn(amount = amount, refId = refId, targetSource = sof.moneySource)
     }
 
     private fun onTransferSof(refId: String?) {
         openMiniAppUseCase.onActionToMiniApp(refId, params = emptyMap(), String.serializer())
             .onEach {
                 sofSlice.state.value.list.firstOrNull { sof -> sof.moneySource == EnumMoneySource.MoMo }
                     ?.let { momo ->
                         shouldSuggestFirstSof = false
                         changeSof(momo)
 
                     }
                 onCloseSheet()
             }.launchIn(viewModelScope)
     }
 
     private fun onActivate(params: JsonObject?) {
         FirebaseTrace.track(
             ChangeMoneySourceFromPopup(
                 listValue = "installment", listValueDetail = "0"
             )
         )
         val activationParams = buildJsonObject {
             put("activationInfo", params ?: emptyJsonObject())
             put("tracking", buildJsonObject {
                 put("fromFeature", "payment_confirm")
                 put("screenName", "TTAT")
             })
         }
 
         navigator.navigateFeatureCode(
             ServiceFeatureCode.ActivatePayLater.code,
             activationParams,
             SofFromJS.serializer()
         ).onEach { result ->
             try {
                 val newSof = newSofRepository.onReceiveSof(result)
                 newSof?.let {
                     shouldSuggestFirstSof = false
                     changeSof(newSof)
                 }
             } catch (err: PaymentError) {
                 err.handlePaymentError()
             }
             onCloseSheet()
         }.launchIn(viewModelScope)
     }
 
     private fun changeSof(sof: SOFItem) {
         val action = sofSlice.findSof(sof.sourceToken, sof.ID)?.let { foundSof ->
             Action.ChangeSof(foundSof)
         } ?: Action.MoveOrAddSofToFirst(sof)
         sofSlice.dispatch(action)
     }
 
     private fun onDefaultSofAction(refId: String?, params: Map<String, Any?> = emptyMap()) {
         openMiniAppUseCase.onActionToMiniApp(refId, params, SofFromJS.serializer())
             .onEach { result ->
                 try {
                     val newSof = newSofRepository.onReceiveSof(result)
                     newSof?.let {
                         shouldSuggestFirstSof = false
                         changeSof(newSof)
                     }
                 } catch (err: PaymentError) {
                     err.handlePaymentError()
                 }
                 onCloseSheet()
             }.launchIn(viewModelScope)
     }
 }
