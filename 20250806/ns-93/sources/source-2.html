 @Composable
 fun debounceCompose(debounceTime: Long = 500L, action: () -> Unit): () -> Unit {
     return remember(action) {
         var lastClickTime: Long = 0
         val callback: () -> Unit = {
             val current = TimeUtils.getCurrentTimeMillis()
             if (current - lastClickTime >= debounceTime) {
                 lastClickTime = current
                 action()
             }
         }
         return@remember callback
     }
 }
