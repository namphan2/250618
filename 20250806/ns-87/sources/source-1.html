 package vn.momo.compose.payment.payx.presentation.mixin
 
 import kotlinx.serialization.json.buildJsonObject
 import kotlinx.serialization.json.put
 import vn.momo.compose.payment.base.core.composeApi.ILogger
 import vn.momo.compose.payment.base.core.composeApi.INavigator
 
 interface CommonMixin {
     val navigator: INavigator
     val logger: ILogger
     val logName: String
         get() = this::class.simpleName ?: "CommonMixin"
 
     fun logInfo(title: String?, message: Any?, showToast: Boolean = false) {
         if (showToast) {
             navigator.showToastDebugger(buildJsonObject {
                 put("title", "[${logName}]_$title")
                 put("message", "$message")
                 put("duration", 5000)
             })
         }
         logger.i("[${logName}]_$title : $message")
     }
 }
