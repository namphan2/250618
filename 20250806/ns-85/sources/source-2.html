 @Composable
 fun ConfirmContentV2(
     state: TransactionReducer.State,
     hasAppliedPromotion: Boolean,
     viewModel: PaymentConfirmVM,
     sofListState: PayXSofReducer.State,
     rewardState: RewardReducer.RewardState,
     cartData: CartData?,
 ) {
     val (isNoSofAvailable, payLater, promoted) = remember(
         sofListState.list, sofListState.groupOfSof, sofListState.promoteSof
     ) {
         val isNoSofAvailable = sofListState.list.none { it.available }
         val payLater = sofListState.groupOfSof.payLater
         val promoted = sofListState.promoteSof
         Triple(isNoSofAvailable, payLater, promoted)
     }
     val groupSof = sofListState.groupOfSof
     val sofList = groupSof.debit
     val campaignSof = groupSof.campaignSof
     val cta = sofListState.cta
     val screenWidth = getScreenDimensions().screenWidth
     val rewardInline = rewardState.missions
     val payLaterConfig = sofListState.payLaterConfigUi
 
     val surface = AppTheme.current.colors.background.surface
     val blockModifier = Modifier.fillMaxWidth().scaledPadding(horizontal = 12f)
     val padding = Modifier.scaledPadding(horizontal = 12f)
 
     Body(
         modifier = Modifier.scaledPadding(vertical = 12f).animateContentSize(
             animationSpec = linearTween(delayMillis = 0)
         )
     ) {
         blockKey(
             "payment_info", modifier = Modifier.fillMaxWidth().then(padding).shadowBill()
         ) {
             PaymentInfoCarousel(
                 items = state.invoices,
                 shouldShowTid = state.shouldShowTid,
                 hideAmount = state.hideAmount,
                 hasAppliedPromotion = hasAppliedPromotion,
                 screenWidth = screenWidth,
                 onPressDetail = { _ ->
                     viewModel.trackPressPaymentInfoDetail()
                 },
                 onDisplayedBottomSheet = { _ ->
                     viewModel.trackPaymentInfoBottomSheetDisplayed()
                 },
                 onCloseBottomSheet = { _, buttonName ->
                     viewModel.trackClosePaymentInfoBottomSheet(buttonName)
                 }
             )
         }
         cartData?.infoBlock?.let {
             blockKey("infoBlock") {
                 createPayXComponent(listOf(it))
             }
         }
 
         if ((state.promotion.isDisplay && !state.hideAmount) || !state.inlineMessage.isNullOrBlank()) {
             blockKey("promotion") {
                 PickPromotionBlock(
                     promotion = state.promotion,
                     inlineMessage = InlineMessageError(
                         description = state.inlineMessage
                     ),
                     onSelectPromotion = viewModel::navigateChoosePromotion,
                     hideAmount = state.hideAmount,
                     rewardMission = rewardInline,
                     onMissionShowed = {
                         viewModel.rewardSlice.trackMissionShow(
                             moneySource = sofListState.selected?.moneySource,
                             promotions = state.promotion.promoDisplay
                         )
                     })
             }
         }
 
         if (viewModel.props.isAllowPickSource) {
             blockKey("sof", modifier = blockModifier) {
                 SofListView(
                     sofListState = sofListState,
                     sofList = sofList,
                     isNoSof = isNoSofAvailable,
                     campaignSof = campaignSof,
                     number = groupSof.windowSize,
                     onPressSof = viewModel.sofSlice::changeSofIfNotSelected,
                     onPressCta = viewModel.sofSlice::onPressCta,
                     onClickShowAll = {
                         viewModel.onClickShowAll()
                     },
                     isAllowAddSOF = viewModel.props.isAllowAddSource,
                     isAllowAddCard = viewModel.props.isAllowAddCredit,
                     onLoaded = {
                         viewModel.sofSlice.trackSofDisplayed(
                             "list_sof", sofListState.list, sofListState.version
                         )
                     },
                     onAddSof = viewModel::addSof,
                     onAddCreditCard = viewModel::addCard,
                     onToggleBalance = viewModel.sofSlice::toggleHideBalance,
                     ctaList = cta,
                     onPressNotice = viewModel.sofSlice::onPressNotice,
                     isCollapsed = groupSof.isCollapsed,
                     onToggleExpand = { viewModel.sofSlice.toggleExpand(sofListState.version) },
                     transactionFee = null
                 )
             }
             if (payLater.isNotEmpty() || promoted != null) {
                 val hasBadge = !payLaterConfig?.badge.isNullOrEmpty()
                 blockKey(
                     "payLater",
                     modifier = if (hasBadge) blockModifier.padding(top = Spacing.XXS).background(
                         color = surface, shape = RoundedCornerShape(12.dp)
                     ) else null,
                     tagline = {
                         if (hasBadge) {
                             RibbonBadge.Reverse(
                                 data = XSaleBadge(
                                     label = payLaterConfig?.badge ?: "",
                                     type = TaglineType.Promotion
                                 ),
                                 modifier = Modifier.align(Alignment.TopStart)
                                     .scaledOffset(x = 0f, y = -8f)
                             )
                         }
                     },
                     background = {
                         Image(
                             Images.bg_paylater,
                             modifier = Modifier.width(76.dp).height(62.7.dp).align(Alignment.TopEnd)
                                 .scaledOffset(x = 0f, y = -6f),
                             options = Options(
                                 contentScale = ContentScale.Crop
                             )
                         )
                     }) {
                     PayLaterBlock(
                         sofListState = sofListState,
                         payLaters = payLater,
                         promoted = promoted,
                         onPressSof = viewModel.sofSlice::changeSofIfNotSelected,
                         onPressUnlock = viewModel.sofSlice::onUnlock,
                         onSelectPromoted = {
                             viewModel.sofSlice.trackPressInstallment(sofListState.version)
                             viewModel.sofSlice.dispatch(
                                 SofReducer.Action.SelectPromotedSof(it)
                             )
                         },
                         onShowUnlock = { viewModel.sofSlice.trackShowUnlock(sofListState.version) },
                         payLaterMessages = payLaterConfig?.item?.messages
                     )
                 }
             }
         }
         blockKey("smartWidget") {
             SmartWidget(viewModel.props.smartWidget)
         }
     }
 }
