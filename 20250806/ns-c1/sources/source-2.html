 class PromoCodeSlice(
     private val scope: CoroutineScope,
     private val repository: IPromoCodeRepository,
     private val navigator: INavigator,
     private val logger: ILogger,
 ) : ViewModelSlice<PromoCodeReducer.State, PromoCodeReducer.Action, PromoCodeReducer.Effect>(
     initialState = PromoCodeReducer.State(),
     reducer = PromoCodeReducer(),
     scope,
 ) {
     fun onCheckPromoCode(
         params: CheckPromoCodeParams,
         onSuccess: ((generatedVouchers: List<IVoucherBase>) -> Unit)? = {},
         onError: ((String) -> Unit)? = {},
     ) {
         repository
             .checkPromoCode(params)
             .take(1)
             .onStart {
                 logger.i("onCheckPromoCode start $params")
                 dispatch(
                     PromoCodeReducer.Action.SetState(
                         promoCode = params.getPromoCode(),
                         isCheckingPromoCode = true,
                         checkPromoCodeResult = null,
                         checkPromoCodeError = null,
                         generatedPromoVouchers = state.value.generatedPromoVouchers,
                         isPromoCodeApplied = false,
                     )
                 )
                 navigator.showLoading()
             }
             .onEach {
                 logger.i("onCheckPromoCode ${params.getPromoCode()} result $it")
                 val generatedPromoVouchers = it?.recommendVouchers?.plus(
                     it.invalidVouchers ?: emptyList()
                 ) ?: emptyList()
 
                 if (generatedPromoVouchers.isEmpty() && it?.totalVoucherPromotion == 0) {
                     throw Exception(getString(Res.string.devErrorDesc))
                 }
 
                 val totalReceivedVoucher = generatedPromoVouchers.size
                 val totalUnreceivedVoucher = (it?.totalVoucherPromotion ?: 0) - totalReceivedVoucher
 
                 logger.i("onCheckPromoCode ${params.getPromoCode()} totalReceivedVoucher=$totalReceivedVoucher totalUnreceivedVoucher=$totalUnreceivedVoucher")
 
                 val promoCodeNotice = if (totalUnreceivedVoucher > 0) {
                     Pair(
                         getString(Res.string.txtPromoNotice1).replace(
                             "{n}", it?.totalVoucherPromotion.toString()
                         ), getString(Res.string.txtPromoNotice2).replace(
                             "{n}", totalUnreceivedVoucher.toString()
                         )
                     )
                 } else {
                     null
                 }
 
                 val isPromoCodeApplied = generatedPromoVouchers.isNotEmpty()
 
                 if (isPromoCodeApplied) {
                     onSuccess?.invoke(generatedPromoVouchers)
                 }
 
                 dispatch(
                     PromoCodeReducer.Action.SetState(
                         promoCode = params.getPromoCode(),
                         isCheckingPromoCode = false,
                         checkPromoCodeResult = it,
                         checkPromoCodeError = null,
                         generatedPromoVouchers = state.value.generatedPromoVouchers + generatedPromoVouchers,
                         isPromoCodeApplied = isPromoCodeApplied,
                         promoCodeNotice = promoCodeNotice,
                         isBottomSheetVisible = !isPromoCodeApplied,
                     )
                 )
                 navigator.hideLoading()
             }
             .catch {
                 logger.i("onCheckPromoCode error $it")
                 onError?.invoke(it.message ?: getString(Res.string.devErrorDesc))
                 dispatch(
                     PromoCodeReducer.Action.SetState(
                         promoCode = "",
                         isCheckingPromoCode = false,
                         checkPromoCodeResult = null,
                         checkPromoCodeError = it.message,
                         generatedPromoVouchers = state.value.generatedPromoVouchers,
                         isPromoCodeApplied = false,
                         isBottomSheetVisible = true,
                     )
                 )
                 navigator.hideLoading()
             }
             .launchIn(scope)
     }
 
     fun onSetPromoCode(promoCode: String) {
         dispatch(PromoCodeReducer.Action.SetPromoCode(promoCode))
     }
 
     fun onClearPromoCodeError() {
         dispatch(PromoCodeReducer.Action.ClearError)
     }
 
     fun onToggleBottomSheet() {
         dispatch(PromoCodeReducer.Action.ToggleBottomSheet)
     }
 }
