 package vn.momo.compose.payment.singleBill.domain.useCases.voucher
 
 import kotlinx.serialization.json.JsonElement
 import kotlinx.serialization.json.JsonObject
 import kotlinx.serialization.json.buildJsonObject
 import kotlinx.serialization.json.put
 import vn.momo.compose.payment.payx.domain.entities.PromoDisplay
 import vn.momo.compose.payment.payx.domain.repositories.voucher.IVoucherRepository
 import vn.momo.compose.payment.singleBill.data.dataSource.dto.response.SBMomoMsgDTO
 import vn.momo.compose.payment.singleBill.data.dataSource.memory.SBSavedTransaction
 import vn.momo.compose.utils.datamapping.toJsonElement
 
 class GenerateSBChooseVouchersUseCase(
     private val repository: IVoucherRepository,
     private val savedData: SBSavedTransaction,
     ) {
     suspend operator fun invoke(
         promotions: List<PromoDisplay>,
         discountCode: String?,
     ): JsonObject {
         val transaction = buildTransaction(savedData.data)
         return buildJsonObject {
             put("selectedVouchers", repository.convertVoucherItemToJs(promotions).toJsonElement())
             put("discountCode", discountCode)
             put("transaction", transaction)
         }
     }
 
     private fun buildTransaction(
         tranHisMsg: SBMomoMsgDTO?,
     ): JsonElement {
         val transaction = tranHisMsg?.let {
             buildJsonObject {
                 put("ID", it.tranHisMsg?.ID)
                 put("serviceId", it.serviceId)
                 put("originalAmount", it.tranHisMsg?.originalAmount)
                 put("amount", it.tranHisMsg?.amount)
                 put("cbAmount", it.cbAmount)
                 put("moneySource", it.tranHisMsg?.moneySource)
                 put("invoiceId", "")
                 put("paymentType", "")
                 put("enableOptions", buildJsonObject {
                     put("discount", it.tranHisMsg?.enableOptions?.discount)
                     put("voucher", it.tranHisMsg?.enableOptions?.voucher)
                 })
             }
         }
         return transaction.toJsonElement()
     }
 }
