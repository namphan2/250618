     canOpenBank: Boolean,

     val clipboardManager = LocalClipboardManager.current

         clipboardManager.setText(annotatedString = buildAnnotatedString {
             append(text = info.bankAccount.number)
         })
         onCopyBankInfo()
     }
 
     val uiCoroutine = rememberCoroutineScope()
     val logoPainter = rememberImagePainter(info.logo)
     val qrCodePainter = rememberQrCodePainter(
         data = info.qrContent,

             dark = QrBrush.Companion.solid(AppTheme.current.colors.text.default),
             light = QrBrush.Companion.solid(AppTheme.current.colors.background.surface),
             ball = QrBrush.Companion.solid(AppTheme.current.colors.text.default),
             frame = QrBrush.Companion.solid(AppTheme.current.colors.text.default)
         )
     )
     val qrCapture = rememberGraphicsLayer()
 
     Column(
         verticalArrangement = Arrangement.spacedBy(Spacing.M, alignment = Alignment.Bottom)
     ) {
         BottomSheetHeader(
             title = stringResource(Res.string.bankTransTitle),
             iconRight = AsyncIconProp(Images.ic_close_fat, size = 14.dp, onPress = onClose)
         )
         Divider()
         Column(
             modifier = Modifier.fillMaxWidth()
                 .heightIn(max = (0.75 * screenHeight - 190).dp)

                     Text(
                         stringResource(Res.string.bankTransDesc),
                         style = Typography.descriptionDefaultRegular,
                         modifier = Modifier.weight(1f)
                     )
                 }
             }
             VietQRImage(
                 qrCodePainter = qrCodePainter,
                 credit = info.credit,

                 color = AppTheme.current.colors.background.selected,

                 Column(

                             info.bankInfo.name,
                             style = Typography.headerSSemibold,
                             modifier = Modifier.weight(1f)
                         )

                 InfoRow {

                         stringResource(Res.string.totalPayment),

                         text,
