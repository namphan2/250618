 class NapasInputRemoteDataSource(
     private val networkApi: IRemoteService,
     private val clientIp: IClientIpAddress,
 ) {
     fun initCashInByCardInfo(params: InitCashInPayloadDTO): Flow<InitCashInResponseDTO> {
         return clientIp.getIpAddress().flatMapLatest { ipAddress ->
             val message = InitCashInRequestDTO(
                 msgType = "NAPAS_CASHIN_INIT_MSG",
                 momoMsg = params.copy(
                     clientIP = ipAddress
                 ),
                 headers = emptyJsonObject(),
                 subPath = "api/NAPAS_CASHIN_INIT_MSG",
                 subDomain = "owa"
             )
             networkApi.post(message, InitCashInResponseDTO.serializer()).map {
                 it?.response ?: throw NapasInputResponseError.DevError(
                     DevError.NoResponse.code,
                     "Init card response error",
                     cause = Exception(this::class.simpleName),
                 )
             }
         }
 
     }
 }
