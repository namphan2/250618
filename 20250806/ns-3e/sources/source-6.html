 @Composable
 fun LinkList(data: PayXLinkList, onPress: ((cta: PayXCta?) -> Unit)? = null) {
     Column(
         modifier = Modifier
             .fillMaxWidth()
             .background(
                 color = Colors.black_01,
                 RoundedCornerShape(size = Spacing.M)
             )
             .padding(horizontal = Spacing.M)
     ) {
         data.content.forEachIndexed { index, item ->
             Row(
                 modifier = Modifier
                     .fillMaxWidth()
                     .noFeedbackClickable(enabled = item.action?.refId != null) {
                         onPress?.invoke(item.action)
                     }
                     .padding(vertical = Spacing.S),
                 verticalAlignment = Alignment.CenterVertically,
                 horizontalArrangement = Arrangement.SpaceBetween
             ) {
                 item.text?.let {
                     Text(
                         text = item.text,
                         color = item.color,
                         style = item.style ?: Typography.bodyDefaultRegular,
                         overflow = TextOverflow.Ellipsis,
                         maxLines = item.numberOfLine ?: Int.MAX_VALUE
                     )
                 }
                 Icon(
                     source = "16_arrow_chevron_right",
                     size = 16.dp
                 )
             }
             if (index < data.content.size - 1) {
                 Divider()
             }
         }
     }
 }
