 package vn.momo.compose.payment.base.core.composeApi
 
 import kotlinx.serialization.builtins.serializer
 import vn.momo.api.APIs
 import vn.momo.api.IComposeApi
 import vn.momo.api.requestOneShot
 import vn.momo.compose.payment.base.core.composeApi.model.VixResponse
 import vn.momo.compose.utils.datamapping.ModelConvertUtils
 
 class ImageApi(private val api: IComposeApi) {
     suspend fun requestSaveImagePermission(): Boolean {
         val permission = api.requestOneShot(APIs.requestPermission.name, listOf("storage")).let {
             ModelConvertUtils.decode(VixResponse.serializer(String.serializer()), it)?.response
         }
         return permission == "granted" || permission == "limited"
     }
     suspend fun saveImage(base64: String): Boolean {
         return api.requestOneShot(APIs.saveImage.name, listOf(base64)).let {
             ModelConvertUtils.decode(VixResponse.serializer(Boolean.serializer()), it)?.response
         } ?: false
     }
 }
