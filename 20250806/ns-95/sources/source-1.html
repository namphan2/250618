 package vn.momo.compose.payment.payx.presentation.view.components.radio
 
 import androidx.compose.animation.animateColorAsState
 import androidx.compose.animation.core.animateDpAsState
 import androidx.compose.animation.core.tween
 import androidx.compose.foundation.border
 import androidx.compose.foundation.layout.Box
 import androidx.compose.foundation.layout.size
 import androidx.compose.foundation.shape.RoundedCornerShape
 import androidx.compose.runtime.Composable
 import androidx.compose.ui.Modifier
 import androidx.compose.ui.draw.clip
 import androidx.compose.ui.unit.dp
 import androidx.compose.ui.unit.times
 import vn.momo.kits.const.AppTheme
 import vn.momo.kits.const.Radius
 
 @Composable
 fun RadioButton(
     selected: Boolean,
     modifier: Modifier = Modifier,
     enabled: Boolean = true,
 ) {
     val borderWidth = animateDpAsState(
         targetValue = if (selected) 0.3 * RadioIconSize else 0.1 * RadioIconSize,
         animationSpec = tween(durationMillis = RadioAnimationDuration)
     )
 
     val borderColor = animateColorAsState(
         targetValue = when {
             !enabled && selected -> AppTheme.current.colors.background.tonal
             !enabled -> AppTheme.current.colors.text.disable
             selected -> AppTheme.current.colors.primary
             else -> AppTheme.current.colors.text.default
         },
         animationSpec = tween(durationMillis = RadioAnimationDuration)
     )
 
     Box(
         modifier = modifier
             .border(borderWidth.value, borderColor.value, RoundedCornerShape(Radius.M))
             .clip(RoundedCornerShape(Radius.M))
             .size(RadioIconSize)
     )
 }
 
 private val RadioIconSize = 20.dp
 private const val RadioAnimationDuration = 200
