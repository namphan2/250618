 package vn.momo.compose.payment.promotion.presentation.view.components
 
 import androidx.compose.foundation.Canvas
 import androidx.compose.foundation.layout.fillMaxWidth
 import androidx.compose.foundation.layout.height
 import androidx.compose.runtime.Composable
 import androidx.compose.ui.Modifier
 import androidx.compose.ui.geometry.Offset
 import androidx.compose.ui.geometry.Size
 import androidx.compose.ui.graphics.PathEffect
 import androidx.compose.ui.graphics.drawscope.Fill
 import androidx.compose.ui.platform.LocalDensity
 import androidx.compose.ui.unit.dp
 import vn.momo.compose.paymentoffline.presentation.components.toPx
 import vn.momo.kits.const.AppTheme
 import vn.momo.kits.const.Spacing
 
 @Composable
 fun VoucherSeparator() {
     val arcColor = AppTheme.current.colors.background.default
     val pathColor = AppTheme.current.colors.border.default
     val pathEffect = PathEffect.dashPathEffect(floatArrayOf(15f, 10f), 0f)
     val arcRadius = Spacing.S.toPx()
     val strokeWidthPx = with(LocalDensity.current) { 1.dp.toPx() }
 
     Canvas(
         modifier = Modifier
             .fillMaxWidth()
             .height(Spacing.L)
     ) {
         val centerY = size.height / 2
 
         drawArc(
             color = arcColor,
             startAngle = -90f,
             sweepAngle = 180f,
             useCenter = true,
             topLeft = Offset(-arcRadius, centerY - arcRadius),
             size = Size(arcRadius * 2, arcRadius * 2),
             style = Fill
         )
 
         drawLine(
             color = pathColor,
             start = Offset(arcRadius, centerY),
             end = Offset(size.width - arcRadius, centerY),
             pathEffect = pathEffect,
             strokeWidth = strokeWidthPx
         )
 
         drawArc(
             color = arcColor,
             startAngle = 90f,
             sweepAngle = 180f,
             useCenter = true,
             topLeft = Offset(size.width - arcRadius, centerY - arcRadius),
             size = Size(arcRadius * 2, arcRadius * 2),
             style = Fill
         )
     }
 }
