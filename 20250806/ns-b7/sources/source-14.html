 @Composable
 fun VoucherCoinItem(
     voucher: IVoucherBase,
     modifier: Modifier = Modifier,
     selected: Boolean = false,
     onToggleSelect: (IVoucherBase) -> Unit? = {},
     onDisplayed: () -> Unit? = {},
 ) {
     val isInsufficientCoin = voucher.extra?.isInsufficientCoin ?: false
     val backgroundColor by animateColorAsState(
         targetValue = if (selected) AppTheme.current.colors.background.selected else AppTheme.current.colors.background.surface
     )
     val borderColor by animateColorAsState(
         targetValue = if (selected) AppTheme.current.colors.secondary else AppTheme.current.colors.background.surface
     )
     val iconScale by animateFloatAsState(
         targetValue = if (selected) 1f else 0f, // Zoom in/out scale
     )
 
     val prefixName = when {
         !voucher.prefixName.isNullOrBlank() -> voucher.prefixName
         !voucher.extra?.prefixName.isNullOrBlank() -> voucher.extra?.prefixName
         !voucher.extraGiftType?.prefixName.isNullOrBlank() -> voucher.extraGiftType?.prefixName
         else -> null
     }
     val suffixName = when {
         !voucher.suffixName.isNullOrBlank() -> voucher.suffixName
         !voucher.extra?.suffixName.isNullOrBlank() -> voucher.extra?.suffixName
         !voucher.extraGiftType?.suffixName.isNullOrBlank() -> voucher.extraGiftType?.suffixName
         else -> null
     }
 
     LaunchedEffect(voucher.extra?.rewardId) {
         onDisplayed()
     }
 
     Surface(
         modifier = modifier
             .clickable(
                 enabled = true,
                 onClick = { onToggleSelect(voucher) },
                 interactionSource = remember { MutableInteractionSource() },
                 indication = null,
             ),
         shape = RoundedCornerShape(Radius.M)
     ) {
         Column(
             modifier = Modifier
                 .fillMaxSize()
                 .conditional(isInsufficientCoin || voucher.extra?.isInvalid == true) {
                     background(Colors.yellow_09)
                 }
         ) {
             Box(
                 modifier = Modifier
                     .fillMaxSize()
                     .background(color = backgroundColor, shape = RoundedCornerShape(Radius.M))
                     .alpha(if (isInsufficientCoin) 0.5f else 1f)
                     .border(
                         width = 1.dp,
                         color = borderColor,
                         shape = RoundedCornerShape(Radius.M)
                     )
                     .clip(RoundedCornerShape(Radius.M))
                     .padding(Spacing.XS),
             ) {
                 Row(
                     horizontalArrangement = Arrangement.spacedBy(Spacing.S),
                     modifier = Modifier.height(IntrinsicSize.Max)
                 ) {
                     Column(
                         modifier = Modifier
                             .background(Colors.black_02, RoundedCornerShape(Radius.S))
                             .fillMaxHeight()
                             .width(92.dp)
                             .padding(
                                 vertical = Spacing.XS,
                                 horizontal = Spacing.S + 2.dp
                             ),
                         horizontalAlignment = Alignment.CenterHorizontally,
                         verticalArrangement = Arrangement.Center,
                     ) {
                         Image(
                             source = rememberAsyncImagePainter(voucher.icon),
                             modifier = Modifier
                                 .size(Spacing.XXL)
                                 .clip(RoundedCornerShape(Spacing.L))
                         )
                         Text(
                             text = voucher.extra?.voucherServiceName
                                 ?: stringResource(Res.string.choosePromotionScreenTitle),
                             color = Colors.black_17,
                             textAlign = TextAlign.Center,
                             style = Typography.headerXsSemibold,
                             modifier = Modifier.padding(top = Spacing.XS),
                             maxLines = 2,
                             overflow = TextOverflow.Ellipsis,
                         )
                     }
 
                     Column(
                         modifier = Modifier
                             .weight(1f)
                             .heightIn(min = 80.dp)
                             .fillMaxWidth()
                             .padding(vertical = Spacing.XS),
                         verticalArrangement = Arrangement.spacedBy(Spacing.XXS),
                     ) {
                         if (!prefixName.isNullOrBlank()) {
                             Text(
                                 text = prefixName,
                                 style = Typography.headerSSemibold,
                                 color = AppTheme.current.colors.text.default,
                                 maxLines = 2,
                                 modifier = Modifier.padding(end = Spacing.L),
                                 overflow = TextOverflow.Ellipsis,
                             )
                         }
 
                         if (!suffixName.isNullOrBlank()) {
                             Text(
                                 text = suffixName,
                                 style = Typography.descriptionDefaultRegular,
                                 color = AppTheme.current.colors.text.hint,
                                 maxLines = 2,
                                 modifier = Modifier.padding(end = Spacing.L),
                                 overflow = TextOverflow.Ellipsis,
                             )
                         }
 
                         Spacer(modifier = Modifier.weight(1f))
 
                         Row(
                             verticalAlignment = Alignment.CenterVertically,
                             horizontalArrangement = Arrangement.End,
                             modifier = Modifier
                                 .fillMaxWidth()
                                 .padding(horizontal = Spacing.XS)
                         ) {
                             Image(
                                 source = rememberAsyncImagePainter(Images.ic_rewards_coin),
                                 modifier = Modifier
                                     .size(20.dp)
                             )
                             Text(
                                 text = voucher.extra?.salePrice?.toDouble()
                                     ?.formatCurrency(currency = "")
                                     ?: "0",
                                 style = Typography.actionSBold,
                                 color = Colors.gold_02,
                                 overflow = TextOverflow.Ellipsis,
                             )
                             Text(
                                 text = voucher.extra?.originalPrice?.toDouble()
                                     ?.formatCurrency(currency = "")
                                     ?: "0",
                                 style = Typography.descriptionXsRegularStrikethrough,
                                 color = Colors.black_08,
                                 overflow = TextOverflow.Ellipsis,
                                 modifier = Modifier
                                     .padding(start = Spacing.XS)
                             )
                         }
                     }
 
                 }
 
                 Box(
                     modifier = Modifier
                         .size(Spacing.L)
                         .align(Alignment.TopEnd),
                 ) {
                     if (selected) {
                         Icon(
                             source = "notifications_check_circle_full",
                             size = Spacing.L,
                             color = AppTheme.current.colors.primary,
                             modifier = Modifier
                                 .scale(iconScale)
                         )
                     }
                 }
             }
 
             if (isInsufficientCoin || voucher.extra?.isInvalid == true) {
                 Row(
                     modifier = Modifier
                         .fillMaxWidth()
                         .padding(vertical = Spacing.XS, horizontal = Spacing.S),
                     verticalAlignment = Alignment.CenterVertically,
                     horizontalArrangement = Arrangement.spacedBy(
                         space = Spacing.XS,
                         alignment = Alignment.Start
                     ),
                 ) {
                     Image(
                         source = rememberAsyncImagePainter(Images.ic_warning),
                         modifier = Modifier
                             .size(size = Spacing.L),
                     )
                     when {
                         !voucher.extra?.failGiftDesc.isNullOrBlank() -> voucher.extra?.failGiftDesc
                         else -> stringResource(Res.string.voucherCoinInvalidMessage)
                     }?.let {
                         Text(
                             text = it,
                             style = Typography.descriptionXsRegular,
                             maxLines = 1,
                             overflow = TextOverflow.Ellipsis,
                         )
                     }
                 }
             }
         }
     }
 }
