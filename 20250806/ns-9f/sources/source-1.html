 class UiBottomSheet(
     private val navigator: INavigator,
     private val data: PayXBottomSheet,
     override val actionCreator: PayXActionCreator
 ): UiOverlay {
     override suspend fun handle(): PayXAction? {
         val data = navigator.showBottomSheet(buildJsonObject {
             put("type", "customContent")
             put(
                 "props", ModelConvertUtils.encodeToJson(
                     CustomContentBottomSheetProps.serializer(),
                     CustomContentBottomSheetProps(
                         data,
                         true
                     )
                 ) ?: emptyJsonObject()
             )
             put("cancelable", true)
         }).firstOrNull()
         val button = data?.get("button")
         val action = ModelConvertHelper.decode(PayXButtonBottomSheet.serializer(), button?.toString() ?: "") ?: PayXButtonBottomSheet()
         return actionCreator.createAction(action)
     }
 }
