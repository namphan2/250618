 package vn.momo.compose.payment.promotion.presentation.viewModel.reducers
 
 import vn.momo.compose.payment.base.presentation.viewModel.reducer.Reducer
 import vn.momo.compose.payment.promotion.domain.entities.IVoucherBase
 import vn.momo.compose.payment.promotion.presentation.utils.distinctById
 import vn.momo.core.features.annotation.Immutable
 
 class VoucherSearchReducer :
     Reducer<VoucherSearchReducer.State, VoucherSearchReducer.Action, VoucherSearchReducer.Effect> {
     @Immutable
     data class State(
         val searchModalVisivility: Boolean? = false,
         val searchVouchers: List<IVoucherBase> = emptyList(),
         val isSearchingVoucher: Boolean? = false,
         val selectedVouchers: List<IVoucherBase> = emptyList(),
     ) : Reducer.State
 
     @Immutable
     sealed class Action : Reducer.Action {
         data class SetSearchModalVisibility(val isVisible: Boolean) : Action()
         data class SetSearchVouchers(val vouchers: List<IVoucherBase>) : Action()
         data class SelectSearchVoucher(val voucher: IVoucherBase) : Action()
         data class SetSearchVoucherState(val isSearching: Boolean) : Action()
     }
 
     @Immutable
     class Effect() : Reducer.Effect
 
     override fun reduce(prev: State, action: Action): Pair<State, Effect?> {
         return when (action) {
             is Action.SetSearchModalVisibility -> prev.copy(
                 searchModalVisivility = action.isVisible,
             ) to null
 
             is Action.SetSearchVouchers -> prev.copy(
                 searchVouchers = action.vouchers,
                 isSearchingVoucher = false
             ) to null
 
             is Action.SetSearchVoucherState -> prev.copy(
                 isSearchingVoucher = action.isSearching,
             ) to null
 
             is Action.SelectSearchVoucher -> prev.copy(
                 selectedVouchers = (listOf(action.voucher) + prev.selectedVouchers).distinctById(),
             ) to null
         }
     }
 }
