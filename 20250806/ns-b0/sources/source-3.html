 class VoucherPosRepository(
     private val remoteDataSource: VoucherPosDataSource
 ) : IVoucherPosRepository {
     override fun getListVoucherPos(
         pathType: VoucherPathType,
         page: Int?,
         params: ChoosePromotionPosScreenProps,
         location: LocationCoords?,
     ): Flow<ListPage<IVoucherBase>> {
         val request = VoucherPosRequestDTO(
             body = VoucherPosRequestRequestBody(
                 diffMoneySource = pathType == VoucherPathType.PosInvalid,
                 page = page,
                 displayConfig = params.displayConfig ?: params.serviceId,
                 type_ = -1,
                 isLocationSort = true,
                 lat = location?.latitude?.toString(),
                 lng = location?.longitude?.toString(),
                 moneySourceFilter = params.moneySourceFilter?.joinToString(","),
                 cardNumberFilter = params.cardNumberFilter?.toFormattedCardNumbers()
                     ?.joinToString(","),
             )
         )
 
         return remoteDataSource.getRecommendVouchers(request)
             .tracePromotionApi(pathType.traceEvent)
             .map { response ->
                 response?.momoMsg?.let {
                     ListPage(page = it.page ?: 1,
                         limit = it.limit ?: DEFAULT_PAGE_SIZE,
                         totalRecord = it.totalRecord ?: 0,
                         records = it.records?.map { item ->
                             item.toVoucherBase(
                                 VoucherPathType.toLocalType(pathType)
                             )
                         } ?: emptyList())
                 } ?: ListPage(
                     page = 1, limit = DEFAULT_PAGE_SIZE, totalRecord = 0, records = emptyList()
                 )
             }
     }
 
     override fun onSearchVoucherPos(
         searchKey: String, page: Int?, params: ChoosePromotionPosScreenProps
     ): Flow<ListPage<IVoucherBase>> {
         val cardNumber = if (!params.cardNumberFilter.isNullOrEmpty()) {
             params.cardNumberFilter.toFormattedCardNumbers().joinToString(",")
         } else {
             null
         }
 
         val request = VoucherPosSearchRequestDTO(
             body = VoucherPosSearchRequestRequestBody(
                 momoMsg = VoucherPosSearchRequestRequestMomoMsg(
                     page = page,
                     searchKey = searchKey,
                     isPaymentOffline = true,
                     moneySourceFilter = params.moneySourceFilter,
                     cardNumber = cardNumber,
                 )
             )
         )
 
         return remoteDataSource.searchVoucher(request)
             .tracePromotionApi(PromotionApiPerfEvent.GetSearchVouchersPos.event)
             .map { response ->
                 println("onSearchVoucherPos response: $response")
                 response?.momoMsg?.let {
                     ListPage(page = it.page ?: 1,
                         limit = it.limit ?: DEFAULT_PAGE_SIZE,
                         totalRecord = it.totalRecord ?: 0,
                         records = it.records?.map { item ->
                             item.toVoucherBase()
                         } ?: emptyList())
                 } ?: ListPage(
                     page = 1, limit = DEFAULT_PAGE_SIZE, totalRecord = 0, records = emptyList()
                 )
             }
     }
 }
