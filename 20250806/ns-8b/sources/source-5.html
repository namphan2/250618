 @OptIn(ExperimentalMaterial3Api::class)
 @Composable
 fun PayXComposeBottomSheet(
     state: MutableState<Boolean>,
     title: String,
     backgroundColor: Color = Color(0xfff2f2f6),
     footerColor: Color = AppTheme.current.colors.background.surface,
     hasFooter: Boolean = false,
     maxHeightRatio: Double = 0.65,
     onClose: (buttonName: String) -> Unit = {},
     content: @Composable () -> Unit,
 ) {
     val screenHeight = getScreenDimensions().screenHeight
     val sheetState = rememberModalBottomSheetState(
         skipPartiallyExpanded = true,
     )
     val coroutineScope = rememberCoroutineScope()
     val bottomInsetDp = with(LocalDensity.current) {
         BottomSheetDefaults.windowInsets.getBottom(this).toDp().takeIf { it.value > 0 } ?: 38.dp
     }
     val scrollState = rememberScrollState()
 
     fun requestClose(buttonName: String = "icon") {
         coroutineScope.launch { sheetState.hide() }.invokeOnCompletion {
             if (!sheetState.isVisible) {
                 state.value = false
                 onClose(buttonName)
             }
         }
     }
 
     if (state.value) {
         ModalBottomSheet(onDismissRequest = {
             state.value = false
         },
             sheetState = sheetState,
             dragHandle = {
                 BottomSheetDefaults.DragHandle(
                     color = Colors.black_06, width = 40.dp
                 )
             },
             shape = RoundedCornerShape(topStart = 12.dp, topEnd = 12.dp),
             contentWindowInsets = { WindowInsets(0.dp) },
             containerColor = backgroundColor
         ) {
             Column(
                 modifier = Modifier.fillMaxWidth().heightIn(max = (screenHeight * maxHeightRatio).dp)
                     .animateContentSize()
             ) {
                 BottomSheetHeader(
                     modifier = Modifier.padding(horizontal = Spacing.L),
                     title = title,
                     iconRight = AsyncIconProp(
                         Images.ic_close_fat,
                         size = 14.dp,
                         onPress = { requestClose() })
                 )
                 Spacer(modifier = Modifier.height(18.dp))
                 Box(
                     modifier = Modifier
                         .fillMaxWidth()
                         .weight(1f, fill = false)
                         .verticalScroll(scrollState)
                 ) {
                     content()
                 }
                 Spacer(modifier = Modifier.height(12.dp))
                 Box(
                     modifier = Modifier.fillMaxWidth()
                         .background(if (hasFooter) footerColor else backgroundColor)
                 ) {
                     Button(modifier = Modifier.padding(
                         start = Spacing.M,
                         end = Spacing.M,
                         top = Spacing.S,
                         bottom = bottomInsetDp
                     ), title = stringResource(Res.string.close), onClick = {
                         requestClose("primary")
                     })
                 }
             }
         }
     }
 }
