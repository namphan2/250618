 class XSaleReducer :
     Reducer<XSaleReducer.XSaleState, XSaleReducer.XSaleAction, XSaleReducer.XSaleEffect> {
     @Immutable
     data class XSaleState(
         val config: XSaleConfig? = null,
         val selected: XSaleItem? = null
     ) : Reducer.State
 
     @Immutable
     sealed class XSaleAction : Reducer.Action {
         data class UpdateConfig(val config: XSaleConfig?) : XSaleAction()
         data class Select(val item: XSaleItem) : XSaleAction()
         data class Highlight(val id: String) : XSaleAction()
         object Deselect : XSaleAction()
     }
 
     @Immutable
     sealed class XSaleEffect : Reducer.Effect {
         data class AddCart(val itemId: String, val payload: JsonObject) : XSaleEffect()
         data class RemoveCart(val itemId: String) : XSaleEffect()
     }
 
     override fun reduce(
         prev: XSaleState,
         action: XSaleAction
     ): Pair<XSaleState, XSaleEffect?> {
         return when (action) {
             is XSaleAction.UpdateConfig -> prev.copy(config = action.config) to null
             is XSaleAction.Select -> onSelect(prev, action.item)
             is XSaleAction.Highlight -> prev.copy(
                 selected = prev.config?.data?.firstOrNull { it.id == action.id }
             ) to null
 
             is XSaleAction.Deselect -> prev.copy(selected = null) to null
         }
     }
 
     private fun onSelect(prev: XSaleState, item: XSaleItem): Pair<XSaleState, XSaleEffect?> {
         val effect = when {
             item.id == prev.selected?.id -> XSaleEffect.RemoveCart(itemId = item.id)
             item.action is PayXAction.SendMessage -> XSaleEffect.AddCart(
                 itemId = item.id,
                 payload = item.action.payload
             )
 
             else -> null
         }
 
         prev.config?.let { config ->
             FirebaseTrace.track(
                 XSaleItemClicked(
                     serviceId = prev.config.serviceId,
                     purchaseId = PaymentKoinScope.getScope()
                         .get<PaymentSession>().requestPayXData.initialPurchaseId,
                     layoutType = config.trackingLayout,
                     uiItemType = "${config.trackingLayout}1",
                     action = when {
                         item.id == prev.selected?.id -> "deselect"
                         item.action is PayXAction.SendMessage -> "select"
                         else -> ""
                     },
                     itemId = item.id
                 )
             )
         }
         return prev to effect
     }
 }
