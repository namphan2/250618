 @Composable
 fun InvoiceInfoView(data: List<InvoiceInfo>, hideAmount: Boolean = false) {
     Column {
         data.forEachIndexed { index, item ->
             InvoiceItemUiView(
                 modifier = Modifier.padding(vertical = 12.dp),
                 key = item.title,
                 value = item.value,
                 titleTextStyle = item.titleTextStyle,
                 valueTextStyle = item.valueTextStyle,
                 isShowBorderBottom = !hideAmount || index < data.lastIndex
             )
         }
     }
 }
 
 @Composable
 fun InvoiceItemUiView(
     modifier: Modifier = Modifier,
     key: String,
     value: String,
     isShowBorderTop: Boolean? = false,
     isShowBorderBottom: Boolean? = true,
     titleTextStyle: TextCustomStyle? = null,
     valueTextStyle: TextCustomStyle? = null,
 ) {
     val titleColor = titleTextStyle?.getColorFromStyle(Colors.black_12) ?: Colors.black_12
     val valueColor = valueTextStyle?.getColorFromStyle(Colors.black_17) ?: Colors.black_17
 
     if (isShowBorderTop == true) {
         Divider(color = Colors.black_03, thickness = 1.dp)
     }
     BoxWithConstraints(modifier = Modifier.fillMaxWidth().then(modifier)) {
         // ACS-2796 Chia đúng tỷ lệ paymentInfo
         Row(
             modifier = Modifier.fillMaxWidth(),
             verticalAlignment = Alignment.CenterVertically,
             horizontalArrangement = Arrangement.SpaceBetween
         ) {
             Text(
                 text = key,
                 style = Typography.bodyDefaultRegular,
                 color = titleColor,
                 textAlign = TextAlign.Start,
                 modifier = Modifier.weight(4f),
                 maxLines = 2,
                 overflow = TextOverflow.Ellipsis
             )
             Spacer(modifier = Modifier.width(8.dp))
             Text(
                 text = value,
                 style = Typography.headerSSemibold,
                 color = valueColor,
                 textAlign = TextAlign.End,
                 modifier = Modifier.weight(6f),
                 maxLines = 2,
                 overflow = TextOverflow.Ellipsis
             )
         }
     }
     if (isShowBorderBottom == true) {
         Divider(color = Colors.black_03, thickness = 1.dp)
     }
 }
