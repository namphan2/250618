 import androidx.compose.foundation.layout.Arrangement
 import androidx.compose.foundation.layout.Row
 import androidx.compose.foundation.layout.Spacer
 import androidx.compose.foundation.layout.fillMaxWidth
 import androidx.compose.foundation.layout.size
 import androidx.compose.runtime.Composable

 import androidx.compose.ui.Modifier
 import androidx.compose.ui.text.style.TextAlign
 import androidx.compose.ui.unit.dp
 import androidx.compose.ui.unit.max
 import vn.momo.kits.components.Text

 import vn.momo.kits.const.Colors
 
 @Composable
 fun BottomSheetHeader(
     modifier: Modifier = Modifier,
     title: String = "",
     iconLeft: AsyncIconProp? = null,
     iconRight: AsyncIconProp? = null,
 ) {
     val iconSize = max(iconLeft?.size ?: 0.dp, iconRight?.size ?: 0.dp)
     Row(
         modifier = Modifier
             .fillMaxWidth()
             .then(modifier),
         verticalAlignment = Alignment.CenterVertically,
         horizontalArrangement = Arrangement.spacedBy(8.dp)
     ) {
         if (iconLeft != null) {
             AsyncIcon(prop = iconLeft)
             Spacer(modifier = Modifier.size(iconSize - iconLeft.size))
         } else {
             Spacer(modifier = Modifier.size(iconSize))
         }
         Text(
             text = title,
             style = Typography.headerDefaultBold,
             color = Colors.black_17,
             textAlign = TextAlign.Center,
             modifier = Modifier.weight(1F),
         )
         if (iconRight != null) {
             Spacer(modifier = Modifier.size(iconSize - iconRight.size))
             AsyncIcon(prop = iconRight)
         } else {
             Spacer(modifier = Modifier.size(iconSize))
         }
     }
 }
