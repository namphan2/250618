 package vn.momo.compose.payment.payx.presentation.view.components
 
 import androidx.compose.ui.geometry.Offset
 import androidx.compose.ui.geometry.Rect
 import androidx.compose.ui.geometry.Size
 import androidx.compose.ui.graphics.Outline
 import androidx.compose.ui.graphics.Path
 import androidx.compose.ui.graphics.Shape
 import androidx.compose.ui.unit.*
 import kotlin.math.round
 
 //TODO:[TR] migrate to kits
 internal data class DottedShape(
     val step: Dp,
 ) : Shape {
     override fun createOutline(
         size: Size, layoutDirection: LayoutDirection, density: Density
     ) = Outline.Generic(Path().apply {
         val stepPx = with(density) { step.toPx() }
 
         val stepsCount = round((size.width / stepPx) / 2.0).toInt()
         val gapCount = (stepsCount - 1)
 
         val gapPx = ((size.width - (stepsCount * stepPx)) / gapCount)
         val dotSize = Size(width = stepPx, height = size.height)
 
         for (i in 0 until stepsCount) {
             addRect(
                 Rect(
                     offset = Offset(x = ((i * (gapPx + stepPx))), y = 0f), size = dotSize
                 )
             )
         }
         close()
     })
 }
