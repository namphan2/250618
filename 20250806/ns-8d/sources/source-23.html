 @Composable
 fun MoneySourceItem(
     data: SOFItem,
     isHideBalance: Boolean,
     isSelected: Boolean,
     onPress: () -> Unit
 ) {
     val screenDimension = getScreenDimensions().screenWidth
     val itemWidth = screenDimension * 0.35
     PositionedBadgedBox(badge = {
         val badge = data.badge
         if (badge != null && badge.shortTitle.isNotEmpty()) {
             Badge(
                 backgroundColor = if (badge.type == TaglineType.Promotion) Colors.orange_03 else Colors.gold_05,
             ) {
                 Text(
                     text = badge.shortTitle,
                     style = Typography.actionXxsBold,
                     color = Colors.black_01
                 )
             }
         }
     }, offset = { contentSize, badgeSize ->
         IntOffset(x = contentSize.width - badgeSize.width - 12, y = -badgeSize.height / 2)
     }) {
         Box(modifier = Modifier.semantics {
             testTag = "source_funds_${data.moneySource.id}"
             contentDescription = "source_funds_${data.moneySource.id}"
         }.width(itemWidth.dp)
             .background(
                 if (isSelected) Colors.pink_10 else Colors.black_01,
                 shape = RoundedCornerShape(12.dp)
             )
             .border(
                 if (isSelected) BorderStroke(2.dp, Colors.pink_03)
                 else BorderStroke(1.dp, Colors.black_04), shape = RoundedCornerShape(12.dp)
             )
             .noFeedbackClickable(enabled = data.available) { onPress() }
             .padding(start = 8.dp, end = 8.dp, top = 10.dp, bottom = 10.dp)
             .alpha(if (data.available || isSelected) 1f else 0.5f),
             contentAlignment = Alignment.Center) {
             Row(verticalAlignment = Alignment.CenterVertically) {
                 Image(
                     source = data.logo ?: "", //TODO: set default icon
                     modifier = Modifier.size(24.dp)
                 )
                 Spacer(modifier = Modifier.width(8.dp))
                 Column(
                     modifier = Modifier.weight(1f),
                     verticalArrangement = Arrangement.spacedBy(2.dp)
                 ) {
                     Text(
                         text = data.shortName,
                         style = Typography.headerXsSemibold,
                         color = Colors.black_17,
                         maxLines = 1,
                         overflow = TextOverflow.Ellipsis
                     )
                     Text(
                         text = if(isHideBalance && data.shortDesc.isCurrency()) "******" else data.shortDesc,
                         style = Typography.descriptionDefaultRegular,
                         color = Colors.black_12,
                         maxLines = 1,
                         overflow = TextOverflow.Ellipsis
                     )
                 }
             }
         }
     }
 }
