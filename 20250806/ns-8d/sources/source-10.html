 @Composable
 fun DotWave(dotSize: Dp = 10.dp, dotSpacing: Dp = 4.dp) {
     val animationDuration = 300
     val animationHeight = -6f
     val initialAnimationHeight = 0f
 
     val infiniteTransition = rememberInfiniteTransition()
 
     val dot1OffsetY by infiniteTransition.animateFloat(
         initialValue = initialAnimationHeight,
         targetValue = animationHeight,
         animationSpec = infiniteRepeatable(
             animation = tween(animationDuration, easing = LinearEasing),
             repeatMode = RepeatMode.Reverse
         )
     )
     val dot2OffsetY by infiniteTransition.animateFloat(
         initialValue = initialAnimationHeight,
         targetValue = animationHeight,
         animationSpec = infiniteRepeatable(
             animation = tween(
                 animationDuration, easing = LinearEasing
             ),
             repeatMode = RepeatMode.Reverse,
             initialStartOffset = StartOffset(animationDuration / 2)
         )
     )
     val dot3OffsetY by infiniteTransition.animateFloat(
         initialValue = initialAnimationHeight,
         targetValue = animationHeight,
         animationSpec = infiniteRepeatable(
             animation = tween(
                 animationDuration, easing = LinearEasing
             ), repeatMode = RepeatMode.Reverse, initialStartOffset = StartOffset(animationDuration)
         )
     )
 
     val colorTransition = listOf(
         Color(0xFFF067B2), Color(0xFFF58BC5), Color(0xFFF8ADD6)
     )
     val dot1Color by infiniteTransition.animateColor(
         initialValue = colorTransition[0],
         targetValue = colorTransition[1],
         animationSpec = infiniteRepeatable(
             animation = tween(animationDuration, easing = LinearEasing),
             repeatMode = RepeatMode.Reverse
         )
     )
     val dot2Color by infiniteTransition.animateColor(
         initialValue = colorTransition[1],
         targetValue = colorTransition[2],
         animationSpec = infiniteRepeatable(
             animation = tween(
                 animationDuration, easing = LinearEasing
             ),
             repeatMode = RepeatMode.Reverse,
             initialStartOffset = StartOffset(animationDuration / 2)
         )
     )
     val dot3Color by infiniteTransition.animateColor(
         initialValue = colorTransition[2],
         targetValue = colorTransition[0],
         animationSpec = infiniteRepeatable(
             animation = tween(
                 animationDuration, easing = LinearEasing
             ), repeatMode = RepeatMode.Reverse, initialStartOffset = StartOffset(animationDuration)
         )
     )
 
     Row(
         horizontalArrangement = Arrangement.spacedBy(dotSpacing),
         verticalAlignment = Alignment.CenterVertically,
     ) {
         Box(
             modifier = Modifier.offset(y = dot1OffsetY.dp).size(dotSize)
                 .clip(MaterialTheme.shapes.small).background(dot1Color)
         )
         Box(
             modifier = Modifier.offset(y = dot2OffsetY.dp).size(dotSize)
                 .clip(MaterialTheme.shapes.small).background(dot2Color)
         )
         Box(
             modifier = Modifier.offset(y = dot3OffsetY.dp).size(dotSize)
                 .clip(MaterialTheme.shapes.small).background(dot3Color)
         )
     }
 }
