 package vn.momo.compose.payment.promotion.core.utils
 
 import kotlinx.coroutines.CoroutineScope
 import kotlinx.coroutines.flow.MutableStateFlow
 import kotlinx.coroutines.flow.combine
 import kotlinx.coroutines.flow.launchIn
 import kotlinx.coroutines.flow.onEach
 
 /**
  * Handle multiple infinite queries in a sequence with lazy query creation.
  * The queriesProvider will be called when the sequence is first enabled.
  */
 fun <T> sequenceQueriesFactory(
     scope: CoroutineScope,
     queriesProvider: () -> List<InfiniteQuery<T>>,
     initialEnabled: Boolean = true
 ): InfiniteQuery<T> {
     var actualSequenceQuery: InfiniteQuery<T>? = null
     
     val enabledState = MutableStateFlow(initialEnabled)
     val combinedState = MutableStateFlow(InfiniteQueryState<T>())
     
     fun createAndSetupSequence(): InfiniteQuery<T> {
         val queries = queriesProvider()
         if (queries.isEmpty()) {
             throw IllegalArgumentException("Queries list cannot be empty")
         }
         
         println("[SequenceQueriesFactory] Created ${queries.size} queries lazily")
         
         val sequence = createSequenceQuery(scope, queries, true)
         
         // Forward the actual sequence state to our combined state
         sequence.state.onEach { state ->
             combinedState.value = state
         }.launchIn(scope)
         
         return sequence
     }
     
     // Observe enabled state changes
     enabledState.onEach { enabled ->
         when {
             enabled && actualSequenceQuery == null -> {
                 // Create queries and sequence when first enabled
                 actualSequenceQuery = createAndSetupSequence()
                 println("[SequenceQueriesFactory] Lazy sequence enabled with queries")
             }
             !enabled && actualSequenceQuery != null -> {
                 // Disable the actual sequence
                 actualSequenceQuery!!.enabled.value = false
                 println("[SequenceQueriesFactory] Lazy sequence disabled")
             }
             enabled && actualSequenceQuery != null -> {
                 // Re-enable existing sequence
                 actualSequenceQuery!!.enabled.value = true
                 println("[SequenceQueriesFactory] Lazy sequence re-enabled")
             }
         }
     }.launchIn(scope)
     
     return InfiniteQuery(
         state = combinedState,
         onRefetch = { actualSequenceQuery?.onRefetch?.invoke() },
         onFetchNextPage = { actualSequenceQuery?.onFetchNextPage?.invoke() },
         enabled = enabledState
     )
 }
 
 private fun <T> createSequenceQuery(
     scope: CoroutineScope,
     queries: List<InfiniteQuery<T>>,
     initialEnabled: Boolean
 ): InfiniteQuery<T> {
     // Track the currently active query index
     val activeQueryIndex = MutableStateFlow(0)
 
     val enabledState = MutableStateFlow(initialEnabled)
 
     // Initially disable all queries except the first one (and only if globally enabled)
     queries.forEachIndexed { index, query ->
         query.enabled.value = initialEnabled && index == 0
     }
 
     // Combined state from all queries
     val combinedState = MutableStateFlow(InfiniteQueryState<T>())
 
     // Observe enabled state changes
     enabledState.onEach { enabled ->
         if (enabled) {
             // Enable the first query when the sequence is enabled
             val currentIndex = activeQueryIndex.value
             queries.getOrNull(currentIndex)?.enabled?.value = true
             println("[SequenceQueriesFactory] Sequence enabled, activating query: $currentIndex")
         } else {
             // Disable all queries when the sequence is disabled
             queries.forEach { it.enabled.value = false }
             println("[SequenceQueriesFactory] Sequence disabled, all queries stopped")
         }
     }.launchIn(scope)
 
     // Observe all query states and combine them
     val queryStates = queries.map { it.state }
     combine(queryStates) { states ->
         val isAllFetched = states.all { it.isFetched }
         InfiniteQueryState(
             pages = states.flatMap { it.pages },
             isFetching = !isAllFetched,
             isRefetching = states.any { it.isRefetching },
             isFetched = isAllFetched,
         )
     }.onEach { newState ->
         combinedState.value = newState
 
         // Only proceed with query sequencing if enabled
         if (!enabledState.value) return@onEach
 
         // Check if current active query is fully fetched and move to next
         val currentIndex = activeQueryIndex.value
         val currentQuery = queries.getOrNull(currentIndex)
 
         if (currentQuery != null &&
             currentQuery.state.value.isFetched &&
             !currentQuery.state.value.hasNextPage() &&
             currentIndex < queries.size - 1
         ) {
 
             // Move to next query
             val nextIndex = currentIndex + 1
             activeQueryIndex.value = nextIndex
             queries[nextIndex].enabled.value = true
 
             println("[SequenceQueriesFactory] Moving to next query: $nextIndex")
         }
     }.launchIn(scope)
 
     val onRefetch: () -> Unit = {
         println("[SequenceQueriesFactory] Refetching - resetting to first query")
 
         // Reset to first query
         activeQueryIndex.value = 0
 
         // Disable all queries except the first one (and only if globally enabled)
         queries.forEachIndexed { index, query ->
             query.enabled.value = enabledState.value && index == 0
         }
 
         // Trigger refetch on the first query if enabled
         if (enabledState.value) {
             queries.firstOrNull()?.onRefetch?.invoke()
         }
     }
 
     val onFetchNextPage: () -> Unit = {
         val currentIndex = activeQueryIndex.value
         val currentQuery = queries.getOrNull(currentIndex)
 
         if (currentQuery != null) {
             println("[SequenceQueriesFactory] Fetching next page for query: $currentIndex")
             currentQuery.onFetchNextPage.invoke()
         }
     }
 
     return InfiniteQuery(
         state = combinedState,
         onRefetch = onRefetch,
         onFetchNextPage = onFetchNextPage,
         enabled = enabledState
     )
 }
