 @Composable
 fun VoucherDetailScreen(
     viewModel: VoucherDetailVM,
     statusBarHeightPx: Int? = null,
 ) {
     val state by viewModel.state.collectAsState()
 
     var footerButtonProps by remember { mutableStateOf<FooterButtonProps?>(null) }
 
     LaunchedEffect(state.voucherData) {
         footerButtonProps = viewModel.getFooterButtonProps()
     }
 
     val upcomingTime = viewModel.getUpcomingUseTime(state.voucherData?.startDate).let {
         if (it?.first != null) {
             stringResource(it.first, it.second.toString())
         } else {
             null
         }
     }
 
     PromotionComposeContainer(
         statusBarHeightPx = statusBarHeightPx,
         context = MiniAppContext(
             appId = "vn.momo.payment",
             appCode = "instant_voucher_detail",
             appIcon = Images.ic_momo,
             appName = mapOf(
                 "vi" to "Chi tiết quà tặng",
                 "en" to "Voucher detail",
             ),
         )
     ) {
         Screen(
             title = stringResource(Res.string.voucherDetailScreenTitle),
             goBack = { viewModel.screenGoBack() },
             headerType = HeaderType.EXTENDED,
             scrollable = false,
             content = {
                 if (state.isLoading && state.voucherData == null) {
                     Box(
                         modifier = Modifier.fillMaxSize(), contentAlignment = Alignment.Center
                     ) {
 
                     }
                 } else if (state.voucherData == null) {
                     Box(
                         modifier = Modifier.fillMaxSize(), contentAlignment = Alignment.Center
                     ) {
                         EmptyState(
                             title = stringResource(Res.string.voucherDetailNotFound),
                             description = "",
                             modifier = Modifier.padding(Spacing.M)
                         )
                     }
                 } else {
                     InstantVoucherDetail(
                         voucher = state.voucherData!!,
                         footerButtonProps = footerButtonProps,
                         onSupportCenterClick = { viewModel.goToSupportCenter() },
                         upcomingTime = upcomingTime,
                     )
                 }
             },
         )
     }
 }
