 class TransactionResultBtsVM(
     private val transactionResultVM: TransactionResultRevampVM,
     override val navigator: INavigator,
     override val logger: ILogger,
 ) : ViewModel(), IPaymentVM<TransactionResultAction>, CommonMixin, ResultXWidgetMixin {
     lateinit var props: TransactionInfoBottomSheetProps
     override val logName: String = "TR_TransactionResultBtsVM"
     override val widgetEvents: Map<TransactionResultWidget, GrafanaMetric> = mapOf(
         TransactionResultWidget.Service to TRUserEvent.TRBottomSheetServiceWidget.event,
     )
     override var params: TransactionResultParams = TransactionResultParams()
 
     override val grafana: IGrafanaTracer = getInjected<IGrafanaTracer>()
     override val scope = CoroutineScope(Dispatchers.IO)
     override val buffer = mutableMapOf<String, String?>()
     override val jobs = mutableMapOf<String, Job>()
     override val firebase: IFirebaseTracer = getInjected<IFirebaseTracer>()
     override var durationBuffer = mutableMapOf<String, Long?>()
     override var grafanaFlow: String? = "payx"
     override var firebaseFlow: String? = "PayX"
     override var firebaseAppId: String? = "vn.momo.payx"
 
     val state = PState(
         viewModelScope, TransactionResultBtsState(
             ticketList = transactionResultVM.state.value.ticketList
         )
     )
 
     private var isTrackingDone = false
 
     private fun trackInfoBottomSheetDisplayed() {
         if (isTrackingDone) {
             return
         }
         firebaseTrack(TransactionInfoBottomSheetDisplayed())
         isTrackingDone = true
     }
 
     private fun trackingScreenDisplayed() {
         firebaseTrack(TransactionResultScreenDisplayed())
     }
 
     fun onCreate(params: Map<String, Any?>) {
         props =
             ModelConvertUtils.decode(TransactionInfoBottomSheetProps.serializer(), params.tryGetMapNullable("props") ?: emptyMap()) ?: props
         this.params = props.params
         viewModelScope.launch {
             state.state.collect {
                 logInfo("[onCreate] state ticketList", it.ticketList)
             }
         }
         updateTicketList()
         grafanaStart(TRUserEvent.TRBottomSheet.event)
         startGrafanaTraceWidget()
         trackingScreenDisplayed()
         grafanaSuccess(TRUserEvent.TRBottomSheet.event)
     }
 
     override fun onCleared() {
         super.onCleared()
         state.reset()
     }
 
     override fun dispatch(action: TransactionResultAction) {
         when (action) {
             is TransactionResultAction.OnWidgetLoadSuccess -> {
                 onWidgetLoadSuccess(action.widget, transactionResultVM.state.value.ticketList.getOrNull(0)?.transaction)
             }
 
             is TransactionResultAction.OnWidgetLoadFailed -> {
                 onWidgetLoadFailed(action.widget)
             }
 
             is TransactionResultAction.OnWidgetLayoutChanged -> {
                 onWidgetLayoutChanged(action.widget, action.width, action.height, transactionResultVM.state.value.ticketList.getOrNull(0)?.transaction)
             }
 
             else -> {}
         }
     }
 
     private fun updateTicketList() {
         if (state.value.ticketList.isEmpty()) {
             // cover crash if don't have state ticketList
             return
         }
         val currentTicketList = state.value.ticketList
         val ticketIndex = props.ticketIndex
         val ticket = currentTicketList.getOrNull(ticketIndex)
         val transaction = ticket?.transaction
         transaction?.let {
             viewModelScope.launch {
                 val transactionDetailKey = withContext(Dispatchers.IO) {
                     getString(Res.string.transactionDetail)
                 }
                 val transactionId = transaction.transactionId
                 val updatedTextRows = ticket.textRows?.map { textRow ->
                     if (textRow.key == transactionDetailKey) {
                         val tidColor = transactionResultVM.getTidColor(
                             ticket.disableDetailMode, it
                         )
                         textRow.copy(
                             value = transactionId,
                             valueColor = tidColor,
                             action = TextRowAction.GO_TO_DETAIL,
                             rightIconTintColor = tidColor,
                         )
                     } else {
                         textRow
                     }
                 }
                 val updatedTicketList = currentTicketList.mapIndexed { index, ticket ->
                     if (index == ticketIndex) {
                         val updatedWidget = ticket.widget.copy(
                             params = ticket.widget.params.copy(
                                 type = WidgetType.MAIN.type
                             )
                         )
                         ticket.copy(textRows = updatedTextRows, widget = updatedWidget)
                     } else {
                         ticket
                     }
                 }
                 state.update { it.copy(ticketList = updatedTicketList) }
             }
             trackInfoBottomSheetDisplayed()
         }
         logInfo("updateTicketList", ticket)
     }
 
     override fun onWidgetLoadSuccess(widget: TransactionResultWidget, transactionItem: TransactionItem?) {
         super.onWidgetLoadSuccess(widget, transactionItem)
         logInfo("onWidgetLoadSuccess", widget.id)
         if (widget == TransactionResultWidget.Service && !state._state.value.widgetServiceLoaded) {
             state.update {
                 it.copy(
                     widgetServiceLoaded = true
                 )
             }
         }
     }
 
     override fun onWidgetLoadFailed(widget: TransactionResultWidget) {
         super.onWidgetLoadFailed(widget)
         logInfo("onWidgetLoadFailed", widget.id)
         if (widget == TransactionResultWidget.Service && !state._state.value.widgetServiceLoaded) {
             state.update {
                 it.copy(
                     widgetServiceLoaded = true
                 )
             }
         }
     }
 
     override fun onWidgetLayoutChanged(widget: TransactionResultWidget, width: Int?, height: Int?, transactionItem: TransactionItem?) {
         super.onWidgetLayoutChanged(widget, width, height, transactionItem)
         logInfo("onWidgetLayoutChanged", "${widget.id}-${width}-${height}")
     }
