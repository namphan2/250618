 interface IPayXRealmService {

 }
 
 class PayXRealmService(private val api: IComposeApi) : IPayXRealmService {
     override fun realmQuery(
         schema: String,
         query: List<String>,
         sort: String,
         limit: Int
     ): Flow<List<JsonObject?>> {
         return api.requestWithFlow(
             APIs.realmQuery.name,
             listOf(schema, query, sort, limit)

         }
     }
