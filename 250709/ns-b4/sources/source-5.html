 @Composable
 fun HtmlText(
     htmlString: String,
     color: Color? = null,
     style: TextStyle = Typography.bodyDefaultRegular,
     textAlign: TextAlign? = TextAlign.Start,
     fontFamily: String = "SFProText",
     modifier: Modifier = Modifier,
 ) {
     val annotatedString = remember(htmlString) {
         htmlToAnnotatedString(
             html = htmlString,
             compactMode = true,
         )
     }
 
     androidx.compose.material3.Text(
         modifier = modifier.fillMaxWidth(),
         color = color ?: AppTheme.current.colors.text.default,
         text = annotatedString,
         style = style,
         fontSize = scaleSize(style.fontSize.value).sp,
         fontFamily = getFontFamily(fontFamily, style.fontWeight),
         textAlign = textAlign,
     )
 }
 
 fun String.adjustHtmlParagraphSpacing(): String {
     return this
 //        .replace("<p><br /></p>", "<br />")
         .replace(Regex("<p>(.*?)</p>"), "<p>$1</p><br />")
 }
