 package vn.momo.compose.payment.promotion.presentation.viewModel.reducers
 
 import vn.momo.compose.payment.base.presentation.viewModel.reducer.Reducer
 import vn.momo.compose.payment.promotion.domain.entities.CheckPromoCodeResult
 import vn.momo.compose.payment.promotion.domain.entities.IVoucherBase
 import vn.momo.core.features.annotation.Immutable
 
 class PromoCodeReducer :
     Reducer<PromoCodeReducer.State, PromoCodeReducer.Action, PromoCodeReducer.Effect> {
     @Immutable
     data class State(
         val promoCode: String = "",
         val isCheckingPromoCode: Boolean = false,
         val checkPromoCodeResult: CheckPromoCodeResult? = null,
         val checkPromoCodeError: String? = null,
         val generatedPromoVouchers: List<IVoucherBase> = emptyList(),
         val isPromoCodeApplied: Boolean = false,
         val promoCodeApplied: String = "",
         val promoCodeNotice: Pair<String, String>? = null,
         val isBottomSheetVisible: Boolean = false,
     ) : Reducer.State
 
     @Immutable
     sealed class Action : Reducer.Action {
         data class SetPromoCode(
             val promoCode: String,
         ) : Action()
 
         object ClearError : Action()
 
         object ToggleBottomSheet : Action()
 
         data class SetState(
             val promoCode: String,
             val isCheckingPromoCode: Boolean,
             val checkPromoCodeResult: CheckPromoCodeResult?,
             val checkPromoCodeError: String?,
             val generatedPromoVouchers: List<IVoucherBase> = emptyList(),
             val isPromoCodeApplied: Boolean = false,
             val promoCodeNotice: Pair<String, String>? = null,
             val isBottomSheetVisible: Boolean = false,
         ) : Action()
     }
 
     @Immutable
     class Effect() : Reducer.Effect
 
     override fun reduce(prev: State, action: Action): Pair<State, Effect?> {
         return when (action) {
             is Action.SetPromoCode -> prev.copy(
                 promoCode = action.promoCode,
             ) to null
 
             is Action.ClearError -> prev.copy(
                 promoCode = "",
                 checkPromoCodeError = null,
                 isPromoCodeApplied = false,
                 promoCodeNotice = null,
             ) to null
 
             is Action.ToggleBottomSheet -> prev.copy(
                 isBottomSheetVisible = !prev.isBottomSheetVisible,
             ) to null
 
             is Action.SetState -> prev.copy(
                 promoCode = action.promoCode,
                 isCheckingPromoCode = action.isCheckingPromoCode,
                 checkPromoCodeResult = action.checkPromoCodeResult,
                 checkPromoCodeError = action.checkPromoCodeError,
                 generatedPromoVouchers = action.generatedPromoVouchers,
                 isPromoCodeApplied = action.isPromoCodeApplied,
                 promoCodeApplied = if (action.isPromoCodeApplied) action.promoCode else "",
                 promoCodeNotice = action.promoCodeNotice,
                 isBottomSheetVisible = action.isBottomSheetVisible,
             ) to null
         }
     }
 }
