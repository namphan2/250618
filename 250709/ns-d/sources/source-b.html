 package vn.momo.compose.payment.base.core.composeApi
 
 import kotlinx.coroutines.CoroutineScope
 import kotlinx.coroutines.Dispatchers
 import kotlinx.coroutines.IO
 import kotlinx.coroutines.launch
 import vn.momo.api.APIs
 import vn.momo.api.IComposeApi
 import vn.momo.api.requestOneShot
 
 
 interface ILogger {
     fun i(message: String)
     fun w(message: String)
     fun e(message: String)
 }
 
 class PayXLogger(private val api: IComposeApi) : ILogger {
     override fun i(message: String) {
         log("INFO", message)
     }
 
     override fun w(message: String) {
         log("WARNING", message)
     }
 
     override fun e(message: String) {
         log("ERROR", message)
     }
 
     private fun log(level: String, message: String) {
         CoroutineScope(Dispatchers.IO).launch {
             api.requestOneShot(APIs.logFile.name, listOf(message, level))
         }
     }
 
 }
