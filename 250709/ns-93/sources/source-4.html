 fun Modifier.shadowBill(
     color: Color = Color(0xFFE27EB5).copy(alpha = 0.15f),
     blurRadius: Float = 12f,
     offsetY: Dp = 6.dp,
     offsetX: Dp = (-1).dp,
     spread: Dp = 3.dp,
     cornerRadius: Dp = 16.dp,
 ): Modifier = drawBehind {
     drawIntoCanvas { canvas ->
         val paint = Paint()
         val frameworkPaint = paint.asFrameworkPaint()
 
         if (blurRadius > 0f) {
             frameworkPaint.setMaskFilter(blurRadius)
         }
 
         frameworkPaint.color = color.toArgb()
 
         val spreadPx = spread.toPx()
         val offsetXPx = offsetX.toPx()
         val offsetYPx = offsetY.toPx()
 
         val shadowTop = 12f + spreadPx + blurRadius
         val shadowBottom = size.height + spreadPx + blurRadius
 
         val left = -spreadPx + offsetXPx
         val right = size.width + spreadPx + offsetXPx
 
         canvas.drawRoundRect(
             left = left,
             top = shadowTop,
             right = right,
             bottom = shadowBottom,
             radiusX = cornerRadius.toPx(),
             radiusY = cornerRadius.toPx(),
             paint = paint
         )
     }
 }
