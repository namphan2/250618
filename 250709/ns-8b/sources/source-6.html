 @Immutable
 object PayXSofBottomSheet : SofBottomSheetView {
     @Composable
     override fun AddSofCTA(iconBankList: List<String>, onPress: () -> Unit) {
         ActionCard(
             title = stringResource(Res.string.addBank),
             description = stringResource(Res.string.addBankDescription),
             iconLeft = {
                 Image(
                     source = Images.ic_add_bank,
                     modifier = Modifier.size(32.dp)
                 )
             },
             iconRight = {
                 Row(
                     horizontalArrangement = Arrangement.End,
                     verticalAlignment = Alignment.CenterVertically
                 ) {
                     SofIconList(
                         iconBankList,
                         maxItem = 2,
                         extraItem = "${iconBankList.size - 2}"
                     )
                     Icon(
                         source = "arrow_chevron_right_small",
                         size = 24.dp,
                         color = Colors.black_17
                     )
                 }
             },
             cardStyle = ActionCardStyle().copy(border = null),
             modifier = Modifier.noFeedbackClickable {
                 FirebaseTrace.track(
                    ChangeMoneySourceFromPopup(
                         listValue = "add_sof", listValueDetail = "0"
                     )
                 )
                 onPress()
             })
     }
 
     @Composable
     override fun NoSofMessage(isEnoughMoMoBalance: Boolean) {
         if (!isEnoughMoMoBalance) {
             MessageInfo(
                 message = stringResource(Res.string.noSofAvailable),
                 icon = Images.ic_circle_alert,
                 background = Colors.red_10
             )
         }
     }
 
     @Composable
     override fun Footer(
         isNoSofAvailable: Boolean,
         addSofBlock: @Composable () -> Unit
     ) {
         if (!isNoSofAvailable) {
             addSofBlock()
         }
     }
 
     @Composable
     override fun HeaderWhenNoSof(addSofBlock: @Composable () -> Unit) {
         addSofBlock()
     }
 }
