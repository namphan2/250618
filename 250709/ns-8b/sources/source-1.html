 @Composable
 fun AppToAppPaymentBts(
     viewModel: PayXAppToAppVM,
     onClose: () -> Unit
 ) {
     val pagerState = rememberPagerState(0, pageCount = { 2 })
     val uiScope = rememberCoroutineScope()
     val state = viewModel.state.collectAsState()
     val info = state.value.info
 
     val backgroundColor = animateColorAsState(
         targetValue = if (pagerState.currentPage == 0) AppTheme.current.colors.background.surface else AppTheme.current.colors.background.default,
     )
 
     LaunchedEffect(Unit) {
         viewModel.effect.filter { it == VAReducer.Effect.Close }
             .onEach {
                 onClose()
             }
             .take(1)
             .launchIn(this)
         viewModel.effect.filter { it == VAReducer.Effect.OpenBankList }
             .onEach {
                 pagerState.animateScrollToPage(
                     1,
                     animationSpec = linearTween()
                 )
             }
             .launchIn(this)
     }
     PayXBottomSheet(
         customLayout = true,
         modifier = Modifier.background(backgroundColor.value)
     ) {
         HorizontalPager(
             state = pagerState,
             userScrollEnabled = false,
             modifier = Modifier.animateContentSize(
                 linearTween(
                     durationMillis = 300,
                     delayMillis = 100
                 )
             )
         ) { page ->
             when (page) {
                 0 -> BankTransferInfoView(
                     info = info.transaction,
                     remainingTime = state.value.formatedCount,
                     onClose = onClose,
                     onCopyBankInfo = viewModel::toastCopy,
                     onOpenBankCopy = { viewModel.openBank(VAReducer.OpenBankType.Copy) },
                     onDownloadQr = viewModel::onDownloadQr,
                     canOpenBank = info.userBank.isNotEmpty()
                 )
 
                 1 -> SelectBankView(
                     type = state.value.type,
                     bankList = info.userBank,
                     onOpenBank = viewModel::openBankUrl,
                     onClose = {
                         uiScope.launch {
                             pagerState.animateScrollToPage(
                                 0,
                                 animationSpec = linearTween()
                             )
                         }
                     })
             }
         }
     }
 }
