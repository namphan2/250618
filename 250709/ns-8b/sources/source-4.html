 const val PIN_LENGTH = 6
 const val TIMER = 60
 
 @Composable
 fun OTPBottomSheetView(
     viewModel: OTPAuthenticationVM,
 ) {
     val screenWidth = getScreenDimensions().screenWidth
     val inputWidth = screenWidth * 0.6
     val otp = viewModel.otpState.state.collectAsState()
     val errorMessage = viewModel.errorMessageState.state.collectAsState()
     var timer by remember { mutableStateOf(TIMER) }
 
     LaunchedEffect(timer > 0) {
         while (timer > 0) {
             delay(1000)
             timer--
         }
     }
 
     fun onPressed(item: String) {
         if (item == "backspace") {
             viewModel.onBackSpace()
         } else {
             viewModel.onItemPressed(item)
         }
     }
 
     Column(
         modifier = Modifier
             .fillMaxWidth(),
         horizontalAlignment = Alignment.CenterHorizontally
     ) {
         LazyRow(
             modifier = Modifier
                 .padding(vertical = Spacing12)
                 .width(Spacing(inputWidth.toInt()))
                 .height(Spacing60)
                 .clip(shape = RoundedCornerShape(Spacing30))
                 .border(
                     width = Spacing1,
                     color = Colors.CustomKeyboardBorderColor,
                     shape = RoundedCornerShape(Spacing30)
                 )
                 .background(Colors.White)
                 .semantics { contentDescription = "OTPTextInput" },
             horizontalArrangement = Arrangement.Center,
             verticalAlignment = Alignment.CenterVertically
         ) {
             items(PIN_LENGTH) {
                 Column(
                     modifier = Modifier
                         .padding(horizontal = Spacing4),
                     verticalArrangement = Arrangement.Center,
                     horizontalAlignment = Alignment.CenterHorizontally
                 ) {
                     val text = otp.value.toList().getOrNull(it)
                     Text(
                         text = text?.toString() ?: "",
                         style = Typography.bodyDefaultRegular,
                     )
 
                     Column(
                         modifier = Modifier
                             .background(Colors.TextUnderlineBorderbottom)
                             .width(Spacing16)
                             .height(Spacing2)
                     ) {}
                 }
             }
             items(1) {
                 Image(
                     source = Images.ic_circle_close,
                     modifier = Modifier.size(24.dp).noFeedbackClickable {
                         viewModel.onClearOTP()
                     }
                 )
             }
         }
 
         Text(
             text = errorMessage.value,
             style = Typography.bodyDefaultRegular,
             color = Colors.Error,
             modifier = Modifier
                 .padding(horizontal = Spacing10, vertical = Spacing20),
             textAlign = TextAlign.Center,
             maxLines = 3
         )
 
         if (timer > 0) {
             Text(
                 text = stringResource(Res.string.otpDesc, timer),
                 style = Typography.bodyDefaultRegular,
                 modifier = Modifier
                     .padding(horizontal = Spacing40),
                 color = Colors.CustomKeyboardcenterMessage,
                 textAlign = TextAlign.Center
             )
         }
 
         Text(
             text = stringResource(Res.string.resend),
             style = Typography.bodyDefaultRegular,
             color = if (timer <= 0) Colors.Blue04 else Colors.Disabled,
             modifier = Modifier
                 .noFeedbackClickable {
                     if (timer <= 0) {
                         timer = TIMER
                         viewModel.onResendOTP()
                     }
                 }
                 .padding(Spacing12)
         )
 
         NumberKeyboardOTP(onPressed = { onPressed(it) }, screenWidth = Spacing(screenWidth))
     }
 }
 
 @Composable
 fun NumberKeyboardOTP(
     onPressed: (String) -> Unit,
     screenWidth: Dp
 ) {
     val keyWidth = (screenWidth - 2.dp) / 3
     val keyHeight = keyWidth / 2
 
     LazyVerticalGrid(
         modifier = Modifier.fillMaxWidth(),
         columns = GridCells.Fixed(3)
     ) {
         itemsIndexed(PasswordBottomSheetKeys) { _, item ->
             Row(
                 horizontalArrangement = Arrangement.Center,
                 verticalAlignment = Alignment.CenterVertically,
                 modifier = Modifier
                     .height(keyHeight)
                     .background(
                         if (item != PasswordBottomSheetKey.Backspace && item != PasswordBottomSheetKey.None) Colors.White
                         else Colors.CustomKeyboardBackgroundColor
                     )
                     .border(
                         width = Spacing1,
                         color = Colors.CustomKeyboardBorderColor,
                     )
                     .noFeedbackClickable { onPressed(item.value) }
                     .semantics { contentDescription = "key_${item.value}" }
             ) {
                 if (item == PasswordBottomSheetKey.Backspace) {
                     Icon(
                         source = "ic_backspace_24",
                         size = 20.dp,
                         color = Colors.White
                     )
                 } else if (item != PasswordBottomSheetKey.None) {
                     Text(
                         text = item.value,
                         style = Typography.headlineDefaultBold,
                     )
                 }
             }
         }
     }
 
 
 }
