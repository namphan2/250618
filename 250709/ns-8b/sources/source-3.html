 @Composable
 fun CustomContentBottomSheet(
     data: PayXBottomSheet,
     viewModel: CustomContentBottomSheetVM
 ) {
     val screenHeight = getScreenDimensions().screenHeight
 
     fun getBackgroundColor(index: Int, button: PayXButtonBottomSheet): Color {
         if (index == 0) {
             return button.backgroundColor?.toColor() ?: Colors.pink_03
         }
         return button.backgroundColor?.toColor() ?: Colors.black_01
     }
 
     fun getTitleColor(index: Int, button: PayXButtonBottomSheet): Color {
         if (index == 0) {
             return button.color?.toColor() ?: Colors.black_01
         }
         return button.color?.toColor() ?: Colors.pink_03
     }
 
     Column(
         modifier = Modifier.fillMaxWidth()
             .heightIn((screenHeight * 0.5).dp, (screenHeight * 0.7).dp).animateContentSize()
             .clip(shape = RoundedCornerShape(topStart = Spacing.L, topEnd = Spacing.L))
     ) {
         Column(
             modifier = Modifier.weight(1f)
                 .clip(shape = RoundedCornerShape(Radius.M))
                 .background(Colors.black_01)
                 .verticalScroll(state = rememberScrollState()),
             verticalArrangement = Arrangement.spacedBy(Spacing.M)
         ) {
             data.image?.let {
                 Image(
                     source = it,
                     modifier = Modifier.fillMaxWidth().aspectRatio(2f),
                     options = Options(contentScale = ContentScale.FillBounds)
                 )
             }
             Text(
                 text = data.content ?: "",
                 style = Typography.bodyDefaultRegular,
                 color = Colors.black_17,
                 modifier = Modifier.padding(Spacing.M)
             )
         }
         Column(
             modifier = Modifier.fillMaxWidth().background(Colors.black_01)
                 .padding(
                     top = Spacing.S,
                     bottom = Spacing.XL
                 )
         ) {
             data.buttons?.mapIndexed { index, payXButtonBottomSheet ->
                 val background = getBackgroundColor(index, payXButtonBottomSheet)
                 val titleColor = getTitleColor(index, payXButtonBottomSheet)
                 val type = if (index == 0) ButtonType.Primary else ButtonType.Outlined
 
                 MMButton(
                     title = payXButtonBottomSheet.title ?: "",
                     colors = ButtonDefaults.buttonColors(
                         background
                     ),
                     textColor = titleColor,
                     type = type,
                     onClick = {
                         viewModel.onClickButton(payXButtonBottomSheet, index)
                     }
                 )
             }
         }
     }
 }
