 @Composable
 fun PayXBottomSheet(
     modifier: Modifier = Modifier,
     customLayout: Boolean = false,
     @FloatRange(
         from = 0.0,
         to = 1.0
     ) minFraction: Float = DefaultPayXBottomSheet.MIN_HEIGHT_FRACTION,
     @FloatRange(
         from = 0.0,
         to = 1.0
     ) maxFraction: Float = DefaultPayXBottomSheet.MAX_HEIGHT_FRACTION,
     backgroundColor: Color = AppTheme.current.colors.background.default,
     header: (@Composable () -> Unit)? = null,
     content: @Composable ColumnScope.() -> Unit,
 ) {
     val screenHeight = getScreenDimensions().screenHeight
 
     Column(
         modifier = Modifier.fillMaxWidth()
             .heightIn(
                 if (customLayout) Dp.Unspecified else (minFraction * screenHeight).dp,
                 if (customLayout) Dp.Unspecified else (maxFraction * screenHeight).dp
             )
             .conditional(!customLayout) { animateContentSize(linearTween()) }
             .clip(shape = RoundedCornerShape(topStart = Spacing.L, topEnd = Spacing.L))
             .background(backgroundColor)
             .then(modifier),
         horizontalAlignment = Alignment.CenterHorizontally
     ) {
         Box(
             modifier = Modifier.padding(top = Spacing.S, bottom = Spacing.XS)
                 .width(40.dp)
                 .height(4.dp)
                 .background(Colors.black_06, RoundedCornerShape(28.dp))
         )
         Column(
             modifier = Modifier
                 .fillMaxWidth()
                 .padding(horizontal = Spacing.M, vertical = Spacing.L)
         ) {
             header?.invoke()
             header?.let {
                 Spacer(modifier = Modifier.height(16.dp))
             }
             content()
         }
     }
 }
 
 @Stable
 object DefaultPayXBottomSheet {
     const val MIN_HEIGHT_FRACTION = 0.5f
     const val MAX_HEIGHT_FRACTION = 0.8f
 }
