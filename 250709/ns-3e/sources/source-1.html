 @Composable
 fun CartItem(
     data: PayXCartItem,
     onPress: ((cta: PayXCta?) -> Unit)? = null
 ) {
     Row(
         modifier = Modifier
             .fillMaxWidth()
             .then(
                 data.backgroundColor?.takeIf { it.isNotEmpty() }?.let {
                     Modifier
                         .padding(vertical = Spacing.XS)
                         .background(
                             Colors.fromHex(it),
                             shape = RoundedCornerShape(Spacing.M)
                         )
                         .padding(Spacing.S)
                 } ?: Modifier.padding(vertical = Spacing.S)
             ),
         horizontalArrangement = Arrangement.SpaceBetween,
         verticalAlignment = Alignment.CenterVertically
     ) {
         Row(
             verticalAlignment = Alignment.CenterVertically
         ) {
             data.logo?.let {
                 Image(
                     source = it.uri ?: "",
                     modifier = Modifier
                         .padding(end = Spacing.M)
                         .width(it.size?.dp ?: 40.dp)
                         .height(it.size?.dp ?: 40.dp)
                 )
             }
             Column {
                 data.title?.let {
                     Label(it, onPress)
                 }
                 data.description?.let {
                     Label(it, onPress)
                 }
             }
         }
         data.value?.let {
             Label(it, onPress)
         }
     }
 }
