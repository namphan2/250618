 package vn.momo.compose.payment.base.presentation.uiKit.component
 
 import androidx.compose.runtime.Composable
 import androidx.compose.ui.Modifier
 import androidx.compose.ui.text.style.TextOverflow
 import vn.momo.compose.payment.base.domain.entities.overlayData.PayXCta
 import vn.momo.compose.payment.base.domain.entities.uiKit.PayXText
 import vn.momo.compose.payment.payx.presentation.view.components.modifiers.noFeedbackClickable
 import vn.momo.kits.components.Text
 import vn.momo.kits.const.Typography
 
 @Composable
 fun Label(data: PayXText, onPress: ((cta: PayXCta?) -> Unit)? = null) {
     val maxLength = data.maxLength
     val displayText = if (maxLength != null && data.text?.length!! > maxLength) {
         data.text.take(maxLength) + "..."
     } else {
         data.text
     }
     Text(
         text = displayText ?: "",
         color = data.color,
         style = data.style ?: Typography.bodyDefaultRegular,
         overflow = TextOverflow.Ellipsis,
         maxLines = data.numberOfLine ?: Int.MAX_VALUE,
         modifier = Modifier.noFeedbackClickable(enabled = data.action?.refId != null) {
             onPress?.invoke(data.action)
         }
     )
 }
