 @Composable
 fun PaymentQuickPayBottomSheet(
     viewModel: PaymentQuickPayVM,
     closeSheet: () -> Unit,
 ) {
     val slice = viewModel.paymentQuickPaySlice
     val state by slice.state.collectAsState()
     val scope = rememberCoroutineScope()
 
     fun requestClose() {
         closeSheet()
     }
 
     LaunchedEffect(Unit) {
         slice.effect.onEach {
             when (it) {
                 is PaymentQuickPayReducer.QuickPayEffect.DismissConfirmScreen -> {
                     requestClose()
                 }
 
                 else -> {}
             }
         }.launchIn(scope)
     }
 
     fun onClick() {
         slice.dispatch(PaymentQuickPayReducer.QuickPayAction.OnPressConfirm)
         requestClose()
     }
 
     val screenHeight = getScreenDimensions().screenHeight
     Column(
         modifier = Modifier.fillMaxWidth()
             .heightIn((screenHeight * 0.5).dp, (screenHeight * 0.75).dp).animateContentSize()
             .clip(shape = RoundedCornerShape(topStart = 16.dp, topEnd = 16.dp))
             .background(Color(0xfff2f2f6)).padding(horizontal = 12.dp, vertical = 16.dp),
     ) {
         BottomSheetHeader(
             title = stringResource(Res.string.quickpay_header_title),
             iconRight = AsyncIconProp(Images.ic_close_fat, size = 14.dp, onPress = {
                 slice.dispatch(PaymentQuickPayReducer.QuickPayAction.RequestClose)
                 requestClose()
             })
         )
         Spacer(modifier = Modifier.height(16.dp))
         Column(
             modifier = Modifier.fillMaxWidth().verticalScroll(state = rememberScrollState()),
             verticalArrangement = Arrangement.spacedBy(8.dp)
         ) {
             Text(
                 text = state.statusDesc
             )
         }
         Column(
             modifier = Modifier.fillMaxWidth(),
             horizontalAlignment = Alignment.CenterHorizontally
         ) {
             Spacer(modifier = Modifier.height(8.dp))
             Button(
                 title = stringResource(Res.string.confirm),
                 iconLeft = "security_lock",
                 onClick = {
                     scope.launch {
                         onClick()
                     }
                 },
                 type = if (state.enableConfirm) ButtonType.PRIMARY else ButtonType.DISABLED,
                 modifier = Modifier.semantics { testTag = "btn_primary_confirm" }
             )
         }
     }
 }
