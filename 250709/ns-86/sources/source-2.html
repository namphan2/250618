 @OptIn(ExperimentalAnimationApi::class)
 @Composable
 fun MissionRewardView(
     modifier: Modifier = Modifier,
     onShow: () -> Unit = {},
     missionRewardSummary: PayXResult<List<PayXComponent>, Throwable>
 ) {
     val borderColor = AppTheme.current.colors.border.default
     val isShow = missionRewardSummary is PayXResult.Loading
             || (missionRewardSummary is PayXResult.Success
             && missionRewardSummary.data.isNotEmpty())
 
     LaunchedEffect(isShow) {
         if (isShow) {
             onShow()
         }
     }
 
     PayXAnimatedVisibility(
         visible = isShow
     ) {
         Crossfade(
             modifier = Modifier.fillMaxWidth()
                 .scaledPadding(top = 12f)
                 .drawBehind {
                     drawLine(
                         color = borderColor,
                         start = Offset(0f, 0f),
                         end = Offset(size.width, 0f),
                         strokeWidth = 1f
                     )
                 }
                 .scaledPadding(top = 12f)
                 .then(modifier),
             targetState = missionRewardSummary,
             animationSpec = linearTween(durationMillis = 300),
             label = "MissionRewardSlideAnimation"
         ) { summary ->
             when (summary) {
                 is PayXResult.Loading -> {
                     Box(
                         Modifier.width(200.dp).height(24.dp)
                             .clip(shape = RoundedCornerShape(Radius.XS))
                             .skeletonEffect()
                     )
                 }
 
                 is PayXResult.Success -> {
                     Row(
                         verticalAlignment = Alignment.CenterVertically,
                         horizontalArrangement = scaledSpacedBy(4f)
                     ) {
                         createPayXComponent(summary.data)
                     }
                 }
 
                 is PayXResult.Fail -> {}
             }
         }
     }
 }
