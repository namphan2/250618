 @Composable
 fun ScallopedTop(
     modifier: Modifier = Modifier,
     backgroundColor: Color = Colors.black_01,
     scallopRadiusDp: Dp = 6.dp,
     scallopSpacingDp: Dp = Spacing.S,
     horizontalPaddingDp: Dp = Spacing.L,
     solidTopCornerRadiusDp: Dp = Spacing.XXS,
     height: Dp = 8.dp
 ) {
     val density = LocalDensity.current
     val shape = with(density) {
         ScallopedTopShape(
             scallopRadius = scallopRadiusDp.toPx(),
             scallopSpacing = scallopSpacingDp.toPx(),
             horizontalPadding = horizontalPaddingDp.toPx(),
             solidTopCornerRadius = solidTopCornerRadiusDp.toPx()
         )
     }
 
     Box(
         modifier = modifier
             .height(height)
             .background(color = backgroundColor, shape = shape)
     )
 }
 
 class ScallopedTopShape(
     private val scallopRadius: Float,
     private val scallopSpacing: Float,
     private val horizontalPadding: Float,
     private val solidTopCornerRadius: Float,
 ) : Shape {
     override fun createOutline(
         size: Size,
         layoutDirection: LayoutDirection,
         density: Density
     ): Outline {
         val path = Path()
 
         val scallopDiameter = scallopRadius * 2
         val fullScallopWidth = scallopDiameter + scallopSpacing
         val availableWidth = size.width - horizontalPadding * 2
         val count = (availableWidth / fullScallopWidth).toInt() + 1
         val startX = horizontalPadding
 
         // Move to bottom-left corner
         path.moveTo(0f, size.height)
 
         // Left side up
         path.lineTo(0f, solidTopCornerRadius)
         path.arcTo(
             rect = Rect(0f, 0f, solidTopCornerRadius * 2, solidTopCornerRadius * 2),
             startAngleDegrees = 180f,
             sweepAngleDegrees = 90f,
             forceMoveTo = false
         )
 
         // Draw scallops along the top
         for (i in 0 until count) {
             val x = startX + i * fullScallopWidth
             val centerX = x + scallopRadius
             val centerY = 0f
 
             path.arcTo(
                 rect = Rect(
                     centerX - scallopRadius,
                     centerY - scallopRadius,
                     centerX + scallopRadius,
                     centerY + scallopRadius
                 ),
                 startAngleDegrees = 180f,
                 sweepAngleDegrees = -180f,
                 forceMoveTo = false
             )
         }
 
         // Top-right corner
         path.lineTo(size.width - solidTopCornerRadius, 0f)
         path.arcTo(
             rect = Rect(size.width - solidTopCornerRadius * 2, 0f, size.width, solidTopCornerRadius * 2),
             startAngleDegrees = 270f,
             sweepAngleDegrees = 90f,
             forceMoveTo = false
         )
 
         // Right side down
         path.lineTo(size.width, size.height)
 
         // Bottom edge
         path.lineTo(0f, size.height)
 
         path.close()
 
         return Outline.Generic(path)
     }
 }
