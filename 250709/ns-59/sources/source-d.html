 fun XSaleConfigDTO.toModel(): XSaleConfig {

         subTitle = this.subTitle,

 fun XSaleItemDTO?.toModel() = this?.let {
     this.action?.refParams?.takeIf { it.isNotEmpty() }?.let {
         XSaleItem(
             id = this.id ?: randomUUID(),
             title = this.title ?: "",
             subTitle = this.subTitle,
             subTitleColor = ColorUtil.fromHex(this.subTitleColor ?: "#34C759", Colors.green_03),
             badge = this.badge?.toModel(),
             icon = this.icon,
             leading = this.leading?.let {
                 XSaleLeading(title = it.title ?: "", subTitle = it.subTitle)
             },
             action = PayXAction.SendMessage(
                 payload = this.action.refParams
             )
         )
     }
 }
