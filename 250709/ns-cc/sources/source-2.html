 package vn.momo.compose.payment.singleBill.data.repository.confirm
 
 import vn.momo.compose.payment.payx.data.dataSource.dto.response.RemoteDialogDTO
 import vn.momo.compose.payment.singleBill.data.repository.transaction.SBErrorGenerator
 import vn.momo.compose.payment.singleBill.domain.entities.error.SingleBillError
 import vn.momo.compose.payment.singleBill.domain.entities.error.SingleBillErrorHelper
 import vn.momo.compose.utils.emptyJsonObject
 import vn.momo.core.modules.networking.http.ErrorCode
 
 class SBConfirmErrorGenerator(
     private val errorCode: Int?,
     private val errorDesc: String?,
     private val fraudData: RemoteDialogDTO?,
     private val tranId: String? = null
 ) : SBErrorGenerator {
     override suspend fun generateError(): SingleBillError? {
         val errorConfig = SingleBillErrorHelper.getSingleBillErrorCDN()
         val popupError =
             SingleBillError.CustomPopupError.create(
                 errorCode,
                 errorDesc,
                 fraudData,
                 errorConfig,
                 tranId
             )
         return when {
             errorCode == null || errorCode == 0 -> null
             errorCode == 50000 -> SingleBillError.UserCancel(
                 errorCode,
                 errorDesc,
                 emptyJsonObject()
             )
 
             ErrorCode.isSystemError(errorCode) -> SingleBillError.SystemError(
                 errorCode,
                 errorDesc,
                 emptyJsonObject()
             )
 
             popupError != null -> popupError
 
             else -> SingleBillErrorHelper.generateDefaultErrorPopup(errorCode, errorDesc)
         }
     }
 }
