 package vn.momo.compose.payment.payx.domain.entities.transactionresult
 
 import vn.momo.compose.resources.Res
 import vn.momo.compose.resources.paymentMoneySourceFee
 import vn.momo.compose.resources.paymentPromo
 import vn.momo.compose.resources.paymentServiceFee
 import vn.momo.compose.resources.paymentSubTotal
 import vn.momo.compose.resources.paymentTotalItem

 import org.jetbrains.compose.resources.getString
 import vn.momo.compose.utils.formatCurrency
 
 data class TransactionFeeItem(
     var totalAmount: Double = 0.0,
     var conversionFee: Double = 0.0,
     var totalFeeSoF: Double = 0.0,
     var subtotalAmount: Double = 0.0,
     var totalFee: Double = 0.0,
     var totalDiscount: Double = 0.0,
 ) {
     val isBlank: Boolean
         get() = listOf(
             totalAmount,
             conversionFee,
             totalFeeSoF,
             subtotalAmount,
             totalFee,
             totalDiscount
         ).all { it != 0.0 }
 
     constructor(purchaseId: PurchaseInfo?) : this(
         totalAmount = purchaseId?.totalAmount ?: 0.0,
         conversionFee = purchaseId?.conversionFee ?: 0.0,
         totalFeeSoF = purchaseId?.totalFeeSoF ?: 0.0,
         subtotalAmount = purchaseId?.subtotalAmount ?: 0.0,
         totalFee = purchaseId?.totalFee ?: 0.0,
         totalDiscount = purchaseId?.totalDiscount ?: 0.0
     )
 
     private suspend fun getLocalized(key: String): String {
         return when (key) {
             "totalAmount" -> getString(Res.string.paymentTotalItem)
             "subtotalAmount" -> getString(Res.string.paymentSubTotal)
             "totalFeeSoF" -> getString(Res.string.paymentMoneySourceFee)
             "conversionFee" -> getString(Res.string.paymentTransferFee)
             "totalFee" -> getString(Res.string.paymentServiceFee)
             "totalDiscount" -> getString(Res.string.paymentPromo)
             else -> ""
         }
     }
 
     private suspend fun getTextRowItem(key: String): TextRowItem? {
         val textRowItem = TextRowItem()
 
         textRowItem.key = getLocalized(key)
 
         when (key) {
             "totalAmount" -> textRowItem.value = totalAmount.formatCurrency()
             "subtotalAmount" -> textRowItem.value = subtotalAmount.formatCurrency()
             "totalFeeSoF" -> textRowItem.value = totalFeeSoF.formatCurrency()
             "conversionFee" -> textRowItem.value = conversionFee.formatCurrency()
             "totalFee" -> textRowItem.value = totalFee.formatCurrency()
             "totalDiscount" -> {
                 textRowItem.value = "-".plus(totalDiscount.formatCurrency())
                 textRowItem.valueColor = 0xFFf5222d
             }
 
             else -> textRowItem.value = totalAmount.formatCurrency()
         }
 
         if (textRowItem.value.isBlank()
             || textRowItem.value == "-"
             || textRowItem.value == 0.0.formatCurrency()
             || textRowItem.value == "-".plus(0.0.formatCurrency())
         ) {
             return null
         }
 
         return textRowItem
     }
 
 
     private suspend fun toListTextRowItem(keys: Array<String>?): List<TextRowItem> {
         return keys?.asList()?.mapNotNull {
             getTextRowItem(it)
         } ?: emptyList()
     }
 
     suspend fun getListFee(): List<TextRowItem> {
         return this.toListTextRowItem(arrayOf("totalFeeSoF", "conversionFee", "totalFee"))
     }
 
     suspend fun getListDiscount(): List<TextRowItem> {
         return this.toListTextRowItem(arrayOf("totalDiscount"))
     }
 
     suspend fun getTotalAmount(): TextRowItem {
         return TextRowItem(
             getLocalized("totalAmount"),
             totalAmount.formatCurrency(),
         )
     }
 
     suspend fun getSubtotalAmount(): TextRowItem {
         return TextRowItem(
             getLocalized("subtotalAmount"),
             subtotalAmount.formatCurrency(),
         )
     }
