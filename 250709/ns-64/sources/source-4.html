 class DefaultVARepository(
     private val cdnConfig: ICDNConfig,
     private val transactionStatusRepository: ITransactionStatusRepository,
     private val logger: ILogger
 ) : VirtualAccountRepository {
     override fun getBankLogo(bankCode: String): Flow<BankInfo?> {
         return cdnConfig.get(
             "https://static.momocdn.net/app/json/bank/all_bank_info.json",
             ListSerializer(BankInfo.serializer())
         ).map { listBankConfig ->
             val bankMap = listBankConfig?.associateBy { it.bankCode } ?: emptyMap()
             bankMap[bankCode]
         }
     }
 
     override fun listenToBankTransfer(
         purchaseId: String?,
         paymentType: String?
     ): Flow<TxnResponse?> {
         return merge(
             mqttTransactionResult(purchaseId),
             loopingApi(api = getTransactionResultStatusAPI(purchaseId, paymentType))
         )
             .filter { resp ->
                 resp?.momoMsg?.replyMsgs?.isNotEmpty() ?: false
             }
             .take(1)
     }
 
     private fun mqttTransactionResult(purchaseId: String?): Flow<TxnResponse?> {
         return transactionStatusRepository.listenerMQTT().filter { response ->
             val purchaseIdMqtt = response?.momoMsg?.getPurchaseOrder()
             purchaseId == purchaseIdMqtt
         }.onEach {
             purchaseId?.let { id ->
                 FirebaseTrace.track(
                     MqttResponded(
                         purchaseId = id
                     )
                 )
             }
             logger.i("[PayX] VA-observerTransactionStatusMQTT: $it")
         }
     }
 
     private fun loopingApi(api: Flow<TxnResponse?>) = flow {
         while (true) {
             emitAll(api)
             delay(5000)
         }
     }
 
     private fun getTransactionResultStatusAPI(
         purchaseId: String?,
         paymentType: String?
     ): Flow<TxnResponse?> {
         return transactionStatusRepository.getTransactionStatus(
             purchaseId, paymentType, 1
         ).onEach {
             logger.i("[PayX] VA-getTransactionResultStatusAPI: $it")
         }
     }
 }
