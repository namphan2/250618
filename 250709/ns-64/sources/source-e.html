 class DefaultNewSofHandler(
     private val tranType: Int?,
     private val realm: IRealmSofDataSource,
 ) : NewSofRepository {
     override fun onAddBank(data: AddBankResponseDto?): SOFItem? {
         if (data == null) return null
         if (data.isSelectNapas == true && data.sourceNapas != null) {
             val source = data.sourceNapas.toSofListDTO()
             return source.toModel(tranType)
         }
         val source = data.incomeSource?.toSofListDTO()
             ?: return null
 
         return source.toModel(tranType)
     }
 
     override fun onReceiveSof(data: SofFromJS?): SOFItem? {
         return data?.let { jsSof ->
             (jsSof.selectedSource ?: jsSof.source)?.toSofListDTO()?.toModel(tranType)
         }
     }
 
     override fun onAddInternationalCard(data: JsonObject): Flow<SOFItem?> {
         val cardId = data["card"]?.jsonObject?.get("cardId")?.jsonPrimitive?.content ?: run {
             return flow { emit(null) }
         }
         return realm.requestSync().map { list ->
             list.find { it.rowCardId == cardId }?.toModel(tranType)
         }
     }
 }
