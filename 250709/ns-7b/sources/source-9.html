 package vn.momo.compose.payment.payx.domain.useCases.confirm
 
 import kotlinx.coroutines.flow.Flow
 import kotlinx.coroutines.flow.transform
 import vn.momo.compose.payment.base.core.composeApi.INavigator
 import vn.momo.compose.payment.base.domain.entities.ENapasExpectedTypeCallback
 import vn.momo.compose.payment.base.domain.entities.NapasInfo
 import vn.momo.compose.payment.base.domain.entities.SOFItem
 import vn.momo.compose.payment.base.domain.entities.NapasInputCallbackData
 import vn.momo.compose.utils.datamapping.toJsonObject
 
 
 class StartNapasInputUseCase(
     private val navigator: INavigator,
 ) {
     operator fun invoke(currentSof: SOFItem?): Flow<NapasInfo?> {
         val cardType = currentSof?.info?.cardType ?: ""
         val navigateParams = mapOf(
             "cardType" to cardType,
             "cardFullName" to "",
             "cardNumber" to "",
             "cardIssueDate" to "",
             "personalId" to "",
             "isClaimCodeNapas" to false,
             "multiBankData" to emptyMap<String, String>(),
             "isMapNapas" to false,
             "handleOnSubmit" to true,
             "expectedType" to ENapasExpectedTypeCallback.FORM.name
         )
         return navigator.navigateFeatureCode(
             "napas_input_native",
             navigateParams.toJsonObject(),
             NapasInputCallbackData.serializer()
         ).transform { callbackData ->
             if (callbackData?.onBackOtp == true) {
                 emit(null)
             } else if (callbackData?.onSubmit == true) {
                 val card = callbackData.card
                 emit(
                     NapasInfo(
                         card?.cardNumber,
                         card?.cardFullName,
                         card?.cardIssueDate,
                         card?.isSaveToken ?: false,
                         currentSof?.moneySource,
                         currentSof?.info?.partnerCode,
                         currentSof?.info?.cardType,
                         currentSof?.info?.sofState,
                         currentSof?.info?.bankName
                     )
                 )
             }
         }
     }
 }
