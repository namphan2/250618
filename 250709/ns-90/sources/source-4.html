 @Composable
 fun BasicCardHeader(
     title: String,
     titleExtra: @Composable () -> Unit = {},
     action: HeaderAction? = null
 ) {
     Row(
         Modifier.fillMaxWidth(),
         verticalAlignment = Alignment.CenterVertically,
         horizontalArrangement = Arrangement.SpaceBetween,
     ) {
         Row(
             Modifier.weight(1f),
             verticalAlignment = Alignment.CenterVertically,
             horizontalArrangement = scaledSpacedBy(4f),
         ) {
             Text(
                 text = title,
                 style = Typography.actionSBold,
                 color = AppTheme.current.colors.text.default
             )
             Box(Modifier.weight(1f)) {
                 titleExtra()
             }
         }
 
         action?.let {
             Text(
                 text = action.title,
                 style = Typography.actionXsBold,
                 color = AppTheme.current.colors.primary,
                 accessibilityId = action.accessibilityId,
                 modifier = Modifier
                     .noFeedbackClickable(
                         action.enable,
                         onClick = debounceCompose(debounceTime = 500L, action = action.onAction)
                     )
                     .alpha(if (action.enable) 1f else 0.5f)
             )
         }
     }
