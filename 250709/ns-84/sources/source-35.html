 import vn.momo.kits.const.Colors
 import vn.momo.kits.const.Spacing
 import vn.momo.kits.const.Typography
 
 @Composable
 fun SofIconList(
     data: List<String>,
     maxItem: Int = 3,
     itemSize: Int = 32,
     extraItem: String? = null,
 ) {
     Row(
         verticalAlignment = Alignment.CenterVertically
     ) {
         val dataDisplay = data.slice(0 until maxItem.coerceIn(0, data.size))
         dataDisplay.mapIndexed { index, icon ->
             Surface(
                 shape = CircleShape,
                 border = BorderStroke(1.dp, AppTheme.current.colors.border.default),
                 modifier = Modifier
                     .then(
                         if (extraItem != null) {
                             Modifier.offset(x = Spacing.M * (maxItem - index))
                         } else {
                             Modifier.offset(x = Spacing.M * (maxItem - index - 1))
                         }
                     )
             ) {
                 Image(
                     source = icon,
                     modifier = Modifier
                         .size(itemSize.dp)
                         .background(AppTheme.current.colors.background.surface)
                 )
             }
         }
         if (dataDisplay.isNotEmpty() && extraItem != null) {
             Surface(
                 shape = CircleShape,
                 border = BorderStroke(1.dp, AppTheme.current.colors.border.default),
                 modifier = Modifier.size(itemSize.dp)
             ) {
                 Box(
                     contentAlignment = Alignment.Center,
                     modifier = Modifier
                         .fillMaxSize()
                         .background(AppTheme.current.colors.background.selected)
                 ) {
                     Text(text = "${extraItem}+", style = Typography.descriptionDefaultRegular, color = Colors.black_17)
                 }
             }
         }
     }
 }
