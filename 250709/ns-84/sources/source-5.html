 @OptIn(ExperimentalMaterial3Api::class)
 @Composable
 fun FooterV2(
     amount: TotalAmount,
     hideAmount: Boolean,
     hasAppliedPromotion: Boolean,
     enableConfirm: Boolean,
     onConfirm: () -> Unit,
     amountInfo: List<InfoRowContent> = emptyList(),
     cartData: CartData?,
     onToggleShowAmount: (state: String) -> Unit
 ) {
     val confirmButtonType: ButtonType =
         if (enableConfirm) ButtonType.PRIMARY else ButtonType.DISABLED
     val showBottomSheet = remember { mutableStateOf(false) }
 
     LaunchedEffect(showBottomSheet.value) {
         onToggleShowAmount(if (showBottomSheet.value) "expand" else "collapse")
     }
 
     if (hideAmount) {
         ConfirmButton(hideAmount, confirmButtonType, onConfirm)
     } else {
         Column(
             modifier = Modifier.fillMaxWidth(),
             verticalArrangement = scaledSpacedBy(12f)
         ) {
             cartData?.footer?.let {
                 createPayXComponent(listOf(it))
             }
             Row(
                 modifier = Modifier.fillMaxWidth(),
                 verticalAlignment = Alignment.CenterVertically,
                 horizontalArrangement = scaledSpacedBy(12f)
             ) {
                 Column(
                     modifier = Modifier.weight(1f),
                     verticalArrangement = Arrangement.Center
                 ) {
                     Row(
                         verticalAlignment = Alignment.CenterVertically,
                         modifier = Modifier.noFeedbackClickable(onClick = debounceCompose {
                             if (amountInfo.isNotEmpty()) {
                                 showBottomSheet.value = true
                             }
                         })
                     ) {
                         Text(
                             text = amount.total.title ?: stringResource(Res.string.totalPayment),
                             style = Typography.descriptionDefaultRegular,
                             color = AppTheme.current.colors.text.default,
                         )
                         if (amountInfo.isNotEmpty()) {
                             Icon(
                                 source = "arrow_chevron_up_small",
                                 size = 24.dp,
                                 color = AppTheme.current.colors.text.default,
                             )
                         }
                     }
                     Column {
                         val isPromotionApplied =
                             amount.totalBeforePromotion.isShow && hasAppliedPromotion
                         if (amount.total.isShow) {
                             AutoResizedText(
                                 text = amount.total.format,
                                 style = Typography.headerMBold,
                                 maxLines = 1,
                                 textAlign = TextAlign.End,
                                 color = if (isPromotionApplied) AppTheme.current.colors.success.primary else AppTheme.current.colors.text.default,
                             )
                         }
                         if (isPromotionApplied) {
                             AutoResizedText(
                                 text = amount.totalBeforePromotion.format,
                                 style = Typography.bodyDefaultRegular,
                                 textDecoration = TextDecoration.LineThrough,
                                 maxLines = 1,
                                 textAlign = TextAlign.End,
                             )
                         }
                     }
                 }
                 ConfirmButton(hideAmount, confirmButtonType, onConfirm)
             }
         }
     }
     AmountBts(showBottomSheet, amountInfo)
 }
 
 @Composable
 private fun ConfirmButton(
     hideAmount: Boolean,
     confirmButtonType: ButtonType,
     onConfirm: () -> Unit
 ) {
     val buttonModifier = Modifier.semantics {
         testTag = "btn_primary_confirm"
         contentDescription = "btn_primary_confirm"
     }
 
     Button(
         title = stringResource(Res.string.confirm),
         onClick = onConfirm,
         type = confirmButtonType,
         modifier = buttonModifier,
         isFull = hideAmount,
         iconLeft = "security_lock"
     )
 }
