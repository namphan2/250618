 @Composable
 fun SofGroupView(
     group: SOFGroup,
     selected: SOFItem?,
     isHideBalance: Boolean,
     onSelect: (SOFItem) -> Unit = {},
     modifier: Modifier = Modifier,
     parentTrail: (@Composable () -> Unit)? = null,
     onPressParent: () -> Unit = {},
     badge: String?
 ) {
     val density = LocalDensity.current
     val screenWidth = getScreenDimensions().screenWidth
     val itemMaxWidth = (screenWidth - scaleSize(24f)) / 3
     val scrollState = rememberScrollState()
 
     val index by remember(group.children, selected) {
         derivedStateOf {
             group.children.indexOfFirst {
                 it.isMatchSource(selected?.sourceToken, selected?.ID)
             }
         }
     }
 
     LaunchedEffect(index) {
         if (index >= 0) {
             scrollState.animateScrollTo(with(density) { ((itemMaxWidth + 8).dp * index).toPx() }.toInt())
         }
     }
 
     Column(
         modifier = Modifier
             .background(color = Colors.black_02, shape = RoundedCornerShape(12.dp))
             .then(modifier),
         verticalArrangement = scaledSpacedBy(8f)
     ) {
         SofCard.Default(
             sof = group.parent,
             isSelected = group.isSelected(selected?.uId),
             trailing = parentTrail,
             badge = badge,
             onPress = onPressParent
         )
         Row(
             Modifier.scaledPadding(start = 8f, bottom = 8f)
                 .horizontalScroll(scrollState),
             horizontalArrangement = scaledSpacedBy(8f),
             verticalAlignment = Alignment.CenterVertically
         ) {
             group.children.onEach { childSof ->
                 key(childSof.uId) {
                     val isSelected = childSof.isMatchSource(selected?.sourceToken, selected?.ID)
                     SofCard.MiniSof(
                         modifier = Modifier
                             .widthIn(max = itemMaxWidth.dp),
                         sof = childSof,
                         isSelected = isSelected,
                         isHideBalance = isHideBalance,
                         onPress = {
                             onSelect(childSof)
                         })
                 }
             }
         }
     }
 }
