 package vn.momo.compose.payment.promotion.core.utils
 
 import vn.momo.compose.payment.base.core.composeApi.firebaseEvent.IFirebaseTracer
 import vn.momo.compose.payment.base.core.service.trace.model.journey.EventName
 import vn.momo.compose.payment.promotion.core.model.IPromotionEventParams
 import vn.momo.compose.utils.datamapping.ModelConvertUtils
 import vn.momo.compose.utils.datamapping.toJsonObject
 import vn.momo.koin.KoinHelper
 
 object PromotionFirebaseTrace {
     private val tracer: IFirebaseTracer = KoinHelper.getInjected<IFirebaseTracer>()
 
     fun trackEvent(eventName: EventName, params: IPromotionEventParams) {
         val jsonParams = params.let {
             ModelConvertUtils.encodeToMap(IPromotionEventParams.serializer(), it)
         }?.filterValues { it != null }?.toJsonObject()
 
         tracer.track(
             eventName = eventName.id,
             eventParams = jsonParams,
         )
     }
 }
