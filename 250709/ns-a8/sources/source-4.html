 class VoucherRepository(
     private val multiBillRemoteDataSource: VoucherMultiBillDataSource,
     private val singleBillRemoteDataSource: VoucherSingleBillDataSource,
 ) : IVoucherRepository {
     override fun getVoucherList(
         pathType: VoucherPathType,
         page: Int?,
         params: ChoosePromotionScreenProps,
         user: String?,
     ): Flow<ListPage<IVoucherBase>> {
         if (pathType === VoucherPathType.SingleBillRecommend) {
             return getVoucherRecommendSingleBill(
                 pathType = pathType,
                 page = page,
                 params = params,
                 user = user,
             ).tracePromotionApi(pathType.traceEvent)
         }
 
         if (listOf(
                 VoucherPathType.SingleBillGiftMask,
                 VoucherPathType.SingleBillInvalidGiftMask,
                 VoucherPathType.SingleBillInvalid
             ).contains(pathType)
         ) {
             return getVoucherSingleBill(
                 pathType = pathType,
                 page = page,
                 params = params,
                 user = user,
             ).tracePromotionApi(pathType.traceEvent)
         }
 
         return getVoucherMultiBill(
             pathType = pathType,
             page = page,
             params = params
         ).tracePromotionApi(pathType.traceEvent)
     }
 
     override fun getVoucherRecommendSingleBill(
         pathType: VoucherPathType,
         page: Int?,
         params: ChoosePromotionScreenProps,
         user: String?,
     ): Flow<ListPage<IVoucherBase>> {
         val request = PostVoucherSingleBillRequestDTO(
             subPath = pathType.path,
             subDomain = "api",
             body = VoucherSingleBillRequestBody(
                 msgType = pathType.msgType,
                 momoMsg = PostVoucherSingleBillRequestMomoMsg(
                     page = page,
                     limit = DEFAULT_PAGE_SIZE,
                     serviceId = params.transaction?.serviceId,
                     originalAmount = params.transaction?.originalAmount,
                     moneySourceFilter = listOfNotNull(params.transaction?.moneySource),
                     cardNumber = params.cardNumber?.toFormattedCardNumbers(),
                     user = user,
                 )
             )
         )
 
         return singleBillRemoteDataSource.getRecommendVouchers(request)
             .tracePromotionApi(pathType.traceEvent)
             .map { response ->
                 response?.momoMsg?.let {
                     ListPage(
                         page = it.page ?: 1,
                         limit = it.limit ?: DEFAULT_PAGE_SIZE,
                         totalRecord = it.totalRecord ?: 0,
                         records = it.records?.map { item ->
                             item.toVoucherBase(
                                 VoucherPathType.toLocalType(pathType)
                             )
                         } ?: emptyList()
                     )
                 } ?: ListPage(
                     page = 1,
                     limit = DEFAULT_PAGE_SIZE,
                     totalRecord = 0,
                     records = emptyList()
                 )
             }
     }
 
     override fun getVoucherSingleBill(
         pathType: VoucherPathType,
         page: Int?,
         params: ChoosePromotionScreenProps,
         user: String?,
     ): Flow<ListPage<IVoucherBase>> {
         val request = GetVoucherSingleBillRequestDTO(

                 originalAmount = params.transaction?.originalAmount,

                 cardNumber = params.cardNumber?.toFormattedCardNumbers()?.joinToString(","),
                 user = user,
             )
         )
 
         return singleBillRemoteDataSource.getVouchers(request)
             .tracePromotionApi(pathType.traceEvent)
             .map { response ->
                 response?.momoMsg?.let {
                     ListPage(
                         page = it.page ?: 1,
                         limit = it.limit ?: DEFAULT_PAGE_SIZE,
                         totalRecord = it.totalRecord ?: 0,
                         records = it.records?.map { item ->
                             item.toVoucherBase(
                                 VoucherPathType.toLocalType(pathType)
                             )
                         } ?: emptyList()
                     )
                 } ?: ListPage(
                     page = 1,
                     limit = DEFAULT_PAGE_SIZE,
                     totalRecord = 0,
                     records = emptyList()
                 )
             }
     }

     override fun getListVoucherCoin(
         page: Int?,
         params: ChoosePromotionScreenProps
     ): Flow<VoucherCoinListPage> {
         val request = VoucherCoinRequestDTO(
             body = VoucherCoinRequestBody(
                 momoMsg = VoucherCoinRequestMomoMsg(
                     page = page,
                     limit = DEFAULT_PAGE_SIZE,
                     moneySourceFilter = params.moneySourceFilter,
                     cardNumber = params.cardNumber,
                     storeId = params.productInfo?.storeId,
                     merchantId = params.productInfo?.merchantId,
                     moneySourceInfo = params.invoiceInfo?.getOrNull(0)?.moneySourceInfo,
                     moneySource = params.invoiceInfo?.getOrNull(0)?.moneySource ?: 0,
                     items = params.billData?.items,
                 )
             )
         )
 
         return multiBillRemoteDataSource.getListVoucherCoin(request)
             .tracePromotionApi(PromotionApiPerfEvent.GetCoinVouchers.event)
             .map { response ->
                 response?.result?.let {
                     if (!it) {
                         throw Exception(response.errorDesc)
                     }
                 }
 
                 response?.momoMsg?.let {
                     VoucherCoinListPage(
                         page = it.page ?: 1,
                         limit = it.limit ?: DEFAULT_PAGE_SIZE,
                         totalRecord = it.totalRecord ?: 0,
                         records = it.records?.map { item ->
                             item.toVoucherCoin()
                         } ?: emptyList(),
                         coinInfo = it.coinInfo,
                     )
                 } ?: VoucherCoinListPage(
                     page = 1,
                     limit = DEFAULT_PAGE_SIZE,
                     totalRecord = 0,
                     records = emptyList(),
                     coinInfo = null,
                 )
             }
     }
 
     override fun claimVoucherCoin(rewardId: String): Flow<ClaimVoucherCoinRefExtra?> {
         val request = ClaimVoucherCoinRequestDTO(
             body = ClaimVoucherCoinRequestBody(
                 itemId = rewardId
             )
         )
 
         return multiBillRemoteDataSource.claimVoucherCoin(request)
             .tracePromotionApi(PromotionApiPerfEvent.ClaimCoinVoucher.event)
             .map { response ->
                 response?.data?.refExtra?.let { rawExtra ->
                     ModelConvertUtils.decode(ClaimVoucherCoinRefExtra.serializer(), rawExtra)
                 }
             }
     }
