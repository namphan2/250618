     private fun getAccessibilityLabel(sof: SOFItem): String {
         val bankNameSuffix = when (sof.moneySource) {
             EnumMoneySource.Bank -> sof.fullName
             EnumMoneySource.Card -> sof.info.bankName ?: ""
             else -> ""
         }
         val cardSchemeSuffix =
             if (sof.moneySource == EnumMoneySource.Napas) "NAPAS" else sof.info.cardSchema.orEmpty()
         val payLaterInstallmentSuffix = if (sof.isInstallment) "installment" else ""
         val listKey = listOf(
             "btn_sof",
             sof.moneySource.id.toString(),
             bankNameSuffix,
             cardSchemeSuffix,
             payLaterInstallmentSuffix
         ).filter { it.isNotEmpty() }
         return listKey.joinToString("_").lowercase()
     }

     @Composable
     fun Default(

             description = sof.shortDesc.takeIf { it.isNotEmpty() } ?: sof.fullDesc,
             leading = {
                 Image(source = sof.logo ?: "", modifier = Modifier.size(24.dp).alpha(alpha))
             },
             extra = {
                 Row(
                     horizontalArrangement = Arrangement.spacedBy(8.dp),
                     verticalAlignment = Alignment.CenterVertically
                 ) {
                     Box(Modifier.weight(1f)) {
                         val badgeString = sof.badge?.shortTitle?.takeIf { it.isNotBlank() }
                             ?: sof.badge?.title?.takeIf { it.isNotBlank() }
 
                         badgeString?.also {
                             Box(

     @Composable
     private fun SofAction(
         sof: SOFItem,
         isSelected: Boolean,
         onPress: () -> Unit = {},
     ) {
         val sofAction = sof.action
         if (sofAction?.isShow == true && (sofAction.action is SofActionType.Activate
                     || sofAction.action is SofActionType.CashIn
                     || sofAction.action is SofActionType.SofTransfer)
         ) {
             Button(
                 modifier = Modifier.alpha(1f),
                 size = Size.SMALL,
                 type = ButtonType.TONAL,
                 isFull = false,
                 title = sofAction.title,
                 onClick = onPress
             )
         } else {
             RadioButton(
                 selected = isSelected,
                 modifier = Modifier.alpha(if (sof.pressable) 1f else 0.5f)
             )
         }
     }
