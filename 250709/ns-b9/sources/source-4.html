 @OptIn(FlowPreview::class)
 class VoucherSearchSlice(
     private val scope: CoroutineScope,
     private val repository: IVoucherPosRepository,
     private val navigator: INavigator,
     private val params: ChoosePromotionPosScreenProps,
     private val logger: ILogger,
     private val promotionTracer: IPromotionVMTracking,
 ) : ViewModelSlice<VoucherSearchReducer.State, VoucherSearchReducer.Action, VoucherSearchReducer.Effect>(
     initialState = VoucherSearchReducer.State(),
     reducer = VoucherSearchReducer(),
     scope,
 ) {
     private val _searchQuery = MutableStateFlow("")
     val searchKey = mutableStateOf("")
 
     init {
         scope.launch {
             _searchQuery
                 .debounce(500) // Wait for 300ms of inactivity
                 .distinctUntilChanged() // Ignore duplicate queries
                 .collect { searchKey ->
                     val keyword = searchKey.trim()
                     if (keyword.isNotEmpty()) {
                         promotionTracer.trackVoucherPOSSearchKeywordInputted(
                             keyword
                         )
 
                         repository.onSearchVoucherPos(
                             keyword, 1, params
                         )
                             .onEach {
                                 logger.i("[VoucherSearchSlice] records.size ${it.records.size}")
                                 dispatch(
                                     VoucherSearchReducer.Action.SetSearchVouchers(
                                         it.records
                                     )
                                 )
                             }.catch {
                                 logger.i("[VoucherSearchSlice] error $it")
                                 dispatch(
                                     VoucherSearchReducer.Action.SetSearchVoucherState(false)
                                 )
                                 navigator.showToast(
                                     buildJsonObject {
                                         put("title", it.message)
                                         put("type", "error")
                                         put("duration", 3000)
                                     }, String.serializer()
                                 )
                             }.launchIn(this)
                     } else {
                         dispatch(VoucherSearchReducer.Action.SetSearchVouchers(emptyList()))
                     }
                 }
         }
     }
 
     fun onVoucherCollected(collectedVoucher: IVoucherBase) {
         dispatch(
             VoucherSearchReducer.Action.SetSearchVouchers(
                 state.value.searchVouchers.map { item ->
                     if (item.extra?.collectId == collectedVoucher.extra?.collectId) {
                         collectedVoucher
                     } else {
                         item
                     }
                 }
             )
         )
     }
 
     fun hideSearchModal() {
         setSearchModalVisibility(false)
         onSearch("")
     }
 
     fun setSearchModalVisibility(isVisible: Boolean) {
         dispatch(VoucherSearchReducer.Action.SetSearchModalVisibility(isVisible))
     }
 
     fun onSearch(key: String) {
         _searchQuery.value = key
         searchKey.value = key
         if (key.isNotEmpty()) {
             dispatch(
                 VoucherSearchReducer.Action.SetSearchVoucherState(true)
             )
         }
     }
 
     fun onSelectVoucher(voucher: IVoucherBase) {
         dispatch(VoucherSearchReducer.Action.SelectSearchVoucher(voucher))
         setSearchModalVisibility(false)
     }
 }
