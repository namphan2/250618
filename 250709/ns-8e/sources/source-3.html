 @Composable
 fun SelectBankView(
     type: VAReducer.OpenBankType,
     bankList: List<DeepLinkBankInfo>,
     onOpenBank: (String?) -> Unit,
     onClose: () -> Unit,
 ) {
     val screenHeight = getScreenDimensions().height
 
     Column {
         BottomSheetHeader(
             title = stringResource(Res.string.openBankTitle),
             iconRight = AsyncIconProp(Images.ic_close_fat, size = 14.dp, onPress = onClose)
         )
         Spacer(modifier = Modifier.height(Spacing.L))
         Column(
             modifier = Modifier.fillMaxWidth()
                 .heightIn(max = (0.8 * screenHeight - 72).dp)
                 .verticalScroll(rememberScrollState()),
             verticalArrangement = Arrangement.spacedBy(Spacing.M)
         ) {
             Spacer(modifier = Modifier.height(1.dp))
             Surface(
                 shape = RoundedCornerShape(Radius.S),
                 border = BorderStroke(0.5.dp, color = Colors.blue_07),
                 color = Colors.blue_10,
             ) {
                 Column(
                     modifier = Modifier.padding(Spacing.M),
                     verticalArrangement = Arrangement.spacedBy(Spacing.XXS)
                 ) {
                     Text(
                         if (type == VAReducer.OpenBankType.QR) stringResource(Res.string.qrDownloaded)
                         else stringResource(Res.string.bankAccountCopied),
                         style = Typography.labelDefaultMedium,
                         modifier = Modifier.fillMaxWidth()
                     )
                     Text(
                         stringResource(Res.string.bankTransGuide),
                         style = Typography.descriptionDefaultRegular,
                         modifier = Modifier.fillMaxWidth()
                     )
                 }
             }
             val background = AppTheme.current.colors.background.surface
             Surface(
                 shape = RoundedCornerShape(Radius.M),
                 color = background,
             ) {
                 val strokeColor = AppTheme.current.colors.border.default
                 Column {
                     bankList.onEachIndexed { idx, bank ->
                         key(bank.id) {
                             Surface(
                                 color = background,
                                 modifier = Modifier.fillMaxWidth()
                                     .padding(horizontal = Spacing.M)
                                     .conditional(idx < bankList.size - 1) {
                                         bottomBorder(
                                             1.dp,
                                             strokeColor
                                         )
                                     }
                                     .padding(vertical = Spacing.M)
                             ) {
                                 Row(
                                     horizontalArrangement = Arrangement.spacedBy(Spacing.S),
                                     verticalAlignment = Alignment.CenterVertically
                                 ) {
                                     bank.logo?.let {
                                         Image(
                                             source = bank.logo,
                                             modifier = Modifier.size(32.dp),
                                             options = Options(alignment = Alignment.Center)
                                         )
                                     }
 
                                     Text(
                                         bank.name,
                                         style = Typography.headerSSemibold,
                                         modifier = Modifier.weight(1f),
                                         maxLines = 1,
                                         overflow = TextOverflow.Ellipsis
                                     )
                                     Button(
                                         title = stringResource(Res.string.openBankingApp),
                                         type = ButtonType.TONAL,
                                         size = Size.SMALL,
                                         isFull = false,
                                         onClick = {
                                             onOpenBank(bank.appScheme)
                                         })
                                 }
                             }
                         }
                     }
 
                 }
             }
         }
     }
 }
