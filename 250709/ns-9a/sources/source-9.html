 @Composable
 fun TransactionResultRevampScreen(viewModel: TransactionResultRevampVM) {
     val transactionResultState by viewModel.state.state.collectAsState()
     val firstTransaction = remember(transactionResultState.ticketList) {
         transactionResultState.ticketList.firstOrNull()
     }
     val (showNBABlock, showExpenseWidget, showCornerstoneWidget) = remember(
         firstTransaction
     ) {
         val transaction = firstTransaction?.transaction
         val isSuccess = transaction?.isSuccess == true
         val isFailed = transaction?.isFailed == true
         Triple(isSuccess, !isFailed, isSuccess)
     }
 
     val allNextBestActionList = remember(
         transactionResultState.nextBestActionList, transactionResultState.apiNextBestActionList
     ) {
         transactionResultState.nextBestActionList + transactionResultState.apiNextBestActionList
     }
 
     LaunchedEffect(firstTransaction, transactionResultState.gamificationData) {
         firstTransaction?.transaction?.takeIf { it.isSuccess }?.let { transaction ->
             delay(100)
             viewModel.dispatch(TransactionResultAction.GetNextBestActionData(transaction.serviceId.orEmpty()))
             if (transactionResultState.gamificationData.isEmpty) {
                 viewModel.dispatch(TransactionResultAction.GetGamificationData(transaction.transactionId))
             }
         }
     }
 
     BackHandler(true, null)
     Screen(title = stringResource(Res.string.transactionResult),
         titlePosition = TitlePosition.CENTER,
         headerType = HeaderType.EXTENDED,
         scrollable = true,
         headerRight = {
             HeaderIcon("basic_home") {
                 viewModel.dispatch(TransactionResultAction.GoHome)
             }
         },
         content = {
             FeeBlocks(transactionResultState.fees)
             transactionResultState.ticketList.fastForEachIndexed { index, item ->
                 StatusTransactionResultScreen(
                     item, index, nextBestActionContent = {
                         NextBestActionBlockV2(
                             dispatcher = viewModel::dispatch,
                             allNextBestActionList,
                             transactionResultState.gamificationData,
                             modifier = Modifier.padding(top = Spacing.L),
                             show = showNBABlock && transactionResultState.abTestNBA == ABTestTagname.Treatment2.code && transactionResultState.showOtherAddons
                         )
                         NextBestActionBlockV3(
                             dispatcher = viewModel::dispatch,
                             allNextBestActionList,
                             transactionResultState.gamificationData,
                             modifier = Modifier.padding(top = Spacing.L),
                             show = showNBABlock && transactionResultState.abTestNBA == ABTestTagname.Treatment3.code && transactionResultState.showOtherAddons
                         )
                     }, viewModel::dispatch
                 )
             }
             NextBestActionBlockV1(
                 dispatcher = viewModel::dispatch,
                 allNextBestActionList,
                 transactionResultState.gamificationData,
                 modifier = Modifier.padding(bottom = Spacing.L),
                 show = showNBABlock && transactionResultState.abTestNBA == ABTestTagname.Treatment1.code && transactionResultState.showOtherAddons
             )
             ExpenseWidget(firstTransaction,
                 show = showExpenseWidget && transactionResultState.widgetServiceLoaded && transactionResultState.showOtherAddons,
                 {
                     viewModel.dispatch(
                         TransactionResultAction.OnWidgetLoadSuccess(
                             TransactionResultWidget.Expense
                         )
                     )
                 },
                 {
                     viewModel.dispatch(
                         TransactionResultAction.OnWidgetLoadFailed(
                             TransactionResultWidget.Expense
                         )
                     )
                 },
                 { width, height ->
                     viewModel.dispatch(
                         TransactionResultAction.OnWidgetLayoutChanged(
                             TransactionResultWidget.Expense, width, height
                         )
                     )
                 })
 
             CornerstoneWidget(firstTransaction,
                 0,
                 show = showCornerstoneWidget && transactionResultState.widgetServiceLoaded && transactionResultState.showOtherAddons,
                 {
                     viewModel.dispatch(
                         TransactionResultAction.OnWidgetLoadSuccess(
                             TransactionResultWidget.Cornerstone0
                         )
                     )
                 },
                 {
                     viewModel.dispatch(
                         TransactionResultAction.OnWidgetLoadFailed(
                             TransactionResultWidget.Cornerstone0
                         )
                     )
                 },
                 { event ->
                     viewModel.dispatch(
                         TransactionResultAction.OnWidgetLayoutChanged(
                             TransactionResultWidget.Cornerstone0, event.width, event.height
                         )
                     )
                 })
             CornerstoneWidget(firstTransaction,
                 1,
                 show = showCornerstoneWidget && transactionResultState.widgetServiceLoaded && transactionResultState.showOtherAddons,
                 {
                     viewModel.dispatch(
                         TransactionResultAction.OnWidgetLoadSuccess(
                             TransactionResultWidget.Cornerstone1
                         )
                     )
                 },
                 {
                     viewModel.dispatch(
                         TransactionResultAction.OnWidgetLoadFailed(
                             TransactionResultWidget.Cornerstone1
                         )
                     )
                 },
                 { event ->
                     viewModel.dispatch(
                         TransactionResultAction.OnWidgetLayoutChanged(
                             TransactionResultWidget.Cornerstone1, event.width, event.height
                         )
                     )
                 })
 
         })
 }
