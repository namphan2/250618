 @Composable
 fun StatusTransactionResultScreen(
     data: TransactionTicketItem,
     index: Int = 0,
     nextBestActionContent: @Composable () -> Unit = {},
     dispatcher: (TransactionResultAction) -> Unit
 ) {
     val transaction = data.transaction ?: TransactionItem()
     val title = stringResource(
         when (transaction.transactionStatus) {
             TransactionStatus.SUCCESS -> Res.string.successfullyTransaction
             TransactionStatus.FAILED -> Res.string.failedRevampTransaction
             TransactionStatus.PROCESSING, TransactionStatus.PENDING -> Res.string.processingTransaction
         }
     )
     val showShareButton = transaction.isSuccess
     val messageInfo = data.messageInfo?.takeIf { it.isNotBlank() } ?: transaction.desc
     val messageInfoErrorCode = buildString {
         messageInfo?.takeIf { it.isNotBlank() }?.let {
             append(it)
             if (transaction.error != null && !transaction.isSuccess) {
                 append(" (${stringResource(Res.string.errorCode)}: ${transaction.error})")
             }
         }
     }
     val showMessageInfo = messageInfoErrorCode.isNotBlank() && !transaction.isSuccess
     val showAmount = transaction.isValidAmount
     val showOriginalAmount = transaction.isPaymentSDK && transaction.isValidOriginalAmount
 
     RevampTransactionTicket(header = {
         TransactionStatusCard(title = title,
             amountText = transaction.amountText,
             showAmount = showAmount,
             originalAmountText = transaction.originalAmountText,
             showOriginalAmount = showOriginalAmount,
             showShareButton = showShareButton,
             onPressShare = {
                 dispatcher.invoke(
                     TransactionResultAction.OnPressShare(transaction)
                 )
             },
             leftIcon = { StatusIconRevamp(data = transaction.transactionStatus) })
     }) {
         Spacer(modifier = Modifier.height(Spacing.S))
         PayXAnimatedVisibility(visible = showMessageInfo) {
             Column {
                 Information(
                     description = messageInfoErrorCode, state = InformationState.WARNING
                 )
                 Spacer(modifier = Modifier.height(Spacing.S))
             }
         }
         TransactionInfo(data = data, index = index, dispatcher = dispatcher)
 
         TransactionResultWidget(Modifier.padding(top = Spacing.S), true, data, {
             dispatcher.invoke(TransactionResultAction.OnWidgetLoadSuccess())
         }, {
             dispatcher.invoke(TransactionResultAction.OnWidgetLoadFailed())
         }, { width: Int, height: Int ->
             dispatcher.invoke(
                 TransactionResultAction.OnWidgetLayoutChanged(
                     width = width, height = height
                 )
             )
         })
         nextBestActionContent()
         Spacer(Modifier.height(Spacing.M))
         RevampActionButtonGroup(
             Modifier,
             primaryButton = data.revampPrimaryButton,
             secondaryButton = data.revampSecondaryButton,
             dispatcher
         )
     }
 }
