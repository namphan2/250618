 package vn.momo.compose.payment.payx.domain.useCases.transactionresult
 
 import vn.momo.compose.payment.payx.domain.entities.transactionresult.TransactionTicketItem
 import vn.momo.compose.payment.payx.domain.entities.transactionresult.TxnResponse
 import vn.momo.compose.payment.payx.domain.repositories.transactionresult.IServiceWidgetRepository
 import vn.momo.compose.payment.payx.domain.repositories.transactionresult.ITransactionResultRepository
 
 class GetTicketListUseCase(
     private val transactionRepository: ITransactionResultRepository,
     private val widgetRepository: IServiceWidgetRepository
 ) {
     suspend operator fun invoke(
         response: TxnResponse?, defaultAppId: String?, versionFlow: String? = null, version: String? = null
     ): List<TransactionTicketItem> {
         val ticketList = transactionRepository.getTicketList(response)
         val widgetList = widgetRepository.getWidgetList(response, versionFlow, version)
 
         return ticketList.mapIndexed { index, ticket ->
             val widgetInfo = widgetList[index]
             widgetInfo.appId = widgetInfo.appId.ifBlank { defaultAppId ?: "" }
             ticket.widget = widgetInfo
             ticket
         }
     }
 
 }
