 @Composable
 fun SplitBillComponent(
     item: SplitBillItem,
     onPress: (type: String) -> Unit
 ) {
     Column(
         modifier = Modifier
             .fillMaxWidth()
             .clip(RoundedCornerShape(Spacing.M))
             .background(Colors.black_01)
             .padding(Spacing.M)
     ) {
         Row(
             verticalAlignment = Alignment.CenterVertically
         ) {
             Image(
                 source = item.icon,
                 modifier = Modifier.width(20.dp).height(20.dp).padding(end = Spacing.XS),
                 options = Options(contentScale = ContentScale.FillBounds)
             )
             Text(
                 text = item.title,
                 style = Typography.headerSSemibold,
                 color = Colors.black_17
             )
         }
         if (item.items.isNotEmpty()) {
             Text(
                 text = item.desc,
                 style = Typography.descriptionDefaultRegular,
                 color = Colors.black_15,
                 modifier = Modifier.padding(vertical = Spacing.S)
             )
             item.items.mapIndexed { idx, i ->
                 SplitBillSelection(i, onPress)
                 if (idx < item.items.size - 1) {
                     Spacer(modifier = Modifier.height(Spacing.S))
                 }
             }
         } else {
             Text(
                 text = "Bạn có thể chuyển tiền vào ngày mai nếu chưa sẵn sàng xác thực khuôn mặt bằng camera.",
                 style = Typography.descriptionDefaultRegular,
                 color = Colors.black_15,
                 modifier = Modifier.padding(vertical = Spacing.S)
             )
         }
 
     }
 }
 
 @Composable
 fun SplitBillSelection(
     item: SplitBillSelectionItem,
     onPress: (type: String) -> Unit
 ) {
     Row(
         modifier = Modifier
             .fillMaxWidth()
             .border(
                 width = 1.dp,
                 color = item.getBorderColor(),
                 shape = RoundedCornerShape(Spacing.M)
             )
             .padding(Spacing.M),
         horizontalArrangement = Arrangement.SpaceBetween,
         verticalAlignment = Alignment.CenterVertically
     ) {
         Row {
             if (listOf(
                     SplitBillSelectionType.Available,
                     SplitBillSelectionType.Unavailable
                 ).contains(item.type)
             ) {
                 Radio(
                     value = true,
                     groupValue = item.type == SplitBillSelectionType.Available,
                     disabled = item.type == SplitBillSelectionType.Unavailable
                 )
             }
             Column(
                 modifier = Modifier
                     .padding(start = Spacing.S)
             ) {
                 Text(
                     text = item.title,
                     style = Typography.bodyDefaultRegular,
                     color = item.getTextColor(),
                     modifier = Modifier.padding(bottom = Spacing.XS)
                 )
                 Text(
                     text = item.desc,
                     style = Typography.bodyDefaultRegular,
                     color = item.getTextColor(),
                 )
             }
         }
         if (item.type == SplitBillSelectionType.Unavailable) {
             Column(
                 modifier = Modifier
                     .clip(RoundedCornerShape(Spacing.XL))
                     .background(Colors.black_03)
                     .padding(horizontal = Spacing.XS, vertical = Spacing.XXS)
             ) {
                 Text(
                     text = item.rightCtaName,
                     style = Typography.descriptionXsRegular,
                     color = Colors.black_12,
                 )
             }
         } else {
             Button(
                 title = item.rightCtaName,
                 size = Size.SMALL,
                 type = if (item.type == SplitBillSelectionType.Remain) ButtonType.OUTLINE else ButtonType.PRIMARY,
                 isFull = false,
                 onClick = { onPress(item.type.name) }
             )
         }
     }
 }
