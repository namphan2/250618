 @Composable
 fun createPayXComponent(list: List<PayXComponent>) {
     val uiKitActionHandler by PaymentKoinScope.getScope(PAYMENT_BASE)
         .inject<UIKitActionHandler>()
 
     fun pushUIEvent(action: PayXCta?) {
         uiKitActionHandler.uiEvent.tryEmit(action)
     }
 
     fun pushCheckBoxEvent(effect: CheckoutBoxEvent) {
         uiKitActionHandler.checkBoxEvent.tryEmit(effect)
     }
 
     list.forEach { component ->
         when (component) {
             is PayXChip -> Chip(data = component) {
                 pushUIEvent(it)
             }
 
             is PayXText -> Label(data = component) {
                 pushUIEvent(it)
             }
 
             is PayXCheckBox -> CheckBoxUI(data = component,
                 onPressText = { pushUIEvent(it) },
                 onCheckedChange = { pushCheckBoxEvent(it) }
             )
 
             is PayXLinkList -> LinkList(data = component) {
                 pushUIEvent(it)
             }
 
             is PayXCartItem -> CartItem(data = component) {
                 pushUIEvent(it)
             }
 
             else -> {}
         }
     }
 }
