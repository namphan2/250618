 package vn.momo.compose.payment.payx.presentation.viewModel.reducers
 
 import vn.momo.compose.payment.base.domain.entities.transaction.TotalAmount
 import vn.momo.compose.payment.base.presentation.viewModel.reducer.Reducer
 import vn.momo.compose.payment.base.domain.entities.CartData
 import vn.momo.compose.payment.payx.domain.entities.FraudInfo
 import vn.momo.compose.payment.payx.domain.entities.InvoiceItem
 import vn.momo.compose.payment.payx.domain.entities.PromotionData
 import vn.momo.core.features.annotation.Immutable
 
 class TransactionReducer :
     Reducer<TransactionReducer.State, TransactionReducer.Action, TransactionReducer.Effect> {
     @Immutable
     data class State(
         val isLoading: Boolean = false,
         val enableConfirm: Boolean = false,
         val skipLoading: Boolean? = false,
         val fraudInfo: FraudInfo? = null,
         val purchaseId: String = "",
         val sourceMessage: String? = "",
         val promotion: PromotionData = PromotionData(),
         val invoices: List<InvoiceItem> = emptyList(),
         val amount: TotalAmount = TotalAmount(),
         val inlineMessage: String? = null,
         val shouldShowTid: Boolean = true,
         val showConfirmScreen: Boolean = true,
         val hideAmount: Boolean = false,
         val serviceCartData: CartData? = null,
     ) : Reducer.State
 
     @Immutable
     sealed class Action : Reducer.Action {
         data class SetLoading(val value: Boolean) : Action()
         data class SetTransaction(
             val data: State
         ) : Action()
 
         data class SetEnableConfirm(val value: Boolean) : Action()
         data class SetShowTid(val value: Boolean) : Action()
         data class SetSkipLoading(val value: Boolean?) : Action()
     }
 
     @Immutable
     class Effect() : Reducer.Effect
 
     override fun reduce(prev: State, action: Action): Pair<State, Effect?> {
         return when (action) {
             is Action.SetLoading -> prev.copy(
                 isLoading = action.value
             ) to null
 
             is Action.SetTransaction -> prev.copy(
                 isLoading = false,
                 enableConfirm = action.data.enableConfirm,
                 skipLoading = action.data.skipLoading,
                 fraudInfo = action.data.fraudInfo,
                 purchaseId = action.data.purchaseId,
                 promotion = action.data.promotion,
                 sourceMessage = action.data.sourceMessage,
                 invoices = action.data.invoices,
                 amount = action.data.amount,
                 inlineMessage = action.data.inlineMessage,
                 showConfirmScreen = action.data.showConfirmScreen,
                 hideAmount = action.data.hideAmount,
                 serviceCartData = action.data.serviceCartData.takeIf { it != null }
                     ?: prev.serviceCartData
             ) to null
 
             is Action.SetEnableConfirm -> prev.copy(enableConfirm = action.value) to null
             is Action.SetShowTid -> prev.copy(shouldShowTid = action.value) to null
             is Action.SetSkipLoading -> prev.copy(
                 skipLoading = (prev.skipLoading ?: false) && (action.value ?: true)
             ) to null // ruleBE: merge skipLoading and skipLoadingExtraOrder
         }
     }
 }
