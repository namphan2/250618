 const val privacyLink = "https://momo.vn/chinh-sach-quyen-rieng-tu"
 
 @Composable
 fun PrivacyDescription() {
     val textWithLink = buildAnnotatedString {
         val defaultText = SpanStyle(color = AppTheme.current.colors.text.hint)
         val linkText = SpanStyle(color = AppTheme.current.colors.interactive.primary)
         val webTitle = stringResource(Res.string.term)
         val privacyDesc1 = stringResource(Res.string.privacyDesc1)
         val riskConsentDesc = stringResource(Res.string.riskConsentTitle)
         val privacyDesc2 = stringResource(Res.string.privacyDesc2)
         val privacyDesc3 = stringResource(Res.string.privacyDesc3)
         withStyle(defaultText) {
             append("$privacyDesc1 ")
         }
         withLink(LinkAnnotation.Url(privacyLink, TextLinkStyles(style = linkText)) {
             val url = (it as LinkAnnotation.Url).url
             FirebaseTrace.track(RiskConsentHyperlinkClicked())
             KoinHelper.get<IRouteApi>().openWeb(url, title = webTitle)
         }) {
             append("$riskConsentDesc ")
         }
         withStyle(defaultText) {
             append("$privacyDesc2 \n")
             append(privacyDesc3)
         }
     }
     BasicText(textWithLink, style = Typography.descriptionDefaultRegular)
 }
