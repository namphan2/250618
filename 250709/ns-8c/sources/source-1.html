 @Composable
 fun ConsentBts(
     viewModel: UserConsentVM,
     onClose: () -> Unit,
 ) {
     val screenHeight = getScreenDimensions().screenHeight
     val state = viewModel.state.collectAsState()
     val scrollState = rememberScrollState()
     val consentType = remember { viewModel.info.type }
 
     val indicator = WindowInsets.systemBars.asPaddingValues().calculateBottomPadding()
     val bottomPadding = min(indicator, 21.dp) + Spacing.S
 
     LaunchedEffect(Unit) {
         GrafanaTrace.count(PayXUserEvent.ShowConsent.event)
         viewModel.trackDisplayed()
     }
 
     Column(
         modifier = Modifier.fillMaxWidth().animateContentSize()
     ) {
         val footerButtonHeight = bottomPadding + Spacing.S + Size.LARGE.value.height
         Column(
             modifier = Modifier.heightIn(max = (0.5 * screenHeight).dp - footerButtonHeight)
                 .verticalScroll(scrollState)
         ) {
             when (consentType) {
                 is ConsentType.NFC -> NFCConsent(
                     info = consentType.bankInfo,
                     consent = state.value.nfc
                 )
 
                 is ConsentType.Risk -> RiskConsent(
                     requiredState = state.value.riskRequired,
                     onChangeRequire = viewModel::onRiskRequiredToggle,
                     optionalState = state.value.riskOptional,
                     onChangeOptional = viewModel::onRiskOptionalToggle
                 )
 
                 is ConsentType.Mixed -> NFCAndRiskConsent(
                     state = state.value,
                     info = consentType.bankInfo
                 )
 
                 ConsentType.None -> {}
             }
         }
 
         Box(
             modifier = Modifier.fillMaxWidth()
                 .padding(top = Spacing.S, bottom = bottomPadding)
         ) {
             Button(
                 onClick = {
                     viewModel.updateConsent(state.value)
                     onClose()
                 },
                 type = if (viewModel.checkStatus(state.value)) ButtonType.PRIMARY else ButtonType.DISABLED,
                 size = Size.LARGE,
                 title = when (consentType) {
                     is ConsentType.NFC, is ConsentType.Mixed -> stringResource(Res.string.agreeConsent)
                     else -> stringResource(Res.string.confirm)
                 }
             )
         }
     }
 }
