 @OptIn(ExperimentalMaterial3Api::class)
 @Composable
 fun PromoCodeBottomSheet(
     isVisible: Boolean = false,
     onDismiss: () -> Unit = {},
     sheetState: SheetState,
 
     promoCode: String = "",
     promoCodeApplied: String = "",
     onChange: (String) -> Unit = {},
     onSubmit: (String) -> Unit = {},
     isChecking: Boolean = false,
     errorMessage: String? = null,
     onValueCleared: () -> Unit = {},
     isApplied: Boolean = false,
 ) {
     val isKeyboardVisible = keyboardVisibilityAsState()
     val screenDimension = getScreenDimensions()
     val maxHeight = screenDimension.screenHeight * 0.5f
     val focusRequester = remember { FocusRequester() }
     val isCancelAction = isApplied && promoCode == promoCodeApplied
 
     LaunchedEffect(isVisible, sheetState.currentValue) {
         if (isVisible && sheetState.currentValue == SheetValue.Expanded) {
             focusRequester.requestFocus()
         }
     }
 
     PromotionBottomSheet(
         isVisible = isVisible,
         onDismiss = {
             onDismiss()
             if (!isApplied) {
                 onChange("")
             }
         },
         sheetState = sheetState,
         headerTitle = stringResource(Res.string.promoCodeInputLabel),
         headerTransparent = true,
         contentWindowInsets = {
             if (isKeyboardVisible.value) {
                 WindowInsets(bottom = 0.dp)
             } else {
                 BottomSheetDefaults.windowInsets
             }
         },
     ) {
         Column(
             Modifier
                 .heightIn(max = maxHeight.dp)
                 .background(color = AppTheme.current.colors.background.default)
         ) {
             PromoCodeInput(
                 promoCode = promoCode,
                 onChange = onChange,
                 isChecking = isChecking,
                 onSubmit = onSubmit,
                 errorMessage = errorMessage,
                 onValueCleared = onValueCleared,
                 isApplied = isApplied,
                 withInlineButton = false,
                 modifier = Modifier.padding(Spacing.M),
                 focusRequester = focusRequester,
             )
 
             Footer(
                 withShadow = false,
                 withBottomPadding = false,
             ) {
                 Button(
                     title = stringResource(
                         if (isCancelAction) {
                             Res.string.cancelLabel
                         } else Res.string.checkPromoCode
                     ),
                     onClick = {
                         if (isCancelAction) {
                             onValueCleared()
                         } else {
                             onSubmit(promoCode)
                         }
                     },
                     type = if (isChecking || promoCode.isBlank()) {
                         ButtonType.DISABLED
                     } else {
                         ButtonType.PRIMARY
                     },
                 )
             }
         }
     }
 }
