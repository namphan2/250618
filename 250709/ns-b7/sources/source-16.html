 @OptIn(ExperimentalMaterial3Api::class)
 @Composable
 fun VoucherList(
     vouchers: List<IVoucherBase>,
     selectedVouchers: List<IVoucherBase>? = emptyList(),
     isFetching: Boolean? = false,
     isRefetching: Boolean? = false,
     shouldDisplayStackingVoucherTag: Boolean? = false,
     collectingVoucherId: String? = null,
     claimingRewardId: String? = null,
     hideEmptyState: Boolean? = false,
 
     onRefetch: () -> Unit? = {},
     onFetchNextPage: () -> Unit? = {},
     onVoucherClick: (voucher: IVoucherBase, index: Int) -> Unit? = { _, _ -> },
     onVoucherSelect: (voucher: IVoucherBase, index: Int) -> Unit? = { _, _ -> },
     onVoucherCollect: (voucher: IVoucherBase, index: Int) -> Unit? = { _, _ -> },
     onVoucherClaim: (voucher: IVoucherBase, index: Int) -> Unit? = { _, _ -> },
     onVoucherUseNow: (voucher: IVoucherBase, index: Int) -> Unit? = { _, _ -> },
     onVoucherDisplayed: (voucher: IVoucherBase, index: Int) -> Unit? = { _, _ -> },
     checkVoucherAvailability: (IVoucherBase) -> Boolean = { true },
     checkVoucherStackable: (IVoucherBase) -> Boolean = { true },
 
     listHeader: @Composable (() -> Unit)? = null,
     listFooter: @Composable (() -> Unit)? = null,
     emptyComponent: @Composable (() -> Unit)? = null,
     scrollState: LazyListState = rememberLazyListState(),
     modifier: Modifier = Modifier,
     pullToRefreshEnabled: Boolean = true,
 
     sourceScreen: PromotionSourceScreen? = null,
 ) {
 
     @Composable
     fun LazyColumnContent() {
         LazyColumn(
             modifier = Modifier.fillMaxSize(),
             state = scrollState,
             verticalArrangement = Arrangement.spacedBy(Spacing.S),
             contentPadding = PaddingValues(vertical = Spacing.M)
         ) {
             if (listHeader != null) {
                 item {
                     listHeader.invoke()
                 }
             }
 
             if (vouchers.isNotEmpty()) {
                 itemsIndexed(
                     items = vouchers,
 //                    key = { _, voucher ->
 //                        when {
 //                            !voucher.ID.isNullOrBlank() -> voucher.ID
 //                            !voucher.extra?.collectId.isNullOrBlank() -> voucher.extra?.collectId
 //                            else -> Random.nextInt()
 //                        }!!
 //                    }
                 ) { index, voucher ->
                     Row(
                         Modifier.animateItem(
                             fadeInSpec = tween(durationMillis = 250),
                             fadeOutSpec = tween(durationMillis = 100),
                             placementSpec = tween(durationMillis = 250)
                         )
                     ) {
                         VoucherItem(
                             voucher = voucher,
                             onClick = { onVoucherClick(voucher, index) },
                             onSelect = { onVoucherSelect(voucher, index) },
                             onCollect = { onVoucherCollect(voucher, index) },
                             onClaim = { onVoucherClaim(voucher, index) },
                             onUseNow = { onVoucherUseNow(voucher, index) },
                             onDisplayed = { onVoucherDisplayed(voucher, index) },
                             selected = selectedVouchers?.any { voucher.ID == it.ID } ?: false,
                             isSelectable = checkVoucherAvailability(voucher),
                             isStackable = shouldDisplayStackingVoucherTag == true
                                     && checkVoucherStackable(voucher),
                             isCollecting = collectingVoucherId != null && collectingVoucherId == voucher.extra?.collectId,
                             isClaiming = claimingRewardId != null && claimingRewardId == voucher.extra?.rewardId,
                             sourceScreen = sourceScreen,
                         )
                     }
                 }
             } else if (isFetching == false && hideEmptyState == false) {
                 item {
                     if (emptyComponent != null) emptyComponent.invoke()
                     else EmptyState(showPromotionGuide = true)
                 }
             }
 
             if (isFetching == true) {
                 item {
                     Column(
                         verticalArrangement = Arrangement.spacedBy(Spacing.S),
                         modifier = Modifier.fillMaxSize()
                     ) {
                         repeat(5) {
                             VoucherSkeleton()
                         }
                     }
                 }
             }
 
             item {
                 listFooter?.invoke()
             }
         }
     }
 
     if (!pullToRefreshEnabled) {
         Box(
             modifier = modifier.fillMaxSize()
         ) {
             LazyColumnContent()
         }
     } else {
         PullToRefreshBox(modifier = modifier.fillMaxSize(),
             isRefreshing = isRefetching == true,
             onRefresh = { onRefetch() }) {
             LazyColumnContent()
         }
     }
 
     LaunchedEffect(scrollState) {
         snapshotFlow {
             val visibleItems = scrollState.layoutInfo.visibleItemsInfo
             val totalItems = scrollState.layoutInfo.totalItemsCount
             visibleItems.lastOrNull()?.index to totalItems
         }.collect { (lastVisibleItemIndex, totalItems) ->
             if (lastVisibleItemIndex != null && totalItems > 0 && lastVisibleItemIndex >= totalItems - 2) {
                 onFetchNextPage()
             }
         }
     }
 }
