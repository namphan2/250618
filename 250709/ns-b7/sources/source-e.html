 /**
  * Simplify version of Kit's ToolkitHeaderRight
  * to allow customize onDismiss callbackData
  */
 @Composable
 fun PromotionToolkit(
     onDismiss: (() -> Unit)? = null,
 ) {
     val api = PlatformApi.current as? ComposeApi
     val context = ApplicationContext.current
 
     println("[PromotionToolkit] context=$context")
 
     fun onPressHelpCenter() {
         val paramMap = mapOf(
             "appId" to context?.appId,
             "code" to context?.appCode,
             "name" to context?.appName,
             "icon" to context?.appIcon,
             "description" to context?.description
         )
         api?.request("showHelpCenter", paramMap) {
             println("[PromotionToolkit] onPressHelpCenter callback: $it")
         }
     }
 
     fun onPressClose() {
         if (onDismiss != null) {
             onDismiss()
         } else {
             api?.request("dismiss", "") { _ -> }
         }
     }
 
     val toolkitWidth = if (context !== null) 65.dp else 28.dp
 
     Row(
         verticalAlignment = Alignment.CenterVertically
     ) {
         Row(
             verticalAlignment = Alignment.CenterVertically,
             horizontalArrangement = Arrangement.Center,
             modifier = Modifier
                 .padding(start = Spacing.S)
                 .border(0.2.dp, Color(0x33000000), shape = RoundedCornerShape(14.dp))
                 .width(toolkitWidth)
                 .height(28.dp)
                 .clip(shape = RoundedCornerShape(14.dp))
                 .background(Color(0x99FFFFFF))
         ) {
             if (context != null) {
                 Icon(
                     source = "help_center",
                     size = 20.dp,
                     modifier = Modifier.padding(4.dp).clickable {
                         onPressHelpCenter()
                     })
                 Box(
                     modifier = Modifier
                         .width(0.5.dp)
                         .height(12.dp)
                         .background(Colors.black_20)
                 )
             }
 
             Icon(
                 source = "16_navigation_close_circle",
                 size = 20.dp,
                 modifier = Modifier.padding(4.dp).clickable { onPressClose() }
             )
         }
     }
 }
