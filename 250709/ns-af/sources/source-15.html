 @Composable
 fun VoucherItem(
     voucher: IVoucherBase,
     onClick: () -> Unit? = {},
     modifier: Modifier = Modifier,
     selected: Boolean = false,
     isSelectable: Boolean = false,
     isStackable: Boolean = false,
     onSelect: () -> Unit? = {},
     onCollect: () -> Unit? = {},
     onUseNow: () -> Unit? = {},
     onDisplayed: () -> Unit? = {},
     isCollecting: Boolean = false,
     sourceScreen: PromotionSourceScreen? = null,
 ) {
     val buildInfo = Providers.providesBuildInfo()
     val isJustGenerated =
         voucher.generatedAt != null && voucher.generatedAt!! > TimeUtils.getCurrentTimeMillis() - 3_000L
     val isHighlighted = remember { mutableStateOf(false) }
     LaunchedEffect(isJustGenerated) {
         if (isJustGenerated) {
             isHighlighted.value = true
             delay(4000L)
             isHighlighted.value = false
         }
     }
 
     LaunchedEffect(voucher.ID) {
         onDisplayed()
     }
 
     val errorMessage = when {
         !voucher.extra?.failGiftDesc.isNullOrBlank() -> voucher.extra?.failGiftDesc
         !voucher.giftConditionDesc.isNullOrBlank() -> voucher.giftConditionDesc
         voucher.localType == VoucherLocalType.Invalid
                 || voucher.localType == VoucherLocalType.InvalidGiftMask ->
             stringResource(Res.string.invalidGiftCondition)
 
         else -> null
     }
 
     val isInvalid = !isSelectable || errorMessage != null
 
     val animatedBackgroundColor by animateColorAsState(
         targetValue = if (isHighlighted.value) AppTheme.current.colors.background.selected else AppTheme.current.colors.background.surface
     )
     val animatedBorderColor by animateColorAsState(
         targetValue = if (isHighlighted.value) AppTheme.current.colors.secondary else AppTheme.current.colors.background.surface
     )
     val animatedAlpha by animateFloatAsState(
         targetValue = if (isInvalid) 0.6f else 1f,
     )
 
     val shouldShowPaymentMethod = listOf(PromotionSourceScreen.POS, PromotionSourceScreen.POS_SEARCH)
             .contains(sourceScreen) && !voucher.extra?.paymentMethod.isNullOrBlank()
 
     Surface(
         modifier = modifier
             .clickable(
                 enabled = true,
                 onClick = { onClick() },
                 interactionSource = remember { MutableInteractionSource() },
                 indication = null,
             ),
         shape = RoundedCornerShape(Radius.M)
     ) {
         Column(
             modifier = Modifier
                 .fillMaxSize()
                 .conditional(errorMessage != null) {
                     background(Colors.yellow_09)
                 }
         ) {
             Row(
                 modifier = Modifier
                     .fillMaxSize()
                     .background(
                         color = animatedBackgroundColor,
                         shape = RoundedCornerShape(Radius.M)
                     )
                     .border(
                         width = 1.dp,
                         color = animatedBorderColor,
                         shape = RoundedCornerShape(Radius.M)
                     )
                     .padding(Spacing.XS),
                 verticalAlignment = Alignment.CenterVertically,
                 horizontalArrangement = Arrangement.spacedBy(Spacing.S),
             ) {
                 Row(
                     modifier = Modifier.weight(1f).height(IntrinsicSize.Max)
                         .alpha(animatedAlpha),
                     horizontalArrangement = Arrangement.spacedBy(Spacing.S),
                 ) {
                     Column(
                         modifier = Modifier
                             .scaledWidth(92f)
                             .fillMaxHeight()
                             .background(Colors.black_02, RoundedCornerShape(Radius.S))
                             .padding(
                                 vertical = Spacing.XS,
                                 horizontal = Spacing.S + 2.dp
                             ),
                         horizontalAlignment = Alignment.CenterHorizontally,
                         verticalArrangement = Arrangement.Center,
                     ) {
                         Image(
                             source = rememberAsyncImagePainter(voucher.icon),
                             modifier = Modifier
                                 .size(Spacing.XXL)
                                 .clip(RoundedCornerShape(Spacing.L))
                         )
                         Text(
                             text = voucher.extra?.voucherServiceName
                                 ?: stringResource(Res.string.choosePromotionScreenTitle),
                             color = Colors.black_17,
                             textAlign = TextAlign.Center,
                             style = Typography.headerXsSemibold,
                             modifier = Modifier.padding(top = Spacing.XS),
                             maxLines = 2,
                             overflow = TextOverflow.Ellipsis,
                         )
                     }
 
                     Column(
                         modifier = Modifier
                             .weight(1f)
                             .heightIn(min = 80.dp)
                             .fillMaxWidth()
                             .padding(vertical = Spacing.XS),
                         verticalArrangement = Arrangement.spacedBy(Spacing.XS),
                     ) {
                         if (voucher.prefixName?.isNotEmpty() == true) {
                             Text(
                                 text = voucher.prefixName ?: "",
                                 style = Typography.headerSSemibold,
                                 color = AppTheme.current.colors.text.default,
                                 maxLines = 2,
                                 overflow = TextOverflow.Ellipsis,
                             )
                         }
 
                         if (voucher.suffixName?.isNotEmpty() == true) {
                             Text(
                                 text = voucher.suffixName ?: "",
                                 style = Typography.descriptionDefaultRegular,
                                 color = AppTheme.current.colors.text.hint,
                                 maxLines = 2,
                                 overflow = TextOverflow.Ellipsis,
                             )
                         }
 
                         if (isStackable && !shouldShowPaymentMethod) {
                             VoucherMetaBadge(
                                 modifier = Modifier.padding(top = Spacing.XXS),
                                 label = if (buildInfo.isDebug) {
                                     voucher.groups?.joinToString(", ") ?: ""
                                 } else {
                                     stringResource(Res.string.stackingVoucherTag)
                                 }
                             )
                         }
 
                         if (shouldShowPaymentMethod) {
                             VoucherMetaBadge(
                                 modifier = Modifier.padding(top = Spacing.XXS),
                                 label = voucher.extra?.paymentMethod ?: ""
                             )
                         }
                     }
                 }
 
                 Box(
                     modifier = Modifier.padding(end = Spacing.M).alpha(animatedAlpha)
                 ) {
                     voucher.localType.let { voucherLocalType ->
                         if (voucherLocalType == VoucherLocalType.GiftMask || voucherLocalType == VoucherLocalType.InvalidGiftMask) {
                             Button(
                                 title = stringResource(isCollecting.let {
                                     if (it) Res.string.collectingVoucherLabel else Res.string.collectVoucherLabel
                                 }),
                                 onClick = { onCollect() },
                                 size = Size.SMALL,
                                 type = if (isCollecting) {
                                     ButtonType.SECONDARY
                                 } else {
                                     ButtonType.OUTLINE
                                 },
                                 isFull = false,
                             )
                         } else {
                             if (sourceScreen == PromotionSourceScreen.POS_SEARCH) {
                                 if (!isInvalid)
                                     Button(
                                         title = stringResource(Res.string.txtUseNow),
                                         onClick = { onUseNow() },
                                         size = Size.SMALL,
                                         type = if (isInvalid) {
                                             ButtonType.DISABLED
                                         } else {
                                             ButtonType.TEXT
                                         },
                                         isFull = false,
                                         modifier = Modifier.offset(x = Spacing.S)
                                     )
                             } else {
                                 CheckBox(
                                     checked = selected,
                                     onCheckedChange = { onSelect() },
                                     disabled = isInvalid,
                                 )
                             }
                         }
                     }
                 }
             }
 
             if (errorMessage != null) {
                 Row(
                     modifier = Modifier
                         .fillMaxWidth()
                         .background(
                             Colors.yellow_09,
                             shape = RoundedCornerShape(bottomStart = Radius.M, bottomEnd = Radius.M)
                         )
                         .padding(vertical = Spacing.XS, horizontal = Spacing.S),
                     verticalAlignment = Alignment.CenterVertically,
                     horizontalArrangement = Arrangement.spacedBy(
                         space = Spacing.XS,
                         alignment = Alignment.Start
                     ),
                 ) {
                     Image(
                         source = rememberAsyncImagePainter(Images.ic_warning),
                         modifier = Modifier
                             .size(size = Spacing.L),
                     )
                     Text(
                         text = errorMessage,
                         style = Typography.descriptionXsRegular,
                         maxLines = 1,
                         overflow = TextOverflow.Ellipsis,
                     )
                 }
             }
         }
     }
 }
