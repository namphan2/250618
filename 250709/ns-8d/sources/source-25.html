 @Composable
 fun NextBestActionBlockV2(
     dispatcher: (TransactionResultAction) -> Unit,
     nextBestActionList: List<NextBestActionItem> = listOf(),
     gamification: GamificationResponse,
     modifier: Modifier = Modifier,
     show: Boolean = true
 ) {
     var visibleContent by remember { mutableStateOf(false) }
     val density = LocalDensity.current
 
     LaunchedEffect(gamification, nextBestActionList) {
         if ((gamification.isNotEmpty || nextBestActionList.isNotEmpty()) && !visibleContent) {
             visibleContent = true
         }
     }
 
     val scrollState = rememberScrollState()
     AnimatedVisibility(visible = visibleContent && show, enter = slideInHorizontally {
         with(density) { 500.dp.roundToPx() }
     } + fadeIn(
         initialAlpha = 0.3f
     ), modifier = modifier) {
         Row(
             Modifier.horizontalScroll(scrollState),
             horizontalArrangement = Arrangement.spacedBy(Spacing.M)
         ) {
             nextBestActionList.forEach {
                 NextBestActionItemV2(title = it.title ?: "",
                     subtitle = it.body ?: "",
                     imageUrl = it.icon ?: "",
                     onPress = {
                         dispatcher.invoke(
                             TransactionResultAction.OnPressBestAction(
                                 it
                             )
                         )
                     })
             }
             gamification.noCoinResult?.onEach {
                 NextBestActionItemV2(title = it.title ?: "",
                     subtitle = it.body ?: "",
                     imageUrl = it.iconUrl ?: "",
                     onPress = {
                         dispatcher.invoke(
                             TransactionResultAction.OnPressGamificationCard(
                                 it
                             )
                         )
                     })
             }
         }
     }
 }
