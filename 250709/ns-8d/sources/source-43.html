 package vn.momo.compose.payment.payx.presentation.view.components
 
 import androidx.compose.ui.geometry.CornerRadius
 import androidx.compose.ui.geometry.Offset
 import androidx.compose.ui.geometry.Rect
 import androidx.compose.ui.geometry.RoundRect
 import androidx.compose.ui.geometry.Size
 import androidx.compose.ui.graphics.Outline
 import androidx.compose.ui.graphics.Path
 import androidx.compose.ui.graphics.PathOperation
 import androidx.compose.ui.graphics.Shape
 import androidx.compose.ui.unit.Density
 import androidx.compose.ui.unit.Dp
 import androidx.compose.ui.unit.LayoutDirection
 import kotlin.math.round
 
 internal class TransactionTicketShape(
     private val offsetY: Float?,
     private val offsetX: Float?,
     private val offsetTopDp: Dp,
     private val cornerRadiusDp: Dp,
     private val dotSizeDp: Dp,
     private val statusSizeDp: Dp,
     private val trapezoidButtonWidthDp: Dp,
     private val trapezoidButtonHeightDp: Dp,
 ) : Shape {
 
     override fun createOutline(
         size: Size,
         layoutDirection: LayoutDirection,
         density: Density,
     ) = Outline.Generic(run path@{
         val cornerRadius = with(density) { cornerRadiusDp.toPx() }
         val dotSize = with(density) { dotSizeDp.toPx() }
         val offsetTop = with(density) { offsetTopDp.toPx() }
         val trapezoidButtonWidth = with(density) { trapezoidButtonWidthDp.toPx() }
         val trapezoidButtonHeight = with(density) { trapezoidButtonHeightDp.toPx() }
         val statusSize = with(density) { statusSizeDp.toPx() }
 
         val rectSize = Size(size.width, size.height - trapezoidButtonHeight - offsetTop)
 
         val rectOffset = Offset(0f, offsetTop)
 
         val rect = Rect(rectOffset, rectSize)
 
         val mainPath = Path().apply {
             addRoundRect(RoundRect(rect, CornerRadius(cornerRadius)))
         }
 
         var resultPath: Path = mainPath
         if (offsetY != null) {
             val cutoutPath =
                 Path().apply {
                     val circleSize = Size(dotSize, dotSize)
                     val visiblePart = 0.5f
                     val leftOval = Rect(
                         offset = Offset(
                             x = 0 - circleSize.width * (1 - visiblePart),
                             y = offsetY - circleSize.height / 2
                         ), size = circleSize
                     )
                     val rightOval = Rect(
                         offset = Offset(
                             x = rect.width - circleSize.width * visiblePart,
                             y = offsetY - circleSize.height / 2
                         ), size = circleSize
                     )
                     addOval(leftOval)
                     addOval(rightOval)
                 }
 
             resultPath = Path().apply {
                 op(mainPath, cutoutPath, PathOperation.Difference)
                 close()
             }
         }
 
         if (offsetX == null) return@path resultPath
 
         val buttonPath = Path().apply {
             val trapezoidBottomWidth = round(trapezoidButtonWidth * 0.7f) // Narrower at the bottom
 
             val trapezoidTopLeft =
                 Offset(
                     offsetX - round(trapezoidButtonWidth / 2f),
                     size.height - trapezoidButtonHeight
                 )
             val trapezoidTopRight =
                 Offset(
                     offsetX + round(trapezoidButtonWidth / 2f),
                     size.height - trapezoidButtonHeight
                 )
             val trapezoidBottomLeft =
                 Offset(round(offsetX - trapezoidBottomWidth / 2f), size.height)
             val trapezoidBottomRight =
                 Offset(round(offsetX + trapezoidBottomWidth / 2f), size.height)
 
             val trapezoidCenterRight = Offset(
                 round((trapezoidBottomRight.x + trapezoidTopRight.x) / 2),
                 round((trapezoidBottomRight.y + trapezoidTopRight.y) / 2)
             )
             val trapezoidCenterLeft = Offset(
                 round((trapezoidBottomLeft.x + trapezoidTopLeft.x) / 2),
                 round((trapezoidBottomLeft.y + trapezoidTopLeft.y) / 2)
             )
 
             moveTo(trapezoidTopLeft.x, trapezoidTopLeft.y)
             lineTo(trapezoidTopRight.x, trapezoidTopRight.y)
             lineTo(trapezoidCenterRight.x, trapezoidCenterRight.y)
 
             quadraticTo(
                 trapezoidBottomRight.x,
                 trapezoidBottomRight.y,
                 trapezoidBottomRight.x - round(trapezoidBottomWidth / 8),
                 trapezoidBottomRight.y
             )
 
             lineTo(
                 trapezoidBottomLeft.x + round(trapezoidBottomWidth / 8), trapezoidBottomLeft.y
             )
 
             moveTo(trapezoidTopLeft.x, trapezoidTopLeft.y)
             lineTo(trapezoidCenterLeft.x, trapezoidCenterLeft.y)
 
             quadraticTo(
                 trapezoidBottomLeft.x,
                 trapezoidBottomLeft.y,
                 trapezoidBottomLeft.x + round(trapezoidBottomWidth / 8),
                 trapezoidBottomLeft.y
             )
 
             close()
 
             addOval(Rect(Offset(offsetX - statusSize / 2, 0f), Size(statusSize, statusSize)))
         }
         resultPath = Path().apply {
             op(resultPath, buttonPath, PathOperation.Union)
         }
         return@path resultPath
     })
 }
