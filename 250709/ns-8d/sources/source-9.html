 @Composable
 fun CheckBox(
     modifier: Modifier = Modifier,
     checked: Boolean = false,
     onCheckedChange: () -> Unit,
     value: String = "",
     values: List<PayXText>? = emptyList(),
     isDisabled: Boolean = false,
     onPressText: ((cta: PayXCta?) -> Unit)? = null
 ) {
     val (borderColor, backgroundColor) = when {
         isDisabled && checked -> AppTheme.current.colors.background.tonal to AppTheme.current.colors.background.tonal
         isDisabled -> AppTheme.current.colors.border.disable to AppTheme.current.colors.background.surface
         checked -> AppTheme.current.colors.primary to AppTheme.current.colors.primary
         else -> AppTheme.current.colors.text.default to AppTheme.current.colors.background.surface
     }
 
     Row(
         modifier = modifier.noFeedbackClickable(enabled = !isDisabled, onClick = onCheckedChange)
             .alpha(if (isDisabled) 0.6f else 1f),
         horizontalArrangement = Arrangement.spacedBy(Spacing.S)
     ) {
         Box(
             modifier = Modifier.size(20.dp)
                 .background(backgroundColor, RoundedCornerShape(Radius.XS))
                 .border(Spacing.XXS, borderColor, RoundedCornerShape(Radius.XS))
         ) {
             if (checked) {
                 Icon(
                     source = "ic_checked",
                     size = 20.dp,
                     color = Colors.black_01
                 )
             }
         }
         if (values?.size!! > 0) {
             HyperText(values, 2, onPressText)
         } else {
             Text(
                 style = Typography.bodyDefaultRegular,
                 text = value,
             )
         }
     }
 }
