 @Composable
 fun SofCardItem(
     sof: SOFItem,
     isHideBalance: Boolean,
     isSelected: Boolean,
     onPress: () -> Unit = {},
     onPressCTA: () -> Unit = {}
 ) {
     fun getAccessibilityLabel(): String {
         val bankNameSuffix = when (sof.moneySource) {
             EnumMoneySource.Bank -> sof.fullName
             EnumMoneySource.Card -> sof.info.bankName ?: ""
             else -> ""
         }
         val cardSchemeSuffix =
             if (sof.moneySource == EnumMoneySource.Napas) "NAPAS" else sof.info.cardSchema.orEmpty()
         val payLaterInstallmentSuffix = if (sof.isInstallment) "installment" else ""
         val listKey = listOf(
             "btn_sof",
             sof.moneySource.id.toString(),
             bankNameSuffix,
             cardSchemeSuffix,
             payLaterInstallmentSuffix
         ).filter { it.isNotEmpty() }
         return listKey.joinToString("_").lowercase()
     }
 
     @Composable
     fun RightSofCTA() {
         if (sof.action?.isShow != true) {
             return
         }
         val sofAction = sof.action
 
         Row(
             horizontalArrangement = Arrangement.spacedBy(8.dp),
             verticalAlignment = Alignment.CenterVertically
         ) {
             Text(
                 text = if (isHideBalance && sofAction.title.isCurrency()) "******" else sofAction.title,
                 style = Typography.actionXsBold,
                 color = if (sofAction.action is SofActionType.Select) Colors.black_17 else Colors.pink_05,
                 modifier = Modifier.noFeedbackClickable(onClick = onPressCTA).alpha(1f),
             )
 
             if (sof.isInstallment && isSelected) {
                 Icon(
                     source = "ic_check_24",
                     size = 24.dp,
                     color = AppTheme.current.colors.success.primary,
                 )
             }
             when (sofAction.action) {
                 SofActionType.HaveNextStep -> Icon(
                     source = "arrow_chevron_right_small", size = 24.dp, color = Colors.black_12
                 )
 
                 else -> return
             }
         }
 
     }
 
     ActionCard(title = sof.fullName.takeIf { it.isNotEmpty() } ?: sof.shortName,
         description = sof.action?.description?.takeIf { it.isNotEmpty() }
             ?: sof.fullDesc.takeIf { it.isNotEmpty() } ?: sof.shortDesc,
         iconLeft = {
             Image(
                 source = sof.logo ?: "", modifier = Modifier.size(32.dp)
             )
         },
         showIconNotificationInvestment = sof.moneySource == EnumMoneySource.Investment,
         iconRight = { RightSofCTA() },
         cardStyle = ActionCardStyle().copy(
             border = if (isSelected && sof.pressable) BorderStroke(
                 2.dp, Colors.pink_05
             ) else null,
             alpha = if (sof.pressable) 1f else 0.5f,
             backgroundColor = if (isSelected && sof.pressable) Colors.pink_11 else Colors.black_01
         ),
         modifier = Modifier.noFeedbackClickable(enabled = sof.pressable, onClick = onPress),
         titleModifier = Modifier.semantics { contentDescription = getAccessibilityLabel() }
 
     )
 }
