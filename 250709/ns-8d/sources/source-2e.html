 @Composable
 fun RecommendMoneyPay(
     backgroundColor: Color,
     btnColor: Color,
     overlayColor: Color,
     logo: String,
     title: String,
     isDisableBtn: Boolean,
     btnIconName: String? = null,
     btnTitle: String,
     titleFont: String,
     buttonFont: String,
     onPressRecommend: () -> Unit
 ) {
     var rowWidth by remember { mutableStateOf(0) }
     val isShowButton = btnIconName?.isNotEmpty() == true || btnTitle.isNotEmpty()
     Box(
         modifier = Modifier
             .fillMaxWidth()
             .background(color = backgroundColor, shape = RoundedCornerShape(12.dp))
             .noFeedbackClickable(enabled = !isDisableBtn) { onPressRecommend() }
     ) {
         Image(
             source = Images.bg_banner_recommend_money_pay,
             modifier = Modifier.matchParentSize(),
             options = Options(
                 alignment = Alignment.BottomEnd,
                 colorFilter = ColorFilter.tint(overlayColor),
                 contentScale = ContentScale.Fit,
             )
         )
         Row(
             modifier = Modifier
                 .fillMaxWidth()
                 .onGloballyPositioned { coor ->
                     rowWidth = coor.size.width
                 }
                 .padding(8.dp),
             verticalAlignment = Alignment.CenterVertically,
         ) {
             Row(
                 verticalAlignment = Alignment.CenterVertically,
                 modifier = Modifier.weight(1f).then(if(isShowButton) Modifier else Modifier.fillMaxWidth())
             ) {
                 logo.takeIf { it.isNotEmpty() }?.let {
                     Image(
                         source = it,
                         modifier = Modifier.size(24.dp)
                     )
                 }
                 Text(
                     text = title,
                     style = titleFont.toTypography() ?: Typography.bodyDefaultRegular,
                     modifier = Modifier.padding(start = 8.dp, end = 8.dp),
                     maxLines = 4,
                     overflow = TextOverflow.Ellipsis
                 )
             }
             if (isShowButton) {
                 Row(
                     verticalAlignment = Alignment.CenterVertically,
                     horizontalArrangement = Arrangement.End,
                     modifier = Modifier.widthIn(max = (0.3f * rowWidth).dp)
                 ) {
                     btnIconName.takeIf { it?.isNotEmpty() == true }?.let {
                         MMImage(
                             painter = getResource(it),
                             contentDescription = null,
                             colorFilter = ColorFilter.tint(btnColor),
                             width = 20.dp,
                             height = 20.dp,
                         )
                         Spacer(modifier = Modifier.width(8.dp))
                     }
                     btnTitle.takeIf { it.isNotEmpty() }?.let {
                         Text(
                             text = it,
                             style = buttonFont.toTypography() ?: Typography.headerSSemibold,
                             color = btnColor,
                             maxLines = 1,
                             overflow = TextOverflow.Ellipsis
                         )
                     }
                 }
             }
         }
     }
 }
