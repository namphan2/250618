 class DefaultRewardRepository(
     private val orderExtraRepository: OrderExtraRepository
 ) : RewardRepository {
     private val reward =
         MutableStateFlow<PayXResult<RewardConfig?, Throwable>>(Fail(error = Throwable()))
     override val rewardConfig: StateFlow<PayXResult<RewardConfig?, Throwable>> =
         reward.asStateFlow()
 
     override suspend fun getConfig(sofList: List<SOFItem>) {
         coroutineScope {
             orderExtraRepository.extraConfig
                 .onEach { result ->
                     when (result) {
                         is PayXResult.Loading, null -> reward.update { PayXResult.Loading }
                         is Success -> {
                             val dto = result.data?.mission
                             reward.update {
                                 Success(
                                     RewardConfig(
                                         rewardBySof = getCoinsWithSof(dto?.rewardInfo),
                                         missionSummaryUi = generateDynamicUi(dto?.voucherInfo?.voucherInlineMessage),
                                         missionIds = dto?.voucherInfo?.vouchers?.mapNotNull { it.giftTypeId }
                                             ?: emptyList()
                                     )
                                 )
                             }
                         }
 
                         is Fail -> reward.update { Fail(error = result.error) }
                     }
                 }
                 .takeWhile { result -> result !is Success && result !is Fail }
                 .collect {}
         }
 
     }
 
     private fun getCoinsWithSof(
         rewards: List<RewardInfoDTO?>?
     ): Map<MoneySource, Int> {
         val remoteReward = rewards?.filterNotNull()
         if (remoteReward?.isNotEmpty() != true) return emptyMap()
         return remoteReward
             .mapNotNull { reward ->
                 val moneySource = reward.moneySource?.let { EnumMoneySource.from(it) }
                 val coins = reward.totalRewardCoins ?: 0
                 if (moneySource != null && coins > 0) {
                     moneySource to coins
                 } else null
 
             }
             .toMap()
     }
 
     private fun generateDynamicUi(json: JsonElement? = null): List<PayXComponent> {
         return when (json) {
             is JsonArray -> json.mapNotNull { element ->
                 val dto =
                     TextDTO.fromJsonElement(element)?.takeIf { it.value?.isNotEmpty() == true }
                         ?: ChipJson.fromJsonElement(element)
                 when (dto) {
                     is TextDTO -> dto.toModel(
                         default = PayXText(
                             color = Colors.black_17,
                             style = Typography.descriptionDefaultRegular,
                             numberOfLine = 1
                         )
                     )
 
                     is ChipDTO -> dto.toModel(
                         default = PayXChip(
                             color = Colors.pink_10, icon = null, label = PayXText(
                                 color = Colors.pink_03,
                                 style = Typography.labelSMedium,
                                 numberOfLine = 1
                             )
                         )
                     )
 
                     else -> null
                 }
             }
 
             is JsonPrimitive -> json.contentOrNull?.takeIf { it.isNotBlank() }?.let {
                 listOf(
                     PayXText(
                         text = it,
                         color = Colors.black_17,
                         style = Typography.bodyDefaultRegular,
                         numberOfLine = 1
                     )
                 )
             } ?: emptyList()
 
             else -> emptyList()
         }
     }
