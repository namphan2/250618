 import vn.momo.compose.payment.payx.data.dataSource.remote.extraTransaction.OrderExtraRemote
 import vn.momo.compose.payment.payx.domain.entities.PayXResult
 import vn.momo.compose.payment.payx.domain.repositories.orderExtra.OrderExtraRepository
 
 class DefaultOrderExtraRepository(
     private val dataSource: OrderExtraRemote
 ) : OrderExtraRepository {
     private val config =
         MutableStateFlow<PayXResult<OrderExtraConfigResponseDTO?, Throwable>?>(null)
     override val extraConfig: StateFlow<PayXResult<OrderExtraConfigResponseDTO?, Throwable>?> =
         config.asStateFlow()
 
     private var updateJob: Job? = null
 
     override suspend fun getCheckoutConfig() {
         coroutineScope {
             dataSource.getConfig().onEach { response ->
                 config.update { PayXResult.Success(data = response) }
             }.catch { exc ->
                 config.update { PayXResult.Fail(data = null, exc) }
             }.launchIn(this)
         }
 
     }
 
     override suspend fun updateConfig() {
         coroutineScope {
             updateJob?.cancel()
             config.update { PayXResult.Loading }
             updateJob = dataSource.updateConfig().onEach { response ->
                 config.update { PayXResult.Success(data = response) }
             }.catch { exc ->
                 config.update { PayXResult.Fail(data = null, exc) }
             }.launchIn(this)
         }
     }
 }
