 object PayXActionHandler {
     suspend operator fun invoke(action: PayXAction?): OverlayEffect? {
         val navigator: INavigator = KoinHelper.getInjected()
         return when (action) {
             PayXAction.AddBank -> OverlayEffect.AddSof
 
             PayXAction.GoToCashIn -> OverlayEffect.CashIn
 
             PayXAction.GoHome -> {
                 navigator.dismissAll()
                 null
             }
 
             is PayXAction.OpenDeeplink -> {
                 navigator.openUrl(action.url)
                 null
             }
 
             is PayXAction.OpenPhoneDialer -> {
                 navigator.openPhoneDial(action.phoneNumber)
                 null
             }
 
             PayXAction.OpenPopUpSof -> OverlayEffect.OpenSof
 
             is PayXAction.OpenWeb -> {
                 navigator.openAppBrowser(
                     title = action.webTitle,
                     url = action.refId
                 )
                 null
             }
 
             is PayXAction.StartFeature -> {
                 navigator.navigateFeatureCode(
                     action.refId, action.refParams
                         ?: emptyJsonObject(),
                     ErrorPopupCallback.serializer()
                 ).map {
                     return@map shouldForceReInit(action.refId, action.refParams, it)
                 }
                     .firstOrNull()
                     ?.let {
                         if (it) OverlayEffect.ReinitTransaction else null
                     }
             }
 
             is PayXAction.TabBarTransaction -> {
                 goToTransactionHistoryTab().launchIn(CoroutineScope(Dispatchers.IO))
                 null
             }
 
             else -> null
         }
     }
 }
