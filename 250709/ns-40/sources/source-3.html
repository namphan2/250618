 @Composable
 fun SplitBillBottomSheet(
     items: List<SplitBillItem>,
     viewModel: SplitBillViewModel
 ) {
     val scrollState = rememberScrollState()
     Column(
         modifier = Modifier
             .verticalScroll(scrollState)
             .fillMaxSize()
     ) {
         items.mapIndexed { idx, item ->
             SplitBillComponent(item) { type ->
                 viewModel.onContinue(type)
             }
             if (idx < items.size - 1) {
                 Spacer(modifier = Modifier.height(Spacing.L))
             }
         }
     }
 }
