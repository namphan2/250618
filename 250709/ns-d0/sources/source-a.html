 @Composable
 fun TransferSofList(
     list: PersistentList<Sof>,
     fee: Int?,
     isNoSof: Boolean,
     isHideBalance: Boolean,
     selected: SOFItem?,
     number: Int = 2,
     expanded: Boolean = false,
     onPressSof: (SOFItem) -> Unit,
     onPressCta: (SOFItem) -> Unit,
     isAllowAddSOF: Boolean = true,
     onAddSof: () -> Unit = {},
     iconBanks: PersistentList<String>,
     ctaList: PersistentList<SofCTAItem>,
     onPressNotice: (SofCTAItem) -> Unit
 ) {
     val displayedItems = list.take(number)
     val hiddenItems = list.drop(number)
     Column(verticalArrangement = scaledSpacedBy(8f)) {
         if (isNoSof) {
             MessageInformation(
                 title = stringResource(Res.string.noSof),
                 color = Colors.blue_07,
                 backgroundColor = Colors.blue_10
             )
         }
 
         ctaList.forEach { cta ->
             SofCard.CtaItem(item = cta.action, onPress = { onPressNotice(cta) })
         }
 
         displayedItems.forEach { sof ->
             P2pSofItemView(
                 sof,
                 fee,
                 isHideBalance = isHideBalance,
                 selected = selected,
                 onPress = onPressSof,
                 onPressCta = onPressCta
             )
         }
         if (list.size <= number) {
             key("addBank") {
                 SofCard.AddBank(
                     title = stringResource(Res.string.addBankDescP2p),
                     onPress = onAddSof,
                     iconBanks = iconBanks
                 )
             }
         }
         if (hiddenItems.isNotEmpty()) {
             PayXAnimatedVisibility(visible = expanded) {
                 Column(verticalArrangement = scaledSpacedBy(8f)) {
                     hiddenItems.forEachIndexed { idx, sof ->
                         P2pSofItemView(
                             sof,
                             fee,
                             isHideBalance = isHideBalance,
                             selected = selected,
                             onPress = onPressSof,
                             onPressCta = onPressCta
                         )
                     }
                     if (isAllowAddSOF) {
                         key("addBank") {
                             SofCard.AddBank(
                                 title = stringResource(Res.string.addBankDescP2p),
                                 onPress = onAddSof,
                                 iconBanks = iconBanks
                             )
                         }
                     }
                 }
             }
         }
     }
 }
 
 @Composable
 private fun P2pSofItemView(
     sof: Sof,
     fee: Int?,
     isHideBalance: Boolean,
     selected: SOFItem?,
     onPress: (SOFItem) -> Unit,
     onPressCta: (SOFItem) -> Unit,
 ) {
     when (sof) {
         is SOFItem -> key(sof.uId) {
             SofCard.Default(
                 sof = sof,
                 isHideBalance = isHideBalance,
                 isSelected = sof.isMatchSource(
                     selected?.sourceToken,
                     selected?.ID
                 ),
                 trailing = buildTrailing(fee?.toDouble(), sof),
                 onPress = { onPress(sof) },
                 onPressCTA = { onPressCta(sof) }
             )
         }
 
         is SOFGroup -> key(sof.parent.uId) {
             SofGroupView(
                 group = sof,
                 selected,
                 onSelect = onPress,
                 isHideBalance = isHideBalance,
                 parentTrail = buildTrailing(fee?.toDouble(), sof.parent)
             )
         }
     }
 }
 
 @Composable
 fun buildTrailing(fee: Double?, sof: SOFItem): (@Composable () -> Unit)? {
     return when {
         sof.isSource(EnumMoneySource.Investment) -> null
         else -> fee?.takeIf { it > 0 }
             ?.formatCurrency()
             ?.let { formatted ->
                 {
                     Text(
                         text = stringResource(Res.string.transferFee)
                             .replace("{amount}", formatted),
                         style = Typography.descriptionXsRegular
                     )
                 }
             }
     }
