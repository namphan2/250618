 package vn.momo.compose.payment.singleBill.presentation.viewModel.reducer
 
 import vn.momo.compose.payment.base.domain.entities.transaction.TotalAmount
 import vn.momo.compose.payment.base.presentation.viewModel.reducer.Reducer
 import vn.momo.compose.payment.payx.domain.entities.InvoiceItem
 import vn.momo.compose.payment.payx.domain.entities.PromotionData
 import vn.momo.compose.payment.singleBill.domain.model.SBTransaction
 import vn.momo.core.features.annotation.Immutable
 
 class SingleBillTransactionReducer :
     Reducer<SingleBillTransactionReducer.State, SingleBillTransactionReducer.Action, SingleBillTransactionReducer.Effect> {
     @Immutable
     data class State(
         val isLoading: Boolean = true,
         val enableConfirm: Boolean = false,
         val invoice: List<InvoiceItem> = emptyList(),
         val amount: TotalAmount = TotalAmount(),
         val warningMsg: String = "",
         val promotion: PromotionData = PromotionData(),
         val tranId: String? = "",
     ) : Reducer.State
 
     @Immutable
     sealed class Action : Reducer.Action {
         data class SetLoading(val value: Boolean) : Action()
         data class SetTransaction(val data: SBTransaction) : Action()
         data class SetEnableConfirm(val value: Boolean) : Action()
         data class SetShowTid(val value: Boolean) : Action()
     }
 
     @Immutable
     class Effect() : Reducer.Effect
 
     override fun reduce(prev: State, action: Action): Pair<State, Effect?> {
         return when (action) {
             is Action.SetTransaction -> prev.copy(
                 isLoading = false,
                 enableConfirm = action.data.enableConfirm,
                 invoice = listOf(action.data.invoice),
                 amount = action.data.totalAmount,
                 warningMsg = action.data.warningMsg,
                 promotion = action.data.promotion ?: PromotionData(),
                 tranId = action.data.tranId,
             ) to null
 
             is Action.SetEnableConfirm -> prev.copy(enableConfirm = action.value) to null
             is Action.SetShowTid -> prev to null
             is Action.SetLoading -> prev.copy(isLoading = action.value) to null
         }
     }
 }
