 class TransactionResultLoadingVM(
     private val koinScope: Scope,
     private val handleNavigateTabUseCase: HandleNavigateTabUseCase,
     private val trLoadingRepository: ITransactionResultLoadingRepository,
     private val app2AppRepository: IApp2AppRepository,
     override val navigator: INavigator,
     override val logger: ILogger,
     private val appId: String?,
     private val miniAppWidgetId: String?,
 ) : ViewModel(), IPaymentVM<TransactionResultLoadingAction>, CommonMixin {
     override val logName: String = "[TR]_TransactionResultLoadingVM"
 
     val state = PState(viewModelScope, TransactionResultLoadingState())
     private var params: TransactionResultLoadingParams = TransactionResultLoadingParams()
     private val trStatusSlice by koinScope.inject<TransactionResultStatusSlice> {
         parametersOf(
             viewModelScope
         )
     }
 
     fun onCreate(params: Map<String, Any?>) {
         logInfo("onCreate-params", params)
         this.params = trLoadingRepository.formatParams(params)
         handleSyncOTPInfo()
         handleSyncBank()
     }
 
     override fun onCleared() {
         trStatusSlice.clearJob()
         super.onCleared()
     }
 
     override fun dispatch(action: TransactionResultLoadingAction) {
         when (action) {
             TransactionResultLoadingAction.Close -> navigator.dismiss()
             TransactionResultLoadingAction.HidePopup -> hidePopup()
             TransactionResultLoadingAction.ShowPopup -> showPopup()
             TransactionResultLoadingAction.NavigateToTransHis -> navigateTransHis()
             else -> {}
         }
     }
 
     private fun hidePopup() {
         state.update {
             it.copy(showPopup = false)
         }
     }
 
     private fun showPopup() {
         state.update {
             it.copy(showPopup = true)
         }
     }
 
     private fun updateMode(value: TRLoadingMode) {
         state.update {
             it.copy(
                 mode = value
             )
         }
     }
 
     private fun updateBankData(value: TRLoadingBankData) {
         state.update {
             it.copy(bankData = value)
         }
     }
 
     private fun navigateTransHis() {
         trStatusSlice.clearJob()
         handleNavigateTabUseCase.invoke("tabbar_transaction", "transaction_history", viewModelScope)
     }
 
     private fun handleSyncBank() {
         trLoadingRepository.getBankData(this.params.bankCode ?: "").onEach {
             logInfo("handleSyncBank", it)
             updateBankData(it)
         }.catch {
             logInfo("handleSyncBank-error", it.message)
         }.launchIn(viewModelScope)
     }
 
     private fun handleSyncOTPInfo() {
         app2AppRepository.syncOTPInfo(
             this.params.shouldEncryptDeeplinkData,
             this.params.tidBank ?: "",
             this.params.token ?: "",
         ).onEach {
             logInfo("handleSyncOTPInfo", it)
             listenTransactionResult(it.purchaseId, it.tidBank, it.firstTranId, it.tids)
         }.catch {
             logInfo("handleSyncOTPInfo-error", it.message)
             updateMode(TRLoadingMode.GET_OTP_INFO_FAIL)
         }.launchIn(viewModelScope)
     }
 
     private fun listenTransactionResult(
         purchaseId: String?, tidBank: String?, firstTranId: String?, tids: String?
     ) {
         trStatusSlice.effect.onEach {
             when (it) {
                 is TransactionResultStatusReducer.TransactionResultStatusEffect.OnTxnResponseEmitted -> {
                     logInfo("listenTransactionResult", it.txnResponse)
                     handleTransactionResultResponse(it.txnResponse)
                 }
 
                 else -> {
                     // NOTHING
                 }
             }
         }.launchIn(viewModelScope)
         trStatusSlice.observerTransactionStatusMQTT(purchaseId ?: tidBank ?: firstTranId)
         trStatusSlice.getTransactionResultStatusAPI(purchaseId = purchaseId, tidBank = tidBank, tids = tids, subDomain = "api")
     }
 
     private fun handleTransactionResultResponse(txnResponse: TxnResponse?) {
         txnResponse?.let {
             val mode = when {
                 it.isNetworkError -> TRLoadingMode.FAIL.also {
                     logInfo(
                         "handleTransactionResultResponse", "FAIL"
                     )
                 }
 
                 it.isTimeout -> TRLoadingMode.TIMEOUT.also {
                     logInfo(
                         "handleTransactionResultResponse", "TIMEOUT"
                     )
                 }
 
                 it.result == true -> TRLoadingMode.SUCCESS.also {
                     logInfo("handleTransactionResultResponse", "SUCCESS")
                     goToTransactionResult(txnResponse)
                 }
 
                 else -> TRLoadingMode.UNKNOWN.also {
                     logInfo(
                         "handleTransactionResultResponse", "UNKNOWN"
                     )
                 }
             }
             updateMode(mode)
             FirebaseTrace.track(
                 TRLModeStatusUpdated(
                     messageStatus = mode.value,
                     errorCode = txnResponse.errorCode?.toInt(),
                     miniappId = appId
                 )
             )
         }
     }
 
     private fun goToTransactionResult(txnResponse: TxnResponse?) {
         logInfo("goToTransactionResult", "")
         trStatusSlice.clearJob()
         val params = TransactionResultParams(
             response = txnResponse,
             miniAppWidgetId = miniAppWidgetId,
             appId = appId ?: "vn.momo.compose"
         )
         navigator.navigateFeatureCode(
             "transaction_result_revamp_native", ModelConvertUtils.encodeToJson(
                 TransactionResultParams.serializer(), params
             ) as? JsonObject ?: emptyJsonObject()
         )
     }
 }
