 enum class PositionSeparator {
     TOP,
     BOTTOM,
     BOTH,
 }
 
 @Composable
 fun <T> ListSeparatorContainer(
     list: List<T>,
     modifier: Modifier = Modifier,
     itemContent: @Composable (index: Int, item: T) -> Unit,
 ) {
     Column(modifier = modifier) {
         list.forEachIndexed { index, item ->
             if (index < list.lastIndex) ItemSeparatorContainer(1, PositionSeparator.BOTTOM) {
                 itemContent(index, item)
             } else itemContent(index, item)
         }
     }
 }
 
 @Composable
 fun ItemSeparatorContainer(
     weight: Int,
     position: PositionSeparator = PositionSeparator.BOTH,
     content: @Composable () -> Unit,
 ) {
 
     val modifierSeparator = Modifier
         .fillMaxWidth()
         .padding(start=Spacing.XXL)
         .height(weight.dp)
         .background(color = Colors.black_03)
 
     Column {
         if (position == PositionSeparator.TOP || position == PositionSeparator.BOTH)
             Box(modifier = modifierSeparator)
         content()
         if (position == PositionSeparator.BOTTOM || position == PositionSeparator.BOTH)
             Box(modifier = modifierSeparator)
     }
 }
