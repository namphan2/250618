 @Composable
 fun TransferPaymentInfoV2(
     invoices: List<InvoiceItem>,
     headerInfo: HeaderInfo? = null,
 ) {
     val backgroundColor = AppTheme.current.colors.background.surface
 
     invoices.forEach { item ->
         val invoiceInfos = buildList {
             addAll(item.info)
             if (item.visibleRefCode) {
                 add(
                     InvoiceInfo(
                         title = stringResource(Res.string.transaction_id),
                         value = item.transRefCode?.toString() ?: "",
                         key = "tid"
                     )
                 )
             }
         }
         Column(
             modifier = Modifier.fillMaxWidth().clip(
                 shape = RoundedCornerShape(
                     bottomStart = Spacing.M,
                     bottomEnd = Spacing.M,
                     topStart = Spacing.XXS,
                     topEnd = Spacing.XXS
                 ),
             )
         ) {
             ScallopedTop(
                 modifier = Modifier.fillMaxWidth(),
             )
             Column(
                 modifier = Modifier.fillMaxWidth().background(
                     backgroundColor, shape = RoundedCornerShape(
                         bottomStart = Spacing.M,
                         bottomEnd = Spacing.M,
                     )
                 ).scaledPadding(horizontal = 12f),
             ) {
                 PaymentTransferInfoHeader(
                     title = headerInfo?.title,
                     description = headerInfo?.description,
                     image = headerInfo?.image,
                 )
                 invoiceInfos.forEachIndexed { _, item ->
                     PaymentInfoRow(
                         title = item.title,
                         value = item.value,
                         titleTextStyle = item.titleTextStyle,
                         valueTextStyle = item.valueTextStyle,
                     )
                 }
                 Spacer(modifier = Modifier.height(Spacing.XS))
             }
         }
     }
 }
