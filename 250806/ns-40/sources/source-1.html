 @OptIn(ExperimentalMaterial3Api::class)
 @Composable
 fun BaseComposeBottomSheet(
     state: MutableState<Boolean>,
     title: String,
     backgroundColor: Color = AppTheme.current.colors.background.default,
     footerColor: Color = AppTheme.current.colors.background.surface,
     headerColor: Color = AppTheme.current.colors.background.default,
     maxHeightRatio: Double = 0.65,
     onClose: () -> Unit = {},
     content: @Composable () -> Unit,
     footer: (@Composable () -> Unit)? = null,
 ) {
     val screenHeight = getScreenDimensions().screenHeight
 
     val coroutineScope = rememberCoroutineScope()
     val scrollState = rememberScrollState()
     val sheetState =
         rememberModalBottomSheetState(skipPartiallyExpanded = true, confirmValueChange = {
             scrollState.value == 0
         })
     val footerActualColor = if (footer != null) footerColor else backgroundColor
 
     fun requestClose() {
         coroutineScope.launch { sheetState.hide() }.invokeOnCompletion {
             if (!sheetState.isVisible) {
                 state.value = false
                 onClose()
             }
         }
     }
 
     if (state.value) {
         ModalBottomSheet(
             onDismissRequest = { requestClose() },
             sheetState = sheetState,
             dragHandle = {
                 Box(
                     modifier = Modifier.fillMaxWidth().background(headerColor).padding(top = 8.dp, bottom = 4.dp),
                     contentAlignment = Alignment.Center
                 ) {
                     Surface(
                         color = Colors.black_06,
                         shape = MaterialTheme.shapes.extraLarge
                     ) {
                         Box(Modifier.size(width = 40.dp, height = 4.dp))
                     }
                 }
             },
             shape = RoundedCornerShape(topStart = 12.dp, topEnd = 12.dp),
             containerColor = footerActualColor
         ) {
             val density = LocalDensity.current
             val fullHeight = (screenHeight * maxHeightRatio).dp
             val imeInsets = WindowInsets.ime
             val imeBottom = imeInsets.getBottom(LocalDensity.current)
             val keyboardHeightDp = with(density) { imeBottom.toDp() }
             val contentHeight = fullHeight - keyboardHeightDp
             Column(
                 modifier = Modifier
                     .fillMaxWidth()
                     .heightIn(max = fullHeight)
                     .background(backgroundColor)
                     .animateContentSize()
             ) {
                 BottomSheetHeader(
                     modifier = Modifier.padding(Spacing.L),
                     title = title,
                     iconRight = AsyncIconProp(
                         Images.ic_close_fat,
                         size = 14.dp,
                         onPress = { requestClose() }
                     )
                 )
                 Spacer(modifier = Modifier.height(1.dp).fillMaxWidth().background(Colors.black_04))
                 Box(
                     modifier = Modifier
                         .fillMaxWidth()
                         .weight(1f, fill = false)
                         .verticalScroll(scrollState)
                 ) {
                     content()
                 }
 
                 Box(
                     modifier = Modifier
                         .fillMaxWidth()
                         .background(footerActualColor)
                         .padding(
                             start = Spacing.M,
                             end = Spacing.M,
                             top = Spacing.S,
                             bottom = Spacing.XL,
                         )
                 ) {
                     footer?.invoke()
                 }
             }
         }
     }
 }
