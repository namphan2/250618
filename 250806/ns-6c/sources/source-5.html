 class TransactionStatusRepository(
     private val transactionStatusRemoteDataSource: GetTransactionStatusRemoteDataSource,
     private val subject: IObserverService,
     private val logger: ILogger,
 ) : ITransactionStatusRepository {
     val appFeatureRepository by lazy { KoinHelper.getInjected<IAppFeatureConfig>() }
     companion object {
         const val MQTT_UPDATE_PAYMENT_STATUS = "MQTT_UPDATE_PAYMENT_STATUS"
         const val DEFAULT_RETRY_TIMES = 5
         val DEFAULT_DELAY_TIMES = longArrayOf(1000, 1000, 1000, 1500)
         const val GET_STATUS_SUB_DOMAIN_DEFAULT = "api"
     }
 
     override fun getTransactionStatus(
         purchaseId: String?,
         paymentType: String?,
         retryTimes: Int?,
         retryTimeout: Map<String, Double>?,
         subDomain: String?,
         tidBank: String?,
         tids: String?,
     ): Flow<TxnResponse> = flow {
         logger.i("[TR]_[getTransactionStatus]_config-retryTimes : $retryTimes-retryTimeout: $retryTimeout")
         val delayTimes = longArrayOf(0) + (retryTimeout?.values?.map { it.toLong() }?.toLongArray()
             ?: DEFAULT_DELAY_TIMES)
         val type = paymentType ?: "IN_APP"
         var times = retryTimes ?: DEFAULT_RETRY_TIMES
         val nonNullSubDomain = getNonNullSubDomain(subDomain)
         repeat(times) {
             val message = generateMessage(purchaseId, type, nonNullSubDomain, tidBank, tids)
             val response = transactionStatusRemoteDataSource.get(message)
                 .flowOn(Dispatchers.IO)
                 .firstOrNull()
             response?.let {
                 emit(it)
             }
             val delayTime = delayTimes.getOrNull(delayTimes.size - times) ?: delayTimes.first()
             logger.i("[TR]_[getTransactionStatus] retryTimes : $times $delayTime")
             times -= 1
             delay(delayTime)
         }
     }
 
     private suspend fun getNonNullSubDomain(subDomain: String?): String {
         return subDomain
             ?: appFeatureRepository.getById("transaction_result_config", TransactionConfigFeature.serializer())
                 .firstOrNull()?.param?.apiGetStatus?.subDomain?: GET_STATUS_SUB_DOMAIN_DEFAULT
     }
 
 
     private fun generateMessage(
         purchaseId: String?,
         paymentType: String,
         subDomain: String,
         tidBank: String?,
         tids: String?,
     ): GetTransactionStatusRequestDTO {
         val headers = mapOf(
             "payment_type" to paymentType
         )
         val queryParams = buildList {
             if (!purchaseId.isNullOrBlank()) add("purchaseId=$purchaseId")
             if (!tidBank.isNullOrBlank()) add("tidBank=$tidBank")
             if (!tids.isNullOrBlank()) add("tids=$tids")
         }.joinToString("&")
         val path = "payment-platform/v1/transaction/status" +
                 if (queryParams.isNotEmpty()) "?$queryParams" else ""
 
         return GetTransactionStatusRequestDTO(
             path = path,
             subDomain = subDomain,
             headers = headers.toJsonObject()
         )
     }
 
     override fun listenerMQTT(): Flow<TxnResponse?> {
         return subject.observe(MQTT_UPDATE_PAYMENT_STATUS, MomoMsg.serializer())
             .map { TxnResponse(momoMsg = it) }.flowOn(Dispatchers.IO)
     }
 }
