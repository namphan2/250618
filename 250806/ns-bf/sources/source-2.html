 class PromotionVMTracking : IPromotionVMTracking {
     private var initialMountTime: Long = 0L
     private var screenProps: ChoosePromotionScreenProps? = null
 
     /**
      * Initialize tracking with mount time and screen props
      */
     override fun initializeTracking(screenProps: ChoosePromotionScreenProps?) {
         initialMountTime = TimeUtils.getCurrentTimeMillis()
         this.screenProps = screenProps
     }
 
     /**
      * Calculate duration from screen mount time to current time
      */
     override fun calculateDuration(): Long {
         return TimeUtils.getCurrentTimeMillis() - initialMountTime
     }
 
     /**
      * Extract order ID from stored screen props based on flow type
      * For multi-bill: screenProps.purchaseId
      * For single-bill: screenProps.transaction.invoiceId
      */
     override fun extractOrderId(): String? {
         return when (screenProps?.isSingleBillFlow) {
             true -> {
                 screenProps?.transaction?.ID
             }
 
             else -> {
                 screenProps?.purchaseId
             }
         }
     }
 
     /**
      * Create enhanced event params with duration and order ID using stored screen props
      */
     override fun enhanceEventParams(
         baseParams: PromotionEventParams?,
         includeDuration: Boolean?,
         includeOrderId: Boolean?,
     ): PromotionEventParams? {
         if (baseParams == null) return null
 
         return baseParams.copy(
             duration = if (includeDuration != false) calculateDuration() else null,
             orderId = if (includeOrderId != false) extractOrderId() else null
         )
     }
 
     override fun shouldEnhanceParams(screen: PromotionSourceScreen): Boolean {
         return when (screen) {
             PromotionSourceScreen.SINGLE_BILL,
             PromotionSourceScreen.MULTI_BILL -> true
 
             else -> false
         }
     }
 
     override fun trackVoucherScreenDisplayed(sourceScreen: PromotionSourceScreen) {
         val eventParams = enhanceEventParams(
             PROMOTION_TRACER_SERVICE_MAPPING[sourceScreen],
             includeDuration = false
         )
         if (eventParams != null) {
             PromotionFirebaseTrace.trackEvent(
                 eventName = EventName.SCREEN_DISPLAYED,
                 params = eventParams
             )
         }
     }
 
     override fun trackVoucherScreenInteracted(
         sourceScreen: PromotionSourceScreen
     ) {
         val eventParams = enhanceEventParams(PROMOTION_TRACER_SERVICE_MAPPING[sourceScreen])
         if (eventParams != null) {
             PromotionFirebaseTrace.trackEvent(
                 eventName = EventName.SCREEN_INTERACTED,
                 params = eventParams
             )
         }
     }
 
     override fun trackVoucherScreenViewed(sourceScreen: PromotionSourceScreen) {
         val eventParams = enhanceEventParams(
             PROMOTION_TRACER_SERVICE_MAPPING[sourceScreen]?.copy(length = TrackingConst.SCREEN_VIEWED_LENGTH),
             includeDuration = false
         )
         if (eventParams != null) {
             PromotionFirebaseTrace.trackEvent(
                 eventName = EventName.SCREEN_VIEWED,
                 params = eventParams
             )
         }
     }
 
     override fun trackVoucherComponentDisplayed(
         sourceScreen: PromotionSourceScreen,
         giftType: String,
         campaignId: String?,
         slot: Int,
         status: String,
     ) {
         val eventParams = enhanceEventParams(
             PROMOTION_TRACER_SERVICE_MAPPING[sourceScreen]?.copy(
                 componentName = "voucher",
                 componentType = "list",
                 giftType = giftType,
                 campaignId = campaignId,
                 slot = slot,
                 status = status
             ),
             includeDuration = shouldEnhanceParams(sourceScreen)
         )
         if (eventParams != null) {
             PromotionFirebaseTrace.trackEvent(
                 eventName = EventName.COMPONENT_DISPLAYED,
                 params = eventParams
             )
         }
     }
