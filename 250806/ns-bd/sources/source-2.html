 @OptIn(ExperimentalMaterial3Api::class)
 @Composable
 fun ChoosePromotionScreen(
     viewModel: ChoosePromotionVM,
     statusBarHeightPx: Int? = null,
 ) {
     val listState = rememberLazyListState(0)
     val scrollState = listState.toScrollState()
 
     val state by viewModel.state.collectAsState()
     val promoCodeState by viewModel.promoCodeSlice.state.collectAsState()
     val displayVouchers = viewModel.displayVouchersState.collectAsState()
 
     val voucherCoinState by viewModel.voucherCoinSlice.state.collectAsState()
     val voucherCoinQuery = viewModel.voucherCoinSlice.voucherCoinQuery
     val voucherCoinQueryState = voucherCoinQuery.state.collectAsState()
 
     val collectVoucherState by viewModel.collectVoucherSlice.state.collectAsState()
 
     val shouldHidePromoCode = viewModel.params.isSingleBillFlow == true
             && viewModel.params.transaction?.enableOptions?.discount != true
 
     val shouldHideFooter = viewModel.params.isSingleBillFlow == true
             && viewModel.params.transaction?.enableOptions?.voucher != true
 
     val footerButtonProps by viewModel.footerButtonProps.collectAsState()
 
     val voucherCoinBottomSheetState = rememberModalBottomSheetState(
         skipPartiallyExpanded = true,
     )
 
     val voucherCoinList = voucherCoinQueryState.value.pages
         .flatMap { it.records }
         .distinctBy { it.extra?.rewardId } ?: emptyList()
 
     val shouldHideEmptyState =
         voucherCoinState.coinInfo?.numberCoins != null && voucherCoinList.isNotEmpty() && voucherCoinState.coinInfo?.numberCoins!! >= 0
 
     val shouldShowVoucherCoin = viewModel.params.isSingleBillFlow != true
             && voucherCoinList.isNotEmpty()
 
     PromotionComposeContainer(
         statusBarHeightPx = statusBarHeightPx,
         context = MiniAppContext(
             appId = "vn.momo.payment",
             appCode = if (viewModel.params.isSingleBillFlow != true) {
                 "choose_promotion_v2"
             } else {
                 "select_promotion"
             },
             appIcon = Images.ic_momo,
             appName = mapOf(
                 "vi" to "Chọn hoặc nhập mã",
                 "en" to "Choose Voucher or Input Code",
             ),
         )
     ) {
         Screen(
             title = stringResource(Res.string.choosePromotionScreenTitle),
             goBack = { viewModel.screenGoBack() },
             headerType = HeaderType.EXTENDED,
             scrollable = false,
             scrollState = scrollState,
             content = {
                 Column(Modifier.fillMaxSize()) {
                     VoucherList(
                         vouchers = displayVouchers.value,
                         selectedVouchers = state.selectedVouchers,
                         isFetching = state.isFetching,
                         isRefetching = state.isRefetching,
                         shouldDisplayStackingVoucherTag = viewModel.shouldDisplayStackingVoucherTag,
                         hideEmptyState = shouldHideEmptyState,
                         onRefetch = { viewModel.onRefetch() },
                         onFetchNextPage = { viewModel.onFetchNextPage() },
                         onVoucherClick = { voucher, index ->
                             viewModel.onVoucherClick(
                                 voucher,
                                 index
                             )
                         },
                         onVoucherSelect = { voucher, index ->
                             viewModel.onToggleVoucherSelect(
                                 voucher,
                                 index
                             )
                         },
                         onVoucherCollect = { voucher, index ->
                             viewModel.onCollectVoucher(
                                 voucher as VoucherBase,
                                 index
                             )
                         },
                         onVoucherClaim = { voucher, index ->
                             viewModel.onClaimVoucherCoin(
                                 voucher,
                                 index,
                             )
                         },
                         onVoucherDisplayed = { voucher, index ->
                             viewModel.onVoucherDisplayed(
                                 voucher,
                                 index
                             )
                         },
                         collectingVoucherId = collectVoucherState.collectingVoucherId,
                         claimingRewardId = voucherCoinState.claimingId,
                         checkVoucherAvailability = { viewModel.checkVoucherAvailabilityByGroup(it) },
                         checkVoucherStackable = { viewModel.checkVoucherStackable(it) },
                         listHeader = {
                             Column(
                                 verticalArrangement = Arrangement.spacedBy(Spacing.S)
                             ) {
                                 if (!shouldHidePromoCode) {
                                     PromoCodeInput(
                                         promoCode = promoCodeState.promoCode,
                                         onChange = { viewModel.promoCodeSlice.onSetPromoCode(it) },
                                         isChecking = promoCodeState.isCheckingPromoCode,
                                         onSubmit = { viewModel.onCheckPromoCode(it) },
                                         errorMessage = promoCodeState.checkPromoCodeError,
                                         onValueCleared = { viewModel.promoCodeSlice.onClearPromoCodeError() },
                                         isApplied = promoCodeState.isPromoCodeApplied,
                                         promoCodeApplied = promoCodeState.promoCodeApplied,
                                     )
                                     if (promoCodeState.promoCodeNotice != null) {
                                         PromoCodeNotice(
                                             title = promoCodeState.promoCodeNotice!!.first,
                                             description = promoCodeState.promoCodeNotice!!.second
                                         )
                                     }
                                 }
                                 if (shouldShowVoucherCoin) {
                                     VoucherCoinBlock(
                                         coinInfo = voucherCoinState.coinInfo,
                                         onClick = {
                                             viewModel.trackVoucherCoinBlockClicked()
                                             viewModel.voucherCoinSlice.toggleVoucherCoinBottomSheet(
                                                 true
                                             )
                                         },
                                         onDisplayed = { isNotEnoughCoin ->
                                             viewModel.trackVoucherCoinBlockDisplayed(isNotEnoughCoin, state.shouldFlatVoucherReward)
                                         },
                                         compact = state.shouldFlatVoucherReward,
                                     )
                                 }
                             }
                         },
                         scrollState = listState,
                         modifier = Modifier
                             .weight(1f)
                             .padding(horizontal = Spacing.M)
                     )
 
                     if (!shouldHideFooter) {
                         Footer(
                             content = {
                                 Button(
                                     title = footerButtonProps.title
                                         ?: stringResource(Res.string.applyButtonLabel),
                                     onClick = { footerButtonProps.onClick?.invoke() },
                                     type = footerButtonProps.type ?: ButtonType.DISABLED,
                                 )
                             }
                         )
                     }
 
                     if (shouldShowVoucherCoin) {
                         VoucherCoinBottomSheet(
                             isVisible = voucherCoinState.isVoucherCoinBottomSheetVisible,
                             onDismiss = {
                                 viewModel.voucherCoinSlice.toggleVoucherCoinBottomSheet(
                                     false
                                 )
                                 viewModel.trackVoucherCoinCloseBottomSheetClicked()
                             },
                             sheetState = voucherCoinBottomSheetState,
                             coinInfo = voucherCoinState.coinInfo,
                             vouchers = voucherCoinList,
                             selectedVoucher = voucherCoinState.selectedVoucherCoin,
                             onToggleVoucherSelect = {
                                 viewModel.voucherCoinSlice.toggleVoucherCoinSelect(
                                     it
                                 )
                             },
                             isFetching = voucherCoinQueryState.value.isFetching,
                             onFetchNextPage = { voucherCoinQuery.onFetchNextPage.invoke() },
                             onClaimVoucher = { voucher, index ->
                                 viewModel.onClaimVoucherCoin(
                                     voucher,
                                     index,
                                 )
                             },
                             isClaiming = voucherCoinState.isClaiming,
                             onVoucherDisplayed = { voucher ->
                                 viewModel.trackVoucherCoinListItemDisplayed(
                                     giftType = voucher.typeId ?: "",
                                     campaignId = voucher.extra?.campaignId ?: voucher.campaign,
                                     status = if (voucher.extra?.isInsufficientCoin == true) {
                                         "invalid"
                                     } else {
                                         "valid"
                                     },
                                 )
                             }
                         )
                     }
                 }
             },
         )
     }
 
 }
