 @Composable
 fun PaymentSingleBillScreen(viewModel: PaymentSingleBillVM) {
     val state by viewModel.state.collectAsState()
     val sofListState by viewModel.sofSlice.state.collectAsState()
     val version = sofListState.version
 
     fun goBack() {
         viewModel.goBack()
     }
 
     PaymentScreenContainer(
         title = stringResource(Res.string.paymentTitle),
         screenType = viewModel.props.screenType ?: ScreenType.FULL_SCREEN,
         goBack = ::goBack,
         version = version,
         content = {
             when {
                 state.isLoading -> ContentLoadingSkeleton(version = version)
                 version.code == 2 -> PaymentTransferContentV2(viewModel)
                 else -> PaymentSingleBillContentV1(viewModel)
             }
         },
         footer = {
             when  {
                 state.isLoading -> FooterLoading()
                 version.code == 2 -> TransferFooter(
                     amount = state.amount,
                     enableConfirm = state.enableConfirm,
                     onConfirm = viewModel::processConfirm,
                     invoice = state.invoice
                 )
 
                 else -> SBFooter(
                     enableConfirm = state.enableConfirm,
                     amount = state.amount,
                     warningMsg = state.warningMsg,
                     onConfirm = viewModel::processConfirm
                 )
             }
         }
     )
 }
