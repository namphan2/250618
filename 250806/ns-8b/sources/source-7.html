 @Composable
 fun PaymentInfoBottomSheet(
     state: MutableState<Boolean>,
     invoiceInfos: List<InvoiceInfo> = emptyList(),
     invoice: InvoiceItem,
     hasAppliedPromotion: Boolean,
     hideAmount: Boolean = false,
     onOpen: () -> Unit = {},
     onClose: (String) -> Unit = {}
 ) {
     LaunchedEffect(state.value) {
         if (state.value) {
             onOpen()
         }
     }
 
     PayXComposeBottomSheet(
         state = state,
         title = stringResource(Res.string.paymentInformationDetail),
         hasFooter = true,
         maxHeightRatio = 0.75,
         onClose = onClose
     ) {
         Column {
             Divider()
             Spacer(modifier = Modifier.height(Spacing.M))
             Box(
                 modifier = Modifier.padding(horizontal = Spacing.M).clip(
                     shape = RoundedCornerShape(Spacing.M)
                 ).fillMaxWidth()
             ) {
                 Column {
                     invoiceInfos.forEachIndexed { index, item ->
                         val backgroundColor =
                             if (index % 2 == 0) Colors.black_01 else Colors.black_02
                         PaymentInfoRow(
                             title = item.title,
                             value = item.value,
                             titleTextStyle = item.titleTextStyle,
                             valueTextStyle = item.valueTextStyle,
                             modifier = Modifier.background(backgroundColor)
                                 .padding(vertical = Spacing.S, horizontal = Spacing.M)
                         )
                     }
                     if (!hideAmount) {
                         val backgroundColor =
                             if (invoiceInfos.size % 2 == 0) Colors.black_01 else Colors.black_02
                         PaymentInfoTotalAmount(
                             totalAmountInvoice = invoice.totalAmountInvoice,
                             subTotalAmountInvoice = invoice.subTotalAmountInvoice,
                             isShowSubTotal = invoice.totalAmountInvoice != invoice.subTotalAmountInvoice && hasAppliedPromotion,
                             titleColor = Colors.black_17,
                             titleTextStyle = Typography.labelDefaultMedium,
                             modifier = Modifier.background(backgroundColor)
                                 .padding(vertical = Spacing.S, horizontal = Spacing.M)
                         )
                     }
                 }
             }
         }
     }
 }
