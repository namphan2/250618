 @Composable
 fun TransactionInfoBottomSheet(
     sharedViewModel: TransactionResultRevampVM,
     viewModel: TransactionResultBtsVM,
     props: TransactionInfoBottomSheetProps,
     closeSheet: () -> Unit,
 ) {
     val screenHeight = getScreenDimensions().screenHeight
     val state by viewModel.state.state.collectAsState()
     val ticket =
         if (state.ticketList.isNotEmpty()) state.ticketList[props.ticketIndex] else TransactionTicketItem()
     val transaction = ticket.transaction ?: TransactionItem()
     val title = when (transaction.transactionStatus) {
         TransactionStatus.SUCCESS -> stringResource(Res.string.successfullyTransaction)
         TransactionStatus.FAILED -> stringResource(Res.string.failedTransaction)
         else -> stringResource(Res.string.processingTransaction)
     }
     val showShareButton = transaction.isSuccess
     val showAmount = transaction.amount != null
     val showOriginalAmount =
         transaction.isPaymentSDK && transaction.originalAmount != null && transaction.amount != null && transaction.originalAmount!! > transaction.amount!!
 
     Column(
         modifier = Modifier
             .height((screenHeight * 0.65).dp).animateContentSize()
     ) {
         Column(
             modifier = Modifier
                 .fillMaxWidth()
                 .verticalScroll(state = rememberScrollState()),
             verticalArrangement = Arrangement.spacedBy(8.dp)
         ) {
             RevampTransactionTicket(header = {
                 TransactionStatusCard(title = title,
                     amountText = transaction.amountText,
                     showAmount = showAmount,
                     originalAmountText = transaction.originalAmountText,
                     showOriginalAmount = showOriginalAmount,
                     showShareButton = showShareButton,
                     onPressShare = {
                         sharedViewModel.dispatch(vn.momo.compose.payment.payx.presentation.viewModel.action.TransactionResultAction.OnPressShare(transaction))
                     },
                     leftIcon = { StatusIconRevamp(data = transaction.transactionStatus) })
             }) {
                 Spacer(modifier = Modifier.height(Spacing.S))
                 TransactionInfo(
                     data = ticket,
                     index = 0,
                     dispatcher = sharedViewModel::dispatch,
                     closeSheet = closeSheet
                 )
                 TransactionResultWidget(Modifier.padding(top = Spacing.S), true, ticket, {
                     viewModel.dispatch(vn.momo.compose.payment.payx.presentation.viewModel.action.TransactionResultAction.OnWidgetLoadSuccess())
                 }, {
                     sharedViewModel.dispatch(vn.momo.compose.payment.payx.presentation.viewModel.action.TransactionResultAction.OnWidgetLoadFailed())
                 }, { width, height ->
                     viewModel.dispatch(
                         vn.momo.compose.payment.payx.presentation.viewModel.action.TransactionResultAction.OnWidgetLayoutChanged(
                             width = width, height = height
                         )
                     )
                 })
                 Spacer(Modifier.height(Spacing.M))
             }
         }
     }
 }
