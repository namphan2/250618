 @Composable
 fun PaymentSingleBillContentV1(
     viewModel: PaymentSingleBillVM
 ) {
     val state = viewModel.state.collectAsState()
     val sofState = viewModel.sofSlice.state.collectAsState()
 
     val hasAppliedPromotion by remember {
         derivedStateOf {
             state.value.promotion.promoDisplay.any { !it.promoId.isNullOrBlank() }
         }
     }
 
     Column(
         modifier = Modifier
             .fillMaxWidth()
             .background(Colors.black_01)
             .padding(vertical = Spacing.XL, horizontal = Spacing.M),
         verticalArrangement = Arrangement.spacedBy(Spacing.M)
     ) {
         if (viewModel.props.isAllowPickSource) {
             SourceOfFunds(
                 sofListState = sofState.value,
                 loadingState = false,
                 bundleSof = sofState.value.selected,
                 onClickHideBalance = viewModel.sofSlice::toggleHideBalance,
                 isAllowAddSOF = true,
                 isAllowAddCard = false,
                 onClickShowAll = viewModel::onClickShowAll,
                 addSofItem = {
                     CarouselItem.AddBankSof(
                         modifier = Modifier.fillMaxHeight(),
                         title = stringResource(Res.string.addBank),
                         onPress = viewModel::addSof
                     )
                 },
                 onAddSof = viewModel::addSof,
                 onAddCard = viewModel::addCard,
                 onSelectSof = viewModel.sofSlice::changeSofIfNotSelected,
                 onLoaded = {
                     viewModel.sofSlice.trackSofDisplayed(
                         "sof_carousel",
                         sofState.value.list,
                         UiVersion()
                     )
                 }
             )
             sofState.value.moneyPayCTA?.takeIf { it.type != RecommendType.None }
                 ?.let { moneyPay ->
                     RecommendMoneyPay(
                         backgroundColor = moneyPay.theme?.bg?.toColor()
                             ?: Colors.indigo_10,
                         btnColor = when {
                             moneyPay.isDisabled == true -> moneyPay.theme?.btnInactive?.toColor()
                             else -> moneyPay.theme?.btnActive?.toColor()
                         } ?: Colors.indigo_05,
                         overlayColor = moneyPay.theme?.overlay?.toColor()
                             ?: Colors.indigo_07,
                         logo = moneyPay.logo ?: "",
                         title = moneyPay.title ?: "",
                         isDisableBtn = moneyPay.isDisabled ?: false,
                         btnIconName = moneyPay.icon,
                         btnTitle = moneyPay.btnTitle ?: "",
                         titleFont = moneyPay.theme?.titleFont ?: "",
                         buttonFont = moneyPay.theme?.buttonFont ?: "",
                         onPressRecommend = {
                             viewModel.sofSlice.selectRecommendMoneyPay(
                                 moneyPay,
                                 null
                             )
                         }
                     )
                 }
         }
         Spacer(modifier = Modifier.height(1.dp))
         TransactionDetail(
             loading = false,
             invoices = state.value.invoice,
             shouldShowTid = true,
             onChangeExpandState = viewModel::onToggleCartDetail,
             disableExpand = true,
             footer = { item ->
                 Column {
                     InvoiceFeeSummary(
                         fee = item.fee,
                         feeInfo = item.feeInfo,
                         feeInfoHighlight = item.feeInfoHighlight ?: ""
                     )
                     if (item.visibleRefCode) {
                         InvoiceItemUiView(
                             key = stringResource(Res.string.transaction_id),
                             value = item.transRefCode?.toString() ?: "",
                             isShowBorderBottom = false
                         )
                     }
                     if (state.value.promotion.isDisplay) {
                         SBChooseVouchers(
                             promotion = state.value.promotion,
                             hasAppliedPromotion = hasAppliedPromotion,
                             onSelectPromotion = viewModel::navigateChoosePromotion,
                             onCancelPromotion = viewModel::cancelPromotion
                         )
                     }
                 }
             }
         )
         TrustBanner()
     }
 }
