 data class InlineMessageError(
     val description: String? = null,
     val showIcon: Boolean = true,
     val showIconClose: Boolean = true
 )
 
 @Composable
 fun PickPromotionBlock(
     promotion: PromotionData,
     inlineMessage: InlineMessageError?,
     onSelectPromotion: (status: PromotionStatus) -> Unit,
     hideAmount: Boolean,
     onMissionShowed: () -> Unit,
     rewardMission: PayXResult<List<PayXComponent>, Throwable>? = null,
     errorIcon: PromotionStatus? = null
 ) {
     Column(
         modifier = Modifier.scaledPadding(12f),
     ) {
         if (promotion.isDisplay && !hideAmount) {
             PromotionPicker(
                 promoDisplay = promotion.promoDisplay,
                 onNavigatePromotion = onSelectPromotion,
                 modifier = Modifier,
                 errorIcon = errorIcon
             )
         }
 
         inlineMessage?.description?.takeIf { it.isNotBlank() }?.also {
             MessageInformation(
                 description = it,
                 showIcon = inlineMessage.showIcon,
                 showIconClose = inlineMessage.showIconClose,
                 state = InformationState.WARNING,
                 modifier = Modifier.scaledPadding(top = 8f)
             )
         }
 
         rewardMission?.let {
             MissionRewardView(
                 onShow = onMissionShowed,
                 missionRewardSummary = it
             )
         }
     }
 }
