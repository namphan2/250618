 @Composable
 fun InfoCard(
     modifier: Modifier = Modifier,
     title: String,
     titleStyle: InfoTextStyle = InfoTextStyle(
         font = Typography.headerXsSemibold,
         color = AppTheme.current.colors.text.default
     ),
     titleExtra: @Composable () -> Unit = {},
     description: String,
     descriptionStyle: InfoTextStyle = InfoTextStyle(
         font = Typography.descriptionXsRegular,
         color = AppTheme.current.colors.text.hint
     ),
     leading: @Composable () -> Unit,
     trailing: @Composable () -> Unit,
     extra: @Composable () -> Unit,
     pressable: Boolean = true,
     blur: Boolean = false,
     onPress: () -> Unit = {},
 ) {
     val alpha = if (blur) 0.5f else 1f
     Row(
         modifier = Modifier.fillMaxWidth().then(modifier)
             .noFeedbackClickable(enabled = pressable, onClick = onPress),
         verticalAlignment = Alignment.CenterVertically,
         horizontalArrangement = scaledSpacedBy(8f)
     ) {
         leading()
         Row(
             Modifier.weight(1f).alpha(alpha),
             verticalAlignment = Alignment.CenterVertically,
             horizontalArrangement = scaledSpacedBy(8f),
         ) {
             Column {
                 Row(
                     horizontalArrangement = scaledSpacedBy(4f),
                     verticalAlignment = Alignment.CenterVertically
                 ) {
                     Text(
                         text = title,
                         style = titleStyle.font,
                         modifier = titleStyle.modifier,
                         maxLines = 1,
                         overflow = TextOverflow.Ellipsis,
                         color = titleStyle.color
                     )
                     titleExtra()
                 }
 
                 description.takeIf { it.isNotBlank() }?.also {
                     Text(
                         modifier = descriptionStyle.modifier,
                         text = description,
                         style = descriptionStyle.font,
                         color = descriptionStyle.color,
                         maxLines = 1,
                         overflow = TextOverflow.Ellipsis
                     )
                 }
             }
             Box(
                 modifier = Modifier.weight(1f)
             ) {
                 extra()
             }
         }
 
         trailing()
     }
 }

 fun MiniCard(
     modifier: Modifier = Modifier,
     border: BorderStroke? = null,
     title: String,
     titleStyle: InfoTextStyle = InfoTextStyle(
         font = Typography.headerSSemibold,
         color = AppTheme.current.colors.text.default

         font = Typography.descriptionXsRegular,
         color = AppTheme.current.colors.text.hint
     ),
     description: String,
     descriptionStyle: InfoTextStyle = InfoTextStyle(
         font = Typography.descriptionXsRegular,
         color = AppTheme.current.colors.text.hint

     image: Any? = null,
     badge: SofBadge? = null

     Box(modifier) {
         Surface(
             shape = RoundedCornerShape(12.dp),
             border = border
         ) {
             Column(Modifier.scaledPadding(8f)) {
                 Row(Modifier.fillMaxWidth(), verticalAlignment = Alignment.CenterVertically) {
                     Text(
                         text = title,
                         style = titleStyle.font,
                         color = titleStyle.color,
                         maxLines = 1,
                         overflow = TextOverflow.Ellipsis
                     )
                     subTitle?.takeIf { it.isNotBlank() }?.also {
                         Text(
                             text = subTitle,
                             modifier = Modifier.weight(1f),
                             style = subTitleStyle.font,
                             color = subTitleStyle.color,
                             textDecoration = subTitleStyle.textDecoration,
                             maxLines = 1,
                             overflow = TextOverflow.Ellipsis
                         )
                     }
                 }
                 Text(
                     text = description,
                     modifier = Modifier.fillMaxWidth(),
                     style = descriptionStyle.font,

             )
         }
     }
