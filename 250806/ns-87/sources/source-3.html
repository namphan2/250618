 interface TraceMixin {
     val logger: ILogger
     val params: TransactionResultParams
 
     // Grafana
     val grafana: IGrafanaTracer
     val scope: CoroutineScope
     val buffer: MutableMap<String, String?>?
     val jobs: MutableMap<String, Job>
     var grafanaFlow: String?
 
     // Firebase
     val firebase: IFirebaseTracer
     var durationBuffer: MutableMap<String, Long?>
     var firebaseFlow: String?
     var firebaseAppId: String?
 
     fun grafanaStart(event: GrafanaMetric) {
         jobs[event.label] = grafana.start(
             StartTraceParameter(
                 grafanaFlow ?: event.flow,
                 event.step,
                 appId = params.appId ?: "vn.momo.compose",
                 serviceCode = params.response?.momoMsg?.replyMsgs?.get(0)?.tranHisMsg?.serviceId
             )
         ).onEach {
             if (!it.traceId.isNullOrBlank()) {
                 buffer?.put(event.label, it.traceId)
             } else {
                 logger.i("[TR]GrafanaTrace-grafanaStart: traceId null ${event.label}")
             }
         }.catch {
             logger.i("[TR]GrafanaTrace-grafanaStart: error ${event.label} ${it.message}")
         }.launchIn(scope)
     }
 
     fun grafanaSuccess(event: GrafanaMetric, params: TraceParameter = TraceParameter()) {
         grafanaStop(event, true, params)
     }
 
     fun grafanaFail(event: GrafanaMetric, params: TraceParameter = TraceParameter()) {
         grafanaStop(event, false, params)
     }
 
     fun grafanaCount(event: GrafanaMetric) {
         scope.launch {
             grafana.count(
                 CountTraceParameter(
                     flow = grafanaFlow ?: event.flow,
                     step = event.step,
                     appId = params.appId ?: "vn.momo.compose",
                     serviceCode = params.response?.momoMsg?.replyMsgs?.get(0)?.tranHisMsg?.serviceId
                 )
             )
         }
     }
 
     fun grafanaStop(event: GrafanaMetric, status: Boolean, _params: TraceParameter) {
         scope.launch {
             jobs[event.label]?.join()
             buffer?.get(event.label)?.let { traceId ->
                 if (status) {
                     grafana.success(
                         traceId, _params.copy(
                             flow = grafanaFlow ?: event.flow,
                             step = event.step,
                             appId = params.appId ?: "vn.momo.compose",
                             serviceCode = params.response?.momoMsg?.replyMsgs?.get(0)?.tranHisMsg?.serviceId
                         )
                     )
                 } else {
                     grafana.fail(
                         traceId, _params.copy(
                             flow =  grafanaFlow ?: event.flow,
                             step = event.step,
                             appId = params.appId ?: "vn.momo.compose",
                             serviceCode = params.response?.momoMsg?.replyMsgs?.get(0)?.tranHisMsg?.serviceId
                         )
                     )
                 }
                 buffer?.remove(event.label)
                 jobs.remove(event.label)
             } ?: run {
                 logger.e("[ResultX] GrafanaTrace.stop:: no traceId $event $status $_params")
             }
         }
     }
 
     fun grafanaStop(
         traceId: String, event: GrafanaMetric, status: Boolean, _params: TraceParameter
     ) {
         if (traceId.isEmpty()) {
             logger.e("[ResultX] GrafanaTrace.stopWithTraceId:: no traceId $event $status $_params")
             return
         }
         scope.launch {
             if (status) {
                 grafana.success(
                     traceId, _params.copy(
                         flow = grafanaFlow ?: event.flow,
                         step = event.step,
                         appId = params.appId ?: "vn.momo.compose",
                         serviceCode = params.response?.momoMsg?.replyMsgs?.get(0)?.tranHisMsg?.serviceId
                     )
                 )
             } else {
                 grafana.fail(
                     traceId, _params.copy(
                         flow = grafanaFlow ?: event.flow,
                         step = event.step,
                         appId = params.appId ?: "vn.momo.compose",
                         serviceCode = params.response?.momoMsg?.replyMsgs?.get(0)?.tranHisMsg?.serviceId
                     )
                 )
             }
         }
     }
 
     fun firebaseTrack(event: JourneyEvent) {
         event.serviceCode = params.response?.momoMsg?.replyMsgs?.get(0)?.tranHisMsg?.serviceId ?: ""
         event.serviceType =
             params.response?.momoMsg?.replyMsgs?.get(0)?.tranHisMsg?.serviceType ?: ""
         event.flowType = params.flowType ?: ""
         event.flow = firebaseFlow ?: "PayX"
         event.appId = firebaseAppId ?: "vn.momo.payx"
         event.orderId = params.response?.momoMsg?.getPurchaseOrder()
         when (event) {
             is TransactionResultRevampScreenDisplayed -> {
                 durationBuffer[event.screenName] = TimeUtils.getCurrentTimeMillis()
                 trackEvent(event)
             }
 
             is TransactionResultRevampScreenInteracted -> {
                 event.timeLoad = durationBuffer[event.screenName]?.let { TimeUtils.getCurrentTimeMillis() - it } ?: 0L
                 trackEvent(event)
             }
 
             else -> {
                 trackEvent(event)
             }
         }
     }
 
     private fun trackEvent(event: JourneyEvent) {
         firebase.track(
             event.eventName,
             ModelConvertUtils.encodeToJson(JourneyEvent.serializer(), event)?.jsonObject
         )
     }
 
     fun grafanaStopE2E(traceId: String?) {
         traceId?.let {
             grafana.success(it, TraceParameter())
         }
     }
 
     fun grafanaStopTraceIdsParams() {
         this.params.traceIds.forEach {
             grafanaStopE2E(it)
         }
     }
 }
