 fun JsonElement?.getLocalized(): String = when (this) {
     null -> ""
     is JsonObject -> this.localized()
     is JsonPrimitive -> this.contentOrNull?.localized() ?: ""
     else -> this.toString().localized()
 }
 
 fun List<IVoucherBase>?.distinctById(): List<IVoucherBase> {
     return this?.distinctBy {
         when {
             !it.ID.isNullOrBlank() -> it.ID
             !it.extra?.collectId.isNullOrBlank() -> it.extra?.collectId
             !it.extra?.rewardId.isNullOrBlank() -> it.extra?.rewardId
             else -> Random.nextInt()
         }
     } ?: emptyList()
 }
 
 fun List<IVoucherBase>.flattenToGroups(): List<VoucherGroup> {
     return this
         .asSequence()
         .filter { it.localType != VoucherLocalType.Invalid }
         .mapNotNull { it.groups }
         .flatten()
         .filterNotNull()
         .distinct()
         .toList()
 }
 
 @Composable
 fun LazyListState.toScrollState(): ScrollState {
     val scrollState = rememberScrollState()
     LaunchedEffect(this) {
         snapshotFlow { firstVisibleItemScrollOffset to firstVisibleItemIndex }
             .collect { (offset, index) ->
                 if (index == 0) {
                     scrollState.scrollTo(offset)
                 }
             }
     }
     return scrollState
 }
