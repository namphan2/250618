 class SBOverlay(
     override val navigator: INavigator,
     val data: UiOverlayData,
 ) : PayXOverlayHandler {
     override fun createUi(): UiOverlay? {
         return when (data) {
             is PayXDialog -> UiDialog(
                 navigator, data, TransactionOverlayAction()
             )
 
             is PayXToastMessage -> UiToastMessage(
                 navigator, data, actionCreator = TransactionOverlayAction()
             )
 
             SofBottomSheetData -> {
                 // TODO
                 null
             }
 
             is PayXBottomSheet -> UiBottomSheet(
                 navigator, data, TransactionOverlayAction()
             )
 
             else -> null
         }
     }
 
     override suspend fun handle(): OverlayEffect? {
         val action = createUi()?.handle()
         return PayXActionHandler(action)
     }
 }
