 @Immutable
 class XSaleLayout {
     companion object {
         @Composable
         fun Carousel(
             modifier: Modifier = Modifier,
             data: List<XSaleItem>,
             selected: XSaleItem?,
             onPress: (XSaleItem) -> Unit,
         ) {
             if (data.isEmpty()) {
                 return
             }
             val horizontalScrollState = rememberLazyListState()
 
             LaunchedEffect(selected) {
                 data.indexOfFirst { it.id == selected?.id }.let {
                     if (it >= 0) {
                         horizontalScrollState.animateScrollToItem(it)
                     }
                 }
             }
 
             LazyRow(
                 horizontalArrangement = Arrangement.spacedBy(8.dp),
                 state = horizontalScrollState,
                 modifier = Modifier
                     .fillMaxWidth()
                     .semantics { contentDescription = "xSaleHorizontalScrollView" }
                     .then(modifier),
             ) {
                 items(data, key = { it.id }) {
                     XSaleItemView.Carousel(
                         data = it,
                         isSelected = it.id == selected?.id,
                         onPress = { onPress(it) }
                     )
                     Spacer(modifier = Modifier.width(Spacing.S))
                 }
             }
         }
 
         @Composable
         fun ListView(
             modifier: Modifier = Modifier,
             data: List<XSaleItem>,
             selected: XSaleItem?,
             onPress: (XSaleItem) -> Unit,
         ) {
             if (data.isEmpty()) {
                 return
             }
             Column(
                 verticalArrangement = scaledSpacedBy(12f),
                 modifier = Modifier.fillMaxWidth().then(modifier)
             ) {
                 data.forEach {
                     key(it.id) {
                         XSaleItemView.Card(
                             data = it,
                             isSelected = it.id == selected?.id,
                             onPress = { onPress(it) })
                     }
                 }
             }
         }
     }
 }
