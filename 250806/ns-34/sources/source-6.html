 @Serializable
 data class CartData(
     val dialog: PayXDialog? = null,
     @SerialName("infoBlock") var _infoBlock: JsonElement? = null,
     @SerialName("footer") var _footer: JsonElement? = null
 ) {
     val infoBlock: PayXComponent?
         get() = PaymentJson.UiDecode.decode<ComponentDTO>(_infoBlock?.toString() ?: "")
             .toPayXComponentUI()
 
     val footer: PayXComponent?
         get() = PaymentJson.UiDecode.decode<ComponentDTO>(_footer?.toString() ?: "")
             .toPayXComponentUI()
 }
 
 fun ComponentDTO?.toPayXComponentUI(): PayXComponent? {
     return when (this) {
         is CheckBoxDTO -> this.toModel()
         is TextDTO -> this.toModel(PayXText())
         is IconDTO -> this.toModel(PayXIcon(""))
         is LinkListDTO -> this.toModel()
         is CartItemDTO -> this.toModel()
         else -> null
     }
 }
