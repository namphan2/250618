 fun ComponentDTO?.toPayXComponentUI(): PayXComponent? {
     return when (this) {
         is CheckBoxDTO -> this.toModel()
         is TextDTO -> this.toModel(PayXText())
         is IconDTO -> this.toModel(PayXIcon(""))
         is LinkListDTO -> this.toModel()
         is CartItemDTO -> this.toModel()
         else -> null
     }
 }
