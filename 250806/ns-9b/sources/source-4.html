 class PayXAppToAppVM(
     val props: VirtualBankBottomSheet,
     private val navigator: INavigator,
     private val imageApi: ImageApi,
     repository: VirtualAccountRepository,
     private val paymentSession: PaymentSession,
     private val routeApi: IRouteApi
 ) : ViewModel<VAReducer.State, VAReducer.Action, VAReducer.Effect>(
     initialState = VAReducer.State(
         info = props,
         count = props.transaction.expiredTime ?: -1,
     ),
     reducer = VAReducer()
 ) {
     private val targetTime = TimeUtils.getCurrentTimeMillis() + (props.transaction.expiredTime ?: 0)
 
     init {
         FirebaseTrace.track(VADisplayed(purchaseId = props.transaction.purchaseId))
 
         repository.getBankLogo(props.transaction.bankInfo.id)
             .onEach { newBank ->
                 dispatch(VAReducer.Action.SetUserBankLogo(newBank?.logo))
             }
             .launchIn(viewModelScope)
 
         viewModelScope.launch {
             while (true) {
                 dispatch(VAReducer.Action.UpdateTime(targetTime - TimeUtils.getCurrentTimeMillis()))
                 delay(1000)
                 if (state.value.count < 0) {
                     break
                 }
             }
         }
 
         repository.listenToBankTransfer(
             props.transaction.purchaseId,
             props.transaction.paymentType
         )
             .onEach { resp ->
                 val params = TransactionResultParams(
                     response = resp,
                     miniAppWidgetId = paymentSession.requestPayXData.miniAppWidgetId,
                     appId = paymentSession.requestPayXData.appId ?: "vn.momo.compose"
                 )
                 navigator.navigateFeatureCode(
                     "transaction_result_revamp_native",
                     ModelConvertUtils.encodeToJson(
                         TransactionResultParams.serializer(),
                         params
                     ) as? JsonObject ?: emptyJsonObject()
                 )
                 emitEffect(VAReducer.Effect.Close)
             }
             .launchIn(viewModelScope)
     }
 
     fun toastCopy() {
         FirebaseTrace.track(
             VAButtonClicked(
                 buttonName = "copy",
                 purchaseId = props.transaction.purchaseId
             )
         )
         viewModelScope.launch {
             navigator.showToast(buildJsonObject {
                 put(
                     "title",
                     getString(Res.string.accountCopied)
                 )
                 put("icon", "https://static.momocdn.net/app/img/payment/ic-check-circle-green.png")
             }, String.serializer())
                 .launchIn(this)
         }
     }
 
     fun onDownloadQr(base64: String) {
         FirebaseTrace.track(
             VAButtonClicked(
                 buttonName = "download",
                 purchaseId = props.transaction.purchaseId
             )
         )
         viewModelScope.launch {
             val permission = imageApi.requestSaveImagePermission()
             if (!permission) {
                 navigator.showToast(buildJsonObject {
                     put(
                         "title",
                         getString(Res.string.downloadQrFailed)
                     )
                     put("type", "warning")
                 }, String.serializer()).launchIn(this)
                 return@launch
             }
             val result = imageApi.saveImage(base64)
             if (result) {
                 navigator.showToast(buildJsonObject {
                     put(
                         "title",
                         getString(Res.string.qrDownloaded2)
                     )
                     put(
                         "icon",
                         "https://static.momocdn.net/app/img/payment/ic-check-circle-green.png"
                     )
                 }, String.serializer()).launchIn(this)
                 openBank(VAReducer.OpenBankType.QR)
             } else {
                 navigator.showToast(buildJsonObject {
                     put(
                         "title",
                         getString(Res.string.downloadQrFailed)
                     )
                     put("type", "warning")
                 }, String.serializer()).launchIn(this)
             }
         }
     }
 
     fun openBank(type: VAReducer.OpenBankType) {
         val bankList = state.value.info.userBank
         if (bankList.isEmpty()) return
         if (bankList.size == 1) {
             openBankUrl(bankList[0].appScheme)
             return
         }
         dispatch(VAReducer.Action.SetType(type))
     }
 
     fun openBankUrl(url: String?) {
         viewModelScope.launch {
             url?.let {
                 val resp = routeApi.openUrl(url)
                 if (!resp) {
                     navigator.showToast(buildJsonObject {
                         put(
                             "title",
                             getString(Res.string.cannotOpenBank)
                         )
                         put("type", "warning")
                     }, String.serializer()).launchIn(this)
                 }
             } ?: navigator.showToast(buildJsonObject {
                 put(
                     "title",
                     getString(Res.string.cannotOpenBank)
                 )
                 put("type", "warning")
             }, String.serializer()).launchIn(this)
         }
     }
 }
