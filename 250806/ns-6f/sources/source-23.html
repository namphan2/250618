 /**
 Data class representing an invoice item in the payment system.
  * @property itemId The unique identifier for the invoice item
  * @property subItemId The sub-identifier for the invoice item (for data only)
  * @property serviceName The name of the service, defaults to empty string (UI: header line invoice)
  * @property fee The service fee amount, defaults to 0 (UI: content invoice). Relative: [visibleFee]
  * @property info The list of invoice information details (UI: payment info rows)
  * @property serviceId The service identifier
  * @property serviceType The type of service
  * @property originalAmount The original amount before any modifications (UI: invoice strikethrough amount)
  * @property amount The current amount after modifications
  * @property totalAmountInvoice The total invoice amount including all charges (UI: invoice footer)
  * @property subTotalAmountInvoice The subtotal amount of the invoice (UI: invoice footer)
  * @property discountCampaignAmount The discount amount from campaigns
  * @property visibleFee Controls fee visibility, defaults to true (UI)
  * @property isXSale Indicates if this is an XSale transaction
  * @property paymentTid The payment transaction ID (UI)
  * @property visibleTid Computed property that determines if transaction ID should be visible based on ID length
  **/

     val itemId: String = "",
