 package vn.momo.compose.payment.promotion.presentation.viewModel.reducers
 
 import vn.momo.compose.payment.base.presentation.viewModel.reducer.Reducer
 import vn.momo.compose.payment.promotion.domain.entities.IVoucherBase
 import vn.momo.compose.payment.promotion.domain.entities.LocationCoords
 import vn.momo.core.features.annotation.Immutable
 
 class ChoosePromotionReducer :
     Reducer<ChoosePromotionReducer.State, ChoosePromotionReducer.Action, ChoosePromotionReducer.Effect> {
     @Immutable
     data class State(
         val vouchers: List<IVoucherBase> = emptyList(),
         val isFetching: Boolean = true,
         val isRefetching: Boolean = false,
         val hasNextPage: Boolean = false,
 
         val selectedVouchers: List<IVoucherBase> = emptyList(),
 
         val localPage: Int = 1, // for POS screen pagination
 
         val userLocation: LocationCoords? = null,
         val isLocationLoaded: Boolean? = false,
 
         val shouldFlatVoucherReward: Boolean = false, // AB test treatment_flatt_vc_reward
     ) : Reducer.State
 
     @Immutable
     sealed class Action : Reducer.Action {
         data class SetVoucherState(
             val vouchers: List<IVoucherBase>,
             val isFetching: Boolean,
             val isRefetching: Boolean,
             val hasNextPage: Boolean = false,
         ) : Action()
 
         data class SetSelectedVouchers(val vouchers: List<IVoucherBase>) : Action()
 
         data class SetLocalPage(val page: Int) : Action()
 
         data class SetUserLocation(
             val location: LocationCoords? = null,
             val isLocationLoaded: Boolean? = false,
         ) : Action()
 
         data class SetShouldFlatVoucherReward(
             val shouldFlatVoucherReward: Boolean,
         ) : Action()
     }
 
     @Immutable
     class Effect() : Reducer.Effect
 
     override fun reduce(prev: State, action: Action): Pair<State, Effect?> {
         return when (action) {
             is Action.SetVoucherState -> prev.copy(
                 vouchers = action.vouchers,
                 isFetching = action.isFetching,
                 isRefetching = action.isRefetching,
                 hasNextPage = action.hasNextPage,
             ) to null
 
             is Action.SetSelectedVouchers -> prev.copy(
                 selectedVouchers = action.vouchers,
             ) to null
 
             is Action.SetLocalPage -> prev.copy(
                 localPage = action.page,
             ) to null
 
             is Action.SetUserLocation -> prev.copy(
                 userLocation = action.location,
                 isLocationLoaded = action.isLocationLoaded,
             ) to null
 
             is Action.SetShouldFlatVoucherReward -> prev.copy(
                 shouldFlatVoucherReward = action.shouldFlatVoucherReward,
             ) to null
         }
     }
 }
