 @Composable
 fun ConfirmScreenContent(
     viewModel: PaymentConfirmVM,
     sofListState: PayXSofReducer.State,
     state: TransactionReducer.State,
     hasAppliedPromotion: Boolean,
     xSaleState: XSaleReducer.XSaleState,
     autoDebitState: AutoDebitReducer.AutoDebitState,
     onAnimationEnd: () -> Unit,
     detectXSalePosition: (LayoutCoordinates) -> Unit
 ) {
     Column(
         modifier = Modifier
             .fillMaxWidth()
             .background(Colors.black_01)
             .padding(vertical = Spacing.XL, horizontal = Spacing.M),
         verticalArrangement = Arrangement.spacedBy(Spacing.M)
     ) {
         if (viewModel.props.isAllowPickSource) {
             SourceOfFunds(
                 sofListState = sofListState,
                 loadingState = state.isLoading,
                 bundleSof = sofListState.selected,
                 onClickHideBalance = viewModel.sofSlice::toggleHideBalance,
                 isAllowAddSOF = viewModel.props.isAllowAddSource,
                 onClickShowAll = viewModel::onClickShowAll,
                 onAddSof = viewModel::addSof,
                 onAddCard = viewModel::addCard,
                 onSelectSof = viewModel.sofSlice::changeSofIfNotSelected,
                 onLoaded = {
                     viewModel.sofSlice.trackSofDisplayed(
                         "sof_carousel",
                         sofListState.list,
                         UiVersion()
                     )
                 },
             )
 
             PayXAnimatedVisibility(
                 visible = sofListState.moneyPayCTA?.type != null && sofListState.moneyPayCTA?.type != RecommendType.None,
             ) {
                 sofListState.moneyPayCTA.also { moneyPay ->
                     RecommendMoneyPay(
                         backgroundColor = moneyPay?.theme?.bg?.toColor()
                             ?: Colors.indigo_10,
                         btnColor = when {
                             moneyPay?.isDisabled == true -> moneyPay.theme?.btnInactive?.toColor()
                             else -> moneyPay?.theme?.btnActive?.toColor()
                         } ?: Colors.indigo_05,
                         overlayColor = moneyPay?.theme?.overlay?.toColor()
                             ?: Colors.indigo_07,
                         logo = moneyPay?.logo ?: "",
                         title = moneyPay?.title ?: "",
                         isDisableBtn = moneyPay?.isDisabled ?: false,
                         btnIconName = moneyPay?.icon,
                         btnTitle = moneyPay?.btnTitle ?: "",
                         titleFont = moneyPay?.theme?.titleFont ?: "",
                         buttonFont = moneyPay?.theme?.buttonFont ?: "",
                         onPressRecommend = {
                             viewModel.sofSlice.selectRecommendMoneyPay(
                                 moneyPay!!,
                                 state.purchaseId
                             )
                         }
                     )
                 }
             }
 
             PayXAnimatedVisibility(
                 visible = sofListState.recommendInstallmentState?.display == true,
             ) {
                 InstallmentPackagesView(
                     state = sofListState.recommendInstallmentState,
                     onPressClose = viewModel.sofSlice::onCloseInstallmentPackage,
                     onPressReadMore = viewModel::openInstallmentInstruction,
                     onPressItem = viewModel::selectInstallmentPackage,
                     onPressToggle = viewModel.sofSlice::toggleCollapseInstallment
                 )
             }
 
         }
         Spacer(modifier = Modifier.height(1.dp))
         TransactionDetail(
             state.isLoading,
             state.invoices,
             hasAppliedPromotion = hasAppliedPromotion,
             initialExpanded = xSaleState.config?.data.isNullOrEmpty() && autoDebitState.items.isEmpty(),
             shouldShowTid = state.shouldShowTid,
             hideAmount = state.hideAmount,
             onChangeExpandState = viewModel::onToggleCartDetail,
             onAnimationEnd = onAnimationEnd,
             footer = { item ->
                 if (!state.hideAmount) {
                     InvoiceTotalAmount(
                         totalAmountInvoice = item.totalAmountInvoice,
                         subTotalAmountInvoice = item.subTotalAmountInvoice,
                         isShowSubTotal = item.totalAmountInvoice != item.subTotalAmountInvoice && hasAppliedPromotion
                     )
                 }
             }
         )
         AutoDebitView(
             state = autoDebitState,
             dispatch = viewModel.autoDebitSlice::dispatch
         )
         XSaleContainer(
             xSaleState = xSaleState,
             dispatch = viewModel.xSaleSlice::dispatch,
             modifier = Modifier.onGloballyPositioned(detectXSalePosition)
         )
         SmartWidget(viewModel.props.smartWidget)
         TrustBanner()
     }
 }
