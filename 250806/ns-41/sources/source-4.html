 enum class ScreenType {
     FULL_SCREEN,
     BOTTOM_SHEET,
 }
 
 @Composable
 fun PaymentScreenContainer(
     title: String,
     screenType: ScreenType = ScreenType.FULL_SCREEN,
     content: @Composable () -> Unit,
     footer: @Composable (() -> Unit)? = null,
     goBack: (() -> Unit)? = null,
     version: UiVersion,
 ) {
     val scrollState = rememberScrollState()
     val showBottomSheet = remember { mutableStateOf(false) }
     val screenWidth = getScreenDimensions().screenWidth
 
     LaunchedEffect(Unit) {
         showBottomSheet.value = true
     }
 
     when (screenType) {
         ScreenType.BOTTOM_SHEET -> {
             BaseComposeBottomSheet(
                 state = showBottomSheet,
                 title = title,
                 content = content,
                 footer = footer,
                 onClose = { goBack?.invoke() },
                 maxHeightRatio = 0.7,
             )
         }
 
         ScreenType.FULL_SCREEN -> {
             Screen(
                 headerRight = { },
                 useAvoidKeyboard = false,
                 scrollState = scrollState,
                 title = title,
                 goBack = goBack,
                 content = content,
                 footer = footer,
                 layoutOffset = if (version.code == 2) 0.dp else 56.dp,
                 animatedHeader = if (version.code == 2) AnimatedHeader(
                     isSurface = false,
                     composable = {
                         Image(
                             source = Images.bg_ttat_revamp,
                             modifier = Modifier.fillMaxWidth().height((screenWidth * 9 / 16).dp),
                             options = Options(contentScale = ContentScale.Crop)
                         )
                     }) else null,
             )
         }
     }
 }
