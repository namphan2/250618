 @Composable
 fun PaymentInput(
     text: String = "",
     floatingValue: String = "",
     floatingValueColor: Color = AppTheme.current.colors.text.hint,
     floatingIcon: String = "",
     floatingIconColor: Color = AppTheme.current.colors.text.default,
     onClickFloating: () -> Unit = {},
     placeholder: String = "",
     size: InputSize = InputSize.SMALL,
     onChangeText: (String) -> Unit = {},
     hintText: String = "",
     error: String = "",
     errorSpacing: Boolean = false,
     disabled: Boolean = false,
     readOnly: Boolean = false,
     secureTextEntry: Boolean = false,
     icon: String = "",
     iconColor: Color = AppTheme.current.colors.text.default,
     onRightIconPressed: () -> Unit = {},
     leadingIcon: String = "",
     leadingIconColor: Color = AppTheme.current.colors.text.hint,
     onFocus: () -> Unit = {},
     onBlur: () -> Unit = {},
     loading: Boolean = false,
     required: Boolean = false,
     fontWeight: InputFontWeight = InputFontWeight.REGULAR,
     keyboardType: KeyboardType,
     visualTransformation: VisualTransformation? = null,
     modifier: Modifier = Modifier,
 ) {
     // Handling state based on the `BasicTextField` concept
     var textFieldValueState by remember { mutableStateOf(TextFieldValue(text = text)) }
     val textFieldValue = textFieldValueState.copy(text = text)
     var lastTextValue by remember(text) { mutableStateOf(text) }
 
     var isFocused by remember { mutableStateOf(false) }
     var passHidden by remember { mutableStateOf(false) }
     val disabledColor = AppTheme.current.colors.text.disable
     var textColor = if (disabled) Colors.black_09 else AppTheme.current.colors.text.default
     val placeholderColor = Colors.black_09
     val bgInput = if (disabled) Colors.black_02 else Colors.black_01
 
     var iconTintColor = iconColor
 
     if (disabled) {
         textColor = disabledColor
     }
 
     val testId = if (disabled) "input_${floatingValue}_disabled" else "input_$floatingValue"
 
     Column(modifier = Modifier.
     setAutomationId(testId)
     ) {
         BasicTextField(
             value = text,
             enabled = !disabled && !readOnly,
             readOnly = readOnly,
             singleLine = true,
             textStyle = TextStyle(
                 color = textColor,
                 fontSize = 16.sp,
                 lineHeight = 24.sp,
                 fontWeight = fontWeight.value
             ),
             visualTransformation = if (secureTextEntry && !passHidden) PasswordVisualTransformation() else  visualTransformation ?: VisualTransformation.None,
             keyboardOptions = KeyboardOptions.Default.copy(keyboardType = keyboardType),
             modifier = modifier.height(size.values.height).onFocusChanged { focusState ->
                 if (!isFocused && focusState.isFocused) {
                     onFocus()
                 } else if (isFocused && !focusState.isFocused) {
                     onBlur()
                 }
                 isFocused = focusState.isFocused
             },
             onValueChange = { newValue ->
                 textFieldValueState.copy(text = newValue)
                 val stringChangedSinceLastInvocation = lastTextValue != newValue
                 lastTextValue = newValue
 
                 if (stringChangedSinceLastInvocation) {
                     onChangeText(newValue)
                 }
             },
             decorationBox = { innerTextField ->
                 // Floating Icon
                 if (floatingValue.isNotEmpty() || floatingIcon.isNotEmpty()) {
                     Box(
                         modifier = Modifier.wrapContentSize()
                             .offset(y = (-size.values.height / 2), x = (Spacing.S))
                             .background(AppTheme.current.colors.background.surface)
                             .zIndex(10f).clickable {
                                 onClickFloating.invoke()
                             }
                         ,
                     ) {
                         Row(
                             modifier = Modifier
                                 .padding(horizontal = Spacing.S),
                             verticalAlignment = Alignment.Bottom
                         ) {
                             Text(
                                 floatingValue,
                                 style = Typography.labelSMedium,
                                 color = floatingValueColor
                             )
                             if (required) {
                                 Text(
                                     "*",
                                     style = Typography.labelSMedium,
                                     color = AppTheme.current.colors.error.primary,
                                 )
                             }
                             if (floatingIcon.isNotEmpty()) {
                                 Icon(
                                     source = floatingIcon,
                                     modifier = Modifier.padding(start = Spacing.XS),
                                     size = 16.dp,
                                     color = floatingIconColor
                                 )
                             }
                         }
                     }
                 }
                 //input box wrapper
                 Box(
                     modifier = Modifier.fillMaxWidth()
                         .border(
                             1.dp,
                             getBorderColor(isFocused, error, disabled),
                             RoundedCornerShape(size.values.borderRadius)
                         ).background(bgInput, RoundedCornerShape(size.values.borderRadius)),
                     contentAlignment = Alignment.CenterStart
                 ) {
                     Row(
                         modifier = Modifier.padding(horizontal = Spacing.M),
                         verticalAlignment = Alignment.CenterVertically
                     ) {
                         //leading icon
                         if (leadingIcon.isNotEmpty()) {
                             Icon(
                                 source = leadingIcon,
                                 modifier = Modifier.padding(end = Spacing.M),
                                 size = if (size.name == InputSize.SMALL.name) 24.dp else 32.dp,
                                 color = leadingIconColor
                             )
                         }
                         Box(Modifier.weight(1f)) {
                             if (textFieldValue.text.isEmpty()) {
                                 Text(
                                     text = placeholder,
                                     style = TextStyle(
                                         fontSize = 16.sp,
                                         lineHeight = 24.sp,
                                         fontWeight = fontWeight.value
                                     ),
                                     color = placeholderColor
                                 )
                             }
                             innerTextField()
                         }
                         if (isFocused && textFieldValue.text.isNotEmpty()) {
                             Row {
                                 Spacer(Modifier.width(Spacing.XS))
                                 Icon(
                                     source = "24_navigation_close_circle_full",
                                     size = 16.dp,
                                     color = AppTheme.current.colors.text.hint,
                                     modifier = Modifier.clickable(
                                         onClick = {
                                             onChangeText("")
                                             textFieldValueState = TextFieldValue(text = "")
                                         },
                                         interactionSource = remember { MutableInteractionSource() },
                                         indication = null
                                     )
                                 )
                             }
                         }
                         if (secureTextEntry && passHidden) {
                             val hidePassword = { onRightIconPressed(); passHidden = !passHidden}
                             RenderRightIcon(loading, "24_security_eye_open", iconTintColor,
                                 hidePassword)
                         } else if (secureTextEntry && !passHidden) {
                             val showPassword = { onRightIconPressed(); passHidden = !passHidden}
                             RenderRightIcon(loading, "24_security_eye_off", iconTintColor,
                                 showPassword)
                         } else {
                             RenderRightIcon(loading, icon, iconTintColor,
                                 onRightIconPressed)
                         }
                     }
                 }
             },
         )
         // show error
         ErrorView(error, errorSpacing, hintText)
     }
 }
