 @Composable
 fun EmptyState(
     title: String = stringResource(Res.string.promotionEmptyStateTitle),
     description: String = stringResource(Res.string.promotionEmptyStateDescription),
     imageUri: String = Images.empty_state_default,
     showPromotionGuide: Boolean = false,
     modifier: Modifier = Modifier,
 ) {
     Column(
         modifier = Modifier

             modifier = Modifier.fillMaxWidth(),
             shape = RoundedCornerShape(Spacing.M)

             Column(
                 modifier = Modifier

                     .background(Colors.black_01)

             ) {
                 Image(
                     source = rememberAsyncImagePainter(imageUri),
                     modifier = Modifier.size(120.dp),
                 )
                 Text(
                     text = title,
                     color = Colors.black_17,
                     textAlign = TextAlign.Center,
                     style = Typography.headerSSemibold,
                     modifier = Modifier.padding(top = Spacing.L)
                 )
                 if (description.isNotEmpty()) {
                     AnnotatedText(
                         text = description,
                         color = Colors.black_17,

                         modifier = Modifier.padding(top = Spacing.XS)

             }
         }
