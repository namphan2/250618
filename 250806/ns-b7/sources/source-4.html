 package vn.momo.compose.payment.promotion.presentation.view.components
 
 import androidx.compose.foundation.background
 import androidx.compose.foundation.layout.Box
 import androidx.compose.foundation.layout.WindowInsets
 import androidx.compose.foundation.layout.asPaddingValues
 import androidx.compose.foundation.layout.fillMaxWidth
 import androidx.compose.foundation.layout.padding
 import androidx.compose.foundation.layout.systemBars
 import androidx.compose.runtime.Composable
 import androidx.compose.ui.Modifier
 import androidx.compose.ui.unit.dp
 import androidx.compose.ui.unit.min
 import vn.momo.compose.utils.keyboardVisibilityAsState
 import vn.momo.kits.const.AppTheme
 import vn.momo.kits.const.Colors
 import vn.momo.kits.const.Spacing
 import vn.momo.kits.modifier.conditional
 import vn.momo.kits.modifier.shadow
 
 @Composable
 fun Footer(
     withShadow: Boolean? = true,
     withBottomPadding: Boolean? = true,
     content: @Composable (() -> Unit)? = null,
 ) {
     val isKeyboardVisible = keyboardVisibilityAsState()
     val systemBarsHeight = WindowInsets.systemBars.asPaddingValues().calculateBottomPadding()
     val bottomPadding = if (isKeyboardVisible.value) {
         0.dp
     } else {
         min(systemBarsHeight, 21.dp)
     }
 
     Box(
         Modifier
             .conditional(withShadow != false) {
                 shadow(
                     Colors.black_20.copy(alpha = 0.1f),
                     borderRadius = 0.dp,
                     10f,
                     offsetX = 0.dp,
                     offsetY = (-2).dp,
                     spread = 1.dp
                 )
             }
             .background(AppTheme.current.colors.background.surface)
             .conditional(withBottomPadding != false) {
                 padding(bottom = bottomPadding)
             }
     ) {
         Box(
             Modifier.fillMaxWidth().padding(vertical = Spacing.S, horizontal = Spacing.M)
         ) {
             content?.invoke()
         }
     }
 }
