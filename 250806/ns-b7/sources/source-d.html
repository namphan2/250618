 @Composable
 fun PromotionPosSearchModal(
     viewModel: ChoosePromotionPosVM,
 ) {
     val focusRequester = remember { FocusRequester() }
     var isInitUiDone by remember { mutableStateOf(false) }
     val listState = rememberLazyListState()
     val scrollState = listState.toScrollState()
     val coroutineScope = rememberCoroutineScope()
 
     val searchState by viewModel.voucherSearchSlice.state.collectAsState()
     val keyboardController = LocalSoftwareKeyboardController.current
 
     val collectVoucherState by viewModel.collectVoucherSlice.state.collectAsState()
 
     LaunchedEffect(isInitUiDone, searchState.searchModalVisivility) {
         if (isInitUiDone && searchState.searchModalVisivility == true) {
             focusRequester.requestFocus()
         }
     }
 
     LaunchedEffect(listState.isScrollInProgress) {
         if (listState.isScrollInProgress) {
             coroutineScope.launch {
                 keyboardController?.hide()
             }
         }
     }
 
     AnimatedVisibility(
         visible = searchState.searchModalVisivility == true,
         enter = fadeIn() + slideInHorizontally(initialOffsetX = { it }),
         exit = fadeOut() + slideOutHorizontally { it }
     ) {
         Screen(
             inputSearchProps = InputSearchProps(
                 modifier = Modifier
                     .focusRequester(focusRequester)
                     .onGloballyPositioned {
                         isInitUiDone = true
                     }
                     .onFocusChanged { focusState ->
                         if (focusState.isFocused) {
                             keyboardController?.show()
                         }
                     },
                 text = viewModel.voucherSearchSlice.searchKey,
                 placeholder = stringResource(Res.string.posSearchVoucherPlaceholder1),
                 showButtonText = true,
                 icon = "",
                 buttonText = stringResource(Res.string.cancelTxt),
                 onPressButtonText = {
                     keyboardController?.hide()
                     viewModel.voucherSearchSlice.hideSearchModal()
                     viewModel.trackVoucherPOSSearchCancelButtonClicked()
                 },
                 onChangeText = {
                     viewModel.voucherSearchSlice.onSearch(it)
                 },
                 onClearPress = {
                     viewModel.voucherSearchSlice.onSearch("")
                 }),
             scrollable = false,
             headerType = HeaderType.EXTENDED,
             scrollState = scrollState
         ) {
             VoucherList(
                 vouchers = searchState.searchVouchers,
                 modifier = Modifier.padding(horizontal = Spacing.M),
                 scrollState = listState,
                 isFetching = searchState.isSearchingVoucher,
                 pullToRefreshEnabled = false,
                 emptyComponent = {
                     if (viewModel.voucherSearchSlice.searchKey.value.isEmpty()) PromotionSearchGuide()
                     else EmptyState(
                         title = stringResource(Res.string.promotionPosEmptySearchHeader),
                         description = stringResource(Res.string.promotionPosEmptySearchDesc),
                         imageUri = Images.empty_state_default,
                         showPromotionGuide = false,
                     )
                 },
                 onVoucherClick = { voucher, index ->
                     viewModel.onVoucherClick(
                         voucher,
                         PromotionSourceScreen.POS_SEARCH,
                         index
                     )
                 },
                 onVoucherCollect = { voucher, index ->
                     viewModel.onCollectVoucher(
                         PromotionSourceScreen.POS_SEARCH,
                         voucher as VoucherBase,
                         index
                     )
                 },
                 collectingVoucherId = collectVoucherState.collectingVoucherId,
                 onVoucherUseNow = { voucher, index -> viewModel.onUseNowVoucher(voucher, index) },
                 onVoucherDisplayed = { voucher, index ->
                     viewModel.onVoucherDisplayed(
                         sourceScreen = PromotionSourceScreen.POS_SEARCH,
                         voucher = voucher,
                         index = index,
                     )
                 },
                 sourceScreen = PromotionSourceScreen.POS_SEARCH,
             )
         }
     }
 }
