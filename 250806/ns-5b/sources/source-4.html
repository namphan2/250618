 @Serializable
 data class MBConfirmResponseDTO(
     val momoMsg: MoMoMsgConfirmResponseDTO? = null,
     override val result: Boolean? = false,
     override val msgType: String? = null,
     override val errorCode: Int? = null,
     override val errorDesc: String? = null,
     override val extra: JsonObject? = null,
     val errorType: String? = null,
     val popup: OverlayDialogDTO? = null,
     val informMessage: InformMessageDTO? = null,
 ) : PaymentNetWorkResponse {
     override fun toJson(): JsonElement? = ModelConvertUtils.encodeToJson(serializer(), this)
 
     fun toModel(): ConfirmResponse {
         return ConfirmResponse(
             momoMsg?.toModel(),
             extra ?: emptyJsonObject(),
             inlineMessage,
             overlayData
         )
     }
 
     private val informType: String?
         get() = errorType?.split(".")?.getOrNull(1)
 
     private val inlineMessage
         get() = errorDesc.takeIf { informType == InformType.InlineMessage.type }
 
     val overlayData: UiOverlayData?
         get() {
             val popupDirectDebit = getDirectDebitDialog(extra, errorCode)
             if (popupDirectDebit != null) {
                 return popupDirectDebit
             }
             val factory: UiOverlayDataFactory? = when (informType) {
                 InformType.Dialog.type -> PayXDialogDataFactory(popup, errorDesc, errorCode)
                 InformType.ToastMessage.type -> PayXToastMessageDataFactory(informMessage)
                 InformType.SofBottomSheet.type -> PayXSofBottomSheetDataFactory()
                 InformType.CustomBottomSheet.type -> when {
                     informMessage?.bottomSheet?.type == "VAQR" -> informMessage.bottomSheet.vaPaymentInfo?.let { vaData ->
                         if (vaData.qrCode.isNullOrEmpty() && vaData.bankAccountNo.isNullOrEmpty()) null
                         else VABottomSheetDataFactory(vaData, momoMsg)
                     }
 
                     else -> PayXCustomBottomSheetDataFactory(informMessage)
                 }
 
                 else -> null
             }
             return factory?.create()
         }
 }
