     protected fun quickActive(refId: String?, refParams: JsonObject?) {
         trackPressOpenPayLater(state.value.version)
         if (refId.isNullOrEmpty()) return
         val extraParams = when (refId) {
             ERefIdRecommendMoneyPay.QuickActivePayLater.stringValue -> buildJsonObject {
                 put("fromFeature", "sof_banner_quick_activation")
                 put("featureName", "TTAT")
             }
 
             else -> emptyJsonObject()
         }
         val params =
             if (refParams.isNullOrEmpty()) extraParams else extraParams + refParams
         navigator.navigateFeatureCode(
             refId, JsonObject(params), JsonObject.serializer()
         ).onEach { response ->
             response?.let { data ->
                 val approvalResult = data["approvalResult"]?.toString() ?: ""
                 if (listOf(
                         EQuickActiveStatus.Active, EQuickActiveStatus.Reject
                     ).contains(
                         EQuickActiveStatus.values()
                             .firstOrNull { status -> status.stringValue == approvalResult }
                             ?: EQuickActiveStatus.Unknown)
                 ) {
                     if (state.value.version.code == 1) {
                         dispatch(Action.SetMoneyPayCTA(null))
                         data["selectedSource"]?.jsonObject?.get("sourceToken")?.jsonPrimitive?.contentOrNull?.let { sourceToken ->
                             findSof(sourceToken, sofID = null)?.let {
                                 changeSofIfNotSelected(
                                     it,

                     } else {
                         dispatch(Action.SetPromoteSof(null))
                         (data["selectedSource"] as? JsonObject)?.let { rawSof ->
                             ModelConvertUtils.decode(
                                 SofDTO.serializer(), rawSof

                         }?.toModel()?.also { newActive ->

         }.launchIn(scope)

     private fun getMoneyPayV1(selectedSof: SOFItem?, purchaseId: String?) {
         recommendMoneyPayRepository.getRecommendMoneyPay().onEach { recommendData ->
             when (recommendData.type) {
                 RecommendType.QuickActiveMoneySource -> {
                     val rmpCTA = recommendData.generateCTA()
                     trackQuickActive(UiVersion())
                     dispatch(
                         Action.SetMoneyPayCTA(
                             rmpCTA.copy(
                                 btnTitle = rmpCTA.btnTitle
                             )
                         )
                     )
                 }
 
                 RecommendType.PromoteMoneySource -> {
                     val rmpCTA = recommendData.generateCTA()
                     val foundSof = state.value.list.find {
                         it.isMatchSource(rmpCTA.sourceToken, rmpCTA.sourceToken)
                     }
                     if (foundSof != null && (foundSof.available || foundSof.canAction)) {
                         FirebaseTrace.track(
                             LoadPromoteSofBanner()
                         )
                         val defaultTitle = getString(
                             Res.string.selectPromoteSource
                         )
                         dispatch(
                             Action.SetMoneyPayCTA(
                                 rmpCTA.copy(
                                     btnTitle = defaultTitle
                                 )
                             )
                         )
                     }
                 }
 
                 RecommendType.Installment -> {
                     onRecommendInstallment(recommendData, selectedSof)
                 }
 
                 RecommendType.NavigateActionLaw40 -> {
                     FirebaseTrace.track(
                         LoadLaw40Banner(
                             refId = recommendData.recommendMoneyInfo?.refId,
                             type = recommendData.type.stringValue,
                             orderId = purchaseId
                         )
                     )
                     dispatch(Action.SetMoneyPayCTA(recommendData.generateCTA()))
                 }
 
                 null, RecommendType.None -> {
                     dispatch(Action.ResetPromoteSof)
                 }
             }
         }.catch { err -> //TODO: track log
             println(err)
         }.launchIn(scope)
     }
