 @Composable
 fun BoxComponent(
     modifier: Modifier = Modifier,
     displayName: String,
     description: String,
     isSelected: Boolean = true,
     isDisable: Boolean = false,
     onClick: () -> Unit
 ) {
     val backgroundColor = if (isDisable) {
         Colors.black_04
     } else {
         Colors.black_01
     }
     val borderColor = if (isSelected) {
         Colors.pink_03
     } else {
         Colors.black_04
     }
     val textColor = if (isDisable) {
         Colors.black_09
     } else {
         Colors.black_17
     }
 
     Box(
         modifier = modifier
             .clip(RoundedCornerShape(Radius.S))
             .background(backgroundColor)
             .border(
                 width = 1.dp,
                 color = borderColor,
                 shape = RoundedCornerShape(Radius.S)
             )
 
             .padding(horizontal = Spacing.S)
             .clickable(enabled = !isDisable && !isSelected) { onClick() },
 
 
         ) {
         Column(
             modifier = Modifier
                 .fillMaxWidth().height(ROW_HEIGHT),
             verticalArrangement = Arrangement.Center
         ) {
 
             Text(
                 text = displayName,
                 color = textColor,
                 style = Typography.actionXsBold
             )
             Text(
                 text = description,
                 color = textColor,
                 style = Typography.descriptionXsRegular
             )
 
 
         }
         if (isSelected) {
             Image(
                 source = Images.ic_selected_sof,
                 modifier = Modifier
                     .size(20.dp)
                     .absoluteOffset(x = 8.dp, y = 0.dp)
                     .align(Alignment.TopEnd)
                     .height(20.dp),
 
                 options = Options(contentScale = ContentScale.Fit)
             )
         }
     }
 }
 
 @Composable
 fun CollapsibleBoxGrid(
     boxes: List<InstallmentPackageState>,
     expanded: Boolean = false,
     onPressItem: (item: InstallmentPackageState) -> Unit,
     onPressToggle: () -> Unit = {},
 ) {
     val rotationState by animateFloatAsState(
         targetValue = if (expanded) 180f else 0f, label = "InstallmentRotate"
     )
     val displayBoxes = if (boxes.isNotEmpty()) boxes.chunked(2) else emptyList()
     val height by animateDpAsState(
         targetValue = if (expanded) TWO_ROW_HEIGHT else ONE_ROW_HEIGHT,
         label = "Height"
     )
     Column(
         modifier = Modifier
             .fillMaxWidth(),
         horizontalAlignment = Alignment.CenterHorizontally
     ) {
         Column(
             modifier = Modifier
                 .fillMaxWidth()
                 .height(height)
         ) {
             displayBoxes.forEach { row ->
                 Row(
                     modifier = Modifier
                         .fillMaxWidth(),
                     horizontalArrangement = Arrangement.spacedBy(5.dp)
                 ) {
                     row.forEach { item ->
                         BoxComponent(
                             displayName = item.displayName,
                             description = item.description,
                             isDisable = item.status == InstallmentStatus.Disable,
                             isSelected = item.status == InstallmentStatus.ActiveAndSelected,
                             modifier = Modifier.weight(1f),
                             onClick = { onPressItem(item) }
                         )
                     }
                 }
                 Spacer(modifier = Modifier.height(SPACING_ROW))
             }
         }
 
 
         // Display the button only if there are more than 2 boxes
         if (boxes.size > 2) {
             Row(
                 modifier = Modifier
                     .padding(top = 8.dp)
                     .clickable {
 //                        expanded = !expanded
                         onPressToggle()
                     },
                 horizontalArrangement = Arrangement.spacedBy(5.dp),
                 verticalAlignment = Alignment.CenterVertically
             ) {
                 Text(
                     text = if (expanded) stringResource(Res.string.hidePackageInstallment) else stringResource(
                         Res.string.showPackageInstallment
                     ),
                     color = Colors.pink_03,
                     style = Typography.actionXsBold,
                 )
                 Icon(
                     source = "arrow_chevron_down",
                     size = 16.dp,
                     color = Colors.pink_03,
                     modifier = Modifier
                         .alpha(ContentAlpha.high)
                         .rotate(rotationState)
                         .padding(start = 4.dp),
                 )
             }
         }
     }
 }
