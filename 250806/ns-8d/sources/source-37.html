 @Composable
 fun SourceOfFunds(
     sofListState: SofState,
     loadingState: Boolean,
     bundleSof: SOFItem?,
     onClickHideBalance: () -> Unit,
     isAllowAddSOF: Boolean = true,
     isAllowAddCard: Boolean = true,
     onClickShowAll: () -> Unit = { },
     onAddSof: () -> Unit = {},
     addSofItem: @Composable (() -> Unit)? = null,
     onAddCard: () -> Unit = {},
     onSelectSof: (SOFItem) -> Unit = {},
     onLoaded: () -> Unit
 ) {
     val screenDimension = getScreenDimensions().screenWidth
     val itemWidth = screenDimension * 0.35
     val canShowIcon by remember(sofListState.list) {
         derivedStateOf { sofListState.list.any { it.shortDesc.isCurrency() } }
     }
 
     LaunchedEffect(Unit) {
         onLoaded()
     }
 
     Column {
         Row(
             Modifier.fillMaxWidth(),
             horizontalArrangement = Arrangement.SpaceBetween,
         ) {
             Row(
                 verticalAlignment = Alignment.CenterVertically,
                 modifier = Modifier.noFeedbackClickable { onClickHideBalance() }) {
                 Text(
                     text = stringResource(Res.string.sofTitle),
                     style = Typography.headerDefaultBold,
                     color = Colors.black_17
                 )
                 Spacer(modifier = Modifier.width(8.dp))
                 if (canShowIcon) {
                     Icon(
                         source = if (sofListState.isHideBalance) Images.ic_eye_close else Images.ic_eye_open,
                         size = 24.dp,
                         modifier = Modifier.semantics {
                             testTag = "icon_eye"
                             contentDescription = "icon_eye"
                         }.clickable {
                             onClickHideBalance()
                         }
                     )
                 }
             }
             Text(
                 text = stringResource(Res.string.viewAll),
                 style = Typography.actionSBold,
                 color = Colors.pink_03,
                 accessibilityId = "linkFunds",
                 modifier = Modifier
                     .noFeedbackClickable(
                         sofListState.list.isNotEmpty(),
                         onClick = debounceCompose(debounceTime = 500L) {
                             onClickShowAll()
                         })
             )
         }
         Spacer(modifier = Modifier.height(12.dp))
         if (loadingState) {
             Row(
                 horizontalArrangement = Arrangement.spacedBy(8.dp),
                 modifier = Modifier.fillMaxWidth()
             ) {
                 repeat(3) {
                     Box(
                         modifier = Modifier.size(width = itemWidth.dp, height = 48.dp)
                             .clip(shape = RoundedCornerShape(12.dp))
                             .skeletonEffect(1500)
                     )
                 }
             }
         } else if (sofListState.list.isNotEmpty()) {
             MoneySourceCarousel(
                 sofListState.list,
                 isHideBalance = sofListState.isHideBalance,
                 bundleSof,
                 tailView = {
                     if (isAllowAddSOF) {
                         addSofItem?.invoke() ?: CarouselItem.AddBankSof(
                             modifier = Modifier.fillMaxHeight(),
                             onPress = onAddSof
                         )
                     }
                     if (isAllowAddCard) {
                         CarouselItem.AddInternationalCard(
                             modifier = Modifier.fillMaxHeight(),
                             onPress = onAddCard
                         )
                     }
                 },
                 onPressItem = onSelectSof
             )
         }
     }
 }
