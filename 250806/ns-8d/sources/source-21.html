 package vn.momo.compose.payment.payx.presentation.view.components
 
 import androidx.compose.animation.AnimatedVisibility
 import androidx.compose.foundation.layout.Column
 
 import androidx.compose.runtime.Composable
 import androidx.compose.runtime.getValue
 import androidx.compose.runtime.mutableStateOf
 import androidx.compose.runtime.remember
 import androidx.compose.runtime.setValue
 import androidx.compose.ui.Modifier
 import vn.momo.kits.components.Information
 import vn.momo.kits.components.InformationState
 
 @Composable
 fun MessageInformation(
     title: String? = null,
     description: String = "Description",
     state: InformationState = InformationState.DEFAULT,
     image: String? = null,
     onPressCTA: (() -> Unit)? = null,
     showIcon: Boolean = true,
     CTA: String? = null,
     showIconClose: Boolean = true,
     onPressClose: (() -> Unit)? = null,
     modifier: Modifier = Modifier
 ) {
     var messageState by remember(description) { mutableStateOf(description) }
 
     AnimatedVisibility(visible = messageState.isNotEmpty()) {
         Column(modifier = modifier) {
             Information(
                 title,
                 description = messageState,
                 state,
                 image,
                 onPressCTA,
                 showIcon,
                 CTA,
                 showIconClose,
                 onPressClose = {
                     messageState = ""
                     if (onPressClose != null) {
                         onPressClose()
                     }
                 }
             )
         }
     }
 }
