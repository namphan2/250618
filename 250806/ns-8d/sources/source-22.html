 import vn.momo.compose.utils.getScreenDimensions
 
 const val ITEM_PADDING = 8
 
 @Composable
 fun MoneySourceCarousel(
     data: List<SOFItem>,
     isHideBalance: Boolean,
     selected: SOFItem?,
     onPressItem: (SOFItem) -> Unit,
     tailView: @Composable () -> Unit = {},
 ) {
     val horizontalScrollState = rememberScrollState()
     val density = LocalDensity.current
     val screenDimension = getScreenDimensions().screenWidth
     val itemWidth = screenDimension * 0.35
 
     val index by remember(data, selected) {
         derivedStateOf {
             data.indexOfFirst {
                 it.isMatchSource(selected?.sourceToken, selected?.ID)
             }
         }
     }
 
     LaunchedEffect(index) {
         if (index >= 0) {
             horizontalScrollState.animateScrollTo(with(density) { ((itemWidth + ITEM_PADDING).dp * index).toPx() }.toInt())
         }
     }
 
     Row(
         horizontalArrangement = Arrangement.spacedBy(8.dp),
         modifier = Modifier
             .fillMaxWidth()
             .height(IntrinsicSize.Min)
             .horizontalScroll(state = horizontalScrollState)
             .semantics { contentDescription = "fundHorizontalScrollView" }
     ) {
         data.forEach {
             key(it.uId) {
                 MoneySourceItem(data = it,
                     isHideBalance = isHideBalance,
                     isSelected = it.isMatchSource(selected?.sourceToken, selected?.ID),
                     onPress = { onPressItem(it) })
             }
         }
         tailView()
     }
 }
