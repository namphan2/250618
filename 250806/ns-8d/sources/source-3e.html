 @Composable
 fun TransactionDetail(
     loading: Boolean,
     invoices: List<InvoiceItem>,
     hasAppliedPromotion: Boolean = true,
     initialExpanded: Boolean = true,
     shouldShowTid: Boolean,
     hideAmount: Boolean = false,
     onChangeExpandState: (String) -> Unit = {},
     onAnimationEnd: () -> Unit = {},
     disableExpand: Boolean = false,
     expandIcon: @Composable () -> Unit = {
         Icon(
             source = "arrow_chevron_down",
             size = 24.dp,
             modifier = Modifier.alpha(ContentAlpha.medium)
         )
     },
     footer: @Composable ((InvoiceItem) -> Unit),
 ) {
     Column(
         modifier = Modifier.fillMaxWidth()
             .animateContentSize(
                 animationSpec = linearTween(delayMillis = 0),
                 finishedListener = { _, _ -> onAnimationEnd() }
             ),
         verticalArrangement = Arrangement.spacedBy(12.dp)
     ) {
         Text(
             text = stringResource(Res.string.transactionDetail),
             style = Typography.headerDefaultBold,
             color = Colors.black_17
         )
 
         if (loading) {
             Box(
                 modifier = Modifier.fillMaxWidth().height(340.dp)
                     .clip(shape = RoundedCornerShape(12.dp)).skeletonEffect(1500)
             )
         } else {
             invoices.forEach { item ->
                 key(item.itemId) {
                     val isPayForService = item.serviceName.isNotBlank()
                     ExpandableCard(
                         title = if (isPayForService) stringResource(Res.string.payForService) else "",
                         contentModifier = Modifier.padding(horizontal = 12.dp, vertical = 8.dp),
                         defaultState = initialExpanded,
                         header = {
                             if (isPayForService) {
                                 InvoiceInfoView(
                                     data = listOf(
                                         InvoiceInfo(
                                             title = stringResource(Res.string.service),
                                             value = item.serviceName,
                                             key = "serviceName"
                                         )
                                     )
                                 )
                             }
                         },
                         content = {
                             val invoiceInfos = buildList {
                                 addAll(item.info)
                                 if (shouldShowTid && item.visibleTid) {
                                     add(
                                         InvoiceInfo(
                                             title = stringResource(Res.string.transaction_id),
                                             value = item.paymentTid.toString(),
                                             key = "tid"
                                         )
                                     )
                                 }
                                 if (item.visibleFee == true && !hideAmount) {
                                     add(
                                         InvoiceInfo(
                                             title = stringResource(Res.string.invoiceFee),
                                             value = if (item.fee > 0) item.fee.toDouble()
                                                 .formatCurrency()
                                             else stringResource(Res.string.free),
                                             key = "fee"
                                         )
                                     )
                                 }
                             }
 
                             InvoiceInfoView(data = invoiceInfos, hideAmount = hideAmount)
                             if (!hideAmount) {
                                 Divider(color = Colors.black_04, thickness = 1.dp)
                             }
                         },
                         icon = expandIcon,
                         disableExpand = disableExpand,
                         footer = { footer(item) },
                         onChange = { newState ->
                             onChangeExpandState(if (newState) "expand" else "collapse")
                         })
                 }
             }
         }
     }
 }
