 @Composable
 fun ExpandableCard(
     modifier: Modifier? = null,
     defaultState: Boolean = true,
     title: String,
     description: String? = null,
     logo: String? = null,
     expandedBorderColor: Color = Colors.black_04,
     collapsedBorderColor: Color = Colors.black_04,
     cardTitleColor: Color = AppTheme.current.colors.background.default,
     header: @Composable (() -> Unit)? = null,
     content: @Composable () -> Unit,
     footer: @Composable ((state: Boolean) -> Unit)? = null,
     onChange: ((newState: Boolean) -> Unit)? = null,
     headerSeparator: Boolean = false,
     contentModifier: Modifier = Modifier,
     icon: @Composable (() -> Unit)? = null,
     disableExpand: Boolean? = false,
 ) {
     var expandedState by remember { mutableStateOf(defaultState) }
 
     LaunchedEffect(defaultState) {
         expandedState = defaultState
     }
 
     val containerModifier = modifier ?: Modifier.border(
         BorderStroke(
             1.dp,
             if (expandedState) expandedBorderColor else collapsedBorderColor
         ), RoundedCornerShape(12.dp)
     )
 
     Column(
         modifier = Modifier
             .fillMaxWidth()
             .then(containerModifier)
     ) {
         if (title.isNotBlank()) {
             CardTitle(
                 expandedState = expandedState,
                 title = title,
                 description = description,
                 icon = if (disableExpand == false) icon else null,
                 logo = logo,
                 cardTitleColor = cardTitleColor,
                 onPress = {
                     if (disableExpand == false) {
                         onChange?.invoke(!expandedState)
                         expandedState = !expandedState
                     }
                 })
         }
 
         Column(
             modifier = contentModifier.fillMaxWidth()
         ) {
             header?.let { header() }
             PayXAnimatedVisibility(visible = expandedState) {
                 Column {
                     content()
                 }
             }
 
             footer?.let { footer(expandedState) }
         }
     }
 }
 
 @Composable
 fun CardTitle(
     expandedState: Boolean,
     title: String,
     onPress: () -> Unit,
     description: String? = null,
     logo: String? = null,
     cardTitleColor: Color = AppTheme.current.colors.background.default,
     icon: @Composable (() -> Unit)? = null,
 ) {
     val rotationState by animateFloatAsState(
         targetValue = if (expandedState) 180f else 0f, label = "CardTitleRotate",
         animationSpec = linearTween()
     )
     Row(
         verticalAlignment = Alignment.CenterVertically,
         modifier = Modifier
             .clip(shape = RoundedCornerShape(12.dp, 12.dp, 0.dp, 0.dp))
             .background(cardTitleColor)
             .padding(12.dp)
             .semantics {
                 testTag = "collapse_info_detail"
                 contentDescription = "collapse_info_detail"
             }
             .noFeedbackClickable { onPress() },
         horizontalArrangement = Arrangement.spacedBy(8.dp)
     ) {
         logo?.let {
             Image(source = logo, modifier = Modifier.size(40.dp))
         }
         Column(modifier = Modifier.weight(1f)) {
             Text(
                 text = title,
                 style = Typography.actionSBold,
                 color = Colors.black_17,
                 modifier = Modifier.fillMaxWidth(),
                 maxLines = 2,
                 overflow = TextOverflow.Ellipsis
             )
             description?.let {
                 Text(
                     text = description,
                     style = Typography.descriptionDefaultRegular,
                     color = Colors.black_12,
                     modifier = Modifier.fillMaxWidth(),
                     maxLines = 1,
                     overflow = TextOverflow.Ellipsis
                 )
             }
         }
         icon?.let {
             Box(modifier = Modifier.rotate(rotationState)) {
                 it.invoke()
             }
         }
     }
