 @Composable
 fun GamificationBlock(
     dispatcher: (TransactionResultAction) -> Unit,
     ticket: TransactionTicketItem?,
     data: GamificationResponse,
     nextBestActionContent: @Composable () -> Unit = {},
     hasNextBestAction: Boolean = false,
     show: Boolean = false,
 ) {
     var visibleContent by remember { mutableStateOf(false) }
     var size by remember { mutableStateOf(IntSize.Zero) }
     val density = LocalDensity.current
 
     LaunchedEffect(ticket, data) {
         if (ticket?.transaction?.isSuccess == true && data.isEmpty) {
             ticket.transaction?.let {
                 delay(200)
                 dispatcher(TransactionResultAction.GetGamificationData(it.transactionId))
             }
         }
 
         if ((data.isNotEmpty || hasNextBestAction) && !visibleContent && show) {
             visibleContent = true
         }
     }
 
 
     val scrollState = rememberScrollState()
 
     Column(Modifier.fillMaxWidth().onSizeChanged {
         size = it
     }) {
         AnimatedVisibility(
             visible = false, enter = slideInVertically(
                 initialOffsetY = { fullHeight -> fullHeight },
                 animationSpec = tween(durationMillis = 500, easing = LinearOutSlowInEasing)
             )
         ) {
             Column {
                 Text(
                     data.headingText?.takeIf { it.isNotBlank() } ?: stringResource(Res.string.titleGameCampaign),
                     style = Typography.actionDefaultBold,
                     modifier = Modifier.padding(start = Spacing.M))
                 Spacer(Modifier.height(Spacing.M))
             }
         }
 
         AnimatedVisibility(
             visible = visibleContent,
             enter = slideInHorizontally {
                 with(density) { 500.dp.roundToPx() }
             } + fadeIn(
                 initialAlpha = 0.3f
             ),
         ) {
 //            Spacer(Modifier.width(Spacing.M))
             Row(
                 Modifier.horizontalScroll(scrollState),
                 horizontalArrangement =  Arrangement.spacedBy(Spacing.M)
             ) {
                 Spacer(Modifier.width(Spacing.XXS))
                 nextBestActionContent()
                 data.result?.onEach {
                     GamificationCard(it, size, dispatcher)
 //                    Spacer(Modifier.width(Spacing.M))
                 }
                 Spacer(Modifier.width(Spacing.XXS))
             }
         }
 
         if (visibleContent) {
             Spacer(Modifier.height(Spacing.M))
         }
     }
 }
 
 @Composable
 fun GamificationCard(
     data: GamificationResultData, parentSize: IntSize, dispatcher: (TransactionResultAction) -> Unit
 ) {
     val density = LocalDensity.current
     val imageBitmap = getImageBimap("img_background_gamification.webp")
     val shape = RoundedCornerShape(CornerSize(Radius.M))
     val imageBackgroundColor = Color(0xffe8e8e8)
 
     val width = if (parentSize.width != 0) {
         with(density) {
             (parentSize.width * 2 / 3).toDp()
         }
     } else {
         250.dp
     }
 
     Row(
         Modifier.widthIn(min = width, max = width).height(72.dp).clip(shape).background(Color.White)
             .drawBehind {
                 drawImage(
                     image = imageBitmap, topLeft = Offset(
                         x = size.width - imageBitmap.width.toFloat(), y = 0f
                     )
                 )
             }.padding(Spacing.M)
     ) {
 
         val painter = when {
             data.iconUrl?.contains("https://") == true -> rememberImagePainter(data.iconUrl)
             else -> getResource("ic_common_logo_momo.webp")
         }
 
         Image(
             painter,
             contentDescription = null,
             Modifier.size(48.dp).clip(RoundedCornerShape(CornerSize(Radius.S)))
                 .background(imageBackgroundColor).padding(0.dp)
         )
 
         Spacer(Modifier.width(Spacing.S))
 
         Column(
             modifier = Modifier.weight(7f).fillMaxHeight().padding(Spacing.XS),
             verticalArrangement = Arrangement.SpaceBetween
         ) {
             data.title?.let {
                 Text(
                     it, style = Typography.headerSSemibold, maxLines = 1, overflow = TextOverflow.Ellipsis
                 )
             }
 
             data.body?.let {
                 Spacer(Modifier.weight(1F))
                 Text(
                     it,
                     style = Typography.descriptionDefaultRegular,
                     maxLines = 1,
                     overflow = TextOverflow.Ellipsis
                 )
             }
         }
 
         Spacer(Modifier.weight(1F))
 
         data.ctaText?.let {
             Row(Modifier.weight(5f)) {
                 Column(
                     Modifier.fillMaxHeight(),
                     verticalArrangement = Arrangement.Center,
                 ) {
                     Button(
                         title = it,
                         type = ButtonType.OUTLINE,
                         onClick = {
                             dispatcher(TransactionResultAction.OnPressGamificationCard(data))
                         },
                         size = Size.SMALL,
                         isFull = false,
                     )
                 }
             }
         }
     }
